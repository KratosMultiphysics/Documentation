312  double Compute3dArea(array_1d<double, 3> PointA, array_1d<double, 3> PointB, array_1d<double, 3> PointC)
313  {
314  double a = MathUtils<double>::Norm3(PointA - PointB);
315  double b = MathUtils<double>::Norm3(PointB - PointC);
316  double c = MathUtils<double>::Norm3(PointC - PointA);
317  double s = (a + b + c) / 2.0;
318  double Area = std::sqrt(s * (s - a) * (s - b) * (s - c));
319  return Area;
320  }
