471  const TContainer& rStructure,
472  const int Rank = -1

474  {

478  using PointTypePointer = typename PointType::Pointer;
479  using PointVector = std::vector<PointTypePointer>;
480 
481  // Creating the points
482  PointVector points;

484  points.reserve(structure_size);
485  const auto it_begin = rStructure.begin();
486  if constexpr (std::is_same<TContainer, ModelPart::NodesContainerType>::value) { // For nodes
487  const bool serial_check = (Rank < 0 || !it_begin->SolutionStepsDataHas(PARTITION_INDEX)) ? true : false;
488  if (serial_check) { // Checking if Rank is defined
489  for (std::size_t i = 0; i < structure_size; ++i) {
490  auto it = it_begin + i;
491  points.push_back(PointTypePointer(new PointType(*(it.base()))));
492  }
493  } else { // When rank is defined nodes are added only if local
494  for (std::size_t i = 0; i < structure_size; ++i) {
495  auto it = it_begin + i;
496  if ( it->FastGetSolutionStepValue(PARTITION_INDEX) == Rank) {
497  points.push_back(PointTypePointer(new PointType(*(it.base()))));
498  }
499  }
500  // Reduce size
501  points.shrink_to_fit();
502  }
503  } else { // For other entities
504  for (std::size_t i = 0; i < structure_size; ++i) {
505  auto it = it_begin + i;
506  points.push_back(PointTypePointer(new PointType(*(it.base()))));
507  }
508  }

525  template<class TContainer, class TSpatialContainer, class TResultType>
526  static void ParallelSearch(
527  const TContainer& rInput,
528  const RadiusArrayType& rRadius,
529  TSpatialContainer& rSearch,
530  TResultType& rResults,
531  VectorDistanceType& rResultsDistance,
532  const int AllocationSize = 1000
533  )
534  {
535  // Some definitions
536  using ObjectType = typename TContainer::value_type;
537  using PointType = PointObject<ObjectType>;
538  using PointTypePointer = typename PointType::Pointer;
539  using PointVector = std::vector<PointTypePointer>;
540  using DistanceVector = std::vector<double>;
541  const std::size_t input_size = rInput.size();
542 
543  // Performing search
544  IndexPartition<std::size_t>(input_size).for_each([&](std::size_t i) {
545  auto it = rInput.begin() + i;
546  PointType aux_point(*(it.base()));
547  PointVector results(AllocationSize);
548  DistanceVector results_distances(AllocationSize);
549  const std::size_t number_of_results = rSearch.SearchInRadius(aux_point, rRadius[i], results.begin(), results_distances.begin(), AllocationSize);
550  if (number_of_results > 0) {
551  auto& r_results = rResults[i];
552  auto& r_results_distance = rResultsDistance[i];
553  r_results.reserve(number_of_results);
554  r_results_distance.reserve(number_of_results);
555  for (std::size_t j = 0; j < number_of_results; ++j) {
556  auto p_point = results[j];
557  r_results.push_back(p_point->pGetObject());
558  r_results_distance.push_back(results_distances[j]);
559  }
560  }
561  });

582  template<typename TPointIteratorType>
583  static std::vector<int> SynchronizePoints(
584  TPointIteratorType itPointBegin,
585  TPointIteratorType itPointEnd,
586  std::vector<double>& rAllPointsCoordinates,
587  std::vector<SignedIndexType>& rAllPointsIds,
588  const DataCommunicator& rDataCommunicator,
589  const int NumberOfPoints,
590  const int TotalNumberOfPoints
591  )
592  {
593  // Get the World Size and rank in MPI
594  const int world_size = rDataCommunicator.Size();
595  const int rank = rDataCommunicator.Rank();
596 
597  // Getting global number of points
598  std::vector<int> points_per_partition(world_size);
599  std::vector<int> send_points_per_partition(1, NumberOfPoints);
600  rDataCommunicator.AllGather(send_points_per_partition, points_per_partition);
601  std::size_t initial_id = 0;
602  if constexpr (!std::is_same<TPointIteratorType, ModelPart::NodeIterator>::value || std::is_same<TPointIteratorType, ModelPart::NodeConstantIterator>::value) {
603  initial_id = std::accumulate(points_per_partition.begin(), points_per_partition.begin() + rank, 0);
604  }
605 
606  // Initialize and resize vectors
607  if (rAllPointsCoordinates.size() != static_cast<unsigned int>(TotalNumberOfPoints * 3)) {
608  rAllPointsCoordinates.resize(TotalNumberOfPoints * 3);
609  }
610  if (rAllPointsIds.size() != static_cast<unsigned int>(TotalNumberOfPoints)) {
611  rAllPointsIds.resize(TotalNumberOfPoints);
612  }
613  std::vector<int> recv_sizes(world_size, 0);
614 
615  // Auxiliary variables
616  std::size_t counter = 0;
617  array_1d<double, 3> coordinates;
618  unsigned int i_coord;
619 
620  // If distributed
621  if (rDataCommunicator.IsDistributed()) {
622  // Initialize local points coordinates
623  std::vector<double> send_points_coordinates(NumberOfPoints * 3);
624  std::vector<IndexType> send_points_ids(NumberOfPoints);
625  for (auto it_point = itPointBegin ; it_point != itPointEnd ; ++it_point) {
626  // In case of considering nodes
627  if constexpr (std::is_same<TPointIteratorType, ModelPart::NodeIterator>::value || std::is_same<TPointIteratorType, ModelPart::NodeConstantIterator>::value) {
628  if (it_point->FastGetSolutionStepValue(PARTITION_INDEX) != rank) {
629  continue; // Skip if not local
630  }
631  }
632  noalias(coordinates) = it_point->Coordinates();
633  if constexpr (std::is_same<TPointIteratorType, ModelPart::NodeIterator>::value || std::is_same<TPointIteratorType, ModelPart::NodeConstantIterator>::value) {
634  // Verify that Index is not bigger than the maximum positive values of SignedIndexType
635  KRATOS_DEBUG_ERROR_IF(it_point->Id() > static_cast<IndexType>(std::numeric_limits<SignedIndexType>::max())) << "Index is bigger than the maximum positive value of SignedIndexType" << std::endl;
636  send_points_ids[counter] = it_point->Id();
637  } else {
638  // Verify that Index is not bigger than the maximum positive values of SignedIndexType
639  KRATOS_DEBUG_ERROR_IF(initial_id + counter > static_cast<IndexType>(std::numeric_limits<SignedIndexType>::max())) << "Index is bigger than the maximum positive value of SignedIndexType" << std::endl;
640  send_points_ids[counter] = initial_id + counter;
641  }
642  for (i_coord = 0; i_coord < 3; ++i_coord) {
643  send_points_coordinates[3 * counter + i_coord] = coordinates[i_coord];
644  }
645  ++counter;
646  }
647 
648  /* Sync coordinates */
649 
650  // Generate vectors with sizes for AllGatherv
651  for (int i_rank = 0; i_rank < world_size; ++i_rank) {
652  recv_sizes[i_rank] = 3 * points_per_partition[i_rank];
653  }
654  std::vector<int> recv_offsets(world_size, 0);
655  for (int i_rank = 1; i_rank < world_size; ++i_rank) {
656  recv_offsets[i_rank] = recv_offsets[i_rank - 1] + recv_sizes[i_rank - 1];
657  }
658 
659  // Invoke AllGatherv
660  rDataCommunicator.AllGatherv(send_points_coordinates, rAllPointsCoordinates, recv_sizes, recv_offsets);
661 
662  /* Sync Ids */
663 
664  // Generate vectors with sizes for AllGatherv
665  for (int i_rank = 0; i_rank < world_size; ++i_rank) {
666  recv_sizes[i_rank] = points_per_partition[i_rank];
667  }
668  for (int i_rank = 1; i_rank < world_size; ++i_rank) {
669  recv_offsets[i_rank] = recv_offsets[i_rank - 1] + recv_sizes[i_rank - 1];
670  }
671 
672  // Invoke AllGatherv
673  std::vector<IndexType> all_points_pure_ids(TotalNumberOfPoints);
674  rDataCommunicator.AllGatherv(send_points_ids, all_points_pure_ids, recv_sizes, recv_offsets);
675 
676  // Copy values
677  IndexPartition<std::size_t>(TotalNumberOfPoints).for_each([&all_points_pure_ids, &rAllPointsIds](std::size_t i) {
678  rAllPointsIds[i] = all_points_pure_ids[i];
679  });
680  } else { // Serial
681  // Assign values
682  for (auto it_point = itPointBegin ; it_point != itPointEnd ; ++it_point) {
683  noalias(coordinates) = it_point->Coordinates();
684  if constexpr (std::is_same<TPointIteratorType, ModelPart::NodeIterator>::value || std::is_same<TPointIteratorType, ModelPart::NodeConstantIterator>::value) {
685  // Verify that Index is not bigger than the maximum positive values of SignedIndexType
686  KRATOS_DEBUG_ERROR_IF(it_point->Id() > static_cast<IndexType>(std::numeric_limits<SignedIndexType>::max())) << "Index is bigger than the maximum positive value of SignedIndexType" << std::endl;
687  rAllPointsIds[counter] = it_point->Id();
688  } else {
689  // Verify that Index is not bigger than the maximum positive values of SignedIndexType
690  KRATOS_DEBUG_ERROR_IF(initial_id + counter > static_cast<IndexType>(std::numeric_limits<SignedIndexType>::max())) << "Index is bigger than the maximum positive value of SignedIndexType" << std::endl;
691  rAllPointsIds[counter] = initial_id + counter;
692  }
693  for (i_coord = 0; i_coord < 3; ++i_coord) {
694  rAllPointsCoordinates[3 * counter + i_coord] = coordinates[i_coord];
695  }
696  ++counter;
697  }
698  }
699 
700  // Return the recv sizes
701  return recv_sizes;
702  }

968  inside_ranks.end()

993  rSearchInfo.Shrink();
