146 #ifdef COROT_POLAR_DECOMPOSITION
147 
148  // reference coordinate system
149  ShellT3_LocalCoordinateSystem a(CreateReferenceCoordinateSystem());
150 
151  // current coordinate system using the 1-2 side alignement
152  ShellT3_LocalCoordinateSystem b(geom[0], geom[1], geom[2]);
153 
154  double aX1 = a.X1();
155  double aY1 = a.Y1();
156  double bX1 = b.X1();
157  double bY1 = b.Y1();
158  double aX2 = a.X2();
159  double aY2 = a.Y2();
160  double bX2 = b.X2();
161  double bY2 = b.Y2();
162  double aX3 = a.X3();
163  double aY3 = a.Y3();
164  double bX3 = b.X3();
165  double bY3 = b.Y3();
166 
167  // now we are in the local coordinate systems (reference and current), i.e. we are looking in the local Z direction
168  // which is the same for both coordinate systems.
169  // now we can compute the 2D deformation gradient between the 2 configurations, at the element center.
170 
171  double C1 = 1/(aX1*aY2 - aX2*aY1 - aX1*aY3 + aX3*aY1 + aX2*aY3 - aX3*aY2);
172  double f11 = C1*(aY1 - aY3)*(bX1 - bX2) - C1*(aY1 - aY2)*(bX1 - bX3);
173  double f12 = C1*(aX1 - aX2)*(bX1 - bX3) - C1*(aX1 - aX3)*(bX1 - bX2);
174  double f21 = C1*(aY1 - aY3)*(bY1 - bY2) - C1*(aY1 - aY2)*(bY1 - bY3);
175  double f22 = C1*(aX1 - aX2)*(bY1 - bY3) - C1*(aX1 - aX3)*(bY1 - bY2);
176 
177  // now we can extrapolate the rotation angle that makes this deformation gradient symmetric.
178  // F = R*U -> find R such that R'*F = U
179  double alpha = std::atan2(f21 - f12, f11 + f22);
180 
181  // this final coordinate system is the one in which
182  // the deformation gradient is equal to the stretch tensor
183  return ShellT3_LocalCoordinateSystem(geom[0], geom[1], geom[2], alpha);
184

223

234  void FinalizeCalculations(const ShellT3_LocalCoordinateSystem& LCS,
235  const VectorType& globalDisplacements,
236  const VectorType& localDisplacements,
237  MatrixType& rLeftHandSideMatrix,
238  VectorType& rRightHandSideVector,
239  const bool RHSrequired,
240  const bool LHSrequired) override
241  {
242  // Get the total rotation matrix (local - to - global)
243  // Note: do NOT include the warpage correction matrix!
244  // Explanation:
245  // The Warpage correction matrix computed by the LocalCoordinateSystem is a Linear Projector.
246  // It should be used in a LinearCoordinateTransformation.
247  // Here instead we already calculate a nonlinear Projector (P = Pu - S * G)!
248 
249  MatrixType T(18, 18);
250  LCS.ComputeTotalRotationMatrix(T);
251 
252  // Form all matrices:
253  // S: Spin-Fitter matrix
254  // G: Spin-Lever matrix
255  // P: Projector (Translational & Rotational)
256 
257  MatrixType P(EICR::Compute_Pt(3));
258  MatrixType S(EICR::Compute_S(LCS.Nodes()));
259  MatrixType G(RotationGradient(LCS));
260  noalias(P) -= prod(S, G);
261 
262  // Compute the projected local forces ( pe = P' * RHS ).
263  // Note: here the RHS is already given as a residual vector -> - internalForces -> (pe = - Ke * U)
264  // so projectedLocalForces = - P' * Ke * U
265 
266  VectorType projectedLocalForces(prod(trans(P), rRightHandSideVector));
267 
268  // Compute the Right-Hand-Side vector in global coordinate system (- T' * P' * Km * U).
269  // At this point the computation of the Right-Hand-Side is complete.
270 
271  noalias(rRightHandSideVector) = prod(trans(T), projectedLocalForces);
272 
273  // Begin the computation of the Left-Hand-Side Matrix :
274 
275  if (!LHSrequired) {
276  return; // avoid useless calculations!
277  }
278 
279  // This is a temporary matrix to store intermediate values
280  // to avoid extra dynamic memory allocations!
281 
282  MatrixType temp(18, 18);
283 
284  // H: Axial Vector Jacobian
285  MatrixType H(EICR::Compute_H(localDisplacements));
286 
287  // Step 1: ( K.M : Material Stiffness Matrix )
288  // Apply the projector to the Material Stiffness Matrix (Ke = P' * Km * H * P)
289  // At this point 'LHS' contains the 'projected' Material Stiffness matrix
290  // in local corotational coordinate system
291 
292  noalias(temp) = prod(rLeftHandSideMatrix, H);
293  noalias(rLeftHandSideMatrix) = prod(temp, P);
294  noalias(temp) = prod(trans(P), rLeftHandSideMatrix);
295  rLeftHandSideMatrix.swap(temp);
296 
297  // Step 2: ( K.GP: Equilibrium Projection Geometric Stiffness Matrix )
298  // First assemble the 'Fnm' matrix with the Spins of the nodal forces.
299  // Actually at this point the 'Fnm' Matrix is the 'Fn' Matrix,
300  // because it only contains the spins of the 'translational' forces.
301  // At this point 'LHS' contains also this term of the Geometric stiffness
302  // (Ke = (P' * Km * H * P) - (G' * Fn' * P))
303 
304  MatrixType Fnm(18, 3, 0.0);
305 
306  EICR::Spin_AtRow(projectedLocalForces, Fnm, 0);
307  EICR::Spin_AtRow(projectedLocalForces, Fnm, 6);
308  EICR::Spin_AtRow(projectedLocalForces, Fnm, 12);
309 
310  noalias(temp) = prod(trans(G), trans(Fnm));
311  noalias(rLeftHandSideMatrix) += prod(temp, P); // note: '+' not '-' because the RHS already has the negative sign
312 
313  // Step 3: ( K.GR: Rotational Geometric Stiffness Matrix )
314  // Add the Spins of the nodal moments to 'Fnm'.
315  // At this point 'LHS' contains also this term of the Geometric stiffness
316  // (Ke = (P' * Km * H * P) - (G' * Fn' * P) - (Fnm * G))
317 
318  EICR::Spin_AtRow(projectedLocalForces, Fnm, 3);
319  EICR::Spin_AtRow(projectedLocalForces, Fnm, 9);
320  EICR::Spin_AtRow(projectedLocalForces, Fnm, 15);
321 
322  noalias(rLeftHandSideMatrix) += prod(Fnm, G); // note: '+' not '-' because the RHS already has the negative sign
323 
324  // Step 4: (Global Stiffness Matrix)
325  // Transform the LHS to the Global coordinate system.
326  // T' * [(P' * Km * H * P) - (G' * Fn' * P) - (Fnm * G)] * T
327  noalias(temp) = prod(rLeftHandSideMatrix, T);
328  noalias(rLeftHandSideMatrix) = prod(trans(T), temp);
329  }
