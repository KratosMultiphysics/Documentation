84  double gravity = 9.81;
85  const double well_radius = 0.075;
86  if ((particle_coordinates[0] * particle_coordinates[0] + particle_coordinates[1] * particle_coordinates[1]) < well_radius * well_radius) {

88  } else if (is_found) {

93  }
