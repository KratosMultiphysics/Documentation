103  {

106  rElement.CalculateOnIntegrationPoints(rVariable, values_on_integration_points,
107  mrModelPart.GetProcessInfo());
108 
109  for (IndexType iNode = 0; iNode < r_this_geometry.PointsNumber(); ++iNode) {
110  // We first initialize the source, which we need to do by getting the first value,
111  // because we don't know the size of the dynamically allocated Vector/Matrix
112  T source = rExtrapolationMatrix(iNode, 0) * values_on_integration_points[0];
113  for (IndexType i_gauss_point = 1; i_gauss_point < values_on_integration_points.size(); ++i_gauss_point) {
114  source += rExtrapolationMatrix(iNode, i_gauss_point) * values_on_integration_points[i_gauss_point];
115  }
116  source /= r_this_geometry[iNode].GetValue(mrAverageVariable);
117 
118  rAtomicAddOperation(r_this_geometry[iNode].FastGetSolutionStepValue(rVariable), source);
119  }
120  }
