76  void ExecuteInitialize() override
77  {

79 
80  const Variable<double>& var = KratosComponents<Variable<double>>::Get(mVariableName);
81  block_for_each(mrModelPart.Nodes(), [&var, this](Node& rNode) {
82  const double pressure = -PORE_PRESSURE_SIGN_FACTOR * mSpecificWeight *
83  (mReferenceCoordinate - rNode.Coordinates()[mGravityDirection]);
84 
85  if (mIsSeepage) {
86  if (pressure < PORE_PRESSURE_SIGN_FACTOR * mPressureTensionCutOff) { // Before 0. was used i.s.o. the tension cut off value -> no effect in any test.
87  rNode.FastGetSolutionStepValue(var) = pressure;
88  if (mIsFixed) rNode.Fix(var);
89  } else {
90  if (mIsFixedProvided) rNode.Free(var);
91  }
92  } else {
93  rNode.FastGetSolutionStepValue(var) =
94  std::min(pressure, PORE_PRESSURE_SIGN_FACTOR * mPressureTensionCutOff);
95  if (mIsFixed) rNode.Fix(var);
96  else if (mIsFixedProvided) rNode.Free(var);
97  }
98  });

100  KRATOS_CATCH("")
