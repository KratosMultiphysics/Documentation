329  static void InvertMatrix(
330  const TMatrix1& rInputMatrix,
331  TMatrix2& rInvertedMatrix,
332  double& rInputMatrixDet,
333  const double Tolerance = ZeroTolerance
334  )

337 
338  const SizeType size = rInputMatrix.size2();
339 
340  if(size == 1) {
341  if(rInvertedMatrix.size1() != 1 || rInvertedMatrix.size2() != 1) {
342  rInvertedMatrix.resize(1,1,false);
343  }
344  rInvertedMatrix(0,0) = 1.0/rInputMatrix(0,0);
345  rInputMatrixDet = rInputMatrix(0,0);
346  } else if (size == 2) {
347  InvertMatrix2(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
348  } else if (size == 3) {
349  InvertMatrix3(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
350  } else if (size == 4) {
351  InvertMatrix4(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
352  } else if (std::is_same<TMatrix1, Matrix>::value) {
353 
354  const SizeType size1 = rInputMatrix.size1();
355  const SizeType size2 = rInputMatrix.size2();
356  if(rInvertedMatrix.size1() != size1 || rInvertedMatrix.size2() != size2) {
357  rInvertedMatrix.resize(size1, size2,false);
358  }
359 
360  Matrix A(rInputMatrix);
361  typedef permutation_matrix<SizeType> pmatrix;
362  pmatrix pm(A.size1());
363  const int singular = lu_factorize(A,pm);
364  rInvertedMatrix.assign( IdentityMatrix(A.size1()));
365  KRATOS_ERROR_IF(singular == 1) << "Matrix is singular: " << rInputMatrix << std::endl;
366  lu_substitute(A, pm, rInvertedMatrix);
367 
368  // Calculating determinant
369  rInputMatrixDet = 1.0;
370 
371  for (IndexType i = 0; i < size1;++i) {
372  IndexType ki = pm[i] == i ? 0 : 1;
373  rInputMatrixDet *= (ki == 0) ? A(i,i) : -A(i,i);
374  }
375  } else { // Bounded-matrix case
376  const SizeType size1 = rInputMatrix.size1();
377  const SizeType size2 = rInputMatrix.size2();
378 
379  Matrix A(rInputMatrix);
380  Matrix invA(rInvertedMatrix);
381 
382  typedef permutation_matrix<SizeType> pmatrix;
383  pmatrix pm(size1);
384  const int singular = lu_factorize(A,pm);
385  invA.assign( IdentityMatrix(size1));
386  KRATOS_ERROR_IF(singular == 1) << "Matrix is singular: " << rInputMatrix << std::endl;
387  lu_substitute(A, pm, invA);
388 
389  // Calculating determinant
390  rInputMatrixDet = 1.0;
391 
392  for (IndexType i = 0; i < size1;++i) {
393  IndexType ki = pm[i] == i ? 0 : 1;
394  rInputMatrixDet *= (ki == 0) ? A(i,i) : -A(i,i);
395  }
396 
397  for (IndexType i = 0; i < size1;++i) {
398  for (IndexType j = 0; j < size2;++j) {
399  rInvertedMatrix(i,j) = invA(i,j);
400  }
401  }
402  }
403 
404  // Checking condition number
405  if (Tolerance > 0.0) { // Check is skipped for negative tolerances
406  CheckConditionNumber(rInputMatrix, rInvertedMatrix, Tolerance);
407  }
408  }

1632  static inline bool GaussSeidelEigenSystem(
1633  const TMatrixType1& rA,
1634  TMatrixType2& rEigenVectorsMatrix,
1635  TMatrixType2& rEigenValuesMatrix,
1636  const double Tolerance = 1.0e-18,
1637  const SizeType MaxIterations = 20
1638  )
1639  {
1640  bool is_converged = false;
1641 
1642  const SizeType size = rA.size1();
1643 
1644  if (rEigenVectorsMatrix.size1() != size || rEigenVectorsMatrix.size2() != size)
1645  rEigenVectorsMatrix.resize(size, size, false);
1646  if (rEigenValuesMatrix.size1() != size || rEigenValuesMatrix.size2() != size)
1647  rEigenValuesMatrix.resize(size, size, false);
1648 
1649  const TMatrixType2 identity_matrix = IdentityMatrix(size);
1650  noalias(rEigenVectorsMatrix) = identity_matrix;
1651  noalias(rEigenValuesMatrix) = rA;
1652 
1653  // Auxiliar values
1654  TMatrixType2 aux_A, aux_V_matrix, rotation_matrix;
1655  double a, u, c, s, gamma, teta;
1656  IndexType index1, index2;
1657 
1658  aux_A.resize(size,size,false);
1659  aux_V_matrix.resize(size,size,false);
1660  rotation_matrix.resize(size,size,false);
1661 
1662  for(IndexType iterations = 0; iterations < MaxIterations; ++iterations) {
1663  is_converged = true;
1664 
1665  a = 0.0;
1666  index1 = 0;
1667  index2 = 1;
1668 
1669  for(IndexType i = 0; i < size; ++i) {
1670  for(IndexType j = (i + 1); j < size; ++j) {
1671  if((std::abs(rEigenValuesMatrix(i, j)) > a ) && (std::abs(rEigenValuesMatrix(i, j)) > Tolerance)) {
1672  a = std::abs(rEigenValuesMatrix(i,j));
1673  index1 = i;
1674  index2 = j;
1675  is_converged = false;
1676  }
1677  }
1678  }
1679 
1680  if(is_converged) {
1681  break;
1682  }
1683 
1684  // Calculation of Rotation angle
1685  gamma = (rEigenValuesMatrix(index2, index2)-rEigenValuesMatrix(index1, index1)) / (2 * rEigenValuesMatrix(index1, index2));
1686  u = 1.0;
1687 
1688  if(std::abs(gamma) > Tolerance && std::abs(gamma)< (1.0/Tolerance)) {
1689  u = gamma / std::abs(gamma) * 1.0 / (std::abs(gamma) + std::sqrt(1.0 + gamma * gamma));
1690  } else {
1691  if (std::abs(gamma) >= (1.0/Tolerance)) {
1692  u = 0.5 / gamma;
1693  }
1694  }
1695 
1696  c = 1.0 / (std::sqrt(1.0 + u * u));
1697  s = c * u;
1698  teta = s / (1.0 + c);
1699 
1700  // Rotation of the Matrix
1701  noalias(aux_A) = rEigenValuesMatrix;
1702  aux_A(index2, index2) = rEigenValuesMatrix(index2,index2) + u * rEigenValuesMatrix(index1, index2);
1703  aux_A(index1, index1) = rEigenValuesMatrix(index1,index1) - u * rEigenValuesMatrix(index1, index2);
1704  aux_A(index1, index2) = 0.0;
1705  aux_A(index2, index1) = 0.0;
1706 
1707  for(IndexType i = 0; i < size; ++i) {
1708  if((i!= index1) && (i!= index2)) {
1709  aux_A(index2, i) = rEigenValuesMatrix(index2, i) + s * (rEigenValuesMatrix(index1, i)- teta * rEigenValuesMatrix(index2, i));
1710  aux_A(i, index2) = rEigenValuesMatrix(index2, i) + s * (rEigenValuesMatrix(index1, i)- teta * rEigenValuesMatrix(index2, i));
1711  aux_A(index1, i) = rEigenValuesMatrix(index1, i) - s * (rEigenValuesMatrix(index2, i) + teta * rEigenValuesMatrix(index1, i));
1712  aux_A(i, index1) = rEigenValuesMatrix(index1, i) - s * (rEigenValuesMatrix(index2, i) + teta * rEigenValuesMatrix(index1, i));
1713  }
1714  }
1715 
1716  noalias(rEigenValuesMatrix) = aux_A;
1717 
1718  // Calculation of the eigeneigen vector matrix V
1719  noalias(rotation_matrix) = identity_matrix;
1720  rotation_matrix(index2, index1) = -s;
1721  rotation_matrix(index1, index2) = s;
1722  rotation_matrix(index1, index1) = c;
1723  rotation_matrix(index2, index2) = c;
1724 
1725  noalias(aux_V_matrix) = ZeroMatrix(size, size);
1726 
1727  for(IndexType i = 0; i < size; ++i) {
1728  for(IndexType j = 0; j < size; ++j) {
1729  for(IndexType k = 0; k < size; ++k) {
1730  aux_V_matrix(i, j) += rEigenVectorsMatrix(i, k) * rotation_matrix(k, j);
1731  }
1732  }
1733  }
1734  noalias(rEigenVectorsMatrix) = aux_V_matrix;
1735  }
1736

1738
