342  template<class TMatrix1, class TMatrix2>
343  static void InvertMatrix(
344  const TMatrix1& rInputMatrix,
345  TMatrix2& rInvertedMatrix,
346  double& rInputMatrixDet,
347  const double Tolerance = ZeroTolerance
348  )
349  {
350  KRATOS_DEBUG_ERROR_IF_NOT(rInputMatrix.size1() == rInputMatrix.size2()) << "Matrix provided is non-square" << std::endl;
351 
352  const SizeType size = rInputMatrix.size2();
353 
354  if(size == 1) {
355  if(rInvertedMatrix.size1() != 1 || rInvertedMatrix.size2() != 1) {
356  rInvertedMatrix.resize(1,1,false);
357  }
358  rInvertedMatrix(0,0) = 1.0/rInputMatrix(0,0);
359  rInputMatrixDet = rInputMatrix(0,0);
360  } else if (size == 2) {
361  InvertMatrix2(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
362  } else if (size == 3) {
363  InvertMatrix3(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
364  } else if (size == 4) {
365  InvertMatrix4(rInputMatrix, rInvertedMatrix, rInputMatrixDet);
366  } else if (std::is_same<TMatrix1, Matrix>::value) {
367 
368  const SizeType size1 = rInputMatrix.size1();
369  const SizeType size2 = rInputMatrix.size2();
370  if(rInvertedMatrix.size1() != size1 || rInvertedMatrix.size2() != size2) {
371  rInvertedMatrix.resize(size1, size2,false);
372  }
373 
374  Matrix A(rInputMatrix);
375  typedef permutation_matrix<SizeType> pmatrix;
376  pmatrix pm(A.size1());
377  const int singular = lu_factorize(A,pm);
378  rInvertedMatrix.assign( IdentityMatrix(A.size1()));
379  KRATOS_ERROR_IF(singular == 1) << "Matrix is singular: " << rInputMatrix << std::endl;
380  lu_substitute(A, pm, rInvertedMatrix);
381 
382  // Calculating determinant
383  rInputMatrixDet = 1.0;
384 
385  for (IndexType i = 0; i < size1;++i) {
386  IndexType ki = pm[i] == i ? 0 : 1;
387  rInputMatrixDet *= (ki == 0) ? A(i,i) : -A(i,i);
388  }
389  } else { // Bounded-matrix case
390  const SizeType size1 = rInputMatrix.size1();
391  const SizeType size2 = rInputMatrix.size2();
392 
393  Matrix A(rInputMatrix);
394  Matrix invA(rInvertedMatrix);
395 
396  typedef permutation_matrix<SizeType> pmatrix;
397  pmatrix pm(size1);
398  const int singular = lu_factorize(A,pm);
399  invA.assign( IdentityMatrix(size1));
400  KRATOS_ERROR_IF(singular == 1) << "Matrix is singular: " << rInputMatrix << std::endl;
401  lu_substitute(A, pm, invA);
402 
403  // Calculating determinant
404  rInputMatrixDet = 1.0;
405 
406  for (IndexType i = 0; i < size1;++i) {
407  IndexType ki = pm[i] == i ? 0 : 1;
408  rInputMatrixDet *= (ki == 0) ? A(i,i) : -A(i,i);
409  }
410 
411  for (IndexType i = 0; i < size1;++i) {
412  for (IndexType j = 0; j < size2;++j) {
413  rInvertedMatrix(i,j) = invA(i,j);
414  }
415  }
416  }
417 
418  // Checking condition number
419  if (Tolerance > 0.0) { // Check is skipped for negative tolerances
420  CheckConditionNumber(rInputMatrix, rInvertedMatrix, Tolerance);
421  }
422  }

1683  static inline bool GaussSeidelEigenSystem(
1684  const TMatrixType1& rA,
1685  TMatrixType2& rEigenVectorsMatrix,
1686  TMatrixType2& rEigenValuesMatrix,
1687  const double Tolerance = 1.0e-18,
1688  const SizeType MaxIterations = 20
1689  )
1690  {
1691  bool is_converged = false;
1692 
1693  const SizeType size = rA.size1();
1694 
1695  if (rEigenVectorsMatrix.size1() != size || rEigenVectorsMatrix.size2() != size)
1696  rEigenVectorsMatrix.resize(size, size, false);
1697  if (rEigenValuesMatrix.size1() != size || rEigenValuesMatrix.size2() != size)
1698  rEigenValuesMatrix.resize(size, size, false);
1699 
1700  const TMatrixType2 identity_matrix = IdentityMatrix(size);
1701  noalias(rEigenVectorsMatrix) = identity_matrix;
1702  noalias(rEigenValuesMatrix) = rA;
1703 
1704  // Auxiliar values
1705  TMatrixType2 aux_A, aux_V_matrix, rotation_matrix;
1706  double a, u, c, s, gamma, teta;
1707  IndexType index1, index2;
1708 
1709  aux_A.resize(size,size,false);
1710  aux_V_matrix.resize(size,size,false);
1711  rotation_matrix.resize(size,size,false);
1712 
1713  for(IndexType iterations = 0; iterations < MaxIterations; ++iterations) {
1714  is_converged = true;
1715 
1716  a = 0.0;
1717  index1 = 0;
1718  index2 = 1;
1719 
1720  for(IndexType i = 0; i < size; ++i) {
1721  for(IndexType j = (i + 1); j < size; ++j) {
1722  if((std::abs(rEigenValuesMatrix(i, j)) > a ) && (std::abs(rEigenValuesMatrix(i, j)) > Tolerance)) {
1723  a = std::abs(rEigenValuesMatrix(i,j));
1724  index1 = i;
1725  index2 = j;
1726  is_converged = false;
1727  }
1728  }
1729  }
1730 
1731  if(is_converged) {
1732  break;
1733  }
1734 
1735  // Calculation of Rotation angle
1736  gamma = (rEigenValuesMatrix(index2, index2)-rEigenValuesMatrix(index1, index1)) / (2 * rEigenValuesMatrix(index1, index2));
1737  u = 1.0;
1738 
1739  if(std::abs(gamma) > Tolerance && std::abs(gamma)< (1.0/Tolerance)) {
1740  u = gamma / std::abs(gamma) * 1.0 / (std::abs(gamma) + std::sqrt(1.0 + gamma * gamma));
1741  } else {
1742  if (std::abs(gamma) >= (1.0/Tolerance)) {
1743  u = 0.5 / gamma;
1744  }
1745  }
1746 
1747  c = 1.0 / (std::sqrt(1.0 + u * u));
1748  s = c * u;
1749  teta = s / (1.0 + c);
1750 
1751  // Rotation of the Matrix
1752  noalias(aux_A) = rEigenValuesMatrix;
1753  aux_A(index2, index2) = rEigenValuesMatrix(index2,index2) + u * rEigenValuesMatrix(index1, index2);
1754  aux_A(index1, index1) = rEigenValuesMatrix(index1,index1) - u * rEigenValuesMatrix(index1, index2);
1755  aux_A(index1, index2) = 0.0;
1756  aux_A(index2, index1) = 0.0;
1757 
1758  for(IndexType i = 0; i < size; ++i) {
1759  if((i!= index1) && (i!= index2)) {
1760  aux_A(index2, i) = rEigenValuesMatrix(index2, i) + s * (rEigenValuesMatrix(index1, i)- teta * rEigenValuesMatrix(index2, i));
1761  aux_A(i, index2) = rEigenValuesMatrix(index2, i) + s * (rEigenValuesMatrix(index1, i)- teta * rEigenValuesMatrix(index2, i));
1762  aux_A(index1, i) = rEigenValuesMatrix(index1, i) - s * (rEigenValuesMatrix(index2, i) + teta * rEigenValuesMatrix(index1, i));

1765  }
1766 
1767  noalias(rEigenValuesMatrix) = aux_A;

1771  rotation_matrix(index2, index1) = -s;
1772  rotation_matrix(index1, index2) = s;
1773  rotation_matrix(index1, index1) = c;
1774  rotation_matrix(index2, index2) = c;
1775 
1776  noalias(aux_V_matrix) = ZeroMatrix(size, size);
1777

1779  for(IndexType j = 0; j < size; ++j) {
1780  for(IndexType k = 0; k < size; ++k) {
1781  aux_V_matrix(i, j) += rEigenVectorsMatrix(i, k) * rotation_matrix(k, j);
1782  }
1783  }
1784  }
1785  noalias(rEigenVectorsMatrix) = aux_V_matrix;
1786  }
