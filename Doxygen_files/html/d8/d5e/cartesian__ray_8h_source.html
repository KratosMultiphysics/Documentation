101  void CollapseIntersectionPoints(double Tolerance){
102 
103  if (mIntersections.size() < 2) {
104  return;
105  }
106  // Sort
107  std::sort(mIntersections.begin(), mIntersections.end());
108  auto i_unique_end = mIntersections.begin();
109  auto i_patch_begin = mIntersections.begin();
110  auto i_patch_end = mIntersections.begin();
111  for(auto i_intersection = mIntersections.begin() + 1 ; i_intersection != mIntersections.end() ; i_intersection++) {
112  if (std::abs(i_intersection->first - i_patch_begin->first) < Tolerance){ // we are in a patch so let's continue until the patch finishes
113  i_patch_end = i_intersection;
114  if(i_intersection + 1 != mIntersections.end()) { // if i_intersection is not the last one we keep continuing
115  continue;
116  }
117  }
118  if((i_patch_begin == i_patch_end) || // The previous patch was only one intersection and we add it.
119  (CheckPassingThroughByExtraRays(i_patch_begin, i_patch_end+1, Tolerance, 2.00*Tolerance))) { // more than one intersection to be checked with extra rays
120  if(i_unique_end != i_patch_begin) {
121  *i_unique_end = std::move(*i_patch_begin);
122  }
123  i_unique_end++;
124  }
125  if((i_intersection != i_patch_end)&&(i_intersection + 1 == mIntersections.end())) { // Adding the last intersection
126  if(i_unique_end != i_intersection) {
127  *i_unique_end = std::move(*i_intersection);
128  }
129  i_unique_end++;
130  }
131  i_patch_begin = i_intersection;
132  i_patch_end = i_intersection;
133  }
134  auto new_size = std::distance(mIntersections.begin(), i_unique_end);
135  mIntersections.resize(new_size);
136  }

267  const std::array<double, 8> delta_u{Delta, Delta, 0.00, -Delta, -Delta, -Delta, 0.00, Delta};

309  static int ComputeTriangleRayIntersection(
310  const TGeometryType& rTriangleGeometry,
311  const array_1d<double,3>& rLinePoint1,
312  const array_1d<double,3>& rLinePoint2,
313  array_1d<double,3>& rIntersectionPoint,
314  const double epsilon = 1e-12) {
315 
316  // This is the adaption of the implemnetation of Moller-Trumbore algorithm provided in:
317  // https://en.wikipedia.org/wiki/M%C3%B6ller%E2%80%93Trumbore_intersection_algorithm
318 
319  const array_1d<double,3> edge1 = rTriangleGeometry[1] - rTriangleGeometry[0];
320  const array_1d<double,3> edge2 = rTriangleGeometry[2] - rTriangleGeometry[0];
321  const array_1d<double,3> line_vector = rLinePoint2 - rLinePoint1;
322  array_1d<double,3> h;
323  MathUtils<double>::CrossProduct(h, line_vector, edge2);
324  double a = inner_prod(edge1,h);
325  if (a > -epsilon && a < epsilon)
326  return 0; // This ray is parallel to this triangle.
327  const double f = 1.0/a;
328  const array_1d<double,3> s = rLinePoint1 - rTriangleGeometry[0];
329  const double u = f * inner_prod(s,h);
330  if (u < -epsilon || u > 1.0 + epsilon)
331  return 0;
332  array_1d<double,3> q;
333  MathUtils<double>::CrossProduct(q, s, edge1);
334  const double v = f * inner_prod(line_vector,q);
335  if (v < -epsilon || u + v > 1.0 + epsilon)
336  return 0;
337  // At this stage we can compute t to find out where the intersection point is on the line.
338  float t = f * inner_prod(edge2,q);
339  if (t > epsilon && t < 1.00/epsilon) // ray intersection
340  {
341  rIntersectionPoint = rLinePoint1 + line_vector * t;
342  return 1;
343  }
344  else // This means that there is a line intersection but not a ray intersection.
345  return 0;
346  }
