70 void ComputeEquivalentF(
71  const Element& rElement,
72  const TVectorType& rStrainTensor,
73  TMatrixType& rF
74  )
75 {
76  const auto& r_geometry = rElement.GetGeometry();
77  const SizeType dimension = r_geometry.WorkingSpaceDimension();
78 
79  if(dimension == 2) {
80  rF(0,0) = 1.0+rStrainTensor(0);
81  rF(0,1) = 0.5*rStrainTensor(2);
82  rF(1,0) = 0.5*rStrainTensor(2);
83  rF(1,1) = 1.0+rStrainTensor(1);
84  } else {
85  rF(0,0) = 1.0+rStrainTensor(0);
86  rF(0,1) = 0.5*rStrainTensor(3);
87  rF(0,2) = 0.5*rStrainTensor(5);
88  rF(1,0) = 0.5*rStrainTensor(3);
89  rF(1,1) = 1.0+rStrainTensor(1);
90  rF(1,2) = 0.5*rStrainTensor(4);
91  rF(2,0) = 0.5*rStrainTensor(5);
92  rF(2,1) = 0.5*rStrainTensor(4);
93  rF(2,2) = 1.0+rStrainTensor(2);
94  }
95 }

105 template<class TMatrixType1, class TMatrixType2>
106 void CalculateB(
107  const GeometricalObject& rElement,
108  const TMatrixType1& rDN_DX,
109  TMatrixType2& rB
110  )
111 {
112  const auto& r_geometry = rElement.GetGeometry();
113  const SizeType number_of_nodes = r_geometry.PointsNumber();
114  const SizeType dimension = rDN_DX.size2();
115 
116  rB.clear();
117 
118  if(dimension == 2) {
119  for ( IndexType i = 0; i < number_of_nodes; ++i ) {
120  const IndexType initial_index = i*2;
121  rB(0, initial_index ) = rDN_DX(i, 0);
122  rB(1, initial_index + 1) = rDN_DX(i, 1);
123  rB(2, initial_index ) = rDN_DX(i, 1);
124  rB(2, initial_index + 1) = rDN_DX(i, 0);
125  }
126  } else if(dimension == 3) {
127  for ( IndexType i = 0; i < number_of_nodes; ++i ) {
128  const IndexType initial_index = i*3;
129  rB(0, initial_index ) = rDN_DX(i, 0);
130  rB(1, initial_index + 1) = rDN_DX(i, 1);
131  rB(2, initial_index + 2) = rDN_DX(i, 2);
132  rB(3, initial_index ) = rDN_DX(i, 1);
133  rB(3, initial_index + 1) = rDN_DX(i, 0);
134  rB(4, initial_index + 1) = rDN_DX(i, 2);
135  rB(4, initial_index + 2) = rDN_DX(i, 1);
136  rB(5, initial_index ) = rDN_DX(i, 2);
137  rB(5, initial_index + 2) = rDN_DX(i, 0);
138  }
139  }
