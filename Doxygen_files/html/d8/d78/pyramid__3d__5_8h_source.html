627  Matrix& rResult,
628  const CoordinatesArrayType& rPoint
629  )
630  {
631  rResult.resize( 5, 3, false );
632  noalias( rResult ) = ZeroMatrix( 5, 3 );
633 
634  rResult( 0, 0 ) = (-0.125) * ( 1 - rPoint[1] ) * ( 1 - rPoint[2] ) ;
635  rResult( 0, 1 ) = (-0.125) * ( 1 - rPoint[0] ) * ( 1 - rPoint[2] ) ;
636  rResult( 0, 2 ) = (-0.125) * ( 1 - rPoint[0] ) * ( 1 - rPoint[1] ) ;
637 
638  rResult( 1, 0 ) = (+0.125) * ( 1 - rPoint[1] ) * ( 1 - rPoint[2] ) ;
639  rResult( 1, 1 ) = (-0.125) * ( 1 + rPoint[0] ) * ( 1 - rPoint[2] ) ;
640  rResult( 1, 2 ) = (-0.125) * ( 1 + rPoint[0] ) * ( 1 - rPoint[1] ) ;
641 
642  rResult( 2, 0 ) = (+0.125) * ( 1 + rPoint[1] ) * ( 1 - rPoint[2] ) ;
643  rResult( 2, 1 ) = (+0.125) * ( 1 + rPoint[0] ) * ( 1 - rPoint[2] ) ;
644  rResult( 2, 2 ) = (-0.125) * ( 1 + rPoint[0] ) * ( 1 + rPoint[1] ) ;
645 
646  rResult( 3, 0 ) = (-0.125) * ( 1 + rPoint[1] ) * ( 1 - rPoint[2] ) ;
647  rResult( 3, 1 ) = (+0.125) * ( 1 - rPoint[0] ) * ( 1 - rPoint[2] ) ;
648  rResult( 3, 2 ) = (-0.125) * ( 1 - rPoint[0] ) * ( 1 + rPoint[1] ) ;

700  double CalculateDistance(
701  const CoordinatesArrayType& rPointGlobalCoordinates,
702  const double Tolerance = std::numeric_limits<double>::epsilon()
703  ) const override
704  {
705  // Point to compute distance to
706  const Point point(rPointGlobalCoordinates);
707 
708  // Check if point is inside
709  CoordinatesArrayType aux_coordinates;

711  return 0.0;
712  }
713 
714  // Compute distance to faces
715  std::array<double, 5> distances;

717  distances[1] = GeometryUtils::PointDistanceToTriangle3D(this->GetPoint(1), this->GetPoint(2), this->GetPoint(4), point);
718  distances[2] = GeometryUtils::PointDistanceToQuadrilateral3D(this->GetPoint(0), this->GetPoint(1), this->GetPoint(2), this->GetPoint(3), point);
719  distances[3] = GeometryUtils::PointDistanceToTriangle3D(this->GetPoint(2), this->GetPoint(3), this->GetPoint(4), point);
720  distances[4] = GeometryUtils::PointDistanceToTriangle3D(this->GetPoint(3), this->GetPoint(0), this->GetPoint(4), point);
721  const auto min = std::min_element(distances.begin(), distances.end());

796  // see ShapeFunctionValue
797  // special function such that it can also be used in the static methods
798  static double ShapeFunctionValueImpl(
799  IndexType ShapeFunctionIndex,
800  const CoordinatesArrayType& rPoint)
801  {
802  switch ( ShapeFunctionIndex )
803  {
804  case 0:
805  return( (0.125) * (1 - rPoint[0]) * (1 - rPoint[1]) * (1 - rPoint[2]) );
806  case 1:
807  return( (0.125) * (1 + rPoint[0]) * (1 - rPoint[1]) * (1 - rPoint[2]) );
808  case 2:
809  return( (0.125) * (1 + rPoint[0]) * (1 + rPoint[1]) * (1 - rPoint[2]) );
810  case 3:
811  return( (0.125) * (1 - rPoint[0]) * (1 + rPoint[1]) * (1 - rPoint[2]) );
812  case 4:
813  return( (0.5) * (1 + rPoint[2]) );
814  default:
815  KRATOS_ERROR << "Wrong index of shape function:" << ShapeFunctionIndex << std::endl;
