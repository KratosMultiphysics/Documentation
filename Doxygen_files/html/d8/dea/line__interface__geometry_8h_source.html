243  [[nodiscard]] PointerVector<Node> CreatePointsOfMidLine() const
244  {
245  const auto points = this->Points();
246  const auto number_of_midline_nodes = std::size_t{points.size() / 2};
247 
248  auto result = PointerVector<Node>{};
249  for (std::size_t i = 0; i < number_of_midline_nodes; ++i) {
250  const auto mid_point = (points[i] + points[i + number_of_midline_nodes]) / 2;
251  result.push_back(make_intrusive<Node>(points[i].Id(), mid_point));
252  }
253  return result;
254  }
