211  GeometriesArrayType GenerateEdges() const override
212  {
213  const auto points = this->Points();
214 
215  // The first edge coincides with the first side of the element
216  auto begin_of_second_side = points.ptr_begin() + (points.size() / 2);
217  const auto nodes_of_first_edge = PointerVector<Node>{points.ptr_begin(), begin_of_second_side};
218 
219  // The second edge coincides with the second side of the element. However, the nodes must be
220  // traversed in opposite direction.
221  auto nodes_of_second_edge = PointerVector<Node>{begin_of_second_side, points.ptr_end()};
222  std::swap(nodes_of_second_edge(0), nodes_of_second_edge(1)); // end nodes
223  std::reverse(nodes_of_second_edge.ptr_begin() + 2, nodes_of_second_edge.ptr_end()); // any high-order nodes
224 
225  auto result = GeometriesArrayType{};
226  result.push_back(std::make_shared<MidGeometryType>(nodes_of_first_edge));
227  result.push_back(std::make_shared<MidGeometryType>(nodes_of_second_edge));
228  return result;
229  }

259 private:
260  [[nodiscard]] PointerVector<Node> CreatePointsOfMidLine() const
261  {
262  const auto points = this->Points();
263  const auto number_of_midline_nodes = std::size_t{points.size() / 2};
264  auto result = PointerVector<Node>{number_of_midline_nodes};
265 
266  auto is_null = [](const auto& rNodePtr) { return rNodePtr == nullptr; };
267  if (std::any_of(points.ptr_begin(), points.ptr_end(), is_null)) {
268  // At least one point is not defined, so the points of the mid-line can't be computed.
269  // As a result, all the mid-line points will be undefined.
270  return result;
271  }
272 
273  auto begin_of_first_side = points.ptr_begin();
274  auto begin_of_second_side = begin_of_first_side + number_of_midline_nodes;
275  auto make_mid_point = [](auto pPoint1, auto pPoint2) {
276  return make_intrusive<Node>(pPoint1->Id(), Point{(*pPoint1 + *pPoint2) / 2});
277  };
278  std::transform(begin_of_first_side, begin_of_second_side, begin_of_second_side,
279  result.ptr_begin(), make_mid_point);
280  return result;
281  }
