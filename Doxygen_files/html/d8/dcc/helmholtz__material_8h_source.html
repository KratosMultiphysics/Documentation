434  void ComputeFilteredDensity(){
435 
436  for(long unsigned int model_i=0;model_i<mpVMModelParts.size();model_i++){
437 
438  //first update control density
439  AddVariable1ToVarible2(mpVMModelParts[model_i],D_CD,CD);

441  //now filter nodal control desnity
442  //first we need to multiply with mass matrix
443  SetVariable(mpVMModelParts[model_i],HELMHOLTZ_VAR_DENSITY,0.0);
444  SetVariable(mpVMModelParts[model_i],HELMHOLTZ_SOURCE_DENSITY,0.0);
445  //now we need to multiply with the mass matrix
446  block_for_each(mpVMModelParts[model_i]->Elements(), [&](auto& elem_i) {
447  VectorType origin_values;

449  MatrixType mass_matrix;

451  VectorType int_vals = prod(mass_matrix,origin_values);
452  AddElementVariableValuesVector(elem_i,HELMHOLTZ_SOURCE_DENSITY,int_vals);
453  });
454 
455  //now we need to apply BCs
456  block_for_each(mpVMModelParts[model_i]->Nodes(), [&](auto& node_i) {
457  auto& r_nodal_variable_hl_var = node_i.FastGetSolutionStepValue(HELMHOLTZ_VAR_DENSITY);

459  r_nodal_variable_hl_var = node_i.FastGetSolutionStepValue(FD);

520  double ProjectForward(double x, Vector x_limits, Vector y_limits, double beta, int penal_fac = 1){
521 
522  double x1=0.0,x2=0.0,y1=0.0,y2=0.0;
523  if(x>=x_limits[x_limits.size()-1]){
524  x1=x_limits[x_limits.size()-2];
525  x2=x_limits[x_limits.size()-1];
526  y1=y_limits[y_limits.size()-2];
527  y2=y_limits[y_limits.size()-1];
528  }
529  else if(x<=x_limits[0]){
530  x1=x_limits[0];
531  x2=x_limits[1];
532  y1=y_limits[0];
533  y2=y_limits[1];
534  }
535  else{
536  for(long unsigned int i=0;i<x_limits.size()-1;i++)
537  if((x>=x_limits[i]) && (x<=x_limits[i+1]))
538  {
539  y1 = y_limits[i];
540  y2 = y_limits[i+1];
541  x1 = x_limits[i];
542  x2 = x_limits[i+1];
543  break;
544  }
545  }
546 
547  double pow_val = -2.0*beta*(x-(x1+x2)/2);
548 
549  return (y2-y1)/(std::pow(1+std::exp(pow_val),penal_fac)) + y1;
550  }

553  double ProjectBackward(double y,Vector x_limits,Vector y_limits,double beta){
554 
555  double x = 0;
556  if(y>=y_limits[y_limits.size()-1])
557  x = x_limits[y_limits.size()-1];
558  else if(y<=y_limits[0])
559  x = x_limits[0];
560  else{
561  for(long unsigned int i=0;i<y_limits.size()-1;i++)
562  if((y>y_limits[i]) && (y<y_limits[i+1]))
563  {
564  double y1 = y_limits[i];
565  double y2 = y_limits[i+1];
566  double x1 = x_limits[i];
567  double x2 = x_limits[i+1];
568  x = ((x2+x1)/2.0) + (1.0/(-2.0*beta)) * std::log(((y2-y1)/(y-y1))-1);
569  break;
570  }
571  else if(y==y_limits[i]){
572  x = x_limits[i];
573  break;
574  }
575  else if(y==y_limits[i+1]){
576  x = x_limits[i+1];
577  break;
578  }
579  }
580  return x;
581  }

584  double x1=0.0,x2=0.0,y1=0.0,y2=0.0;
585  if(x>=x_limits[x_limits.size()-1]){
586  x1=x_limits[x_limits.size()-2];

588  y1=y_limits[y_limits.size()-2];
589  y2=y_limits[y_limits.size()-1];
590  }
591  else if(x<=x_limits[0]){
592  x1=x_limits[0];
593  x2=x_limits[1];
594  y1=y_limits[0];
595  y2=y_limits[1];
596  }
597  else{
598  for(long unsigned int i=0;i<x_limits.size()-1;i++)
599  if((x>=x_limits[i]) && (x<=x_limits[i+1]))
600  {
601  y1 = y_limits[i];

603  x1 = x_limits[i];
604  x2 = x_limits[i+1];
605  break;
606  }
607  }
608 
609  double pow_val = -2.0*beta*(x-(x1+x2)/2);
610  double dydx = (y2-y1) * (1.0/std::pow(1+std::exp(pow_val),penal_fac+1)) * penal_fac * 2.0 * beta * std::exp(pow_val);
611 
612  // if (y2<y1)
613  // dydx *=-1;
614 
615  return dydx;
