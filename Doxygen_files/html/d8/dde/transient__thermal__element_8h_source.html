66  void CalculateLocalSystem(MatrixType& rLeftHandSideMatrix,
67  VectorType& rRightHandSideVector,
68  const ProcessInfo& rCurrentProcessInfo) override
69  {
70  KRATOS_TRY
71 
72  GeometryType::ShapeFunctionsGradientsType dN_dX_container;
73  Vector det_J_container;
74 
75  // ShapreFunctionsIntegrationsPointsGradients does not allow for the line element in 2D/3D
76  // configuration and will produce errors. To circumvent this, the dN_dX_container is
77  // separately computed with correct dimensions for the line element.
78  if (GetGeometry().LocalSpaceDimension() == 1) {
79  GetGeometry().DeterminantOfJacobian(det_J_container, this->GetIntegrationMethod());
80  dN_dX_container = GetGeometry().ShapeFunctionsLocalGradients(this->GetIntegrationMethod());
81  std::transform(dN_dX_container.begin(), dN_dX_container.end(), det_J_container.begin(),
82  dN_dX_container.begin(), std::divides<>());
83  } else {
84  GetGeometry().ShapeFunctionsIntegrationPointsGradients(dN_dX_container, det_J_container,
85  GetIntegrationMethod());
86  }
87 
88  const auto integration_coefficients = CalculateIntegrationCoefficients(det_J_container);
89  const auto conductivity_matrix = CalculateConductivityMatrix(dN_dX_container, integration_coefficients);
90  const auto capacity_matrix = CalculateCapacityMatrix(integration_coefficients);
91 
92  AddContributionsToLhsMatrix(rLeftHandSideMatrix, conductivity_matrix, capacity_matrix,
93  rCurrentProcessInfo[DT_TEMPERATURE_COEFFICIENT]);
94  AddContributionsToRhsVector(rRightHandSideVector, conductivity_matrix, capacity_matrix);
95 
96  KRATOS_CATCH("")
97  }

226  const BoundedMatrix<double, TNumNodes, TNumNodes>& rConductivityMatrix,
227  const BoundedMatrix<double, TNumNodes, TNumNodes>& rCapacityMatrix,

234  void AddContributionsToRhsVector(VectorType& rRightHandSideVector,
235  const BoundedMatrix<double, TNumNodes, TNumNodes>& rConductivityMatrix,
236  const BoundedMatrix<double, TNumNodes, TNumNodes>& rCapacityMatrix) const
237  {
238  const auto capacity_vector =
239  array_1d<double, TNumNodes>{-prod(rCapacityMatrix, GetNodalValuesOf(DT_TEMPERATURE))};
240  rRightHandSideVector = capacity_vector;
241  const auto conductivity_vector =
242  array_1d<double, TNumNodes>{-prod(rConductivityMatrix, GetNodalValuesOf(TEMPERATURE))};
243  rRightHandSideVector += conductivity_vector;
244  }

246  Vector CalculateIntegrationCoefficients(const Vector& rDetJContainer) const
247  {
248  const auto& r_properties = GetProperties();
249  const auto& r_integration_points = GetGeometry().IntegrationPoints(GetIntegrationMethod());
250 
251  auto result = Vector{r_integration_points.size()};
252  for (unsigned int integration_point_index = 0;
253  integration_point_index < r_integration_points.size(); ++integration_point_index) {
254  result[integration_point_index] = r_integration_points[integration_point_index].Weight() *
255  rDetJContainer[integration_point_index];
256  if (GetGeometry().LocalSpaceDimension() == 1) {
257  result[integration_point_index] *= r_properties[CROSS_AREA];
258  }
259  }
260 
261  return result;
262  }

264  BoundedMatrix<double, TNumNodes, TNumNodes> CalculateConductivityMatrix(
265  const GeometryType::ShapeFunctionsGradientsType& rShapeFunctionGradients,
266  const Vector& rIntegrationCoefficients) const
267  {
268  const auto law = CreateThermalLaw();
269  const auto constitutive_matrix = law->CalculateThermalDispersionMatrix(GetProperties());
270 
271  auto result = BoundedMatrix<double, TNumNodes, TNumNodes>{ZeroMatrix{TNumNodes, TNumNodes}};
272  for (unsigned int integration_point_index = 0;
273  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
274  ++integration_point_index) {
275  BoundedMatrix<double, TDim, TNumNodes> Temp =
276  prod(constitutive_matrix, trans(rShapeFunctionGradients[integration_point_index]));
277  result += prod(rShapeFunctionGradients[integration_point_index], Temp) *
278  rIntegrationCoefficients[integration_point_index];
279  }
280 
281  return result;
282  }

284  BoundedMatrix<double, TNumNodes, TNumNodes> CalculateCapacityMatrix(const Vector& rIntegrationCoefficients) const
285  {
286  const auto& r_properties = GetProperties();
287  RetentionLaw::Parameters parameters(r_properties);
288  auto retention_law = RetentionLawFactory::Clone(r_properties);
289  const double saturation = retention_law->CalculateSaturation(parameters);
290  const auto c_water = r_properties[POROSITY] * saturation * r_properties[DENSITY_WATER] *
291  r_properties[SPECIFIC_HEAT_CAPACITY_WATER];
292  const auto c_solid = (1.0 - r_properties[POROSITY]) * r_properties[DENSITY_SOLID] *
293  r_properties[SPECIFIC_HEAT_CAPACITY_SOLID];
294 
295  auto result = BoundedMatrix<double, TNumNodes, TNumNodes>{ZeroMatrix{TNumNodes, TNumNodes}};
296  const auto& r_N_container = GetGeometry().ShapeFunctionsValues(GetIntegrationMethod());
297  for (unsigned int integration_point_index = 0;
298  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
299  ++integration_point_index) {
300  const auto N = Vector{row(r_N_container, integration_point_index)};
301  result += (c_water + c_solid) * outer_prod(N, N) * rIntegrationCoefficients[integration_point_index];
302  }
303 
304  return result;
305  }

317  std::unique_ptr<GeoThermalLaw> CreateThermalLaw() const
318  {
319  const std::size_t number_of_dimensions = GetGeometry().LocalSpaceDimension();
320  std::unique_ptr<GeoThermalLaw> law;
321 
322  if (GetProperties().Has(THERMAL_LAW_NAME)) {
323  const std::string& rThermalLawName = GetProperties()[THERMAL_LAW_NAME];
324  if (rThermalLawName == "GeoThermalDispersionLaw") {
325  law = std::make_unique<GeoThermalDispersionLaw>(number_of_dimensions);
326  } else if (rThermalLawName == "GeoThermalFilterLaw") {
327  law = std::make_unique<GeoThermalFilterLaw>();
328  } else {
329  KRATOS_ERROR << "Undefined THERMAL_LAW_NAME! " << rThermalLawName << std::endl;
330  }
331  } else {
332  law = std::make_unique<GeoThermalDispersionLaw>(number_of_dimensions);
333  }
334  return law;
335  }
