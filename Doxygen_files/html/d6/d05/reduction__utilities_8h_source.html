344 template <class... Reducer>
345 struct CombinedReduction {
346  typedef std::tuple<typename Reducer::value_type...> value_type;
347  typedef std::tuple<typename Reducer::return_type...> return_type;
348 
349  std::tuple<Reducer...> mChild;
350 
351  CombinedReduction() {}
352 
354  return_type GetValue(){
355  return_type return_value;
356  fill_value<0>(return_value);
357  return return_value;
358  }
359 
360  template <int I, class T>
361  typename std::enable_if<(I < sizeof...(Reducer)), void>::type
362  fill_value(T& v) {
363  std::get<I>(v) = std::get<I>(mChild).GetValue();
364  fill_value<I+1>(v);
365  };
366 
367  template <int I, class T>
368  typename std::enable_if<(I == sizeof...(Reducer)), void>::type
369  fill_value(T& v) {}

372  template <class... T>
373  void LocalReduce(const std::tuple<T...> &&v) {
374  // Static recursive loop over tuple elements
375  reduce_local<0>(v);
376  }
377 
379  void ThreadSafeReduce(const CombinedReduction &other) {
380  reduce_global<0>(other);
381  }
382 
383  private:
384 
385  template <int I, class T>
386  typename std::enable_if<(I < sizeof...(Reducer)), void>::type
387  reduce_local(T &&v) {
388  std::get<I>(mChild).LocalReduce(std::get<I>(v));
389  reduce_local<I+1>(std::forward<T>(v));
390  };
