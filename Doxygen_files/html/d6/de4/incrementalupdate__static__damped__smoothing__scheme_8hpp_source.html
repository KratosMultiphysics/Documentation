80  void Initialize(ModelPart& r_model_part) override
81  {
82  KRATOS_TRY
83 
84  mDeltaTime = r_model_part.GetProcessInfo()[DELTA_TIME];
85 
86  // Pure Newmark scheme
87  // Note: this is a quasistatic scheme. The velocity and acceleration are stored without affecting the displacement
88  mbeta = 0.25;
89  mgamma = 0.5;
90 
91  mNewmark0 = ( 1.0 / (mbeta * mDeltaTime * mDeltaTime) );
92  mNewmark1 = ( mgamma / (mbeta * mDeltaTime) );
93  mNewmark2 = ( 1.0 / (mbeta * mDeltaTime) );
94  mNewmark3 = ( 0.5 / (mbeta) - 1.0 );
95  mNewmark4 = ( (mgamma / mbeta) - 1.0 );
96  mNewmark5 = ( mDeltaTime * 0.5 * ( ( mgamma / mbeta ) - 2.0 ) );
97 
98  r_model_part.GetProcessInfo()[RAYLEIGH_ALPHA] = mRayleighAlpha;
99  r_model_part.GetProcessInfo()[RAYLEIGH_BETA] = mRayleighBeta;
100 
101  // Initialize INITIAL_STRESS_TENSOR
102  block_for_each(r_model_part.Nodes(), [](Node& rNode){
103  auto& r_initial_stress = rNode.FastGetSolutionStepValue(INITIAL_STRESS_TENSOR);
104  if (r_initial_stress.size1() != 3 || r_initial_stress.size2() != 3) {
105  r_initial_stress.resize(3,3,false);
106  }
107  r_initial_stress.clear();
108  });
109 
110  mSchemeIsInitialized = true;
111 
112  KRATOS_CATCH("")
113  }
