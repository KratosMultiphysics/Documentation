78  static inline void GetRotationMatrix(
79  MatrixType& rRotationMatrix,
80  const VectorType& rNormalVector,
81  const unsigned int Dimension
82  )
83  {
84  if(Dimension == 2){
85  if (rRotationMatrix.size1() != 2 && rRotationMatrix.size2() != 2)
86  rRotationMatrix.resize(2,2,false);
87  noalias(rRotationMatrix) = IdentityMatrix(Dimension);
88 
89  double aux = rNormalVector[0]*rNormalVector[0] + rNormalVector[1]*rNormalVector[1];
90  aux = std::sqrt(aux);
91  if (std::abs(aux) < std::numeric_limits<double>::epsilon()) aux = std::numeric_limits<double>::epsilon();
92 
93  rRotationMatrix(0,0) = rNormalVector[0]/aux;
94  rRotationMatrix(0,1) = rNormalVector[1]/aux;
95  rRotationMatrix(1,0) = -rNormalVector[1]/aux;
96  rRotationMatrix(1,1) = rNormalVector[0]/aux;
97  }
98  else if (Dimension == 3){
99  if (rRotationMatrix.size1() != 3 && rRotationMatrix.size2() != 3)
100  rRotationMatrix.resize(2,2,false);
101  noalias(rRotationMatrix) = IdentityMatrix(Dimension);
102 
103  double aux = rNormalVector[0]*rNormalVector[0] + rNormalVector[1]*rNormalVector[1] + rNormalVector[2]*rNormalVector[2];
104  aux = std::sqrt(aux);
105  if (std::abs(aux) < std::numeric_limits<double>::epsilon()) aux = std::numeric_limits<double>::epsilon();
106 
107  rRotationMatrix(0,0) = rNormalVector[0]/aux;
108  rRotationMatrix(0,1) = rNormalVector[1]/aux;
109  rRotationMatrix(0,2) = rNormalVector[2]/aux;
110 
111  // Define the new coordinate system, where the first vector is aligned with the normal
112  // To choose the remaining two vectors, we project the first component of the cartesian base to the tangent plane
113  Vector rT1 = ZeroVector(3);
114  rT1(0) = 1.0;
115  rT1(1) = 0.0;
116  rT1(2) = 0.0;
117  double dot = rRotationMatrix(0,0); //this->Dot(rN,rT1);
118 
119  // It is possible that the normal is aligned with (1,0,0), resulting in norm(rT1) = 0
120  // If this is the case, repeat the procedure using (0,1,0)
121  if ( std::abs(dot) > 0.99 )
122  {
123  rT1(0) = 0.0;
124  rT1(1) = 1.0;
125  rT1(2) = 0.0;
126 
127  dot = rRotationMatrix(0,1); //this->Dot(rN,rT1);
128  }
129 
130  // Calculate projection and normalize
131  rT1[0] -= dot*rRotationMatrix(0,0);
132  rT1[1] -= dot*rRotationMatrix(0,1);
133  rT1[2] -= dot*rRotationMatrix(0,2);
134  MPMMathUtilities<double>::Normalize(rT1);
135 
136  rRotationMatrix(1,0) = rT1[0];
137  rRotationMatrix(1,0) = rT1[1];
138  rRotationMatrix(1,0) = rT1[2];
139 
140  // The third base component is choosen as N x T1, which is normalized by construction
141  rRotationMatrix(2,0) = rRotationMatrix(0,1)*rT1[2] - rRotationMatrix(0,2)*rT1[1];
142  rRotationMatrix(2,1) = rRotationMatrix(0,2)*rT1[0] - rRotationMatrix(0,0)*rT1[2];
143  rRotationMatrix(2,2) = rRotationMatrix(0,0)*rT1[1] - rRotationMatrix(0,1)*rT1[0];
144  }
145  else{
146  KRATOS_ERROR << "Dimension given is wrong!" << std::endl;
147  }
148  }

221  static inline void QRFactorization(const MatrixType& A, MatrixType& Q, MatrixType& R)
222  {
223 
224  if(A.size1()!= A.size2())
225  KRATOS_ERROR<<" GIVEN MATRIX IS NOT A SQUARE MATRIX: QRFactorization calculation"<<std::endl;
226 
227  // QR Factorization with Householder-Algorithm
228  unsigned int dimension= A.size1();
229 
230  Vector y(dimension);
231 
232  Vector w(dimension);
233 
234  R.resize(dimension,dimension,false);
235 
236  noalias(R)=ZeroMatrix(dimension, dimension);
237 
238  Q.resize(dimension,dimension,false);
239 
240  noalias(Q)=ZeroMatrix(dimension, dimension);
241 
242  Matrix Help(A.size1(),A.size2());
243  noalias(Help) = A;
244 
245  Matrix unity(dimension,dimension);
246  noalias(unity) = ZeroMatrix(dimension, dimension);
247 
248  for(unsigned int j=0; j<dimension; j++)
249  unity(j,j)=1.0;
250 
251  std::vector<Matrix> HelpQ(dimension-1);
252 
253  std::vector<Matrix> HelpR(dimension-1);
254 
255  for(unsigned int i=0; i< dimension-1; i++)
256  {
257  HelpQ[i].resize(dimension,dimension,false);
258  HelpR[i].resize(dimension,dimension,false);
259  noalias(HelpQ[i])= unity;
260  noalias(HelpR[i])= ZeroMatrix(dimension, dimension);
261  }
262 
263  for(unsigned int iteration=0; iteration< dimension-1; iteration++)
264  {
265  // Vector y
266  for(unsigned int i=iteration; i<dimension; i++)
267  y[i]= Help(i,iteration);
268 
269 
270  // Helpvalue l
271  double normy=0.0;
272 
273  for(unsigned int i=iteration; i<dimension; i++)
274  normy += y[i]*y[i];
275 
276  normy= std::sqrt(normy);
277 
278  double l= std::sqrt((normy*(normy+std::abs(y(iteration))))/2);
279 
280  double k=0.0;
281 
282  if(y[iteration] !=0)
283  k= - y(iteration)/std::abs(y(iteration))*normy;
284  else
285  k= -normy;
286 
287  for(unsigned int i=iteration; i<dimension; i++)
288  {
289  double e=0;
290 
291  if(i==iteration)
292  e=1;
293 
294  w[i]= 1/(2*l)*(y[i]-k*e);
295  }
296 
297  for(unsigned int i=iteration; i<dimension; i++)
298  for(unsigned int j=iteration; j<dimension; j++)
299  HelpQ[iteration](i,j)= unity(i,j)- 2*w[i]*w[j];
300 
301 
302  for(unsigned int i=iteration; i<dimension; i++)
303  for(unsigned int j=iteration; j<dimension; j++)
304  for(unsigned int k=iteration; k<dimension; k++)
305  HelpR[iteration](i,j)+= HelpQ[iteration](i,k)*Help(k,j);
306 
307  Help= HelpR[iteration];
308 
309  }
310 
311  // Assembling R
312  for(unsigned int k=0; k<dimension-1; k++)
313  {
314  for(unsigned int i=k; i<dimension; i++)
315  for(unsigned int j=k; j<dimension; j++)
316  R(i,j) =HelpR[k](i,j);
317 
318  }
319 
320  for(unsigned int k=1; k<dimension-1; k++)
321  {
322  for(unsigned int i=0; i<dimension; i++)
323  for(unsigned int j=0; j<dimension; j++)
324  for(unsigned int l=0; l<dimension; l++)
325  Q(i,j)+= HelpQ[(k-1)](i,l)*HelpQ[k](l,j);
326  noalias(HelpQ[k])=Q;
327  }
328  if(dimension-1==1)
329  noalias(Q)=HelpQ[0];
330 
331  }

343  static inline Vector EigenValues(const Matrix& A, const double rTolerance = 1e-9, const double rZeroTolerance = 1e-9)
344  {
345  unsigned int dimension= A.size1();
346 
347  Matrix Convergence(2,dimension);
348  noalias(Convergence) = ZeroMatrix(2,dimension);
349 
350  double delta = 0.0;
351 
352  double abs = 0.0;
353 
354  Vector Result(dimension);
355  noalias(Result) = ZeroVector(dimension);
356 
357  Matrix HelpA(dimension,dimension);
358  noalias(HelpA) = ZeroMatrix(dimension, dimension);

361  noalias(HelpQ) = ZeroMatrix(dimension, dimension);
362

364  noalias(HelpR) = ZeroMatrix(dimension, dimension);
365 
366  HelpA=A;
367 
368  bool is_converged=false;
369  unsigned int max_iters = 10;
370  unsigned int iter = 0;
371 
372  while(is_converged == false && iter<max_iters )
373  {
374  double shift= HelpA((dimension-1),(dimension-1));
375 
376  for(unsigned int i=0; i<dimension; i++)
377  {
378  HelpA(i,i) = HelpA(i,i)- shift;
379  }
380 
381  MPMMathUtilities<double>::QRFactorization(HelpA, HelpQ, HelpR);
382 
383  HelpA= ZeroMatrix(dimension, dimension);
384 
385  for(unsigned int i=0; i<dimension; i++)
386  {
387  HelpA(i,i) += shift;
388  for(unsigned int j=0; j< dimension; j++)
389  {
390  for(unsigned int k=0; k< dimension; k++)
391  {
392  HelpA(i,j) += HelpR(i,k)*HelpQ(k,j);
393  }
394  }
395  }
396 
397  delta= 0.0;
398 
399  abs = 0.0;
400 
401  for(unsigned int i=0; i<dimension; i++)
402  {
403  Convergence(0,i)=Convergence(1,i);
404  Convergence(1,i)=HelpA(i,i);
405  delta+= (Convergence(1,i)-Convergence(0,i))*(Convergence(1,i)-Convergence(0,i));
406  abs+=(Convergence(1,i))*(Convergence(1,i));
407  }
408 
409  delta= std::sqrt(delta);
410 
411  abs=std::sqrt(abs);
412 
413  if(abs < rZeroTolerance)
414  abs=1.0;
415 
416  if(delta < rZeroTolerance || (delta/abs) < rTolerance)
417  is_converged=true;
418 
419  iter++;
420  }
421 
422 
423  for(unsigned int i=0; i<dimension; i++)
424  {
425  Result[i]= HelpA(i,i);
426 
427  if(std::abs(Result[i]) <rZeroTolerance)
428  Result[i]=0.0;
429  }
430 
431  return Result;

441  static inline Vector EigenValuesDirectMethod(const Matrix& A)
442  {
443  // Given a real symmetric 3x3 matrix A, compute the eigenvalues
444  unsigned int dimension= A.size1();
445  Vector Result(dimension);
446  noalias(Result) = ZeroVector(dimension);
447 
448  const double p1 = A(0,1)*A(0,1) + A(0,2)*A(0,2) + A(1,2)*A(1,2);
449  if (p1 == 0)
450  {//A is diagonal.
451  Result[0] = A(0,0);
452  Result[1] = A(1,1);
453  Result[2] = A(2,2);
454  return Result;
455  }
456 
457  const double q = (A(0,0) + A(1,1) + A(2,2)) / 3.0;
458  const double p2 = (A(0,0) - q) * (A(0,0) - q) + (A(1,1) - q) * (A(1,1) - q) + (A(2,2) - q) * (A(2,2) - q) + 2.0 * p1;
459  const double p = std::sqrt(p2 / 6.0);
460 
461  Matrix B(3,3);
462  const double inv_p = 1.0/p;
463 
464  // B = (1 / p) * (A - q * I) where I is the identity matrix
465  B(0,0) = inv_p * (A(0,0) - q);
466  B(1,1) = inv_p * (A(1,1) - q);
467  B(2,2) = inv_p * (A(2,2) - q);
468  B(0,1) = inv_p * A(0,1);
469  B(1,0) = inv_p * A(1,0);
470  B(0,2) = inv_p * A(0,2);
471  B(2,0) = inv_p * A(2,0);
472  B(1,2) = inv_p * A(1,2);
473  B(2,1) = inv_p * A(2,1);
474 
475  //r = det(B) / 2
476  double r = 0.5 * ( B(0,0)*B(1,1)*B(2,2) + B(0,1)*B(1,2)*B(2,0) + B(1,0)*B(2,1)*B(0,2) - B(2,0)*B(1,1)*B(0,2) - B(1,0)*B(0,1)*B(2,2) - B(0,0)*B(2,1)*B(1,2) );
477 
478  // In exact arithmetic for a symmetric matrix -1 <= r <= 1
479  // but computation error can leave it slightly outside this range.
480  double phi = 0.0;
481  if (r <= -1) { phi = Globals::Pi / 3.0; }
482  else if (r >= 1) { phi = 0.0; }
483  else { phi = acos(r) / 3.0;}
484 
485  // the eigenvalues satisfy eig3 <= eig2 <= eig1
486  Result[0] = q + 2.0 * p * cos(phi);
487  Result[2] = q + 2.0 * p * cos(phi + (2.0 * Globals::Pi / 3.0));
488  Result[1] = 3.0 * q - Result[0] - Result[2]; //% since trace(A) = eig1 + eig2 + eig3
489 
490  return Result;
491  }
