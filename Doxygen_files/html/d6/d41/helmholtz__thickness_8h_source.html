521  double ProjectForward(double x,Vector x_limits,Vector y_limits,double beta, int penal_fac = 1){
522 
523  double x1=0,x2=0,y1=0,y2=0;
524  if(x>=x_limits[x_limits.size()-1]){
525  x1=x_limits[x_limits.size()-2];
526  x2=x_limits[x_limits.size()-1];
527  y1=y_limits[y_limits.size()-2];
528  y2=y_limits[y_limits.size()-1];
529  }
530  else if(x<=x_limits[0]){
531  x1=x_limits[0];
532  x2=x_limits[1];
533  y1=y_limits[0];
534  y2=y_limits[1];
535  }
536  else{
537  for(long unsigned int i=0;i<x_limits.size()-1;i++)
538  if((x>=x_limits[i]) && (x<=x_limits[i+1]))
539  {
540  y1 = y_limits[i];
541  y2 = y_limits[i+1];
542  x1 = x_limits[i];
543  x2 = x_limits[i+1];
544  break;
545  }
546  }
547 
548  double pow_val = -2.0*beta*(x-(x1+x2)/2);
549 
550  if(pow_val>700)
551  pow_val = 700;
552  if(pow_val<-700)
553  pow_val = -700;
554 
555  return (y2-y1)/(std::pow(1+std::exp(pow_val),penal_fac)) + y1;
556  }

559  double ProjectBackward(double y,Vector x_limits,Vector y_limits,double beta){
560 
561  double x = 0;
562  if(y>=y_limits[y_limits.size()-1])
563  x = x_limits[y_limits.size()-1];
564  else if(y<=y_limits[0])
565  x = x_limits[0];
566  else{
567  for(long unsigned int i=0;i<y_limits.size()-1;i++)
568  if((y>y_limits[i]) && (y<y_limits[i+1]))
569  {
570  double y1 = y_limits[i];
571  double y2 = y_limits[i+1];
572  double x1 = x_limits[i];
573  double x2 = x_limits[i+1];
574  x = ((x2+x1)/2.0) + (1.0/(-2.0*beta)) * std::log(((y2-y1)/(y-y1))-1);
575  break;
576  }
577  else if(y==y_limits[i]){
578  x = x_limits[i];
579  break;
580  }
581  else if(y==y_limits[i+1]){
582  x = x_limits[i+1];
583  break;
584  }
585  }
586  return x;
587  }
