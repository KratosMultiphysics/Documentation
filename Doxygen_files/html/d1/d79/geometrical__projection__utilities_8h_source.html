89  template<class TGeometryType>
90  static inline double FastProjectDirection(
91  const TGeometryType& rGeom,
92  const PointType& rPointToProject,
93  PointType& rPointProjected,
94  const array_1d<double,3>& rNormal,
95  const array_1d<double,3>& rVector,
96  const SizeType EchoLevel = 0
97  )
98  {
99  // Zero tolerance
100  const double zero_tolerance = std::numeric_limits<double>::epsilon();
101 
102  // We define the distance
103  double distance = 0.0;
104 
105  const array_1d<double,3> vector_points = rGeom[0].Coordinates() - rPointToProject.Coordinates();
106 
107  if( norm_2( rVector ) < zero_tolerance && norm_2( rNormal ) > zero_tolerance ) {
108  distance = inner_prod(vector_points, rNormal)/norm_2(rNormal);
109  noalias(rPointProjected.Coordinates()) = rPointToProject.Coordinates() + rVector * distance;
110  KRATOS_WARNING_IF("GeometricalProjectionUtilities", EchoLevel > 0) << "WARNING:: Zero projection vector. Projection using the condition vector instead." << std::endl;
111  } else if (std::abs(inner_prod(rVector, rNormal) ) > zero_tolerance) {
112  distance = inner_prod(vector_points, rNormal)/inner_prod(rVector, rNormal);
113  noalias(rPointProjected.Coordinates()) = rPointToProject.Coordinates() + rVector * distance;
114  } else {
115  noalias(rPointProjected.Coordinates()) = rPointToProject.Coordinates();
116  KRATOS_WARNING_IF("GeometricalProjectionUtilities", EchoLevel > 0) << "WARNING:: The line and the plane are coplanar. Something wrong happened " << std::endl;
117  }
118 
119  return distance;
120  }

190  template<class TGeometryType>
191  static inline double FastProjectOnLine(const TGeometryType& rGeometry,
192  const PointType& rPointToProject,
193  PointType& rPointProjected)
194  {
195  const array_1d<double, 3>& r_p_a = rGeometry[0].Coordinates();
196  const array_1d<double, 3>& r_p_b = rGeometry[1].Coordinates();
197  const array_1d<double, 3> ab = r_p_b - r_p_a;
198 
199  const array_1d<double, 3>& p_c = rPointToProject.Coordinates();
200 
201  const double factor = (inner_prod(r_p_b, p_c) - inner_prod(r_p_a, p_c) - inner_prod(r_p_b, r_p_a) + inner_prod(r_p_a, r_p_a)) / inner_prod(ab, ab);
202 
203  rPointProjected.Coordinates() = r_p_a + factor * ab;
204 
205  return norm_2(rPointProjected.Coordinates()-p_c);
206  }

265  template<class TGeometryType, class TPointClass1, class TPointClass2 = TPointClass1>
266  static inline double FastProjectOnLine2D(
267  const TGeometryType& rGeometry,
268  const TPointClass1& rPointToProject,
269  TPointClass2& rPointProjected
270  )
271  {
272  // Node coordinates
273  const array_1d<double, 3>& r_p_a = rGeometry[0];
274  const array_1d<double, 3>& r_p_b = rGeometry[1];
275  const array_1d<double, 3>& r_p_c = rPointToProject;
276 
277  // We compute the normal
278  array_1d<double,3> normal;
279  normal[0] = r_p_b[1] - r_p_a[1];
280  normal[1] = r_p_a[0] - r_p_b[0];
281  normal[2] = 0.0;
282 
283  const double norm_normal = norm_2(normal);
284  KRATOS_ERROR_IF(norm_normal <= std::numeric_limits<double>::epsilon()) << "Zero norm normal: X: " << normal[0] << "\tY: " << normal[1] << std::endl;
285  normal /= norm_normal;
286 
287  const array_1d<double,3> vector_points = r_p_a - r_p_c;
288  const double distance = inner_prod(vector_points, normal);
289  noalias(rPointProjected) = r_p_c + normal * distance;
290 
291  return distance;
292  }

302  template<class TGeometryType>
303  static inline bool ProjectIterativeLine2D(
304  TGeometryType& rGeomOrigin,
305  const array_1d<double,3>& rPointDestiny,
306  array_1d<double,3>& rResultingPoint,
307  const array_1d<double, 3>& rNormal,
308  const double Tolerance = 1.0e-8,
309  double DeltaXi = 0.5
310  )
311  {
312  //rResultingPoint.clear();
313 
314  double old_delta_xi = 0.0;
315 
316  array_1d<double, 3> current_global_coords;
317 
318  KRATOS_DEBUG_CHECK_VARIABLE_IN_NODAL_DATA(NORMAL, rGeomOrigin[0])
319  KRATOS_DEBUG_CHECK_VARIABLE_IN_NODAL_DATA(NORMAL, rGeomOrigin[1])
320 
321  array_1d<array_1d<double, 3>, 2> normals;
322  normals[0] = rGeomOrigin[0].FastGetSolutionStepValue(NORMAL);
323  normals[1] = rGeomOrigin[1].FastGetSolutionStepValue(NORMAL);
324 
325  BoundedMatrix<double,2,2> X;
326  BoundedMatrix<double,2,1> DN;
327  for(unsigned int i=0; i<2;++i) {
328  X(0,i) = rGeomOrigin[i].X();
329  X(1,i) = rGeomOrigin[i].Y();
330  }
331 
332  Matrix J = ZeroMatrix( 1, 1 );
333 
334  //Newton iteration:
335 
336  const unsigned int max_iter = 20;
337 
338  for ( unsigned int k = 0; k < max_iter; ++k ) {
339  array_1d<double, 2> N_origin;
340  N_origin[0] = 0.5 * ( 1.0 - rResultingPoint[0]);
341  N_origin[1] = 0.5 * ( 1.0 + rResultingPoint[0]);
342 
343  array_1d<double,3> normal_xi = N_origin[0] * normals[0] + N_origin[1] * normals[1];
344  normal_xi = normal_xi/norm_2(normal_xi);
345 
346  current_global_coords = ZeroVector(3);
347  for( IndexType i_node = 0; i_node < 2; ++i_node )
348  current_global_coords += N_origin[i_node] * rGeomOrigin[i_node].Coordinates();
349 
350  const array_1d<double,3> VectorPoints = rGeomOrigin.Center() - rPointDestiny;
351  const double distance = inner_prod(VectorPoints, rNormal)/inner_prod(-normal_xi, rNormal);
352  const array_1d<double, 3> current_destiny_global_coords = rPointDestiny - normal_xi * distance;
353 
354  // Derivatives of shape functions
355  Matrix ShapeFunctionsGradients;
356  ShapeFunctionsGradients = rGeomOrigin.ShapeFunctionsLocalGradients(ShapeFunctionsGradients, rResultingPoint );
357 
358  noalias(DN) = prod(X,ShapeFunctionsGradients);
359 
360  noalias(J) = prod(trans(DN),DN); // TODO: Add the non linearity concerning the normal
361 
362  const Vector RHS = prod(trans(DN),subrange(current_destiny_global_coords - current_global_coords,0,2));
363 
364  old_delta_xi = DeltaXi;
365  DeltaXi = RHS[0]/J(0, 0);
366 
367  rResultingPoint[0] += DeltaXi;
368 
369  if (rResultingPoint[0] <= -1.0)
370  rResultingPoint[0] = -1.0;
371  else if (rResultingPoint[0] >= 1.0)
372  rResultingPoint[0] = 1.0;
373 
374  if ( std::abs(DeltaXi - old_delta_xi) < Tolerance )
375  return true;
376  }
377 
378  return false;
379  }
