987  bounded_matrix<double,2,2> check;

992  h += pgeom[1].FastGetSolutionStepValue(NODAL_H);
993  h += pgeom[2].FastGetSolutionStepValue(NODAL_H);

995

1006

1009 
1010  double x0_2 = x0*x0 + y0*y0;
1011  double x1_2 = x1*x1 + y1*y1;

1028  {
1029  return true;
1030  }
