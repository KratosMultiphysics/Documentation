474  double h = 0.0;

477  {
478  double x = i->X();

480  double z = i->Z();
481  double l = (x - xc)*(x - xc);
482  l += (y - yc)*(y - yc);
483  l += (z - zc)*(z - zc);
484 
485  if (l > h) h = l;
486  }
487  h = sqrt(h);
488 
489  if(h > h_max) h_max = h;
490 
491  }
