129  double CalculatePressure(const Node& rNode)
130  {
131  // find top boundary

421  void FillListOfBoundaryNodesFast(std::vector<int>& BoundaryNodes)
422  {
423  const int ID_UNDEFINED = -1;
424  const int N_ELEMENT = 10;
425 
426  std::vector<std::vector<int>> ELementsOfNodes;
427  std::vector<int> ELementsOfNodesSize;
428 
429  int MaxNodeID = GetMaxNodeID();
430 
431  ELementsOfNodes.resize(MaxNodeID);
432  ELementsOfNodesSize.resize(MaxNodeID);
433 
434  for (unsigned int i = 0; i < ELementsOfNodes.size(); ++i) {
435  ELementsOfNodes[i].resize(N_ELEMENT);
436  ELementsOfNodesSize[i] = 0;
437  std::fill(ELementsOfNodes[i].begin(), ELementsOfNodes[i].end(), ID_UNDEFINED);
438  }
439 
440  const unsigned int nElements = mrModelPart.NumberOfElements();
441  ModelPart::ElementsContainerType::iterator it_begin_elements = mrModelPart.ElementsBegin();
442 
443  for (unsigned int i = 0; i < nElements; ++i) {
444  ModelPart::ElementsContainerType::iterator pElemIt = it_begin_elements + i;
445  for (unsigned int iPoint = 0; iPoint < pElemIt->GetGeometry().PointsNumber(); ++iPoint) {
446  int NodeID = pElemIt->GetGeometry()[iPoint].Id();
447  int ElementId = pElemIt->Id();
448 
449  int index = NodeID - 1;
450  ELementsOfNodesSize[index]++;
451  if (ELementsOfNodesSize[index] > N_ELEMENT - 1) {
452  ELementsOfNodes[index].push_back(ElementId);
453  } else {
454  ELementsOfNodes[index][ELementsOfNodesSize[index] - 1] = ElementId;
455  }
456  }
457  }
458 
459  for (unsigned int i = 0; i < nElements; ++i) {
460  ModelPart::ElementsContainerType::iterator pElemIt = it_begin_elements + i;
461 
462  int nEdges = pElemIt->GetGeometry().EdgesNumber();
463  for (int iEdge = 0; iEdge < nEdges; ++iEdge) {
464  const unsigned int nPoints = pElemIt->GetGeometry().GenerateEdges()[iEdge].PointsNumber();
465  std::vector<int> FaceID(nPoints);
466  for (unsigned int iPoint = 0; iPoint < nPoints; ++iPoint) {
467  FaceID[iPoint] = pElemIt->GetGeometry().GenerateEdges()[iEdge].GetPoint(iPoint).Id();
468  }
469 
470  if (!IsMoreThanOneElementWithThisEdgeFast(FaceID, ELementsOfNodes, ELementsOfNodesSize)) {
471  // boundary nodes:
472  for (unsigned int iPoint = 0; iPoint < nPoints; ++iPoint) {
473  auto it = std::find(BoundaryNodes.begin(), BoundaryNodes.end(), FaceID[iPoint]);
474  if (it == BoundaryNodes.end()) {
475  BoundaryNodes.push_back(FaceID[iPoint]);
476  }
477  }
478  }
479  }
480  }
481 
482  KRATOS_ERROR_IF(BoundaryNodes.empty())
483  << "No boundary node is found for interpolate line pressure process" << std::endl;

486  bool IsMoreThanOneElementWithThisEdgeFast(const std::vector<int>& rFaceIDs,
487  const std::vector<std::vector<int>>& rELementsOfNodes,
488  const std::vector<int>& rELementsOfNodesSize) const
489 
490  {
491  const int ID_UNDEFINED = -1;
492  int nMaxElements = 0;
493  for (auto node_id : rFaceIDs) {
494  nMaxElements += rELementsOfNodesSize[node_id - 1];
495  }
496 
497  if (nMaxElements > 0) {
498  std::vector<vector<int>> ElementIDs;
499  ElementIDs.resize(rFaceIDs.size());
500  for (unsigned int i = 0; i < ElementIDs.size(); ++i) {
501  ElementIDs[i].resize(nMaxElements);
502  std::fill(ElementIDs[i].begin(), ElementIDs[i].end(), ID_UNDEFINED);
503  }
504 
505  for (unsigned int iPoint = 0; iPoint < rFaceIDs.size(); ++iPoint) {
506  int NodeID = rFaceIDs[iPoint];
507  int index = NodeID - 1;
508  for (int i = 0; i < rELementsOfNodesSize[index]; ++i) {
509  int iElementID = rELementsOfNodes[index][i];
510  ElementIDs[iPoint][i] = iElementID;
511  }
512  }
513 
514  std::vector<int> SharedElementIDs;
515  for (unsigned int iPoint = 0; iPoint < rFaceIDs.size(); ++iPoint) {
516  for (unsigned int i = 0; i < ElementIDs[iPoint].size(); ++i) {
517  int iElementID = ElementIDs[iPoint][i];
518  bool found = false;
519  if (iElementID != ID_UNDEFINED) {
520  for (unsigned int iPointInner = 0; iPointInner < rFaceIDs.size(); ++iPointInner) {
521  if (iPointInner != iPoint) {
522  // std::any_of followed by breaking out of 2 for loops
523  for (unsigned int j = 0; j < ElementIDs[iPointInner].size(); ++j) {
524  if (ElementIDs[iPointInner][j] == iElementID) found = true;
525  }
526  }
527  }
528  }
529 
530  if (found) {
531  auto it = std::find(SharedElementIDs.begin(), SharedElementIDs.end(), iElementID);
532  if (it == SharedElementIDs.end()) {
533  SharedElementIDs.push_back(iElementID);
534  }
535  }
536  }
537  }
538 
539  return SharedElementIDs.size() > 1;
540  }
541 
542  return false;
543  }
