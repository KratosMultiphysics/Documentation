581  VariableUtils().SetHistoricalVariableToZero(NORMAL, r_controlling_object.Nodes());

584  const auto& r_geom = cond_i.GetGeometry();

589  for(auto& node_i : r_geom)

591  node_i.SetLock();

611  void AdjustFilterSizes()
612  {
613  for(long unsigned int obj_i=0;obj_i<mpVMModelPartsTypes.size();obj_i++)
614  if(mpVMModelPartsTypes[obj_i]=="solid"){
615  ModelPart* mpVMModePart = mpVMModelParts[obj_i];
616  Properties::Pointer p_vm_model_part_property = mpVMModelPartsProperties[obj_i];
617 
618  p_vm_model_part_property->SetValue(HELMHOLTZ_BULK_RADIUS_SHAPE,1.0);
619 
620  double total_volume_strain_energy = 0.0;
621  total_volume_strain_energy = block_for_each<SumReduction<double>>(mpVMModePart->Elements(), [&](auto& elem_i) {
622  double elem_strain_energy = 0.0;
623  elem_i.Calculate(ELEMENT_STRAIN_ENERGY,elem_strain_energy,mpVMModePart->GetProcessInfo());
624  return elem_strain_energy;
625  });
626 
627  double total_surface_strain_energy = 0.0;
628  total_surface_strain_energy = block_for_each<SumReduction<double>>(mpVMModePart->Conditions(), [&](auto& cond_i) {
629  double cond_strain_energy;
630  cond_i.Calculate(ELEMENT_STRAIN_ENERGY,cond_strain_energy,mpVMModePart->GetProcessInfo());
631  return cond_strain_energy;
632  });
633  double bulk_surface_ratio = mTechniqueSettings["surface_bulk_ratio"].GetDouble();
634  double bulk_filter_size = (total_surface_strain_energy)/(bulk_surface_ratio * total_volume_strain_energy);
635  bulk_filter_size = std::pow(bulk_filter_size,0.1/0.1);
636  p_vm_model_part_property->SetValue(HELMHOLTZ_BULK_RADIUS_SHAPE,bulk_filter_size);
637

644  void ComputeInitialControlPoints()
645  {
646  for(long unsigned int model_i =0;model_i<mpVMModelParts.size();model_i++)
647  {
648  ModelPart* mpVMModePart = mpVMModelParts[model_i];
649 
650  ProcessInfo &rCurrentProcessInfo = (mpVMModePart)->GetProcessInfo();
651 
652  SetVariableZero(mpVMModePart,HELMHOLTZ_SOURCE_SHAPE);
653  SetVariableZero(mpVMModePart,CX);
654 
655  // // calculate (K+M) * X element wise, here we treat CX as an auxilary
656  rCurrentProcessInfo[COMPUTE_CONTROL_POINTS_SHAPE] = false;
657  block_for_each(mpVMModePart->Nodes(), [&](auto& node_i) {
658  array_3d& r_nodal_variable_hl_vars = node_i.FastGetSolutionStepValue(HELMHOLTZ_VARS_SHAPE);
659  r_nodal_variable_hl_vars(0) = node_i.X0();
660  r_nodal_variable_hl_vars(1) = node_i.Y0();
661  r_nodal_variable_hl_vars(2) = node_i.Z0();
662  });
663 
664  block_for_each(mpVMModePart->Elements(), [&](auto& elem_i) {
665  VectorType rhs;
666  MatrixType lhs;
667  elem_i.Initialize(mpVMModePart->GetProcessInfo());
668  elem_i.CalculateLocalSystem(lhs,rhs,mpVMModePart->GetProcessInfo());
669  AddElementVariableValuesVector(elem_i,CX,rhs,-1.0);
670  });
671 
672  block_for_each(mpVMModePart->Conditions(), [&](auto& cond_i) {
673  VectorType rhs;
674  MatrixType lhs;
675  cond_i.Initialize(mpVMModePart->GetProcessInfo());
676  cond_i.CalculateLocalSystem(lhs,rhs,mpVMModePart->GetProcessInfo());
677  AddConditionVariableValuesVector(cond_i,CX,rhs,-1.0);
678  });
679 
680 
681  // here we fill the RHS of M * S = (K+M) * X
682  SetVariable1ToVarible2(mpVMModePart,CX,HELMHOLTZ_SOURCE_SHAPE);
683  SetVariableZero(mpVMModePart,HELMHOLTZ_VARS_SHAPE);
684 
685  // apply BC on the RHS and unassign BC
686  block_for_each(mpVMModePart->Nodes(), [&](auto& node_i) {
687  array_3d& r_nodal_variable_hl_vars = node_i.FastGetSolutionStepValue(HELMHOLTZ_VARS_SHAPE);
688 
689  if(node_i.IsFixed(HELMHOLTZ_VARS_SHAPE_X))
690  r_nodal_variable_hl_vars(0) = node_i.X0();
691 
692  if(node_i.IsFixed(HELMHOLTZ_VARS_SHAPE_Y))
693  r_nodal_variable_hl_vars(1) = node_i.Y0();
694 
695  if(node_i.IsFixed(HELMHOLTZ_VARS_SHAPE_Z))
696  r_nodal_variable_hl_vars(2) = node_i.Z0();
697  });
698 
699  // here we compute S = M-1 * (K+M) * X
700  rCurrentProcessInfo[COMPUTE_CONTROL_POINTS_SHAPE] = true;
701  mpStrategies[model_i]->Solve();
702  // here we clear/reset the problem for the clearaty's sake
703  mpStrategies[model_i]->GetStrategy()->Clear();
704  SetVariable1ToVarible2(mpVMModePart,HELMHOLTZ_VARS_SHAPE,CX);
705  rCurrentProcessInfo[COMPUTE_CONTROL_POINTS_SHAPE] = false;
706  }
