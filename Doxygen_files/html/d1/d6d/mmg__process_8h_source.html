370  template<class TContainerType>
371  void SetToZeroEntityData(
372  TContainerType& rNewContainer,
373  const TContainerType& rOldContainer
374  )
375  {
376  // Firts we generate the variable list
377  std::unordered_set<std::string> list_variables;
378  const auto it_begin_old = rOldContainer.begin();
379  auto& data = it_begin_old->GetData();
380  for(auto i = data.begin() ; i != data.end() ; ++i) {
381  list_variables.insert((i->first)->Name());
382  }
383 
384  for (auto& var_name : list_variables) {
385  if (KratosComponents<Variable<bool>>::Has(var_name)) {
386  const Variable<bool>& r_var = KratosComponents<Variable<bool>>::Get(var_name);
387  VariableUtils().SetNonHistoricalVariable(r_var, false, rNewContainer);
388  } else if (KratosComponents<Variable<double>>::Has(var_name)) {
389  const Variable<double>& r_var = KratosComponents<Variable<double>>::Get(var_name);
390  VariableUtils().SetNonHistoricalVariable(r_var, 0.0, rNewContainer);
391  } else if (KratosComponents<Variable<array_1d<double, 3>>>::Has(var_name)) {
392  const Variable<array_1d<double, 3>>& r_var = KratosComponents<Variable<array_1d<double, 3>>>::Get(var_name);
393  const array_1d<double, 3> aux_value = ZeroVector(3);
394  VariableUtils().SetNonHistoricalVariable(r_var, aux_value, rNewContainer);
395  } else if (KratosComponents<Variable<array_1d<double, 4>>>::Has(var_name)) {
396  const Variable<array_1d<double, 4>>& r_var = KratosComponents<Variable<array_1d<double, 4>>>::Get(var_name);
397  const array_1d<double, 4> aux_value = ZeroVector(4);
398  VariableUtils().SetNonHistoricalVariable(r_var, aux_value, rNewContainer);
399  } else if (KratosComponents<Variable<array_1d<double, 6>>>::Has(var_name)) {
400  const Variable<array_1d<double, 6>>& r_var = KratosComponents<Variable<array_1d<double, 6>>>::Get(var_name);
401  const array_1d<double, 6> aux_value = ZeroVector(6);
402  VariableUtils().SetNonHistoricalVariable(r_var, aux_value, rNewContainer);
403  } else if (KratosComponents<Variable<array_1d<double, 9>>>::Has(var_name)) {
404  const Variable<array_1d<double, 9>>& r_var = KratosComponents<Variable<array_1d<double, 9>>>::Get(var_name);
405  const array_1d<double, 9> aux_value = ZeroVector(9);
406  VariableUtils().SetNonHistoricalVariable(r_var, aux_value, rNewContainer);
407  } else if (KratosComponents<Variable<Vector>>::Has(var_name)) {
408  const Variable<Vector>& r_var = KratosComponents<Variable<Vector>>::Get(var_name);
409  Vector aux_value = ZeroVector(it_begin_old->GetValue(r_var).size());
410  VariableUtils().SetNonHistoricalVariable(r_var, aux_value, rNewContainer);
411  } else if (KratosComponents<Variable<Matrix>>::Has(var_name)) {
412  const Variable<Matrix>& r_var = KratosComponents<Variable<Matrix>>::Get(var_name);
413  const Matrix& ref_matrix = it_begin_old->GetValue(r_var);
414  Matrix aux_value = ZeroMatrix(ref_matrix.size1(), ref_matrix.size2());
415  VariableUtils().SetNonHistoricalVariable(r_var, aux_value, rNewContainer);
416  }
417  }
418  }
