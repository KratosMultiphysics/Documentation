84 template <class TContainerType, class TContainerDataIO, MeshType TMeshType>
85 template<class TDataType>
86 void SpecializedContainerExpression<TContainerType, TContainerDataIO, TMeshType>::Evaluate(const Variable<TDataType>& rVariable)
87 {
88  KRATOS_TRY
89 
90  auto& r_container = this->GetContainer();
91  const IndexType number_of_entities = r_container.size();
92 
93  if (number_of_entities > 0) {
94  const auto& r_expression = this->GetExpression();
95 
96  VariableExpressionDataIO<TDataType> variable_flatten_data_io(r_expression.GetItemShape());
97 
98  // initialize the container variables first
99  if constexpr(std::is_same_v<TContainerType, ModelPart::NodesContainerType>) {
100  // initializes ghost nodes as for the later synchronization
101  // only, the nodal non historical values needs to be set unless
102  // they are properly initialized. Otherwise, in synchronization, the variables will
103  // not be there in the ghost nodes hence seg faults.
104 
105  // the vectors and matrices needs to be initialized in historical and non-historical
106  // data containers because they need to be initialized with the correct size for synchronization
107  if constexpr(std::is_same_v<TDataType, Vector> || std::is_same_v<TDataType, Matrix>) {
108  TDataType dummy_value{};
109  variable_flatten_data_io.Assign(dummy_value, r_expression, 0);
110  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::Historical>>) {
111  VariableUtils().SetVariable(rVariable, dummy_value, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
112  } else if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
113  VariableUtils().SetNonHistoricalVariable(rVariable, dummy_value, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
114  }
115  } else {
116  // if it is a static type, then it only needs to be initialized in the non-historical container with zeros.
117  // historical container should be initialized with the default values when the container is created.
118  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
119  VariableUtils().SetNonHistoricalVariableToZero(rVariable, this->GetModelPart().GetCommunicator().GhostMesh().Nodes());
120  }
121  }
122  }
123 
124  IndexPartition<IndexType>(number_of_entities).for_each(TDataType{}, [&r_container, &rVariable, &r_expression, &variable_flatten_data_io](const IndexType Index, TDataType& rValue){
125  variable_flatten_data_io.Assign(rValue, r_expression, Index);
126  TContainerDataIO::SetValue(*(r_container.begin() + Index), rVariable, rValue);
127  });
128 
129  if constexpr(std::is_same_v<TContainerType, ModelPart::NodesContainerType>) {
130  // synchronize nodal values
131  auto& r_communicator = this->GetModelPart().GetCommunicator();
132 
133  if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::Historical>>) {
134  r_communicator.SynchronizeVariable(rVariable);
135  } else if constexpr(std::is_same_v<TContainerDataIO, ContainerDataIO<ContainerDataIOTags::NonHistorical>>) {
136  r_communicator.SynchronizeNonHistoricalVariable(rVariable);
137  }
138  }
139  }
140 
141  KRATOS_CATCH("");
142 }
