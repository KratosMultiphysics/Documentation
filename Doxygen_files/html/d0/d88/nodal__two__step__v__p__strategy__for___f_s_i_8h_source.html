1060  void CalcNodalStrainsAndStressesForInterfaceFluidNode(ModelPart::NodeIterator itNode)
1061  {
1062 
1063  ModelPart &rModelPart = BaseType::GetModelPart();
1064  const unsigned int dimension = rModelPart.ElementsBegin()->GetGeometry().WorkingSpaceDimension();
1065  const ProcessInfo &rCurrentProcessInfo = rModelPart.GetProcessInfo();
1066  const double timeInterval = rCurrentProcessInfo[DELTA_TIME];
1067 
1068  double deviatoricCoeff = 0;
1069 
1070  const double currFirstLame = timeInterval * itNode->FastGetSolutionStepValue(BULK_MODULUS);
1071 
1072  Matrix Fgrad = itNode->FastGetSolutionStepValue(NODAL_DEFORMATION_GRAD);
1073  Matrix FgradVel = itNode->FastGetSolutionStepValue(NODAL_DEFORMATION_GRAD_VEL);
1074  double detFgrad = 1.0;
1075  Matrix InvFgrad = ZeroMatrix(dimension, dimension);
1076  Matrix SpatialVelocityGrad = ZeroMatrix(dimension, dimension);
1077 
1078  if (dimension == 2)
1079  {
1080  MathUtils<double>::InvertMatrix2(Fgrad, InvFgrad, detFgrad);
1081  }
1082  else if (dimension == 3)
1083  {
1084  MathUtils<double>::InvertMatrix3(Fgrad, InvFgrad, detFgrad);
1085  }
1086 
1087  // it computes the spatial velocity gradient tensor --> [L_ij]=dF_ik*invF_kj
1088  SpatialVelocityGrad = prod(FgradVel, InvFgrad);
1089 
1090  if (dimension == 2)
1091  {
1092  itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] = SpatialVelocityGrad(0, 0);
1093  itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1] = SpatialVelocityGrad(1, 1);
1094  itNode->FastGetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2] = 0.5 * (SpatialVelocityGrad(1, 0) + SpatialVelocityGrad(0, 1));
1095 
1096  this->ComputeDeviatoricCoefficientForFluid(itNode, dimension, deviatoricCoeff);
1097 
1098  const double DefVol = itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0] + itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1];
1099 
1100  itNode->GetSolutionStepValue(NODAL_VOLUMETRIC_DEF_RATE) = DefVol;
1101 
1102  double nodalSigmaTot_xx = currFirstLame * DefVol + 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[0];
1103  double nodalSigmaTot_yy = currFirstLame * DefVol + 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[1];
1104  double nodalSigmaTot_xy = 2.0 * deviatoricCoeff * itNode->GetSolutionStepValue(NODAL_SPATIAL_DEF_RATE)[2];
