497  CoordinatesArrayType& PointLocalCoordinates(
498  CoordinatesArrayType& rResult,
499  const CoordinatesArrayType& rPoint
500  ) const override
501  {
502  BoundedMatrix<double,3,3> X;
503  BoundedMatrix<double,3,2> DN;
504  for(unsigned int i=0; i<3;i++) {
505  X(0,i ) = 0.5*(this->GetPoint( i ).X() + this->GetPoint( i+3 ).X());
506  X(1,i ) = 0.5*(this->GetPoint( i ).Y() + this->GetPoint( i+3 ).Y());
507  X(2,i ) = 0.5*(this->GetPoint( i ).Z() + this->GetPoint( i+3 ).Z());
508  }
509 
510  double tol = 1.0e-8;
511  int maxiter = 1000;
512 
513  Matrix J = ZeroMatrix( 2, 2 );
514  Matrix invJ = ZeroMatrix( 2, 2 );
515 
516  //starting with xi = 0
517  rResult = ZeroVector( 3 );
518  Vector DeltaXi = ZeroVector( 2 );
519  array_1d<double,3> CurrentGlobalCoords;
520 
521 
522  //Newton iteration:
523  for ( int k = 0; k < maxiter; k++ )
524  {
525  noalias(CurrentGlobalCoords) = ZeroVector( 3 );
526  this->GlobalCoordinates( CurrentGlobalCoords, rResult );
527 
528  noalias( CurrentGlobalCoords ) = rPoint - CurrentGlobalCoords;
529 
530 
531  //derivatives of shape functions
532  Matrix shape_functions_gradients;
533  shape_functions_gradients = ShapeFunctionsLocalGradients(shape_functions_gradients, rResult );
534  noalias(DN) = prod(X,shape_functions_gradients);
535 
536  noalias(J) = prod(trans(DN),DN);
537  Vector res = prod(trans(DN),CurrentGlobalCoords);
538 
539  //deteminant of Jacobian
540  const double det_j = J( 0, 0 ) * J( 1, 1 ) - J( 0, 1 ) * J( 1, 0 );
541 
542  //filling matrix
543  invJ( 0, 0 ) = ( J( 1, 1 ) ) / ( det_j );
544  invJ( 1, 0 ) = -( J( 1, 0 ) ) / ( det_j );
545  invJ( 0, 1 ) = -( J( 0, 1 ) ) / ( det_j );
546  invJ( 1, 1 ) = ( J( 0, 0 ) ) / ( det_j );
547 
548 
549  DeltaXi( 0 ) = invJ( 0, 0 ) * res[0] + invJ( 0, 1 ) * res[1];
550  DeltaXi( 1 ) = invJ( 1, 0 ) * res[0] + invJ( 1, 1 ) * res[1];
551 
552  rResult[0] += DeltaXi[0];
553  rResult[1] += DeltaXi[1];
554  rResult[2] = 0.0;
555 
556  if ( k>0 && norm_2( DeltaXi ) > 30 )
557  {
558  KRATOS_ERROR << "Computation of local coordinates failed at iteration " << k<< std::endl;
559  }
560 
561  if ( norm_2( DeltaXi ) < tol )
562  {
563  break;
564  }
565  }
566 
567  return( rResult );
568  }
