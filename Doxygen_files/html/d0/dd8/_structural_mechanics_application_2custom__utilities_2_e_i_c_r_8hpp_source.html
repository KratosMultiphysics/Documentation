218  inline static MatrixType Compute_Pt(const SizeType NumNodes)
219  {
220  const RealType a = RealType(NumNodes - 1) / RealType(NumNodes);
221  const RealType b = -1.0 / RealType(NumNodes);
222 
223  const SizeType num_dofs = NumNodes * 6;
224 
225  MatrixType P(IdentityMatrix(num_dofs, num_dofs));
226 
227  for (SizeType i = 0; i < NumNodes; i++) {
228  const SizeType j = i * 6;
229 
230  // diagonal block
231  P(j , j) = a;
232  P(j + 1, j + 1) = a;
233  P(j + 2, j + 2) = a;
234
