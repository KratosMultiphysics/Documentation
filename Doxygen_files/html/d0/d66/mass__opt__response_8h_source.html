120  double CalculateValue() override {
121  double total_mass = 0.0;
122  for(auto& eval_obj : mrResponseSettings["evaluated_objects"]){
123  ModelPart& r_eval_object = mrModel.GetModelPart(eval_obj.GetString());
124  const std::size_t domain_size = r_eval_object.GetProcessInfo()[DOMAIN_SIZE];
125  total_mass += block_for_each<SumReduction<double>>(r_eval_object.Elements(), [&](auto& elem_i) {
126  const bool element_is_active = elem_i.IsDefined(ACTIVE) ? elem_i.Is(ACTIVE) : true;
127  if(element_is_active)
128  return CalculateElementMass(elem_i,domain_size);
129  else
130  return 0.0;
131  });
132  }

136  double CalculateElementMass(Element& elem_i, const std::size_t DomainSize){
137  // We get the element geometry
138  auto& r_this_geometry = elem_i.GetGeometry();
139  const std::size_t local_space_dimension = r_this_geometry.LocalSpaceDimension();
140 
141  double volume_area = 0.0;
142  const IntegrationMethod this_integration_method = r_this_geometry.GetDefaultIntegrationMethod();
143  const GeometryType::IntegrationPointsArrayType& integration_points = r_this_geometry.IntegrationPoints(this_integration_method);
144  for(std::size_t i_point = 0; i_point<integration_points.size(); ++i_point)
145  {
146  Matrix J0;
147  GeometryUtils::JacobianOnInitialConfiguration(r_this_geometry, integration_points[i_point], J0);
148  volume_area += integration_points[i_point].Weight() * MathUtils<double>::GeneralizedDet(J0);
149  }
150 
151 
152  double element_mass = 0;
153  if (local_space_dimension == 2 && DomainSize == 3 && elem_i.GetProperties().Has(THICKNESS) && elem_i.GetProperties().Has(PT))
154  element_mass = volume_area * elem_i.GetProperties().GetValue(PT) * elem_i.GetProperties().GetValue(DENSITY);
155  else if (local_space_dimension == 2 && DomainSize == 3 && elem_i.GetProperties().Has(THICKNESS))
156  element_mass = volume_area * elem_i.GetProperties().GetValue(THICKNESS) * elem_i.GetProperties().GetValue(DENSITY);
157  else if (local_space_dimension == 3 && DomainSize == 3)
158  element_mass = volume_area * elem_i.GetProperties().GetValue(DENSITY);
159  else
160  KRATOS_ERROR << "MassOptResponse::CalculateElementMass: the element is neither shell nor solid element !" << std::endl;
161 
162  return element_mass;
163  }

202  void CalculateElementShapeGradients(Element& elem_i, const std::size_t DomainSize){
203 
204  // We get the element geometry
205  auto& r_this_geometry = elem_i.GetGeometry();
206  const std::size_t number_of_nodes = r_this_geometry.size();
207 
208  // We copy the current coordinates and move the coordinates to the initial configuration
209  std::vector<array_1d<double, 3>> current_coordinates(number_of_nodes);
210  for (std::size_t i_node = 0; i_node < number_of_nodes; ++i_node) {
211  noalias(current_coordinates[i_node]) = r_this_geometry[i_node].Coordinates();
212  noalias(r_this_geometry[i_node].Coordinates()) = r_this_geometry[i_node].GetInitialPosition().Coordinates();
213  }
214  double mass_before_fd = CalculateElementMass(elem_i,DomainSize);
215 
216  for (SizeType i_node = 0; i_node < number_of_nodes; ++i_node){
217  auto& node_grads = r_this_geometry[i_node].FastGetSolutionStepValue(D_MASS_D_X);
218 
219  r_this_geometry[i_node].X() += mDelta;
220  r_this_geometry[i_node].X0() += mDelta;
221  double mass_after_fd = CalculateElementMass(elem_i,DomainSize);
222  node_grads[0] += (mass_after_fd - mass_before_fd) / mDelta;
223  r_this_geometry[i_node].X() -= mDelta;
224  r_this_geometry[i_node].X0() -= mDelta;
225 
226  r_this_geometry[i_node].Y() += mDelta;
227  r_this_geometry[i_node].Y0() += mDelta;
228  mass_after_fd = CalculateElementMass(elem_i,DomainSize);
229  node_grads[1] += (mass_after_fd - mass_before_fd) / mDelta;
230  r_this_geometry[i_node].Y() -= mDelta;
231  r_this_geometry[i_node].Y0() -= mDelta;
232 
233  r_this_geometry[i_node].Z() += mDelta;
234  r_this_geometry[i_node].Z0() += mDelta;
235  mass_after_fd = CalculateElementMass(elem_i,DomainSize);
236  node_grads[2] += (mass_after_fd - mass_before_fd) / mDelta;
237  r_this_geometry[i_node].Z() -= mDelta;
238  r_this_geometry[i_node].Z0() -= mDelta;
239  }
240 
241  // We restore the current configuration
242  for (std::size_t i_node = 0; i_node < number_of_nodes; ++i_node) {
243  noalias(r_this_geometry[i_node].Coordinates()) = current_coordinates[i_node];
244  }

248  void CalculateElementMaterialGradients(Element& elem_i, const std::size_t DomainSize){
249 
250  // We get the element geometry
251  auto& r_this_geometry = elem_i.GetGeometry();
252  const std::size_t number_of_nodes = r_this_geometry.size();
253 
254  double curr_density = elem_i.GetProperties().GetValue(DENSITY);
255  elem_i.GetProperties().SetValue(DENSITY,1.0);
256  double elem_dens_grad = CalculateElementMass(elem_i,DomainSize);
257  elem_i.GetProperties().SetValue(DENSITY,curr_density);
258 
259  for (SizeType i_node = 0; i_node < number_of_nodes; ++i_node){
260  const auto& d_pd_d_fd = r_this_geometry[i_node].FastGetSolutionStepValue(D_PD_D_FD);
261  AtomicAdd(r_this_geometry[i_node].FastGetSolutionStepValue(D_MASS_D_FD), d_pd_d_fd * elem_dens_grad / number_of_nodes);
262  }
