120  double CalculateValue() override {
121  double total_strain_energy = 0.0;
122  for(auto& eval_obj : mrResponseSettings["evaluated_objects"]){
123  ModelPart& r_eval_object = mrModel.GetModelPart(eval_obj.GetString());
124  const ProcessInfo &CurrentProcessInfo = r_eval_object.GetProcessInfo();
125  // Sum all elemental strain energy values calculated as: W_e = u_e^T K_e u_e
126  total_strain_energy += block_for_each<SumReduction<double>>(r_eval_object.Elements(), [&](auto& elem_i) {
127  const bool element_is_active = elem_i.IsDefined(ACTIVE) ? elem_i.Is(ACTIVE) : true;
128  if(element_is_active)
129  {
130  Matrix LHS;
131  Vector RHS;
132  Vector u;
133 
134  // Get state solution relevant for energy calculation
135  const auto& rConstElemRef = elem_i;
136  rConstElemRef.GetValuesVector(u,0);
137 
138  elem_i.CalculateLocalSystem(LHS,RHS,CurrentProcessInfo);
139 
140  // Compute strain energy
141  return 0.5 * inner_prod(u,prod(LHS,u));
142  } else {
143  return 0.0;
144  }
145  });
146  }
147  return total_strain_energy;
148  };

197  void CalculateElementShapeGradients(Element& elem_i, std::string shape_gradien_name, const ProcessInfo &rCurrentProcessInfo){
198 
199  Element::NodesArrayType node_array;
200  for (auto& node_i : elem_i.GetGeometry()){
201  Element::NodeType::Pointer node_p = node_i.Clone();
202  node_array.push_back(node_p);
203  }
204 
205  Element::Pointer p_elem = elem_i.Create(elem_i.Id(),node_array, elem_i.pGetProperties());
206  p_elem->SetData(elem_i.GetData());
207  p_elem->Set(Flags(elem_i));
208  p_elem->Initialize(rCurrentProcessInfo);
209 
210  Vector u;
211  Vector lambda;
212  Vector RHS;
213 
214  // Get state solution
215  p_elem->GetValuesVector(u,0);
216 
217  // Get adjoint variables (Corresponds to 1/2*u)
218  lambda = 0.5*u;
219 
220  // Semi-analytic computation of partial derivative of state equation w.r.t. node coordinates
221  p_elem->CalculateRightHandSide(RHS, rCurrentProcessInfo);
222  int node_iter = 0;
223  for (auto& node_i : p_elem->GetGeometry())
224  {
225  array_3d gradient_contribution(3, 0.0);
226  Vector RHS_perturbed = Vector(RHS.size());
227  Vector derived_RHS = Vector(RHS.size());
228 
229  // x-direction
230  node_i.GetInitialPosition()[0] += mDelta;
231  node_i.Coordinates()[0] += mDelta;
232  p_elem->CalculateRightHandSide(RHS_perturbed, rCurrentProcessInfo);
233  noalias(derived_RHS) = (RHS_perturbed - RHS) / mDelta;
234  node_i.GetInitialPosition()[0] -= mDelta;
235  node_i.Coordinates()[0] -= mDelta;
236  gradient_contribution[0] = inner_prod(lambda, derived_RHS);
237 
238 
239  // y-direction
240  node_i.GetInitialPosition()[1] += mDelta;
241  node_i.Coordinates()[1] += mDelta;
242  p_elem->CalculateRightHandSide(RHS_perturbed, rCurrentProcessInfo);
243  noalias(derived_RHS) = (RHS_perturbed - RHS) / mDelta;
244  node_i.GetInitialPosition()[1] -= mDelta;
245  node_i.Coordinates()[1] -= mDelta;
246  gradient_contribution[1] = inner_prod(lambda, derived_RHS);
247 
248  // z-direction
249  node_i.GetInitialPosition()[2] += mDelta;
250  node_i.Coordinates()[2] += mDelta;
251  p_elem->CalculateRightHandSide(RHS_perturbed, rCurrentProcessInfo);
252  noalias(derived_RHS) = (RHS_perturbed - RHS) / mDelta;
253  node_i.GetInitialPosition()[2] -= mDelta;
254  node_i.Coordinates()[2] -= mDelta;
255  gradient_contribution[2] = inner_prod(lambda, derived_RHS);
256 
257  // Assemble sensitivity to node
258  array_3d& r_nodal_variable = elem_i.GetGeometry()[node_iter].FastGetSolutionStepValue(KratosComponents<Variable<array_1d<double,3>>>::Get(shape_gradien_name));
259  AtomicAdd(r_nodal_variable[0], gradient_contribution[0]);
260  AtomicAdd(r_nodal_variable[1], gradient_contribution[1]);
261  AtomicAdd(r_nodal_variable[2], gradient_contribution[2]);
262  // node_i.FastGetSolutionStepValue(KratosComponents<Variable<array_1d<double,3>>>::Get(shape_gradien_name)) += gradient_contribution;
263  node_iter++;
264  }

339  void CalculateElementMaterialGradients(Element& elem_i, std::string material_gradien_name, const ProcessInfo &rCurrentProcessInfo){
340 
341  // We get the element geometry
342  auto& r_this_geometry = elem_i.GetGeometry();
343  const std::size_t number_of_nodes = r_this_geometry.size();
344 
345  Vector u;
346  Vector lambda;
347 
348  // Get state solution
349  const auto& rConstElemRef = elem_i;
350  rConstElemRef.GetValuesVector(u,0);
351 
352  // Get adjoint variables (Corresponds to 1/2*u)
353  lambda = 0.5*u;
354 
355  Vector d_RHS_d_E;
356  double curr_e = elem_i.GetProperties().GetValue(YOUNG_MODULUS);
357  elem_i.GetProperties().SetValue(YOUNG_MODULUS,1.0);
358  elem_i.CalculateRightHandSide(d_RHS_d_E,rCurrentProcessInfo);
359  elem_i.GetProperties().SetValue(YOUNG_MODULUS,curr_e);
360 
361 
362  for (SizeType i_node = 0; i_node < number_of_nodes; ++i_node){
363  const auto& d_pe_d_fd = r_this_geometry[i_node].FastGetSolutionStepValue(D_PE_D_FD);
364  AtomicAdd(r_this_geometry[i_node].FastGetSolutionStepValue(KratosComponents<Variable<double>>::Get(material_gradien_name)), d_pe_d_fd * inner_prod(d_RHS_d_E,lambda) / number_of_nodes);
365  }
