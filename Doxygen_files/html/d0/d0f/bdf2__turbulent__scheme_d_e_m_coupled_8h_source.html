135  ModelPart& rModelPart,
136  TSystemMatrixType& A,
137  TSystemVectorType& Dx,
138  TSystemVectorType& b) override
139  {
140  ProcessInfo CurrentProcessInfo = rModelPart.GetProcessInfo();
141 
142  this->SetTimeCoefficients(rModelPart.GetProcessInfo());
143 
144  // Base function initializes elements and conditions
145  BaseType::InitializeSolutionStep(rModelPart,A,Dx,b);
146 
147  // Recalculate mesh velocity (to account for variable time step)
148  const double tol = 1.0e-12;
149  const double Dt = rModelPart.GetProcessInfo()[DELTA_TIME];
150  const double OldDt = rModelPart.GetProcessInfo().GetPreviousSolutionStepInfo(1)[DELTA_TIME];
151  if(std::abs(Dt - OldDt) > tol) {
152  const int n_nodes = rModelPart.NumberOfNodes();
153  const Vector& BDFcoefs = rModelPart.GetProcessInfo()[BDF_COEFFICIENTS];
154 
155 #pragma omp parallel for
156  for(int i_node = 0; i_node < n_nodes; ++i_node) {
157  auto it_node = rModelPart.NodesBegin() + i_node;
158  auto& rMeshVel = it_node->FastGetSolutionStepValue(MESH_VELOCITY);
159  const auto& rDisp0 = it_node->FastGetSolutionStepValue(DISPLACEMENT);
160  const auto& rDisp1 = it_node->FastGetSolutionStepValue(DISPLACEMENT,1);
161  const auto& rDisp2 = it_node->FastGetSolutionStepValue(DISPLACEMENT,2);
162  rMeshVel = BDFcoefs[0] * rDisp0 + BDFcoefs[1] * rDisp1 + BDFcoefs[2] * rDisp2;
163  }
164  }

168  void SetTimeCoefficients(ProcessInfo& rCurrentProcessInfo)
169  {
170  KRATOS_TRY;
171 
172  //calculate the BDF coefficients
173  double OldDt;
174  double Dt = rCurrentProcessInfo[DELTA_TIME];
175  double step = rCurrentProcessInfo[STEP];
176  // Initialization of the previous delta time at the beginning of the simulation (when using adaptive delta time)
177  if (rCurrentProcessInfo[MANUFACTURED] && step < 2){
178  OldDt = rCurrentProcessInfo[DELTA_TIME];
179  }
180  else {
181  OldDt = rCurrentProcessInfo.GetPreviousTimeStepInfo(1)[DELTA_TIME];
182  }
183 
184  double Rho = OldDt / Dt;
185  double TimeCoeff = 1.0 / (Dt * Rho * Rho + Dt * Rho);
186 
187  Vector& BDFcoeffs = rCurrentProcessInfo[BDF_COEFFICIENTS];
188  BDFcoeffs.resize(3, false);
189 
190  BDFcoeffs[0] = TimeCoeff * (Rho * Rho + 2.0 * Rho); //coefficient for step n+1 (3/2Dt if Dt is constant)
191  BDFcoeffs[1] = -TimeCoeff * (Rho * Rho + 2.0 * Rho + 1.0); //coefficient for step n (-4/2Dt if Dt is constant)
192  BDFcoeffs[2] = TimeCoeff; //coefficient for step n-1 (1/2Dt if Dt is constant)
193 
194  KRATOS_CATCH("");
195  }

197  void FullProjection(ModelPart& rModelPart)
198  {
199  const ProcessInfo& rCurrentProcessInfo = rModelPart.GetProcessInfo();
200 
201  // Initialize containers
202  const int n_nodes = rModelPart.NumberOfNodes();
203  const int n_elems = rModelPart.NumberOfElements();
204  const array_1d<double,3> zero_vect = ZeroVector(3);
205 #pragma omp parallel for firstprivate(zero_vect)
206  for (int i_node = 0; i_node < n_nodes; ++i_node) {
207  auto ind = rModelPart.NodesBegin() + i_node;
208  noalias(ind->FastGetSolutionStepValue(ADVPROJ)) = zero_vect; // "x"
209  ind->FastGetSolutionStepValue(DIVPROJ) = 0.0; // "x"
210  ind->FastGetSolutionStepValue(NODAL_AREA) = 0.0; // "Ml"
211  }
212 
213  // Newton-Raphson parameters
214  const double RelTol = rModelPart.GetProcessInfo()[RELAXATION_ALPHA] * 1e-4 * rModelPart.NumberOfNodes();
215  const double AbsTol = rModelPart.GetProcessInfo()[RELAXATION_ALPHA] * 1e-6 * rModelPart.NumberOfNodes();
216  const unsigned int MaxIter = 100;
217 
218  // iteration variables
219  unsigned int iter = 0;
220  array_1d<double,3> dMomProj = zero_vect;
221  double dMassProj = 0.0;
222 
223  double RelMomErr = 1000.0 * RelTol;
224  double RelMassErr = 1000.0 * RelTol;
225  double AbsMomErr = 1000.0 * AbsTol;
226  double AbsMassErr = 1000.0 * AbsTol;
227 
228  while( ( (AbsMomErr > AbsTol && RelMomErr > RelTol) || (AbsMassErr > AbsTol && RelMassErr > RelTol) ) && iter < MaxIter)
229  {
230  // Reinitialize RHS
231 #pragma omp parallel for firstprivate(zero_vect)
232  for (int i_node = 0; i_node < n_nodes; ++i_node)
233  {
234  auto ind = rModelPart.NodesBegin() + i_node;
235  noalias(ind->GetValue(ADVPROJ)) = zero_vect; // "b"
236  ind->GetValue(DIVPROJ) = 0.0; // "b"
237  ind->FastGetSolutionStepValue(NODAL_AREA) = 0.0; // Reset because Calculate will overwrite it
238  }
239 
240  // Reinitialize errors
241  RelMomErr = 0.0;
242  RelMassErr = 0.0;
243  AbsMomErr = 0.0;
244  AbsMassErr = 0.0;
245 
246  // Compute new values
247  array_1d<double, 3 > output;
248 #pragma omp parallel for private(output)
249  for (int i_elem = 0; i_elem < n_elems; ++i_elem) {
250  auto it_elem = rModelPart.ElementsBegin() + i_elem;
251  it_elem->Calculate(SUBSCALE_VELOCITY, output, rCurrentProcessInfo);
252  }
253 
254  rModelPart.GetCommunicator().AssembleCurrentData(NODAL_AREA);
255  rModelPart.GetCommunicator().AssembleCurrentData(DIVPROJ);
256  rModelPart.GetCommunicator().AssembleCurrentData(ADVPROJ);
257  rModelPart.GetCommunicator().AssembleNonHistoricalData(DIVPROJ);
258  rModelPart.GetCommunicator().AssembleNonHistoricalData(ADVPROJ);
259 
260  // Update iteration variables
261 #pragma omp parallel for
262  for (int i_node = 0; i_node < n_nodes; ++i_node) {
263  auto ind = rModelPart.NodesBegin() + i_node;
264  const double Area = ind->FastGetSolutionStepValue(NODAL_AREA); // Ml dx = b - Mc x
265  dMomProj = ind->GetValue(ADVPROJ) / Area;
266  dMassProj = ind->GetValue(DIVPROJ) / Area;
267 
268  RelMomErr += std::sqrt(std::pow(dMomProj[0],2) + std::pow(dMomProj[1],2) + std::pow(dMomProj[2],2));
269  RelMassErr += std::fabs(dMassProj);
270 
271  auto& rMomRHS = ind->FastGetSolutionStepValue(ADVPROJ);
272  double& rMassRHS = ind->FastGetSolutionStepValue(DIVPROJ);
273  rMomRHS += dMomProj;
274  rMassRHS += dMassProj;
275 
276  AbsMomErr += std::sqrt(std::pow(rMomRHS[0],2) + std::pow(rMomRHS[1],2) + std::pow(rMomRHS[2],2));
277  AbsMassErr += std::fabs(rMassRHS);
278  }
279 
280  if(AbsMomErr > 1e-10)
281  RelMomErr /= AbsMomErr;
282  else // If residual is close to zero, force absolute convergence to avoid division by zero errors
283  RelMomErr = 1000.0;
284 
285  if(AbsMassErr > 1e-10)
286  RelMassErr /= AbsMassErr;
287  else
288  RelMassErr = 1000.0;
289 
290  iter++;
291  }
292 
293  KRATOS_INFO("BDF2TurbulentSchemeDEMCoupled") << "Performed OSS Projection in " << iter << " iterations" << std::endl;
294  }

329  void UpdateFluidFraction(
330  ModelPart& r_model_part,
331  ProcessInfo& r_current_process_info)
332  {
333  BDF2TurbulentScheme<TSparseSpace, TDenseSpace>::SetTimeCoefficients(r_current_process_info);
334  const Vector& BDFcoefs = r_current_process_info[BDF_COEFFICIENTS];
335  double step = r_current_process_info[STEP];
336 
337  block_for_each(r_model_part.Nodes(), [&](Node& rNode)
338  {
339  double& fluid_fraction_0 = rNode.FastGetSolutionStepValue(FLUID_FRACTION);
340  double& fluid_fraction_1 = rNode.FastGetSolutionStepValue(FLUID_FRACTION_OLD);
341  double& fluid_fraction_2 = rNode.FastGetSolutionStepValue(FLUID_FRACTION_OLD_2);
342 
343  // This condition is needed to avoid large time variation of the porosity at the beginning of the simulation that can induce fluid instabilities
344  if (step <= 2){
345  fluid_fraction_2 = fluid_fraction_0;
346  fluid_fraction_1 = fluid_fraction_0;
347  }
348 
349  rNode.FastGetSolutionStepValue(FLUID_FRACTION_RATE) = BDFcoefs[0] * fluid_fraction_0 + BDFcoefs[1] * fluid_fraction_1 + BDFcoefs[2] * fluid_fraction_2;
350 
351  rNode.GetSolutionStepValue(FLUID_FRACTION_OLD_2) = rNode.GetSolutionStepValue(FLUID_FRACTION_OLD);
352  rNode.GetSolutionStepValue(FLUID_FRACTION_OLD) = rNode.GetSolutionStepValue(FLUID_FRACTION);
353  });
354  }
