65  void CalculateK0Stresses(Element& rElement)
66  {
67  // Get K0 material parameters of this element ( probably there is something more efficient )
68  const Element::PropertiesType& rProp = rElement.GetProperties();
69  ConstitutiveLaw::Pointer pConstitutiveLaw = rProp.GetValue(CONSTITUTIVE_LAW);
70  const int k0_main_direction = rProp[K0_MAIN_DIRECTION];
71  if (k0_main_direction < 0 || k0_main_direction > 1) {
72  KRATOS_ERROR << "undefined K0_MAIN_DIRECTION in ApplyK0ProcedureProcess: " << k0_main_direction << std::endl;
73  }
74 
75  //Check for alternative K0 specifications
76  array_1d<double, 3> k0_vector;
77  if (rProp.Has(K0_NC)) {
78  std::fill(k0_vector.begin(), k0_vector.end(), rProp[K0_NC]);
79  }
80  else if (rProp.Has(INDEX_OF_UMAT_PHI_PARAMETER) && rProp.Has(NUMBER_OF_UMAT_PARAMETERS) && rProp.Has(UMAT_PARAMETERS)) {
81  if (rProp[INDEX_OF_UMAT_PHI_PARAMETER] < 1 || rProp[INDEX_OF_UMAT_PHI_PARAMETER] > rProp[NUMBER_OF_UMAT_PARAMETERS]) {
82  KRATOS_ERROR << "undefined INDEX_OF_UMAT_PHI_PARAMETER in ApplyK0ProcedureProcess: " << rProp[INDEX_OF_UMAT_PHI_PARAMETER] << std::endl;
83  }
84  // is more checking is possible and should that happen here?
85  double phi = rProp[UMAT_PARAMETERS][rProp[INDEX_OF_UMAT_PHI_PARAMETER] - 1];
86  if (phi < 0. || phi > 90.) {
87  KRATOS_ERROR << "friction angle Phi out of range in ApplyK0ProcedureProcess: " << phi << std::endl;
88  }
89  std::fill(k0_vector.begin(), k0_vector.end(), 1.0 - std::sin(MathUtils<>::DegreesToRadians(phi)));
90  }
91  else if (rProp.Has(K0_VALUE_XX) && rProp.Has(K0_VALUE_YY) && rProp.Has(K0_VALUE_ZZ)) {
92  k0_vector[0] = rProp[K0_VALUE_XX];
93  k0_vector[1] = rProp[K0_VALUE_YY];
94  k0_vector[2] = rProp[K0_VALUE_ZZ];
95  }
96  else {
97  KRATOS_ERROR << "Insufficient material data for K0 procedure process: " << std::endl;
98  }
99  const auto PoissonUR = rProp.Has(POISSON_UNLOADING_RELOADING) ? rProp[POISSON_UNLOADING_RELOADING] : 0.;
100 
101  // Determine OCR dependent K0 values ( constant per element! )
102  if ((rProp.Has(K0_NC) || rProp.Has(INDEX_OF_UMAT_PHI_PARAMETER)) && rProp.Has(OCR)) {
103  //Modify for presence of OCR (or POP?) field values
104  k0_vector *= rProp[OCR];
105  array_1d<double, 3> correction(3, (PoissonUR / (1.0 - PoissonUR)) * (rProp[OCR] - 1.0));
106  k0_vector -= correction;
107  }
108  // Get element stress vectorsn
109  const ProcessInfo& rCurrentProcessInfo = this->mrModelPart.GetProcessInfo();
110  std::vector<ConstitutiveLaw::StressVectorType> rStressVectors;
111  rElement.CalculateOnIntegrationPoints(CAUCHY_STRESS_VECTOR, rStressVectors, rCurrentProcessInfo);
112 
113  //Loop over integration points
114  const Element::GeometryType& rGeom = rElement.GetGeometry();
115  const Element::GeometryType::IntegrationPointsArrayType& integration_points = rGeom.IntegrationPoints(rElement.GetIntegrationMethod());
116  for (unsigned int g_point = 0; g_point < integration_points.size(); ++g_point) {
117 
118  // Apply K0 procedure
119  for (int i_dir = 0; i_dir <= 2; ++i_dir) {
120  if (i_dir != k0_main_direction) {
121  rStressVectors[g_point][i_dir] = k0_vector[i_dir] * rStressVectors[g_point][k0_main_direction];
122  }
123  }
124  // Erase shear stresses
125  std::fill(rStressVectors[g_point].begin()+3, rStressVectors[g_point].end(), 0.0);
126  }
127  // Set element integration point stress tensors
128  rElement.SetValuesOnIntegrationPoints(CAUCHY_STRESS_VECTOR, rStressVectors, rCurrentProcessInfo);
129 
130  }
