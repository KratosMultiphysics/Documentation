79  void PredictVariableForNode(Node& rNode, const SecondOrderVectorVariable& rSecondOrderVariables)
80  {
81  const std::vector<std::string> components = {"X", "Y", "Z"};
82 
83  for (const auto& component : components) {
84  const auto& instance_component = VariablesUtilities::GetComponentFromVectorVariable(
85  rSecondOrderVariables.instance.Name(), component);
86 
87  if (!rNode.HasDofFor(instance_component)) continue;
88 
89  const auto& first_time_derivative_component = VariablesUtilities::GetComponentFromVectorVariable(
90  rSecondOrderVariables.first_time_derivative.Name(), component);
91  const auto& second_time_derivative_component = VariablesUtilities::GetComponentFromVectorVariable(
92  rSecondOrderVariables.second_time_derivative.Name(), component);
93 
94  const double previous_variable = rNode.FastGetSolutionStepValue(instance_component, 1);
95  const double current_first_time_derivative =
96  rNode.FastGetSolutionStepValue(first_time_derivative_component, 0);
97  const double previous_first_time_derivative =
98  rNode.FastGetSolutionStepValue(first_time_derivative_component, 1);
99  const double current_second_time_derivative =
100  rNode.FastGetSolutionStepValue(second_time_derivative_component, 0);
101  const double previous_second_time_derivative =
102  rNode.FastGetSolutionStepValue(second_time_derivative_component, 1);
103  if (rNode.IsFixed(second_time_derivative_component)) {
104  rNode.FastGetSolutionStepValue(instance_component) =
105  previous_variable + this->GetDeltaTime() * previous_first_time_derivative +
106  this->GetDeltaTime() * this->GetDeltaTime() *
107  ((0.5 - this->GetBeta()) * previous_second_time_derivative +
108  this->GetBeta() * current_second_time_derivative);
109  } else if (rNode.IsFixed(first_time_derivative_component)) {
110  rNode.FastGetSolutionStepValue(instance_component) =
111  previous_variable +
112  this->GetDeltaTime() * ((this->GetBeta() / this->GetGamma()) *
113  (current_first_time_derivative - previous_first_time_derivative) +
114  previous_first_time_derivative);
115  } else if (!rNode.IsFixed(instance_component)) {
116  rNode.FastGetSolutionStepValue(instance_component) =
117  previous_variable + this->GetDeltaTime() * previous_first_time_derivative +
118  0.5 * this->GetDeltaTime() * this->GetDeltaTime() * previous_second_time_derivative;
119  }
120  }
121  }

266  void AddDynamicsToLHS(LocalSystemMatrixType& LHS_Contribution,
267  LocalSystemMatrixType& M,
268  LocalSystemMatrixType& C,
269  const ProcessInfo& CurrentProcessInfo)
270  {
271  KRATOS_TRY
272 
273  // adding mass contribution
274  if (M.size1() != 0)
275  noalias(LHS_Contribution) +=
276  (1.0 / (this->GetBeta() * this->GetDeltaTime() * this->GetDeltaTime())) * M;
277 
278  // adding damping contribution
279  if (C.size1() != 0)
280  noalias(LHS_Contribution) += (this->GetGamma() / (this->GetBeta() * this->GetDeltaTime())) * C;
281 
282  KRATOS_CATCH("")

285  void AddDynamicsToRHS(Condition& rCurrentCondition,
286  LocalSystemVectorType& RHS_Contribution,
287  LocalSystemMatrixType& M,
288  LocalSystemMatrixType& C,
289  const ProcessInfo& CurrentProcessInfo)
290  {

293  int thread = OpenMPUtils::ThisThread();

296  if (M.size1() != 0) {
297  rCurrentCondition.GetSecondDerivativesVector(mAccelerationVector[thread], 0);
298  noalias(RHS_Contribution) -= prod(M, mAccelerationVector[thread]);
