196

199

201  this->GetDeviatoricCoefficientForFluid(rModelPart, itNode, deviatoricCoeff);
202 
203  if (deviatoricCoeff > deviatoric_threshold && itNode->IsNot(SOLID))
204  {

206  }

209  if (itNode->IsNot(SOLID) || itNode->FastGetSolutionStepValue(INTERFACE_NODE) == true)
210  {
211  volumetricCoeff = timeInterval * itNode->FastGetSolutionStepValue(BULK_MODULUS);
212  }
213 
214  deltaPressure = itNode->FastGetSolutionStepValue(PRESSURE, 0) - itNode->FastGetSolutionStepValue(PRESSURE, 1);
215 
216  LHS_Contribution(0, 0) += nodalVolume / volumetricCoeff;
217 
218  RHS_Contribution[0] += -deltaPressure * nodalVolume / volumetricCoeff;
219 
220  RHS_Contribution[0] += itNode->GetSolutionStepValue(NODAL_VOLUMETRIC_DEF_RATE) * nodalVolume;
221  }
222  }
223 
224  if (itNode->Is(SOLID))
225  {
226  double nodalVolume = itNode->FastGetSolutionStepValue(SOLID_NODAL_VOLUME);
227  double youngModulus = itNode->FastGetSolutionStepValue(YOUNG_MODULUS);
228  double poissonRatio = itNode->FastGetSolutionStepValue(POISSON_RATIO);
229 
230  double deviatoricCoeff = timeInterval * youngModulus / (1.0 + poissonRatio) * 0.5;
231  double volumetricCoeff = timeInterval * poissonRatio * youngModulus / ((1.0 + poissonRatio) * (1.0 - 2.0 * poissonRatio)) + 2.0 * deviatoricCoeff / 3.0;
232 
233  deltaPressure = itNode->FastGetSolutionStepValue(PRESSURE, 0) - itNode->FastGetSolutionStepValue(PRESSURE, 1);
234 
235  LHS_Contribution(0, 0) += nodalVolume / volumetricCoeff;
236 
237  RHS_Contribution[0] += -deltaPressure * nodalVolume / volumetricCoeff;
238

240  }

243

293

296  {
