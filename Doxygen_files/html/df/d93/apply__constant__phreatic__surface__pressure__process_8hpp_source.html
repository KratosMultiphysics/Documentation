87  void ExecuteInitialize() override
88  {
89  KRATOS_TRY
90 
91  const Variable<double>& var = KratosComponents<Variable<double>>::Get(mVariableName);
92  Vector3 direction = ZeroVector(3);
93  direction[mGravityDirection] = 1.0;
94 
95  block_for_each(mrModelPart.Nodes(), [&var, &direction, this](Node& rNode) {
96  double distance = inner_prod(mNormalVector, rNode.Coordinates());
97  const double d = inner_prod(mNormalVector, direction);
98  distance = -(distance - mEqRHS) / d;
99  const double pressure = -PORE_PRESSURE_SIGN_FACTOR * mSpecificWeight * distance;
100  if (mIsSeepage) {
101  if (pressure < PORE_PRESSURE_SIGN_FACTOR * mPressureTensionCutOff) {
102  rNode.FastGetSolutionStepValue(var) = pressure;
103  if (mIsFixed) rNode.Fix(var);
104  } else {
105  if (mIsFixedProvided) rNode.Free(var);
106  }
107  } else {
108  if (mIsFixed) rNode.Fix(var);
109  else if (mIsFixedProvided) rNode.Free(var);
110  rNode.FastGetSolutionStepValue(var) =
111  std::min(pressure, PORE_PRESSURE_SIGN_FACTOR * mPressureTensionCutOff);
112  }
113  });
114 
115  KRATOS_CATCH("")
116  }
