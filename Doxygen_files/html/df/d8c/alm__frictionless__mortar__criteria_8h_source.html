211  r_table << BOLDFONT(FGRN(" Achieved"));

216  r_table << BOLDFONT(FRED(" Not achieved"));
217  else
218  r_table << "Not achieved";
219  }
220  } else {

222  if (BaseType::mOptions.IsNot(ALMFrictionlessMortarConvergenceCriteria::PRINTING_OUTPUT))
