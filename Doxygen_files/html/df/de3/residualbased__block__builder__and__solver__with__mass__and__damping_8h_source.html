158  void Build(typename TSchemeType::Pointer pScheme, ModelPart& rModelPart, TSystemMatrixType& rA, TSystemVectorType& rb) override
159  {
160  KRATOS_TRY
161 
162  KRATOS_ERROR_IF(!pScheme) << "No scheme provided!" << std::endl;
163 
164  // Getting the elements from the model
165  const auto nelements = static_cast<int>(rModelPart.Elements().size());
166 
167  // Getting the array of the conditions
168  const auto nconditions = static_cast<int>(rModelPart.Conditions().size());
169 
170  const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
171  ModelPart::ElementsContainerType::iterator el_begin = rModelPart.ElementsBegin();
172  ModelPart::ConditionsContainerType::iterator cond_begin = rModelPart.ConditionsBegin();
173 
174  // contributions to the system
175  LocalSystemMatrixType lhs_contribution(0, 0);
176  LocalSystemMatrixType mass_contribution(0, 0);
177  LocalSystemMatrixType damping_contribution(0, 0);
178  LocalSystemVectorType rhs_contribution(0);
179 
180  InitializeDynamicMatrix(mMassMatrix, BaseType::mEquationSystemSize, pScheme, rModelPart);
181  InitializeDynamicMatrix(mDampingMatrix, BaseType::mEquationSystemSize, pScheme, rModelPart);
182 
183  // vector containing the localization in the system of the different
184  // terms
185  Element::EquationIdVectorType equation_ids;
186 
187  // Assemble all elements
188  const auto timer = BuiltinTimer();
189 
190 #pragma omp parallel firstprivate(nelements, nconditions, lhs_contribution, mass_contribution, \
191  damping_contribution, rhs_contribution, equation_ids)
192  {
193 #pragma omp for schedule(guided, 512) nowait
194  for (int k = 0; k < nelements; k++) {
195  auto it_elem = el_begin + k;
196 
197  if (it_elem->IsActive()) {
198  // Calculate elemental contribution
199  pScheme->CalculateSystemContributions(*it_elem, lhs_contribution, rhs_contribution,
200  equation_ids, r_current_process_info);
201 
202  // assemble mass and damping matrix
203  it_elem->CalculateMassMatrix(mass_contribution, r_current_process_info);
204  it_elem->CalculateDampingMatrix(damping_contribution, r_current_process_info);
205 
206  if (mass_contribution.size1() != 0) {
207  BaseType::AssembleLHS(mMassMatrix, mass_contribution, equation_ids);
208  }
209  if (damping_contribution.size1() != 0) {
210  BaseType::AssembleLHS(mDampingMatrix, damping_contribution, equation_ids);
211  }
212 
213  // Assemble the elemental contribution
214  BaseType::Assemble(rA, rb, lhs_contribution, rhs_contribution, equation_ids);
215  }
216  }
217 
218 #pragma omp for schedule(guided, 512)
219  for (int k = 0; k < nconditions; k++) {
220  auto it_cond = cond_begin + k;
221 
222  if (it_cond->IsActive()) {
223  // Calculate elemental contribution
224  pScheme->CalculateSystemContributions(*it_cond, lhs_contribution, rhs_contribution,
225  equation_ids, r_current_process_info);
226 
227  // assemble mass and damping matrix
228  it_cond->CalculateMassMatrix(mass_contribution, r_current_process_info);
229  it_cond->CalculateDampingMatrix(damping_contribution, r_current_process_info);
230 
231  if (mass_contribution.size1() != 0) {
232  BaseType::AssembleLHS(mMassMatrix, mass_contribution, equation_ids);
233  }
234  if (damping_contribution.size1() != 0) {
235  BaseType::AssembleLHS(mDampingMatrix, damping_contribution, equation_ids);
236  }
237 
238  // Assemble the elemental contribution
239  BaseType::Assemble(rA, rb, lhs_contribution, rhs_contribution, equation_ids);
240  }
241  }
242  }
243 
244  KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolverWithMassAndDamping", this->GetEchoLevel() >= 1)
245  << "Build time: " << timer.ElapsedSeconds() << std::endl;
246 
247  KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolverWithMassAndDamping",
248  (this->GetEchoLevel() > 2 && rModelPart.GetCommunicator().MyPID() == 0))
249  << "Finished parallel building" << std::endl;
250 
251  KRATOS_CATCH("")
252  }

503  void BuildRHSNoDirichlet(typename TSchemeType::Pointer pScheme, ModelPart& rModelPart, TSystemVectorType& rb)
504  {
505  KRATOS_TRY
506 
507  // Getting the Elements
508  ElementsArrayType& r_elements = rModelPart.Elements();
509 
510  // getting the array of the conditions
511  ConditionsArrayType& r_conditions = rModelPart.Conditions();
512 
513  const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
514 
515  // contributions to the system
516  LocalSystemVectorType rhs_contribution = LocalSystemVectorType(0);
517 
518  // vector containing the localization in the system of the different
519  // terms
520  Element::EquationIdVectorType equation_ids;
521 
522  // assemble all elements
523 
524  const int nelements = static_cast<int>(r_elements.size());
525 #pragma omp parallel firstprivate(nelements, rhs_contribution, equation_ids)
526  {
527 #pragma omp for schedule(guided, 512) nowait
528  for (int i = 0; i < nelements; i++) {
529  typename ElementsArrayType::iterator it = r_elements.begin() + i;
530  // If the element is active
531  if (it->IsActive()) {
532  // calculate elemental Right Hand Side Contribution
533  it->CalculateRightHandSide(rhs_contribution, r_current_process_info);
534  it->EquationIdVector(equation_ids, r_current_process_info);
535 
536  // assemble the elemental contribution
537  BaseType::AssembleRHS(rb, rhs_contribution, equation_ids);
538  }
539  }
540 
541  rhs_contribution.resize(0, false);
542 
543  // assemble all conditions
544  const int nconditions = static_cast<int>(r_conditions.size());
545 #pragma omp for schedule(guided, 512)
546  for (int i = 0; i < nconditions; i++) {
547  auto it = r_conditions.begin() + i;
548  // If the condition is active
549  if (it->IsActive()) {
550  it->CalculateRightHandSide(rhs_contribution, r_current_process_info);
551  it->EquationIdVector(equation_ids, r_current_process_info);
552 
553  // assemble the elemental contribution
554  BaseType::AssembleRHS(rb, rhs_contribution, equation_ids);
555  }
556  }
557  }
558 
559  AddMassAndDampingToRhs(rModelPart, rb);
560 
561  KRATOS_CATCH("")
562  }
