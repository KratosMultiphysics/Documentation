111  void AddDampingToLHS(LocalSystemMatrixType& LHS_Contribution, LocalSystemMatrixType& C, const ProcessInfo& CurrentProcessInfo)
112  {
113  // adding damping contribution
114  if (C.size1() != 0)
115  noalias(LHS_Contribution) += (this->GetGamma() / (this->GetBeta() * this->GetDeltaTime())) * C;
116  }

118  void AddDampingToRHS(Element& rCurrentElement,

120  LocalSystemMatrixType& C,
121  const ProcessInfo& CurrentProcessInfo)
122  {

124 
125  // adding damping contribution

127  rCurrentElement.GetFirstDerivativesVector(mVelocityVector[thread], 0);
128  noalias(RHS_Contribution) -= prod(C, mVelocityVector[thread]);
