49  static void CondenseLeftHandSide(ElementType& rTheElement,
50  MatrixType& rLeftHandSideMatrix,
51  const std::vector<int>& rDofList)
52  {
53  KRATOS_TRY
54  const SizeType num_dofs_condensed = rDofList.size();
55  const SizeType num_dofs_remaining = GetNumDofsElement(rTheElement) - num_dofs_condensed;
56  const double numerical_limit = std::numeric_limits<double>::epsilon();
57 
58  std::vector<MatrixType> sub_matrices =
59  CalculateSchurComplements(rTheElement, rLeftHandSideMatrix, rDofList);
60  // 1.) inverse K22
61  MatrixType K_temp = ZeroMatrix(sub_matrices[3].size1());
62  double detK22 = 0.00;
63  MathUtils<double>::InvertMatrix(sub_matrices[3], K_temp, detK22);
64  KRATOS_ERROR_IF(std::abs(detK22) < numerical_limit)
65  << "Element " << rTheElement.Id() << " is singular !" << std::endl;
66 
67  // 2.) K_cond -> K11 - K12*inv(K22)*K21
68  K_temp = prod(K_temp, sub_matrices[2]);
69  K_temp = prod(sub_matrices[1], K_temp);
70  K_temp = sub_matrices[0] - K_temp;
71 
72  // 3.) Fill rLeftHandSide to maintain same matrix size
73  const std::vector<int> remaining_dof_list = CreateRemainingDofList(rTheElement, rDofList);
74  rLeftHandSideMatrix.clear();
75 
76  SizeType dofA = 0;
77  SizeType dofB = 0;
78  for (SizeType i = 0; i < num_dofs_remaining; ++i) {
79  dofA = remaining_dof_list[i];
80  for (SizeType j = 0; j < num_dofs_remaining; ++j) {
81  dofB = remaining_dof_list[j];
82  rLeftHandSideMatrix(dofA, dofB) = K_temp(i, j);
83  }
84  }
85  KRATOS_CATCH("")
86  }

199  static void ConvertingCondensation(ElementType& rTheElement,
200  Vector& rLocalizedDofVector,
201  Vector& rValues,
202  const std::vector<int>& rDofList,
203  const MatrixType& rLeftHandSideMatrix)
204  {
205  KRATOS_TRY
206  const double numerical_limit = std::numeric_limits<double>::epsilon();
207  const std::vector<int> remaining_dof_list = CreateRemainingDofList(rTheElement, rDofList);
208  const SizeType num_dofs_condensed = rDofList.size();
209 
210  const SizeType num_dofs_element = GetNumDofsElement(rTheElement);
211 
212  const SizeType num_dofs_remaining = num_dofs_element - num_dofs_condensed;
213  std::vector<MatrixType> sub_matrices =
214  CalculateSchurComplements(rTheElement, rLeftHandSideMatrix, rDofList);
215 
216  // 1.) create u1
217  Vector remaining_dofs_disp = ZeroVector(num_dofs_remaining);
218  // Vector all_dofs_disp = ZeroVector(num_dofs_element);
219  // rTheElement.GetValuesVector(all_dofs_disp);
220  // rTheElement.LocalizeVector(all_dofs_disp); // localize global displacement -> element lvl
221  // Note: "rLocalizedDofVector" is what was "all_dofs_disp" previously
222  for (SizeType i = 0; i < num_dofs_remaining; ++i)
223  remaining_dofs_disp[i] = rLocalizedDofVector[remaining_dof_list[i]];
224 
225  // 2.) inverse K22
226  MatrixType K22_inv = ZeroMatrix(sub_matrices[3].size1());
227  double detK22 = 0.00;
228  MathUtils<double>::InvertMatrix(sub_matrices[3], K22_inv, detK22);
229 
230  KRATOS_ERROR_IF(std::abs(detK22) < numerical_limit)
231  << "Element " << rTheElement.Id() << " is singular !" << std::endl;
232 
233  // 3.) u2=inv(K22)*(F2-K21*u1),F2=0->u2=-inv(K22)*K21*u1
234  Vector CondensedDofsDisp = ZeroVector(num_dofs_condensed);
235  CondensedDofsDisp = prod(sub_matrices[2], remaining_dofs_disp);
236  CondensedDofsDisp = -prod(K22_inv, CondensedDofsDisp);
237 
238  // 4.) Fill rValues to maintain same matrix size
239  rValues = ZeroVector(num_dofs_element);
240  for (int i = 0; i < static_cast<int>(num_dofs_element); ++i) {
241  bool check = false;
242  // check if dof i is condensed
243  for (SizeType j = 0; j < num_dofs_condensed; ++j) {
244  if (i == rDofList[j]) {
245  rValues[i] = CondensedDofsDisp[j];
246  check = true;
247  break;
248  }
249  }
250 
251  if (check) continue; // found respective dof -> search for next dof
252  // check remaining dofs
253  else {
254  for (SizeType j = 0; j < num_dofs_remaining; ++j) {
255  if (i == remaining_dof_list[j]) {
256  rValues[i] = remaining_dofs_disp[j];
257  break;
258  }
259  }
260  }
261  }
262  KRATOS_CATCH("")
263  }
