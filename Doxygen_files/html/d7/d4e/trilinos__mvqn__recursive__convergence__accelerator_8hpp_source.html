162  VectorPointerType pProjectedVector)

180  void ApplyJacobian(const VectorPointerType pWorkVector,
181  VectorPointerType pProjectedVector)
182  {
183  KRATOS_TRY;
184 
185  const unsigned int previous_iterations = mJacobianObsMatrixV.size();
186 
187  // Security check for the empty observation matrices case (when no correction has been done in the previous step)
188  if (previous_iterations == 0)
189  {
190  if (mpOldJacobianEmulator != nullptr) // If it is available, consider the previous step Jacobian
191  {
192  mpOldJacobianEmulator->ApplyJacobian(pWorkVector, pProjectedVector);
193  }
194  else // When the JacobianEmulator has no PreviousJacobianEmulator consider minus the identity matrix as inverse Jacobian
195  {
196  TSpace::Assign(*pProjectedVector, -1.0, *pWorkVector);
197  }
198  }
199  else
200  {
201  // Get the domain size
202  unsigned int n_dim = mrInterfaceModelPart.GetProcessInfo()[DOMAIN_SIZE];
203 
204  // Construct the interface map
205  int NumLocalInterfaceDofs = mrInterfaceModelPart.GetCommunicator().LocalMesh().NumberOfNodes() * n_dim;
206  int NumGlobalInterfaceDofs = mrInterfaceModelPart.GetCommunicator().GetDataCommunicator().SumAll(NumLocalInterfaceDofs);
207  int IndexBase = 0; // 0 for C-style vectors, 1 for Fortran numbering
208  Epetra_Map InterfaceMap(NumGlobalInterfaceDofs, NumLocalInterfaceDofs, IndexBase, mrEpetraCommunicator);
209 
210  // Create new vector using given map
211  auto pY(new Epetra_FEVector(InterfaceMap));
212  auto pW(new Epetra_FEVector(InterfaceMap));
213 
214  // Loop to store a std::vector<VectorType> type as Matrix type
215  Epetra_SerialDenseMatrix Vtrans_V(previous_iterations, previous_iterations);
216 
217  for (unsigned int i=0; i<previous_iterations; ++i)
218  {
219  Vtrans_V(i,i) = TSpace::Dot(mJacobianObsMatrixV[i], mJacobianObsMatrixV[i]);
220 
221  for (unsigned int j=i+1; j<previous_iterations; ++j)
222  {
223  Vtrans_V(i,j) = TSpace::Dot(mJacobianObsMatrixV[i], mJacobianObsMatrixV[j]);
224  Vtrans_V(j,i) = Vtrans_V(i,j);
225  }
226  }
227 
228  Epetra_SerialDenseVector Vtrans_r(previous_iterations);
229  Epetra_SerialDenseVector zSystemSol(previous_iterations);
230 
231  for (unsigned int i=0; i<previous_iterations; ++i)
232  {
233  Vtrans_r(i) = TSpace::Dot(mJacobianObsMatrixV[i], (*pWorkVector));
234  }
235 
236  Epetra_SerialDenseSolver EpetraSystemSolver;
237  EpetraSystemSolver.SetMatrix(Vtrans_V);
238  EpetraSystemSolver.SetVectors(zSystemSol,Vtrans_r);
239  EpetraSystemSolver.Solve();
240 
241  TSpace::SetToZero(*pY);
242  for (unsigned int j = 0; j < previous_iterations; ++j)
243  {
244  TSpace::UnaliasedAdd(*pY, zSystemSol(j), mJacobianObsMatrixV[j]);
245  }
246 
247  TSpace::UnaliasedAdd(*pY, -1.0, *pWorkVector);
248 
249  if (mpOldJacobianEmulator == nullptr)
250  {
251  TSpace::Copy(*pY, *pProjectedVector); // Consider minus the identity as previous step Jacobian
252  }
253  else
254  {
255  VectorPointerType pYminus(new VectorType(*pY));
256  TSpace::Assign(*pYminus, -1.0, *pY);
257  mpOldJacobianEmulator->ApplyJacobian(pYminus, pProjectedVector); // The minus comes from the fact that we want to apply r_k - V_k*zQR
258  }
259 
260  // w = W_k*z
261  TSpace::SetToZero(*pW);
262  for (unsigned int j = 0; j < previous_iterations; ++j)
263  {
264  TSpace::UnaliasedAdd(*pW, zSystemSol(j), mJacobianObsMatrixW[j]);
265  }
266 
267  TSpace::UnaliasedAdd(*pProjectedVector, 1.0, *pW);
268 
269  }
270 
271  KRATOS_CATCH( "" );
272

587  void UpdateSolution(const VectorType& rResidualVector,
588  VectorType& rIterationGuess) override
589  {
590  KRATOS_TRY;
591 
592  mProblemSize = TSparseSpace::Size(rResidualVector);
593 
594  VectorPointerType pAuxResidualVector(new VectorType(rResidualVector));
595  VectorPointerType pAuxIterationGuess(new VectorType(rIterationGuess));
596  std::swap(mpResidualVector_1, pAuxResidualVector);
597  std::swap(mpIterationValue_1, pAuxIterationGuess);
598 
599  if (mConvergenceAcceleratorIteration == 0)
600  {
601  if (mConvergenceAcceleratorFirstCorrectionPerformed == false)
602  {
603  // The very first correction of the problem is done with a fixed point iteration
604  TSparseSpace::UnaliasedAdd(rIterationGuess, mOmega_0, *mpResidualVector_1);
605 
606  mConvergenceAcceleratorFirstCorrectionPerformed = true;
607  }
608  else
609  {
610  VectorPointerType pInitialCorrection(new VectorType(rResidualVector));
611 
612  // The first correction of the current step is done with the previous step inverse Jacobian approximation
613  mpCurrentJacobianEmulatorPointer->ApplyPrevStepJacobian(mpResidualVector_1, pInitialCorrection);
614 
615  TSparseSpace::UnaliasedAdd(rIterationGuess, -1.0, *pInitialCorrection); // Recall the minus sign coming from the Taylor expansion of the residual (Newton-Raphson)
616  }
617  }
618  else
619  {
620  // Gather the new observation matrices column information
621  VectorPointerType pNewColV(new VectorType(*mpResidualVector_1));
622  VectorPointerType pNewColW(new VectorType(*mpIterationValue_1));
623 
624  TSparseSpace::UnaliasedAdd(*pNewColV, -1.0, *mpResidualVector_0); // NewColV = ResidualVector_1 - ResidualVector_0
625  TSparseSpace::UnaliasedAdd(*pNewColW, -1.0, *mpIterationValue_0); // NewColW = IterationValue_1 - IterationValue_0
626 
627  // Observation matrices information filling
628  if (mConvergenceAcceleratorIteration <= mProblemSize)
629  {
630  // Append the new information to the existent observation matrices
631  (mpCurrentJacobianEmulatorPointer)->AppendColToV(*pNewColV);
632  (mpCurrentJacobianEmulatorPointer)->AppendColToW(*pNewColW);
633  }
634  else
635  {
636  (mpCurrentJacobianEmulatorPointer)->DropAndAppendColToV(*pNewColV);
637  (mpCurrentJacobianEmulatorPointer)->DropAndAppendColToW(*pNewColW);
638  }
639 
640  // Apply the current step inverse Jacobian emulator to the residual vector
641  VectorPointerType pIterationCorrection(new VectorType(rResidualVector));
642  mpCurrentJacobianEmulatorPointer->ApplyJacobian(mpResidualVector_1, pIterationCorrection);
643 
644  TSparseSpace::UnaliasedAdd(rIterationGuess, -1.0, *pIterationCorrection); // Recall the minus sign coming from the Taylor expansion of the residual (Newton-Raphson)
645  }
646 
647  KRATOS_CATCH( "" );
648  }
