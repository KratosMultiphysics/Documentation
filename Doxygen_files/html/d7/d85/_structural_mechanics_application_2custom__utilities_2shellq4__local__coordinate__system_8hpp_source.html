67  ShellQ4_LocalCoordinateSystem(const Vector3Type& P1global,
68  const Vector3Type& P2global,
69  const Vector3Type& P3global,
70  const Vector3Type& P4global)
71  : mP(4)
72  , mOrientation(3, 3)
73  {
74  // Form the basis vectors alligning the local X direction
75  // with the 1-2 side
76 
77  // compute the central point
78  noalias(mCenter) = P1global;
79  noalias(mCenter) += P2global;
80  noalias(mCenter) += P3global;
81  noalias(mCenter) += P4global;
82  mCenter *= 0.25;
83 
84  // compute the diagonal vectors
85  Vector3Type d13(P3global - P1global);
86  Vector3Type d24(P4global - P2global);
87 
88  // compute the Normal vector at the element center
89  // as the cross product of the 2 diagonals.
90  // While normalizing the normal vector save its norm to compute the area.
91  // Note: the norm should be divided by 2 because its computed from the
92  // cross product of the diagonals, which gives twice the area!
93  Vector3Type e3;
94  MathUtils<RealType>::CrossProduct(e3, d13, d24);
95  mArea = NormalizeVector3(e3);
96  mArea /= 2.0;
97 
98  // compute the local X direction parallel to the projection of the side 1-2 onto
99  // the local xy plane.
100  Vector3Type e1(P2global - P1global);
101  RealType e1_dot_e3 = MathUtils<RealType>::Dot(e1, e3);
102  noalias(e1) -= e1_dot_e3 * e3;
103  NormalizeVector3(e1);
104 
105  // finally compute the local Y direction to be orthogonal to both X and Z local directions
106  Vector3Type e2;
107  MathUtils<RealType>::CrossProduct(e2, e3, e1);
108  NormalizeVector3(e2);
109 
110  // form the 3x3 transformation matrix
111  for (int i = 0; i < 3; i++) {
112  mOrientation(0, i) = e1(i);
113  mOrientation(1, i) = e2(i);
114  mOrientation(2, i) = e3(i);
115  }
116 
117  // transform global coordinates to the local coordinate system
118  for (int i = 0; i < 3; i++) {
119  mP[0](i) = mOrientation(i, 0) * (P1global(0) - mCenter(0)) + mOrientation(i, 1) * (P1global(1) - mCenter(1)) + mOrientation(i, 2) * (P1global(2) - mCenter(2));
120  mP[1](i) = mOrientation(i, 0) * (P2global(0) - mCenter(0)) + mOrientation(i, 1) * (P2global(1) - mCenter(1)) + mOrientation(i, 2) * (P2global(2) - mCenter(2));
121  mP[2](i) = mOrientation(i, 0) * (P3global(0) - mCenter(0)) + mOrientation(i, 1) * (P3global(1) - mCenter(1)) + mOrientation(i, 2) * (P3global(2) - mCenter(2));
122  mP[3](i) = mOrientation(i, 0) * (P4global(0) - mCenter(0)) + mOrientation(i, 1) * (P4global(1) - mCenter(1)) + mOrientation(i, 2) * (P4global(2) - mCenter(2));
123  }
124  }

362  R(i , i+2) = mOrientation(0, 2);
363  R(i+1, i) = mOrientation(1, 0);
364  R(i+1, i+1) = mOrientation(1, 1);

368  R(i+2, i+2) = mOrientation(2, 2);

372  if (this->IsWarped()) {
373  MatrixType W(IdentityMatrix(24));
374  RealType z1 = this->Z1();
375  RealType z2 = this->Z2();
376  RealType z3 = this->Z3();

382  W(12, 16) = -z3;
383  W(13, 15) = z3;
384  W(18, 22) = -z4;
385  W(19, 21) = z4;
