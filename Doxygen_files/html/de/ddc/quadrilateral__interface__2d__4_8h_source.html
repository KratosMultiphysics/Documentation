492  Normal[0] = SecondPoint[1] - FirstPoint[1];

495  Normal /= norm;

498  array_1d<double,2> VectorPoint = ZeroVector(2);
499  VectorPoint[0] = rPoint[0] - FirstPoint[0];

502 
503  if (dist_proy < tol)

506

508  + (rPoint[1] - FirstPoint[1]) * (rPoint[1] - FirstPoint[1]);

510
