152  BlockPartition(TIterator it_begin,
153  TIterator it_end,
154  int Nchunks = ParallelUtilities::GetNumThreads())
155  {
156  static_assert(
157  std::is_same_v<typename std::iterator_traits<TIterator>::iterator_category, std::random_access_iterator_tag>,
158  "BlockPartition requires random access iterators to divide the input range into partitions"
159  );
160  KRATOS_ERROR_IF(Nchunks < 1) << "Number of chunks must be > 0 (and not " << Nchunks << ")" << std::endl;
161 
162  const std::ptrdiff_t size_container = it_end-it_begin;
163 
164  if (size_container == 0) {
165  mNchunks = Nchunks;
166  } else {
167  // in case the container is smaller than the number of chunks
168  mNchunks = std::min(static_cast<int>(size_container), Nchunks);
169  }
170  const std::ptrdiff_t block_partition_size = size_container / mNchunks;
171  mBlockPartition[0] = it_begin;
172  mBlockPartition[mNchunks] = it_end;
173  for (int i=1; i<mNchunks; i++) {
174  mBlockPartition[i] = mBlockPartition[i-1] + block_partition_size;
175  }
176  }

487  {
488  std::vector< std::future<void> > runners(mNchunks);
489  const auto& partition = mBlockPartition;
490  for (int i=0; i<mNchunks; ++i) {
491  runners[i] = std::async(std::launch::async, [&partition, i, &f]()
