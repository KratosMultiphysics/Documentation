36  // NOTE:
37  // Taking into account that we evaluate the shape functions associated to displacements (Nu) at the lobatto integration points of order 1 (at the mid plane of the interface),
38  // and that we need to reduce the dimension of those shape functions in order to interpolate the actual relative displacement at the interface, we must multiply Nu by 2.0. In essence:
39  // 2D-quadrilateral: the displacement is interpolated as in a line, but the pore pressure is interpolated as in a quadrilateral
40  // 3D-wedge: the displacement is interpolated as in a triangle, but the pore pressure is interpolated as in a wedge
41  // 3D-hexahedra: the displacement is interpolated as in a quadrilateral, but the pore pressure is interpolated as in a hexahedra

207  static inline void AddInitialInterfaceStresses2D(Vector& rStressVector,
208  ConstitutiveLaw::Parameters& rValues,
209  const Element::GeometryType& geometry)
210  {
211  // Obtain necessary variables
212  const Vector& N = rValues.GetShapeFunctionsValues();
213  const unsigned int number_of_nodes = geometry.size();
214  const unsigned int dimension = 2;
215  const unsigned int strainSize = 3;
216 
217  // Create components to initialise the stresses
218  Vector nodal_initial_stress_vector(strainSize);
219  Matrix nodal_initial_stress_tensor(dimension, dimension);
220  Vector gp_initial_stress_vector(strainSize);
221  noalias(gp_initial_stress_vector) = ZeroVector(strainSize);
222  Vector LocalInitialStress(dimension);
223  noalias(LocalInitialStress) = ZeroVector(dimension);
224 
225  // Contribution of each of the nodal values to the corresponding GP
226  for (unsigned int i=0; i < number_of_nodes; i++) {
227  const Matrix& r_initial_stress_tensor = geometry[i].GetSolutionStepValue(INITIAL_STRESS_TENSOR);
228  for(unsigned int j=0; j < dimension; j++) {
229  for(unsigned int k=0; k < dimension; k++) {
230  nodal_initial_stress_tensor(j,k) = r_initial_stress_tensor(j,k);
231  }
232  }
233  noalias(nodal_initial_stress_vector) = MathUtils<double>::StressTensorToVector(nodal_initial_stress_tensor);
234 
235  for(unsigned int j=0; j < strainSize; j++){
236  gp_initial_stress_vector[j] += N[i] * nodal_initial_stress_vector[j];
237  }
238  }
239 
240  // Define mid-plane points for quadrilateral_interface_2d_4
241  array_1d<double, 3> pmid0;
242  array_1d<double, 3> pmid1;
243  noalias(pmid0) = 0.5 * (geometry.GetPoint( 0 ) + geometry.GetPoint( 3 ));
244  noalias(pmid1) = 0.5 * (geometry.GetPoint( 1 ) + geometry.GetPoint( 2 ));
245 
246  // Unitary vector in local x direction
247  array_1d<double, 3> Vx;
248  noalias(Vx) = pmid1 - pmid0;
249  double inv_norm_x = 1.0/norm_2(Vx);
250  Vx[0] *= inv_norm_x; // cos
251  Vx[1] *= inv_norm_x; // sin
252 
253  // Define and assign the rotation matrix
254  BoundedMatrix<double,dimension,strainSize> RotationInterface;
255  noalias(RotationInterface) = ZeroMatrix(dimension,strainSize);
256  RotationInterface(0,0) = -Vx[1]*Vx[0];
257  RotationInterface(0,1) = Vx[1]*Vx[0];
258  RotationInterface(0,2) = Vx[0]*Vx[0] - Vx[1]*Vx[1];
259  RotationInterface(1,0) = Vx[1]*Vx[1];
260  RotationInterface(1,1) = Vx[0]*Vx[0];
261  RotationInterface(1,2) = -2*Vx[1]*Vx[0];
262 
263  // Local stress vector
264  noalias(LocalInitialStress) = prod(RotationInterface, trans(gp_initial_stress_vector));
265 
266  // Update the stress vector
267  rStressVector[0] += LocalInitialStress[0];
268  rStressVector[1] += LocalInitialStress[1];
269 
270  }

324  // Utility to compute the 6x6 rotation matrix to rotate the stresses from global to local
325  // The derivation is taken from Cook - Concepts and applications of FEA - Page 274
326  static inline void ObtainRotationMatrix(BoundedMatrix<double,6,6>& RotationMatrixInterface, const Element::GeometryType& Geom)
327  {
328  // Define mid-plane points
329  array_1d<double, 3> pmid0;
330  array_1d<double, 3> pmid1;
331  array_1d<double, 3> pmid2;
332 
333  // Differentiate between prisms and heaxhedra and define the coordinates of the mid-plane
334  const unsigned int number_of_nodes = Geom.size();
335  if (number_of_nodes == 6) {
336  noalias(pmid0) = 0.5 * (Geom.GetPoint( 0 ) + Geom.GetPoint( 3 ));
337  noalias(pmid1) = 0.5 * (Geom.GetPoint( 1 ) + Geom.GetPoint( 4 ));
338  noalias(pmid2) = 0.5 * (Geom.GetPoint( 2 ) + Geom.GetPoint( 5 ));
339  } else if (number_of_nodes == 8) {
340  noalias(pmid0) = 0.5 * (Geom.GetPoint( 0 ) + Geom.GetPoint( 4 ));
341  noalias(pmid1) = 0.5 * (Geom.GetPoint( 1 ) + Geom.GetPoint( 5 ));
342  noalias(pmid2) = 0.5 * (Geom.GetPoint( 2 ) + Geom.GetPoint( 6 ));
343  }
344 
345  // Unitary vector in local x direction
346  array_1d<double, 3> Vx;
347  noalias(Vx) = pmid1 - pmid0;
348  double inv_norm = 1.0/norm_2(Vx);
349  Vx[0] *= inv_norm; // l1
350  Vx[1] *= inv_norm; // l2
351  Vx[2] *= inv_norm; // l3
352 
353  // Unitary vector in local z direction
354  array_1d<double, 3> Vy;
355  noalias(Vy) = pmid2 - pmid0;
356  array_1d<double, 3> Vz;
357  MathUtils<double>::CrossProduct(Vz, Vx, Vy);
358  inv_norm = 1.0/norm_2(Vz);
359  Vz[0] *= inv_norm; // n1
360  Vz[1] *= inv_norm; // n2
361  Vz[2] *= inv_norm; // n3
362 
363  // Unitary vector in local y direction
364  MathUtils<double>::CrossProduct(Vy, Vz, Vx); // Vy = [m1, m2, m3]
365 
366  // Create the inverse of the tranpose matrix as described in Eq. (8.2-8)
367  BoundedMatrix<double,3,3> T11;
368  T11(0,0) = Vx[0]*Vx[0];
369  T11(1,0) = Vx[1]*Vx[1];
370  T11(2,0) = Vx[2]*Vx[2];
371  T11(0,1) = Vy[0]*Vy[0];
372  T11(1,1) = Vy[1]*Vy[1];
373  T11(2,1) = Vy[2]*Vy[2];
374  T11(0,2) = Vz[0]*Vz[0];
375  T11(1,2) = Vz[1]*Vz[1];
376  T11(2,2) = Vz[2]*Vz[2];
377 
378  BoundedMatrix<double,3,3> T12;
379  T12(0,0) = Vx[0]*Vy[0];
380  T12(1,0) = Vx[1]*Vy[1];
381  T12(2,0) = Vx[2]*Vy[2];
382  T12(0,1) = Vy[0]*Vz[0];
383  T12(1,1) = Vy[1]*Vz[1];
384  T12(2,1) = Vy[2]*Vz[2];
385  T12(0,2) = Vz[0]*Vx[0];
386  T12(1,2) = Vz[1]*Vx[1];
387  T12(2,2) = Vz[2]*Vx[2];
388 
389  BoundedMatrix<double,3,3> T21;
390  T21(0,0) = 2*Vx[0]*Vx[1];
391  T21(1,0) = 2*Vx[1]*Vx[2];
392  T21(2,0) = 2*Vx[2]*Vx[0];
393  T21(0,1) = 2*Vy[0]*Vy[1];
394  T21(1,1) = 2*Vy[1]*Vy[2];
395  T21(2,1) = 2*Vy[2]*Vy[0];
396  T21(0,2) = 2*Vz[0]*Vz[1];
397  T21(1,2) = 2*Vz[1]*Vz[2];
398  T21(2,2) = 2*Vz[2]*Vz[0];
399 
400  BoundedMatrix<double,3,3> T22;
401  T22(0,0) = Vx[0]*Vy[1] + Vx[1]*Vy[0];
402  T22(1,0) = Vx[1]*Vy[2] + Vx[2]*Vy[1];
403  T22(2,0) = Vx[2]*Vy[0] + Vx[0]*Vy[2];
404  T22(0,1) = Vy[0]*Vz[1] + Vy[1]*Vz[0];
405  T22(1,1) = Vy[1]*Vz[2] + Vy[2]*Vz[1];
406  T22(2,1) = Vy[2]*Vz[0] + Vy[0]*Vz[2];
407  T22(0,2) = Vz[0]*Vx[1] + Vz[1]*Vx[0];
408  T22(1,2) = Vz[1]*Vx[2] + Vz[2]*Vx[1];
