160  bool SolveSolutionStep() override
161  {
162  // ********** Prediction phase **********
163  KRATOS_INFO("Ramm's Arc Length Strategy")
164  << "ARC-LENGTH RADIUS: " << mRadius / mRadius_0 << " X initial radius" << std::endl;
165 
166  // Initialize variables
167  DofsArrayType& rDofSet = mpBuilderAndSolver->GetDofSet();
168  TSystemMatrixType& mA = *mpA;
169  TSystemVectorType& mDx = *mpDx;
170  TSystemVectorType& mb = *mpb;
171  TSystemVectorType& mf = *mpf;
172  TSystemVectorType& mDxb = *mpDxb;
173  TSystemVectorType& mDxf = *mpDxf;
174  TSystemVectorType& mDxPred = *mpDxPred;
175  TSystemVectorType& mDxStep = *mpDxStep;
176 
177  // initializing the parameters of the iteration loop
178  double NormDx;
179  unsigned int iteration_number = 1;
180  BaseType::GetModelPart().GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
181  mpScheme->InitializeNonLinIteration(BaseType::GetModelPart(), mA, mDx, mb);
182  bool is_converged =
183  mpConvergenceCriteria->PreCriteria(BaseType::GetModelPart(), rDofSet, mA, mDx, mb);
184 
185  TSparseSpace::SetToZero(mA);
186  TSparseSpace::SetToZero(mb);
187  TSparseSpace::SetToZero(mDxf);
188 
189  // Note: This is not so efficient, but I want to solve mA*mDxf=mf without losing mf
190  this->BuildWithDirichlet(mA, mDxf, mb);
191  noalias(mb) = mf;
192  mpBuilderAndSolver->SystemSolve(mA, mDxf, mb);
193 
194  // update results
195  double DLambda = mRadius / TSparseSpace::TwoNorm(mDxf);
196  mDLambdaStep = DLambda;
197  mLambda += DLambda;
198  noalias(mDxPred) = DLambda * mDxf;
199  noalias(mDxStep) = mDxPred;
200  this->Update(rDofSet, mA, mDxPred, mb);
201 
202  // move the mesh if needed
203  if (BaseType::MoveMeshFlag()) BaseType::MoveMesh();
204 
205  // ********** Correction phase (iteration cycle) **********
206  if (is_converged) {
207  mpConvergenceCriteria->InitializeSolutionStep(BaseType::GetModelPart(), rDofSet, mA, mDxf, mb);
208  if (mpConvergenceCriteria->GetActualizeRHSflag()) {
209  TSparseSpace::SetToZero(mb);
210  mpBuilderAndSolver->BuildRHS(mpScheme, BaseType::GetModelPart(), mb);
211  }
212  is_converged =
213  mpConvergenceCriteria->PostCriteria(BaseType::GetModelPart(), rDofSet, mA, mDxf, mb);
214  }
215 
216  while (!is_converged && iteration_number++ < mMaxIterationNumber) {
217  // setting the number of iteration
218  BaseType::GetModelPart().GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
219 
220  mpScheme->InitializeNonLinIteration(BaseType::GetModelPart(), mA, mDx, mb);
221 
222  is_converged = mpConvergenceCriteria->PreCriteria(BaseType::GetModelPart(), rDofSet, mA, mDx, mb);
223 
224  TSparseSpace::SetToZero(mA);
225  TSparseSpace::SetToZero(mb);
226  TSparseSpace::SetToZero(mDxf);
227 
228  // Note: This is not so efficient, but I want to solve mA*mDxf=mf without losing mf
229  this->BuildWithDirichlet(mA, mDxf, mb);
230  noalias(mb) = mf;
231  mpBuilderAndSolver->SystemSolve(mA, mDxf, mb);
232 
233  TSparseSpace::SetToZero(mA);
234  TSparseSpace::SetToZero(mb);
235  TSparseSpace::SetToZero(mDxb);
236 
237  mpBuilderAndSolver->BuildAndSolve(mpScheme, BaseType::GetModelPart(), mA, mDxb, mb);
238 
239  DLambda = -TSparseSpace::Dot(mDxPred, mDxb) / TSparseSpace::Dot(mDxPred, mDxf);
240 
241  noalias(mDx) = mDxb + DLambda * mDxf;
242 
243  // Check solution before update
244  if (mNormxEquilibrium > 1.0e-10) {
245  NormDx = TSparseSpace::TwoNorm(mDx);
246 
247  if ((NormDx / mNormxEquilibrium) > 1.0e3 ||
248  (std::abs(DLambda) / std::abs(mLambda - mDLambdaStep)) > 1.0e3) {
249  is_converged = false;
250  break;
251  }
252  }
253 
254  // update results
255  mDLambdaStep += DLambda;
256  mLambda += DLambda;
257  noalias(mDxStep) += mDx;
258  this->Update(rDofSet, mA, mDx, mb);
259 
260  // move the mesh if needed
261  if (BaseType::MoveMeshFlag()) BaseType::MoveMesh();
262 
263  mpScheme->FinalizeNonLinIteration(BaseType::GetModelPart(), mA, mDx, mb);
264 
265  // *** Check Convergence ***
266  if (is_converged) {
267  if (mpConvergenceCriteria->GetActualizeRHSflag()) {
268  TSparseSpace::SetToZero(mb);
269  mpBuilderAndSolver->BuildRHS(mpScheme, BaseType::GetModelPart(), mb);
270  }
271  is_converged =
272  mpConvergenceCriteria->PostCriteria(BaseType::GetModelPart(), rDofSet, mA, mDx, mb);
273  }
274 
275  } // While
276 
277  // Check iteration_number
278  if (iteration_number >= mMaxIterationNumber) {
279  is_converged = true;
280  // plots a warning if the maximum number of iterations is exceeded
281  if (BaseType::GetModelPart().GetCommunicator().MyPID() == 0) {
282  this->MaxIterationsExceeded();
283  }
284  }
285 
286  // calculate reactions if required
287  if (mCalculateReactionsFlag) {
288  mpBuilderAndSolver->CalculateReactions(mpScheme, BaseType::GetModelPart(), mA, mDx, mb);
289  }
290 
291  BaseType::GetModelPart().GetProcessInfo()[IS_CONVERGED] = is_converged;
292 
293  return is_converged;
294  }

296  void FinalizeSolutionStep() override
297  {
298  KRATOS_TRY
299 
300  unsigned int iteration_number = BaseType::GetModelPart().GetProcessInfo()[NL_ITERATION_NUMBER];
301 
302  // Update the radius
303  mRadius = mRadius * sqrt(double(mDesiredIterations) / double(iteration_number));
304 
305  DofsArrayType& rDofSet = mpBuilderAndSolver->GetDofSet();
306  TSystemMatrixType& mA = *mpA;
307  TSystemVectorType& mDx = *mpDx;
308  TSystemVectorType& mb = *mpb;
309 
310  if (BaseType::GetModelPart().GetProcessInfo()[IS_CONVERGED]) {
311  // Modify the radius to advance faster when convergence is achieved
312  if (mRadius > mMaxRadiusFactor * mRadius_0) mRadius = mMaxRadiusFactor * mRadius_0;
313  else if (mRadius < mMinRadiusFactor * mRadius_0) mRadius = mMinRadiusFactor * mRadius_0;
314 
315  // Update Norm of x
316  mNormxEquilibrium = CalculateReferenceDofsNorm(rDofSet);
317  } else {
318  std::cout << "************ NO CONVERGENCE: restoring equilibrium path ************" << std::endl;
319 
320  TSystemVectorType& mDxStep = *mpDxStep;
321 
322  // update results
323  mLambda -= mDLambdaStep;
324  noalias(mDx) = -mDxStep;
325  this->Update(rDofSet, mA, mDx, mb);
326 
327  // move the mesh if needed
328  if (BaseType::MoveMeshFlag()) BaseType::MoveMesh();
329  }
330 
331  BaseType::GetModelPart().GetProcessInfo()[ARC_LENGTH_LAMBDA] = mLambda;
332  BaseType::GetModelPart().GetProcessInfo()[ARC_LENGTH_RADIUS_FACTOR] = mRadius / mRadius_0;
333 
334  mpScheme->FinalizeSolutionStep(BaseType::GetModelPart(), mA, mDx, mb);
335  mpBuilderAndSolver->FinalizeSolutionStep(BaseType::GetModelPart(), mA, mDx, mb);
336 
337  // Cleaning memory after the solution
338  mpScheme->Clean();
339 
340  if (mReformDofSetAtEachStep) // deallocate the systemvectors
341  {
342  this->ClearStep();
343  }
344 
345  KRATOS_CATCH("")
346  }

489  void UpdateExternalLoads()
490  {
491  // Update External Loads
492  for (unsigned int i = 0; i < mVariableNames.size(); i++) {
493  ModelPart& rSubModelPart = *(mSubModelPartList[i]);
494  const std::string& VariableName = mVariableNames[i];
495 
496  if (KratosComponents<Variable<double>>::Has(VariableName)) {
497  const Variable<double>& var = KratosComponents<Variable<double>>::Get(VariableName);
498 
499 #pragma omp parallel
500  {
501  ModelPart::NodeIterator NodesBegin;
502  ModelPart::NodeIterator NodesEnd;
503  OpenMPUtils::PartitionedIterators(rSubModelPart.Nodes(), NodesBegin, NodesEnd);
504 
505  for (ModelPart::NodeIterator itNode = NodesBegin; itNode != NodesEnd; ++itNode) {
506  double& rvalue = itNode->FastGetSolutionStepValue(var);
507  rvalue *= (mLambda / mLambda_old);
508  }
509  }
510  } else if (KratosComponents<Variable<array_1d<double, 3>>>::Has(VariableName)) {
511  typedef Variable<double> component_type;
512  const component_type& varx =
513  KratosComponents<component_type>::Get(VariableName + std::string("_X"));
514  const component_type& vary =
515  KratosComponents<component_type>::Get(VariableName + std::string("_Y"));
516  const component_type& varz =
517  KratosComponents<component_type>::Get(VariableName + std::string("_Z"));
518 
519 #pragma omp parallel
520  {
521  ModelPart::NodeIterator NodesBegin;
522  ModelPart::NodeIterator NodesEnd;
523  OpenMPUtils::PartitionedIterators(rSubModelPart.Nodes(), NodesBegin, NodesEnd);
524 
525  for (ModelPart::NodeIterator itNode = NodesBegin; itNode != NodesEnd; ++itNode) {
526  double& rvaluex = itNode->FastGetSolutionStepValue(varx);
527  rvaluex *= mLambda / mLambda_old;
528  double& rvaluey = itNode->FastGetSolutionStepValue(vary);
529  rvaluey *= mLambda / mLambda_old;
530  double& rvaluez = itNode->FastGetSolutionStepValue(varz);
531  rvaluez *= mLambda / mLambda_old;
532  }
533  }
534  } else {
535  KRATOS_ERROR
536  << "One variable of the applied loads has a non supported type. Variable: " << VariableName
537  << std::endl;
538  }
539  }
540 
541  // Save the applied Lambda factor
542  mLambda_old = mLambda;
543  }

546  static double CalculateReferenceDofsNorm(DofsArrayType& rDofSet)
547  {
548  auto is_free_dof = [](const auto& rDof) { return rDof.IsFree(); };
549  auto free_dofs = rDofSet | boost::adaptors::filtered(is_free_dof);
550 
551  auto free_dof_values = Vector{boost::size(free_dofs)};
552  auto get_dof_value = [](const auto& rDof) { return rDof.GetSolutionStepValue(); };
553  std::transform(std::begin(free_dofs), std::end(free_dofs), free_dof_values.begin(), get_dof_value);
554 
555  return norm_2(free_dof_values);
556  }
