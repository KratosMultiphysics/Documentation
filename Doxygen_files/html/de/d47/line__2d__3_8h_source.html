437  CoordinatesArrayType& PointLocalCoordinates(
438  CoordinatesArrayType& rResult,
439  const CoordinatesArrayType& rPoint
440  ) const override
441  {
442  BoundedMatrix<double,3,3> X;
443  BoundedMatrix<double,3,1> DN;
444  for(IndexType i=0; i<this->size(); ++i) {
445  const auto& r_node = this->GetPoint(i);
446  X(0, i) = r_node.X();
447  X(1, i) = r_node.Y();
448  X(2, i) = r_node.Z();
449  }
450 
451  static constexpr double MaxNormPointLocalCoordinates = 300.0;
452  static constexpr std::size_t MaxIteratioNumberPointLocalCoordinates = 500;
453  static constexpr double MaxTolerancePointLocalCoordinates = 1.0e-8;
454 
455  Matrix J = ZeroMatrix( 1, 1 );
456  Matrix invJ = ZeroMatrix( 1, 1 );
457 
458  // Starting with xi = 0
459  if (rResult.size() != 3)
460  rResult.resize(3, false);
461  rResult = ZeroVector( 3 );
462  double delta_xi = 0.0;
463  const array_1d<double, 3> zero_array = ZeroVector(3);
464  array_1d<double, 3> current_global_coords;
465 
466  //Newton iteration:
467  for ( IndexType k = 0; k < MaxIteratioNumberPointLocalCoordinates; k++ ) {
468  noalias(current_global_coords) = zero_array;
469  this->GlobalCoordinates( current_global_coords, rResult );
470 
471  noalias( current_global_coords ) = rPoint - current_global_coords;
472 
473  // Derivatives of shape functions
474  Matrix shape_functions_gradients;
475  shape_functions_gradients = ShapeFunctionsLocalGradients(shape_functions_gradients, rResult );
476  noalias(DN) = prod(X, shape_functions_gradients);
477 
478  noalias(J) = prod(trans(DN), DN);
479  const array_1d<double, 1> res = prod(trans(DN), current_global_coords);
480 
481  // The inverted jacobian matrix
482  invJ(0, 0) = 1.0/J( 0, 0 );
483 
484  delta_xi = invJ(0, 0) * res[0];
485 
486  rResult[0] += delta_xi;
487 
488  if ( delta_xi > MaxNormPointLocalCoordinates ) {
489  KRATOS_WARNING_IF("Line2D3", k > 0) << "detJ =\t" << J( 0, 0 ) << " DeltaX =\t" << delta_xi << " stopping calculation. Iteration:\t" << k << std::endl;
490  break;
491  }
492 
493  if ( delta_xi < MaxTolerancePointLocalCoordinates )
494  break;
495  }
496 
497  return rResult;
498  }

523
