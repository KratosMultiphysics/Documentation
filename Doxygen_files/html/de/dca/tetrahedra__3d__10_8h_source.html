674  double ShapeFunctionValue( IndexType ShapeFunctionIndex,
675  const CoordinatesArrayType& rPoint ) const override
676  {
677  double fourthCoord = 1.0 - ( rPoint[0] + rPoint[1] + rPoint[2] );
678 
679  switch ( ShapeFunctionIndex )
680  {
681  case 0:
682  return( fourthCoord*( 2*fourthCoord - 1 ) );
683  case 1:
684  return( rPoint[0]*( 2*rPoint[0] - 1 ) );
685  case 2:
686  return( rPoint[1]*( 2*rPoint[1] - 1 ) );
687  case 3:
688  return( rPoint[2]*( 2*rPoint[2] - 1 ) );
689  case 4:
690  return( 4*fourthCoord*rPoint[0] );
691  case 5:
692  return( 4*rPoint[0]*rPoint[1] );
693  case 6:
694  return( 4*fourthCoord*rPoint[1] );
695  case 7:
696  return( 4*fourthCoord*rPoint[2] );
697  case 8:
698  return( 4*rPoint[0]*rPoint[2] );
699  case 9:
700  return( 4*rPoint[1]*rPoint[2] );
701  default:
702  KRATOS_ERROR << "Wrong index of shape function!" << *this << std::endl;
703  }
704 
705  return 0;
706  }

1148  bool FacesArePlanar() const
1149  {

1152  {{0, 1, 4}, {1, 2, 5}, {2, 0, 6}, {0, 3, 7}, {1, 3, 8}, {2, 3, 9}}};

1154  for (const auto& r_edge : edges) {

1156  const double b = MathUtils<double>::Norm3(r_points[r_edge[1]] - r_points[r_edge[2]]);
1157  const double c = MathUtils<double>::Norm3(r_points[r_edge[2]] - r_points[r_edge[0]]);
1158  if (b + c > a*(1.0+tol) ) {
1159  return false;
1160  }
1161  }
1162  return true;
