997  bool HasIntersection(const BaseType& rOtherGeometry) const override
998  {
999  const double tolerance = std::numeric_limits<double>::epsilon();
1000  // We get the local points
1001  const TPointType& first_point = BaseType::GetPoint(0); //p1

1012  return false;

1026  bool HasIntersection(const Point& rLowPoint, const Point& rHighPoint) const override
1027  {
1028  const double tolerance = std::numeric_limits<double>::epsilon();
1029  // We get the local points
1030  const TPointType& first_point = BaseType::GetPoint(0);
1031  const TPointType& second_point = BaseType::GetPoint(1);
1032 
1033  if ( // If one of the point is inside the box then there is an intersection. If none is inside then we check further.
1034  ( (first_point[0] >= rLowPoint[0] && first_point[0] <= rHighPoint[0])
1035  && (first_point[1] >= rLowPoint[1] && first_point[1] <= rHighPoint[1]) ) // IF the first point is inside the box
1036  ||
1037  ( (second_point[0] >= rLowPoint[0] && second_point[0] <= rHighPoint[0])
1038  && (second_point[1] >= rLowPoint[1] && second_point[1] <= rHighPoint[1]) ) // IF the second point is inside the box
1039  )
1040  return true;
1041 
1042  const double high_x = rHighPoint[0];
1043  const double high_y = rHighPoint[1];
1044  const double low_x = rLowPoint[0];
1045  const double low_y = rLowPoint[1];
1046 
1047  const double denominator = ( second_point[0] - first_point[0] );
1048  const double numerator = (second_point[1] - first_point[1]);
1049  const double slope = std::abs(denominator) > tolerance ? std::abs(numerator) > tolerance ? numerator / denominator : 1.0e-12 : 1.0e12;
1050 
1051  // Intersection with left vertical line of the box that is x = low_x
1052  const double y_1 = slope*( low_x - first_point[0] ) + first_point[1];
1053  if(y_1 >= low_y - tolerance && y_1 <= high_y+tolerance) // If y intersection is between two y bounds there is an intersection
1054  return true;
1055  // Intersection with right vertical line of the box that is x = high_x
1056  const double y_2 = slope*( high_x - first_point[0] ) + first_point[1];
1057  if(y_2 >= low_y - tolerance && y_2 <= high_y+tolerance) // If y intersection is between two y bounds there is an intersection
1058  return true;
1059  // Intersection with bottom horizontal line of the box that is y = low_y
1060  const double x_1 = first_point[0] + ( (low_y - first_point[1]) / slope );
1061  if(x_1 >= low_x-tolerance && x_1 <= high_x+tolerance) // If x intersection is between two x bounds there is an intersection
1062  return true;
1063  // Intersection with top horizontal line of the box that is y = high_y
1064  const double x_2 = first_point[0] + ( (high_y - first_point[1]) / slope );
1065  if(x_2 >= low_x-tolerance && x_2 <= high_x+tolerance) // If x intersection is between two x bounds there is an intersection
1066  return true;
1067 
1068 
1069  return false;
1070  }
