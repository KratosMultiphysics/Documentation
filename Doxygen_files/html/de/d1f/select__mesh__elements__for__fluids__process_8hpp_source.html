632  void ControlSkewedElements2D(bool &accepted,
633  std::vector<double> &normVelocityP,
634  std::vector<array_1d<double, 3>> &nodesVelocities)
635  {
636  KRATOS_TRY
637 
638  const double maxValue = 1.5;
639  const double minValue = 1.0 / maxValue;
640  if (normVelocityP[0] / normVelocityP[1] > maxValue || normVelocityP[0] / normVelocityP[1] < minValue ||
641  normVelocityP[0] / normVelocityP[2] > maxValue || normVelocityP[0] / normVelocityP[2] < minValue ||
642  normVelocityP[1] / normVelocityP[2] > maxValue || normVelocityP[1] / normVelocityP[2] < minValue)
643  {
644  accepted = false;
645  }
646  else
647  {
648  const double cosAngle01 = (nodesVelocities[0][0] * nodesVelocities[1][0] + nodesVelocities[0][1] * nodesVelocities[1][1]) /
649  (std::sqrt(std::pow(nodesVelocities[0][0], 2) + std::pow(nodesVelocities[0][1], 2)) *
650  std::sqrt(std::pow(nodesVelocities[1][0], 2) + std::pow(nodesVelocities[1][1], 2)));
651  const double cosAngle02 = (nodesVelocities[0][0] * nodesVelocities[2][0] + nodesVelocities[0][1] * nodesVelocities[2][1]) /
652  (std::sqrt(std::pow(nodesVelocities[0][0], 2) + std::pow(nodesVelocities[0][1], 2)) *
653  std::sqrt(std::pow(nodesVelocities[2][0], 2) + std::pow(nodesVelocities[2][1], 2)));
654  const double cosAngle12 = (nodesVelocities[1][0] * nodesVelocities[2][0] + nodesVelocities[1][1] * nodesVelocities[2][1]) /
655  (std::sqrt(std::pow(nodesVelocities[1][0], 2) + std::pow(nodesVelocities[1][1], 2)) *
656  std::sqrt(std::pow(nodesVelocities[2][0], 2) + std::pow(nodesVelocities[2][1], 2)));
657 
658  if (std::abs(cosAngle01) < 0.95 || std::abs(cosAngle02) < 0.95 || std::abs(cosAngle12) < 0.95)
659  {
660  accepted = false;
661  }
662  }
663  KRATOS_CATCH("")
664  }

666  void ControlSkewedElements3D(bool &accepted,
667  std::vector<double> &normVelocityP,
668  std::vector<array_1d<double, 3>> &nodesVelocities)
669  {
670  KRATOS_TRY
671  const double maxValue = 2.5;
672  const double minValue = 1.0 / maxValue;
673  if (normVelocityP[0] / normVelocityP[1] < minValue || normVelocityP[0] / normVelocityP[2] < minValue || normVelocityP[0] / normVelocityP[3] < minValue ||
674  normVelocityP[0] / normVelocityP[1] > maxValue || normVelocityP[0] / normVelocityP[2] > maxValue || normVelocityP[0] / normVelocityP[3] > maxValue ||
675  normVelocityP[1] / normVelocityP[2] < minValue || normVelocityP[1] / normVelocityP[3] < minValue ||
676  normVelocityP[1] / normVelocityP[2] > maxValue || normVelocityP[1] / normVelocityP[3] > maxValue ||
677  normVelocityP[2] / normVelocityP[3] < minValue ||
678  normVelocityP[2] / normVelocityP[3] > maxValue)
679  {
680  accepted = false;
681  }
682  else
