106  void ExecuteInitialize() override

110  const Variable<double>& var = KratosComponents<Variable<double>>::Get(mVariableName);
111  block_for_each(mrModelPart.Nodes(), [&var, this](Node& rNode) {
112  const double pressure = PORE_PRESSURE_SIGN_FACTOR * CalculatePressure(rNode);
113 
114  if (mIsSeepage) {
115  if (pressure < PORE_PRESSURE_SIGN_FACTOR * mPressureTensionCutOff) { // Before 0. was used i.s.o. the tension cut off value -> no effect in any test.
116  rNode.FastGetSolutionStepValue(var) = pressure;
117  if (mIsFixed) rNode.Fix(var);
118  } else {
119  if (mIsFixedProvided) rNode.Free(var);
120  }
121  } else {
122  rNode.FastGetSolutionStepValue(var) =
123  std::min(pressure, PORE_PRESSURE_SIGN_FACTOR * mPressureTensionCutOff);
124  if (mIsFixed) rNode.Fix(var);
125  else if (mIsFixedProvided) rNode.Free(var);

153  double CalculatePressure(const Node& rNode) const
154  {
155  double x = rNode.Coordinates()[mHorizontalDirection];
156  x = std::max(x, mMinHorizontalCoordinate);
157  x = std::min(x, mMaxHorizontalCoordinate);
158  const double height = mSlope * (x - mFirstReferenceCoordinate[mHorizontalDirection]) +
159  mFirstReferenceCoordinate[mGravityDirection];
160  const double distance = height - rNode.Coordinates()[mGravityDirection];
