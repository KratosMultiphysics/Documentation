776  void ComputeDihedralAngles(Vector& rDihedralAngles) const override
777  {
778  if(rDihedralAngles.size() != 24) {
779  rDihedralAngles.resize(24, false);
780  }

782  // The three faces that contain the node i can be obtained by doing:
783  // faces[faces_0[i]], faces[faces_1[i]], faces[faces_2[i]]
784  const std::array<unsigned int, 8> faces_0 = {0,0,0,0,5,5,5,5};
785  const std::array<unsigned int, 8> faces_1 = {1,1,3,3,1,1,3,3};
786  const std::array<unsigned int, 8> faces_2 = {4,2,2,4,4,2,2,4};

789  double dihedral_angle_0, dihedral_angle_1, dihedral_angle_2;
790  for (unsigned int i = 0; i < 8; ++i) {
791  const TPointType& r_point_i = this->GetPoint(i);
792  noalias(normal_0) = faces[faces_0[i]].UnitNormal(r_point_i);
793  noalias(normal_1) = faces[faces_1[i]].UnitNormal(r_point_i);
794  noalias(normal_2) = faces[faces_2[i]].UnitNormal(r_point_i);
795  dihedral_angle_0 = std::acos(inner_prod(normal_0, -normal_1));
796  dihedral_angle_1 = std::acos(inner_prod(normal_0, -normal_2));
797  dihedral_angle_2 = std::acos(inner_prod(normal_2, -normal_1));
798  rDihedralAngles[i*3] = dihedral_angle_0;
799  rDihedralAngles[i*3 + 1] = dihedral_angle_1;
800  rDihedralAngles[i*3 + 2] = dihedral_angle_2;
