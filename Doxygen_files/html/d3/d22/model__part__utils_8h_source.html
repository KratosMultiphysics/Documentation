110  static void AddNodesFromOrderedContainer(ModelPart& rTargetModelPart, TIteratorType iNodesBegin, TIteratorType iNodesEnd)
111  {
112  KRATOS_TRY
113  ModelPart::NodesContainerType aux;
114  ModelPart* root_model_part = &rTargetModelPart.GetRootModelPart();
115 
116  for(TIteratorType it = iNodesBegin; it!=iNodesEnd; it++) {
117  aux.push_back( *(it.base()) );
118  }
119  // Add the nodes to the root modelpart
120  root_model_part->Nodes() = JoinOrderedNodesContainerType(root_model_part->Nodes().begin(), root_model_part->Nodes().end(), aux.begin(), aux.end());
121 
122  // Add to all of the leaves
123  ModelPart* current_part = &rTargetModelPart;
124  while(current_part->IsSubModelPart()) {
125  current_part->Nodes() = JoinOrderedNodesContainerType(current_part->Nodes().begin(), current_part->Nodes().end(), aux.begin(), aux.end());
126  current_part = &(current_part->GetParentModelPart());
127  }

316  template<class TIteratorType >
317  static typename ModelPart::NodesContainerType JoinOrderedNodesContainerType(
318  TIteratorType iC1Begin,
319  TIteratorType iC1End,
320  TIteratorType iC2Begin,
321  TIteratorType iC2End
322  )
323  {
324  std::size_t c1length = std::distance(iC1Begin,iC1End);
325  std::size_t c2length = std::distance(iC2Begin,iC2End);
326  TIteratorType blong, elong, bshort, eshort;
327  ModelPart::NodesContainerType aux;
328  aux.reserve(c1length + c2length);
329 
330  // We order c1 and c2 to long and short
331  if(c1length>c2length){
332  blong = iC1Begin; elong = iC1End;
333  bshort = iC2Begin; eshort = iC2End;
334  } else {
335  blong = iC2Begin; elong = iC2End;
336  bshort = iC1Begin; eshort = iC1End;
337  }
338 
339  // If short is empty we return long. If both empty it returns empty aux
340  if(c2length == 0 || c1length == 0){
341  for(TIteratorType it1=blong; it1!=elong; it1++){
342  aux.push_back(*(it1.base()) );
343  }
344  return aux;
345  }
346  TIteratorType it2 = blong;
347  for(TIteratorType it1=bshort; it1!=eshort; it1++) {
348  while(it2!=elong && it2->Id()<it1->Id()) {
349  aux.push_back(*(it2.base()));
350  it2++;
351  }
352  aux.push_back(*(it1.base()) );
353  if(it2!=elong && (it1->Id() == it2->Id())) { //If both are the same, then we need to skip
354  it2++;
355  }
356  }
357  while(it2 != elong) {
358  aux.push_back(*(it2.base()));
359  it2++;
360  }
361 
362  return aux;
363  }
