130  template<class TIteratorType >
131  static typename ModelPart::NodesContainerType JoinOrderedNodesContainerType(
132  TIteratorType iC1Begin, TIteratorType iC1End, TIteratorType iC2Begin, TIteratorType iC2End)
133  {
134  std::size_t c1length = std::distance(iC1Begin,iC1End);
135  std::size_t c2length = std::distance(iC2Begin,iC2End);
136  TIteratorType blong, elong, bshort, eshort;
137  ModelPart::NodesContainerType aux;
138  aux.reserve(c1length + c2length);
139 
140  // We order c1 and c2 to long and short
141  if(c1length>c2length){
142  blong = iC1Begin; elong = iC1End;
143  bshort = iC2Begin; eshort = iC2End;
144  } else {
145  blong = iC2Begin; elong = iC2End;
146  bshort = iC1Begin; eshort = iC1End;
147  }
148 
149  // If short is empty we return long. If both empty it returns empty aux
150  if(c2length == 0 || c1length == 0){
151  for(TIteratorType it1=blong; it1!=elong; it1++){
152  aux.push_back(*(it1.base()) );
153  }

165  }
166  }

170  }
