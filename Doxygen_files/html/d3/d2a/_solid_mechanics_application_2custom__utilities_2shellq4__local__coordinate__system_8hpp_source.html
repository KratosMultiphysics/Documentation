61  ShellQ4_LocalCoordinateSystem(const Vector3Type & P1global,
62  const Vector3Type & P2global,
63  const Vector3Type & P3global,
64  const Vector3Type & P4global)
65  : mP(4)
66  , mOrientation(3, 3)
67  {
68  // Form the basis vectors alligning the local X direction
69  // with the 1-2 side
70 
71  // compute the central point
72  noalias( mCenter ) = P1global;
73  noalias( mCenter ) += P2global;
74  noalias( mCenter ) += P3global;
75  noalias( mCenter ) += P4global;
76  mCenter *= 0.25;
77 
78  // compute the diagonal vectors
79  Vector3Type d13( P3global - P1global );
80  Vector3Type d24( P4global - P2global );
81 
82  // compute the Normal vector at the element center
83  // as the cross product of the 2 diagonals.
84  // While normalizing the normal vector save its norm to compute the area.
85  // Note: the norm should be divided by 2 because its computed from the
86  // cross product of the diagonals, which gives twice the area!
87  Vector3Type e3;
88  MathUtils<RealType>::CrossProduct(e3, d13, d24);
89  mArea = NormalizeVector3( e3 );
90  mArea /= 2.0;
91 
92  // compute the local X direction parallel to the projection of the side 1-2 onto
93  // the local xy plane.
94  Vector3Type e1( P2global - P1global );
95  RealType e1_dot_e3 = MathUtils<RealType>::Dot(e1, e3);
96  noalias( e1 ) -= e1_dot_e3 * e3;
97  NormalizeVector3( e1 );
98 
99  // finally compute the local Y direction to be orthogonal to both X and Z local directions
100  Vector3Type e2;
101  MathUtils<RealType>::CrossProduct(e2, e3, e1);
102  NormalizeVector3( e2 );
103 
104  // form the 3x3 transformation matrix
105  for(int i = 0; i < 3; i++)
106  {
107  mOrientation(0, i) = e1(i);
108  mOrientation(1, i) = e2(i);
109  mOrientation(2, i) = e3(i);
110  }
111 
112  // transform global coordinates to the local coordinate system
113  for(int i = 0; i < 3; i++)
114  {
115  mP[0](i) = mOrientation(i, 0) * (P1global(0) - mCenter(0)) + mOrientation(i, 1) * (P1global(1) - mCenter(1)) + mOrientation(i, 2) * (P1global(2) - mCenter(2));
116  mP[1](i) = mOrientation(i, 0) * (P2global(0) - mCenter(0)) + mOrientation(i, 1) * (P2global(1) - mCenter(1)) + mOrientation(i, 2) * (P2global(2) - mCenter(2));
117  mP[2](i) = mOrientation(i, 0) * (P3global(0) - mCenter(0)) + mOrientation(i, 1) * (P3global(1) - mCenter(1)) + mOrientation(i, 2) * (P3global(2) - mCenter(2));
118  mP[3](i) = mOrientation(i, 0) * (P4global(0) - mCenter(0)) + mOrientation(i, 1) * (P4global(1) - mCenter(1)) + mOrientation(i, 2) * (P4global(2) - mCenter(2));
119  }
120  }

309  KRATOS_ERROR_IF_NOT(TComponent >= 0 && TComponent < 3) <<
