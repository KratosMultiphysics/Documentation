564

566 
567  //Newton iteration:
568  for ( int k = 0; k < maxiter; k++ )
569  {

571  this->GlobalCoordinates( CurrentGlobalCoords, rResult );
572  noalias( CurrentGlobalCoords ) = rPoint - CurrentGlobalCoords;

623  }
624  }
