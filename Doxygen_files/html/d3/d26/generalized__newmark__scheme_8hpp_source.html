63  void FinalizeSolutionStep(ModelPart& rModelPart, TSystemMatrixType& A, TSystemVectorType& Dx, TSystemVectorType& b) override
64  {
65  KRATOS_TRY
66 
67  if (rModelPart.GetProcessInfo()[NODAL_SMOOTHING]) {
68  const unsigned int dim = rModelPart.GetProcessInfo()[DOMAIN_SIZE];
69  const auto stress_tensor_size = dim == N_DIM_3D ? STRESS_TENSOR_SIZE_3D : STRESS_TENSOR_SIZE_2D;
70 
71  // Clear nodal variables
72  block_for_each(rModelPart.Nodes(), [&stress_tensor_size](Node& rNode) {
73  rNode.FastGetSolutionStepValue(NODAL_AREA) = 0.0;
74  Matrix& r_nodal_stress = rNode.FastGetSolutionStepValue(NODAL_CAUCHY_STRESS_TENSOR);
75  if (r_nodal_stress.size1() != stress_tensor_size)
76  r_nodal_stress.resize(stress_tensor_size, stress_tensor_size, false);
77  noalias(r_nodal_stress) = ZeroMatrix(stress_tensor_size, stress_tensor_size);
78  rNode.FastGetSolutionStepValue(NODAL_DAMAGE_VARIABLE) = 0.0;
79  rNode.FastGetSolutionStepValue(NODAL_JOINT_AREA) = 0.0;
80  rNode.FastGetSolutionStepValue(NODAL_JOINT_WIDTH) = 0.0;
81  rNode.FastGetSolutionStepValue(NODAL_JOINT_DAMAGE) = 0.0;
82  });
83 
84  this->FinalizeSolutionStepActiveEntities(rModelPart, A, Dx, b);
85 
86  // Compute smoothed nodal variables
87  block_for_each(rModelPart.Nodes(), [](Node& rNode) {
88  if (const double& nodal_area = rNode.FastGetSolutionStepValue(NODAL_AREA); nodal_area > 1.0e-20) {
89  const double inv_nodal_area = 1.0 / nodal_area;
90  rNode.FastGetSolutionStepValue(NODAL_CAUCHY_STRESS_TENSOR) *= inv_nodal_area;
91  rNode.FastGetSolutionStepValue(NODAL_DAMAGE_VARIABLE) *= inv_nodal_area;
92  }
93 
94  if (const double& nodal_joint_area = rNode.FastGetSolutionStepValue(NODAL_JOINT_AREA);
95  nodal_joint_area > 1.0e-20) {
96  const double inv_nodal_joint_area = 1.0 / nodal_joint_area;
97  rNode.FastGetSolutionStepValue(NODAL_JOINT_WIDTH) *= inv_nodal_joint_area;
98  rNode.FastGetSolutionStepValue(NODAL_JOINT_DAMAGE) *= inv_nodal_joint_area;
99  }
100  });
101  } else {
102  this->FinalizeSolutionStepActiveEntities(rModelPart, A, Dx, b);
103  }
104 
105  KRATOS_CATCH("")
106  }

134  void UpdateScalarTimeDerivative(Node& rNode, const Variable<double>& variable, const Variable<double>& dt_variable) const
135  {
136  if (rNode.IsFixed(dt_variable)) return;
137 
138  const auto delta_variable =
139  rNode.FastGetSolutionStepValue(variable, 0) - rNode.FastGetSolutionStepValue(variable, 1);
140  const auto previous_dt_variable = rNode.FastGetSolutionStepValue(dt_variable, 1);
141 
142  rNode.FastGetSolutionStepValue(dt_variable, 0) =
143  (delta_variable - (1.0 - GetTheta()) * this->GetDeltaTime() * previous_dt_variable) /
144  (GetTheta() * this->GetDeltaTime());
145  }

147  void UpdateVectorFirstTimeDerivative(Node& rNode) const
148  {
149  for (const auto& r_second_order_vector_variable : this->GetSecondOrderVectorVariables()) {
150  if (!rNode.SolutionStepsDataHas(r_second_order_vector_variable.instance)) continue;
151 
152  const array_1d<double, 3> updated_first_derivative =
153  rNode.FastGetSolutionStepValue(r_second_order_vector_variable.first_time_derivative, 1) +
154  (1.0 - GetGamma()) * this->GetDeltaTime() *
155  rNode.FastGetSolutionStepValue(r_second_order_vector_variable.second_time_derivative, 1) +
156  GetGamma() * this->GetDeltaTime() *
157  rNode.FastGetSolutionStepValue(r_second_order_vector_variable.second_time_derivative, 0);
158 
159  NodeUtilities::AssignUpdatedVectorVariableToNonFixedComponents(
160  rNode, r_second_order_vector_variable.first_time_derivative, updated_first_derivative);
161  }
162  }

172  this->GetDeltaTime() * rNode.FastGetSolutionStepValue(
173  r_second_order_vector_variable.first_time_derivative, 1) -
174  (0.5 - GetBeta()) * this->GetDeltaTime() * this->GetDeltaTime() *
175  rNode.FastGetSolutionStepValue(r_second_order_vector_variable.second_time_derivative, 1)) /
176  (GetBeta() * this->GetDeltaTime() * this->GetDeltaTime());
177 
178  NodeUtilities::AssignUpdatedVectorVariableToNonFixedComponents(
179  rNode, r_second_order_vector_variable.second_time_derivative, updated_second_time_derivative);
