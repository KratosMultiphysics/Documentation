124  GeometryIterator AddGeometry(GeometryPointerType pNewGeometry)
125  {
126  auto i = mGeometries.find(pNewGeometry->Id());
127  if (i == mGeometries.end()) {
128  return mGeometries.insert(pNewGeometry);
129  } else if (&(*i) == pNewGeometry.get()) { // check if the pointee coincides
130  return i;
131  } else { // Check if the connectivities coincide
132  // First check for the geometry type
133  KRATOS_ERROR_IF_NOT(TGeometryType::HasSameGeometryType(*i, *pNewGeometry)) << "Attempting to add geometry with Id: " << pNewGeometry->Id() << ". A different geometry with the same Id already exists." << std::endl;
134  // Check that the connectivities are the same
135  // note that we deliberately check the node ids and not the pointer adresses as there might be very rare situations
136 
137  // (e.g., creating nodes bypassing the model part interface) with same connectivities but different pointer addresses
138  for (IndexType i_node = 0; i_node < i->PointsNumber(); ++i_node) {
139  KRATOS_ERROR_IF((*i)[i_node].Id() != (*pNewGeometry)[i_node].Id()) << "Attempting to add a new geometry with Id: " << pNewGeometry->Id() << ". A same type geometry with same Id but different connectivities already exists." << std::endl;
140  }
141  return i;
142  }
143  }
