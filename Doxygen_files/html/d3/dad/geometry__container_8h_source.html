121  GeometryIterator AddGeometry(GeometryPointerType pNewGeometry)
122  {
123  auto i = mGeometries.find(pNewGeometry->Id());
124  if (i == mGeometries.end()) {
125  return mGeometries.insert(pNewGeometry);
126  } else if (&(*i) == pNewGeometry.get()) { // check if the pointee coincides
127  return i;
128  } else { // Check if the connectivities coincide
129  // First check for the geometry type
130  KRATOS_ERROR_IF_NOT(TGeometryType::HasSameGeometryType(*i, *pNewGeometry)) << "Attempting to add geometry with Id: " << pNewGeometry->Id() << ". A different geometry with the same Id already exists." << std::endl;
131  // Check that the connectivities are the same
132  // note that we deliberately check the node ids and not the pointer adresses as there might be very rare situations
133 
134  // (e.g., creating nodes bypassing the model part interface) with same connectivities but different pointer addresses
135  for (IndexType i_node = 0; i_node < i->PointsNumber(); ++i_node) {

137  }
