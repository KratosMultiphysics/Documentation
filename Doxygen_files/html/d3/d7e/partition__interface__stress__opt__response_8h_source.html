108  double CalculateElementStress(Element& elem_i, const ProcessInfo &rCurrentProcessInfo){
109 
110  std::vector<double> gp_weights_vector;
111  std::vector<double> stress_gp_vector;
112  elem_i.CalculateOnIntegrationPoints(INTEGRATION_WEIGHT, gp_weights_vector, rCurrentProcessInfo);
113  elem_i.CalculateOnIntegrationPoints(KratosComponents<Variable<double>>::Get("VON_MISES_STRESS"), stress_gp_vector, rCurrentProcessInfo);
114  double element_density = elem_i.GetProperties().GetValue(DENSITY);
115 
116  double elem_value = 0.0;
117  for(IndexType i = 0; i < gp_weights_vector.size(); i++){
118  elem_value += gp_weights_vector[i] * stress_gp_vector[i] * std::pow((element_density) * (1.0-element_density),2);
119  }
120 
121  return elem_value;

124  double CalculateValue() override {
125  double intg_stress = 0.0;
126  for(auto& eval_obj : mrResponseSettings["evaluated_objects"]){
127  ModelPart& r_eval_object = mrModel.GetModelPart(eval_obj.GetString());
128  const ProcessInfo &CurrentProcessInfo = r_eval_object.GetProcessInfo();
129  // Sum all elemental strain energy values calculated as: W_e = u_e^T K_e u_e
130  intg_stress += block_for_each<SumReduction<double>>(r_eval_object.Elements(), [&](auto& elem_i) {

143  void CalculateGradient() override {
144 
145  KRATOS_TRY;

147  // compute sensitivities

152  VariableUtils().SetHistoricalVariableToZero(D_STRESS_D_FD, controlled_model_part.Nodes());

154  block_for_each(controlled_model_part.Elements(), [&](auto& elem_i) {
155  const bool element_is_active = elem_i.IsDefined(ACTIVE) ? elem_i.Is(ACTIVE) : true;
156  if(element_is_active){
157  auto& r_this_geometry = elem_i.GetGeometry();
158  const std::size_t number_of_nodes = r_this_geometry.size();
159 
160  std::vector<double> gp_weights_vector;
161  elem_i.CalculateOnIntegrationPoints(INTEGRATION_WEIGHT, gp_weights_vector, CurrentProcessInfo);
162  std::vector<double> stress_gp_vector;
163  elem_i.CalculateOnIntegrationPoints(KratosComponents<Variable<double>>::Get("VON_MISES_STRESS"), stress_gp_vector, CurrentProcessInfo);
164  double element_density = elem_i.GetProperties().GetValue(DENSITY);
165 
166  double elem_sens = 0.0;
167  for(IndexType i = 0; i < gp_weights_vector.size(); i++){
168  elem_sens += gp_weights_vector[i] * stress_gp_vector[i] * 2 * std::pow((element_density) * (1.0-element_density),1) * (1.0 - 2.0 * element_density);
169  }
170 
171  for (SizeType i_node = 0; i_node < number_of_nodes; ++i_node){
172  const auto& d_pd_d_fd = r_this_geometry[i_node].FastGetSolutionStepValue(D_PD_D_FD);
173  AtomicAdd(r_this_geometry[i_node].FastGetSolutionStepValue(D_STRESS_D_FD), d_pd_d_fd * elem_sens / number_of_nodes);
174  }
175  }
176  });
