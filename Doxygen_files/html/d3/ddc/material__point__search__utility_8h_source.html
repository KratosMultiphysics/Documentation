59  inline void ConstructNeighbourRelations(
60  GeometryType& rGeom,
61  const ModelPart& rBackgroundGridModelPart
62  )
63  {
64  std::vector<typename Geometry<Node>::Pointer> geometry_neighbours;
65  for (IndexType j = 0; j < rBackgroundGridModelPart.NumberOfElements(); j++)
66  {
67  auto p_geometry_neighbour = (rBackgroundGridModelPart.ElementsBegin() + j)->pGetGeometry();
68  if (p_geometry_neighbour->Id() != rGeom.Id()) // dont add the parent as its own neighbour
69  {
70  for (IndexType n = 0; n < p_geometry_neighbour->size(); n++)
71  {
72  for (IndexType k = 0; k < rGeom.size(); k++)
73  {
74  if (rGeom[k].Id() == (*p_geometry_neighbour)[n].Id()) {
75  // Prevent duplicate additions
76  bool add_entry = true;
77  for (size_t i = 0; i < geometry_neighbours.size(); i++)
78  {
79  if (geometry_neighbours[i]->Id() == p_geometry_neighbour->Id())
80  {
81  add_entry = false;
82  break;
83  }
84  }
85  if (add_entry) {
86  geometry_neighbours.push_back(p_geometry_neighbour);
87  }
88  break;
89  }
90  }
91  }
92  }
93  }
94  rGeom.SetValue(GEOMETRY_NEIGHBOURS, geometry_neighbours);
95  }

117  inline GeometryType& FindGridGeom(
118  GeometryType& rParentGeom,
119  const ModelPart& rBackgroundGridModelPart,
120  const double Tolerance,
121  const array_1d<double, 3>& xg,
122  array_1d<double, 3>& rLocalCoords,
123  const ProcessInfo& rProcessInfo,
124  bool& IsFound
125  )
126  {
127  IsFound = false;
128  if (CheckIsInside(rParentGeom, rLocalCoords, xg, Tolerance)) {
129  IsFound = true;
130  return rParentGeom;
131  } else {
132  // Lock for ParentGeom is required as reading and writing is called in parallel
133  // Lock is set on the node, since the lock option is not available for the geometry
134  rParentGeom(0)->SetLock();
135  if (!rParentGeom.Has(GEOMETRY_NEIGHBOURS)) {
136  ConstructNeighbourRelations(rParentGeom, rBackgroundGridModelPart);
137  }
138  rParentGeom(0)->UnSetLock();
139  auto& geometry_neighbours = rParentGeom.GetValue(GEOMETRY_NEIGHBOURS);
140  for (IndexType k = 0; k < geometry_neighbours.size(); ++k) {
141  if (CheckIsInside(*geometry_neighbours[k], rLocalCoords, xg, Tolerance)) {
142  IsFound = true;
143  return *(geometry_neighbours[k].get());
144  }
145  }
146  }
147  return rParentGeom;
148  }

274  template <std::size_t TDimension>
275  void BinBasedSearchElementsAndConditions(
276  ModelPart& rMPMModelPart,
277  ModelPart& rBackgroundGridModelPart,
278  std::vector<typename Element::Pointer>& rMissingElements,
279  std::vector<typename Condition::Pointer>& rMissingConditions,
280  const std::size_t MaxNumberOfResults,
281  const double Tolerance
282  )
283  {
284  const ProcessInfo& r_process_info = rBackgroundGridModelPart.GetProcessInfo();
285  bool is_pqmpm = (r_process_info.Has(IS_PQMPM))
286  ? r_process_info.GetValue(IS_PQMPM) : false;
287 
288  // Search background grid and make element active
289  Vector N;
290  const int max_result = 1000;
291 
292  #pragma omp parallel
293  {
294  BinBasedFastPointLocator<TDimension> SearchStructure(rBackgroundGridModelPart);
295  SearchStructure.UpdateSearchDatabase();
296  typename BinBasedFastPointLocator<TDimension>::ResultContainerType results(max_result);
297 
298  // Element search and assign background grid
299  #pragma omp for
300  for (int i = 0; i < static_cast<int>(rMissingElements.size()); ++i) {
301  auto element_itr = *(rMissingElements.begin() + i);
302  std::vector<array_1d<double, 3>> xg;
303  element_itr->CalculateOnIntegrationPoints(MP_COORD, xg, rMPMModelPart.GetProcessInfo());
304  typename BinBasedFastPointLocator<TDimension>::ResultIteratorType result_begin = results.begin();
305  Element::Pointer pelem;
306 
307  // FindPointOnMesh find the background element in which a given point falls and the relative shape functions
308  bool is_found = SearchStructure.FindPointOnMesh(xg[0], N, pelem, result_begin, MaxNumberOfResults, Tolerance);
309 
310  if (is_found) {
311  pelem->Set(ACTIVE);
312 
313  if (IsFixExplicitAndOnElementEdge(N, r_process_info) && !is_pqmpm) {
314  // MP is exactly on the edge. Now we give it a little 'nudge'
315  array_1d<double, 3> xg_nudged = array_1d<double, 3>(xg[0]);
316  std::vector<array_1d<double, 3>> mp_vel;
317  element_itr->CalculateOnIntegrationPoints(MP_VELOCITY, mp_vel, rMPMModelPart.GetProcessInfo());
318  xg_nudged += r_process_info[DELTA_TIME] / 1000.0 * mp_vel[0];
319  if (SearchStructure.FindPointOnMesh(xg_nudged, N, pelem, result_begin, MaxNumberOfResults, Tolerance)) {
320  element_itr->SetValuesOnIntegrationPoints(MP_COORD, { xg_nudged }, rMPMModelPart.GetProcessInfo());
321  KRATOS_INFO("MPMSearchElementUtility") << "WARNING: To prevent spurious explicit stresses, Material Point "
322  << element_itr->Id() << " was nudged." << std::endl;
323  } else {
324  is_found = SearchStructure.FindPointOnMesh(xg[0], N, pelem, result_begin, MaxNumberOfResults, Tolerance);
325  KRATOS_INFO("MPMSearchElementUtility") << "WARNING: Material Point " << element_itr->Id()
326  << " lies exactly on an element edge and may give spurious results." << std::endl;
327  }
328  }
329  if (is_pqmpm) {
330  // Updates the quadrature point geometry.
331  (*element_itr).GetGeometry().SetGeometryParent((pelem->pGetGeometry().get()));
332  UpdatePartitionedQuadraturePoint(rBackgroundGridModelPart, xg[0],
333  *element_itr, pelem->pGetGeometry(), Tolerance);
334  } else {
335  auto p_quadrature_point_geometry = element_itr->pGetGeometry();
336  array_1d<double, 3> local_coordinates;
337  pelem->pGetGeometry()->PointLocalCoordinates(local_coordinates, xg[0]);
338  CreateQuadraturePointsUtility<Node>::UpdateFromLocalCoordinates(
339  p_quadrature_point_geometry, local_coordinates,
340  p_quadrature_point_geometry->IntegrationPoints()[0].Weight(), pelem->GetGeometry());
341  }
342  auto& r_geometry = element_itr->GetGeometry();
343  for (IndexType j = 0; j < r_geometry.PointsNumber(); ++j) {
344  r_geometry[j].Set(ACTIVE);
345  }
346  } else {
347  KRATOS_INFO("MPMSearchElementUtility") << "WARNING: Search Element for Material Point: "
348  << element_itr->Id() << " is failed. Geometry is cleared." << std::endl;
349  element_itr->GetGeometry().clear();
350  element_itr->Reset(ACTIVE);
351  element_itr->Set(TO_ERASE);
352  }
353  }
354 
355  // Condition search and assign background grid
356  #pragma omp for
357  for (int i = 0; i < static_cast<int>(rMissingConditions.size()); ++i) {
358  auto condition_itr = *(rMissingConditions.begin() + i);
359  std::vector<array_1d<double, 3>> xg;
360  condition_itr->CalculateOnIntegrationPoints(MPC_COORD, xg, rMPMModelPart.GetProcessInfo());
361 
362  if (xg.size() > 0) {
363  // Only search for particle based BCs!
364  // Grid BCs are still applied on MP_model_part but we don't want to search for them.
365  typename BinBasedFastPointLocator<TDimension>::ResultIteratorType result_begin = results.begin();
366  Element::Pointer pelem;
367 
368  // FindPointOnMesh find the background element in which a given point falls and the relative shape functions
369  bool is_found = SearchStructure.FindPointOnMesh(xg[0], N, pelem, result_begin, MaxNumberOfResults, Tolerance);
370 
371  if (is_found) {
372  auto p_quadrature_point_geometry = condition_itr->pGetGeometry();
373  array_1d<double, 3> local_coordinates;
374  pelem->pGetGeometry()->PointLocalCoordinates(local_coordinates, xg[0]);
375  CreateQuadraturePointsUtility<Node>::UpdateFromLocalCoordinates(
376  p_quadrature_point_geometry, local_coordinates,
377  p_quadrature_point_geometry->IntegrationPoints()[0].Weight(), pelem->GetGeometry());
378 
379  auto& r_geometry = condition_itr->GetGeometry();
380 
381  for (IndexType j = 0; j < r_geometry.PointsNumber(); ++j) {
382  r_geometry[j].Set(ACTIVE);
383  }
384  } else {
385  KRATOS_INFO("MPMSearchElementUtility") << "WARNING: Search Element for Material Point Condition: " << condition_itr->Id()
386  << " is failed. Geometry is cleared." << std::endl;
387  condition_itr->GetGeometry().clear();
388  condition_itr->Reset(ACTIVE);
389  condition_itr->Set(TO_ERASE);
390  }
391  }
392  }
393  }
