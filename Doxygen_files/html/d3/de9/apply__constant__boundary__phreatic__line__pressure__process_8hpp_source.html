100  void ExecuteInitialize() override
101  {
102  KRATOS_TRY
103 
104  const Variable<double>& var = KratosComponents<Variable<double>>::Get(mVariableName);
105 
106  block_for_each(mrModelPart.Nodes(), [&var, this](Node& rNode) {
107  if (mIsFixed) rNode.Fix(var);
108  else if (mIsFixedProvided) rNode.Free(var);
109 
110  double horCoord = rNode.Coordinates()[mHorizontalDirection];
111  horCoord = std::max(horCoord, mMinHorizontalCoordinate);
112  horCoord = std::min(horCoord, mMaxHorizontalCoordinate);
113  const double height = mSlope * (horCoord - mFirstReferenceCoordinate[mHorizontalDirection]) +
114  mFirstReferenceCoordinate[mGravityDirection];
115  const double distance = height - rNode.Coordinates()[mGravityDirection];
116  const double pressure = mSpecificWeight * distance;
117  rNode.FastGetSolutionStepValue(var) = std::max(pressure, 0.0);
118  });
119 
120  KRATOS_CATCH("")
121  }
