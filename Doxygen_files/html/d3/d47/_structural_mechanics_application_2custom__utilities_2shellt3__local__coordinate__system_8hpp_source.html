68  {
69  // Form the basis vectors alligning the local X direction
70  // with the 1-2 side
71 
72  // compute the central point
73  noalias(mCenter) = P1global;
74  noalias(mCenter) += P2global;
75  noalias(mCenter) += P3global;
76  mCenter /= 3.0;
77 
78  // compute the local X direction parallel to the side 1-2
79  Vector3Type e1(P2global - P1global);
80 
81  // compute the temporary local Y direction
82  Vector3Type e2(P3global - P1global);
83 
84  // compute the Normal vector at the element center
85  // While normalizing the normal vector save its norm to compute the area.
86  Vector3Type e3;
87  MathUtils<RealType>::CrossProduct(e3, e1, e2);
88  mArea = NormalizeVector3(e3);
89  mArea /= 2.0;
90 
91  // finally compute the local Y direction to be orthogonal to both X and Z local directions
92  MathUtils<RealType>::CrossProduct(e2, e3, e1);
93 
94  // normalize the X and Y directions
95  NormalizeVector3(e1);
96  NormalizeVector3(e2);
97 
98  // form the 3x3 transformation matrix
99  for (int i = 0; i < 3; i++) {
100  mOrientation(0, i) = e1(i);
101  mOrientation(1, i) = e2(i);
102  mOrientation(2, i) = e3(i);
103  }
104 
105  // transform global coordinates to the local coordinate system
106  for (int i = 0; i < 3; i++) {
107  mP[0](i) = mOrientation(i, 0) * (P1global(0) - mCenter(0)) + mOrientation(i, 1) * (P1global(1) - mCenter(1)) + mOrientation(i, 2) * (P1global(2) - mCenter(2));
108  mP[1](i) = mOrientation(i, 0) * (P2global(0) - mCenter(0)) + mOrientation(i, 1) * (P2global(1) - mCenter(1)) + mOrientation(i, 2) * (P2global(2) - mCenter(2));
