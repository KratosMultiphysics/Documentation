337  virtual void ProjectROM(
338  ModelPart &rModelPart,
339  TSystemMatrixType &rA,
340  TSystemVectorType &rb,
341  TSystemMatrixType &rAComp
342  )
343  {
344  KRATOS_TRY

346  auto& r_root_mp = rModelPart.GetRootModelPart();
347 
348  if (mRightRomBasisInitialized==false){
349  BaseType::mPhiGlobal = ZeroMatrix(BaseBuilderAndSolverType::GetEquationSystemSize(), BaseType::GetNumberOfROMModes());
350  Vector& r_xRom = r_root_mp.GetValue(ROM_SOLUTION_BASE);
351  Vector& r_xBase = r_root_mp.GetValue(SOLUTION_BASE);
352  vector<Matrix>& r_svdPhiMatrices = r_root_mp.GetValue(SVD_PHI_MATRICES);
353  vector<Matrix>& r_nnLayers = r_root_mp.GetValue(NN_LAYERS);
354  Vector& r_refSnapshot = r_root_mp.GetValue(SOLUTION_REFERENCE);
355  RomNNUtility<TSparseSpace,TDenseSpace>::GetXAndDecoderGradient(r_xRom, r_xBase, BaseType::mPhiGlobal, r_svdPhiMatrices, r_nnLayers, r_refSnapshot);
356  mRightRomBasisInitialized = true;
357  }
358  else if (r_root_mp.GetValue(UPDATE_PHI_EFFECTIVE_BOOL)==true){
359  Vector& xromTotal = r_root_mp.GetValue(ROM_SOLUTION_TOTAL);
360  vector<Matrix>& r_svdPhiMatrices = r_root_mp.GetValue(SVD_PHI_MATRICES);
361  vector<Matrix>& r_nnLayers = r_root_mp.GetValue(NN_LAYERS);
362  RomNNUtility<TSparseSpace,TDenseSpace>::GetDecoderGradient(xromTotal, BaseType::mPhiGlobal, r_svdPhiMatrices, r_nnLayers);
363  }
364 
365  BaseType::SetFixedDofsROMBasisToZero();
366 
367  auto a_wrapper = UblasWrapper<double>(rA);
368  const auto& eigen_rA = a_wrapper.matrix();
369  Eigen::Map<EigenDynamicVector> eigen_rb(rb.data().begin(), rb.size());
370  Eigen::Map<EigenDynamicMatrix> eigen_mPhiGlobal(BaseType::mPhiGlobal.data().begin(), BaseType::mPhiGlobal.size1(), BaseType::mPhiGlobal.size2());
371 
372  EigenDynamicMatrix eigen_rA_times_mPhiGlobal = eigen_rA * eigen_mPhiGlobal; //TODO: Make it in parallel.
373 
374  if (BaseType::mHromSimulation) {
375  if (mSolvingTechnique != "normal_equations") {
376  KRATOS_ERROR << "AnnPromLeastSquaresPetrovGalerkinHROM is only valid for 'normal_equations'. The provided solving technique is '" << mSolvingTechnique << "'." << std::endl;
377  }
378 
379  auto a_comp_wrapper = UblasWrapper<double>(rAComp);
380  const auto& eigen_rAComp = a_comp_wrapper.matrix();
381 
382  EigenDynamicMatrix eigen_rAComp_times_mPhiGlobal = eigen_rAComp * eigen_mPhiGlobal; //TODO: Make it in parallel.
383 
384  if (mSolvingTechnique == "normal_equations") {
385  // Compute the matrix multiplication
386  mEigenRomA = eigen_rAComp_times_mPhiGlobal.transpose() * eigen_rA_times_mPhiGlobal; //TODO: Make it in parallel.
387  mEigenRomB = eigen_rAComp_times_mPhiGlobal.transpose() * eigen_rb; //TODO: Make it in parallel.
388  }
389  }
390  else {
391 
392  if (mSolvingTechnique == "normal_equations"){
393  // Compute the matrix multiplication
394  mEigenRomA = eigen_rA_times_mPhiGlobal.transpose() * eigen_rA_times_mPhiGlobal; //TODO: Make it in parallel.
395  mEigenRomB = eigen_rA_times_mPhiGlobal.transpose() * eigen_rb; //TODO: Make it in parallel.
396  }
397  else if (mSolvingTechnique == "qr_decomposition") {
398  mEigenRomA = eigen_rA_times_mPhiGlobal;
399  mEigenRomB = eigen_rb;

579  void BuildWithComplementaryMesh(
580  typename TSchemeType::Pointer pScheme,
581  ModelPart& rModelPart,
582  TSystemMatrixType& A,
583  TSystemVectorType& b)
584  {
585  KRATOS_TRY
586 
587  KRATOS_ERROR_IF(!pScheme) << "No scheme provided!" << std::endl;
588 
589  // // Getting the neighbouring (complementary mesh) and selected elements from the model
590  auto& r_neighboring_and_selected_elems = mNeighbouringAndSelectedElements;
591 
592  const int nelements = r_neighboring_and_selected_elems.size();
593 
594  // // Getting the neighbouring (complementary mesh) and selected conditions from the model
595  auto& r_neighboring_and_selected_conditions = mNeighbouringAndSelectedConditions;
596 
597  const int nconditions = r_neighboring_and_selected_conditions.size();
598 
599  const ProcessInfo& CurrentProcessInfo = rModelPart.GetProcessInfo();
600  ModelPart::ElementsContainerType::iterator el_begin = mNeighbouringAndSelectedElements.begin();
601  ModelPart::ConditionsContainerType::iterator cond_begin = mNeighbouringAndSelectedConditions.begin();
602 
603  //contributions to the system
604  LocalSystemMatrixType LHS_Contribution = LocalSystemMatrixType(0, 0);
605  LocalSystemVectorType RHS_Contribution = LocalSystemVectorType(0);
606 
607  //vector containing the localization in the system of the different terms
608  Element::EquationIdVectorType EquationId;
609 
610  // Assemble all elements
611  const auto timer = BuiltinTimer();
612 
613  #pragma omp parallel firstprivate(nelements,nconditions, LHS_Contribution, RHS_Contribution, EquationId)
614  {
615  # pragma omp for schedule(guided, 512) nowait
616  for (int k = 0; k < nelements; k++) {
617  auto it_elem = el_begin + k;
618 
619  if (it_elem->IsActive()) {
620  // Calculate elemental contribution
621  pScheme->CalculateSystemContributions(*it_elem, LHS_Contribution, RHS_Contribution, EquationId, CurrentProcessInfo);
622 
623  // Assemble lhs the elemental contribution
624  BaseType::Assemble(A, b, LHS_Contribution, RHS_Contribution, EquationId);
625  }
626 
627  }
628 
629  #pragma omp for schedule(guided, 512)
630  for (int k = 0; k < nconditions; k++) {
631  auto it_cond = cond_begin + k;
632 
633  if (it_cond->IsActive()) {
634  // Calculate elemental contribution
635  pScheme->CalculateSystemContributions(*it_cond, LHS_Contribution, RHS_Contribution, EquationId, CurrentProcessInfo);
636 
637  // Assemble lhs the elemental contribution
638  BaseType::Assemble(A, b, LHS_Contribution, RHS_Contribution, EquationId);
639  }
640  }
641  }
642 
643  KRATOS_INFO_IF("AnnPromLeastSquaresPetrovGalerkinROMResidualBasedBlockBuilderAndSolver", BaseType::GetEchoLevel() >= 1) << "Build time: " << timer.ElapsedSeconds() << std::endl;
644 
645  KRATOS_INFO_IF("AnnPromLeastSquaresPetrovGalerkinROMResidualBasedBlockBuilderAndSolver", (BaseType::GetEchoLevel() > 2 && rModelPart.GetCommunicator().MyPID() == 0)) << "Finished parallel building" << std::endl;

647  KRATOS_CATCH("")

650  void FindNeighbouringElementsAndConditions(ModelPart& rModelPart)
651  {
652  mNeighbouringAndSelectedElements.clear();
653  mNeighbouringAndSelectedConditions.clear();
654 
655  // Create sets for storing Ids
656  std::set<int> selected_element_ids;
657  std::set<int> selected_condition_ids;
658 
659  FindGlobalNodalEntityNeighboursProcess<ModelPart::ElementsContainerType> find_nodal_elements_neighbours_process(rModelPart);
660  find_nodal_elements_neighbours_process.Execute();
661  FindGlobalNodalEntityNeighboursProcess<ModelPart::ConditionsContainerType> find_nodal_conditions_neighbours_process(rModelPart);
662  find_nodal_conditions_neighbours_process.Execute();
663 
664  for (auto it_elem = BaseType::mSelectedElements.ptr_begin(); it_elem != BaseType::mSelectedElements.ptr_end(); ++it_elem) {
665  mNeighbouringAndSelectedElements.push_back(*it_elem);
666  selected_element_ids.insert((*it_elem)->Id());
667 
668  const auto& r_geom = (*it_elem)->GetGeometry();
669  const SizeType n_nodes = r_geom.PointsNumber();
670  for (IndexType i_node = 0; i_node < n_nodes; ++i_node) {
671  NodeType::Pointer p_node = r_geom(i_node);
672  auto& neighbour_elements = p_node->GetValue(NEIGHBOUR_ELEMENTS);
673  for (auto& neighbor_elem : neighbour_elements.GetContainer()) {
674  if(selected_element_ids.find(neighbor_elem->Id()) == selected_element_ids.end()) {
675  mNeighbouringAndSelectedElements.push_back(&*neighbor_elem);
676  selected_element_ids.insert(neighbor_elem->Id());
677  }
678  }
679 
680  // Check and add neighbouring conditions
681  auto& neighbour_conditions = p_node->GetValue(NEIGHBOUR_CONDITIONS);
