242  // Each component of rIntegrationPointData is a 4-component vector
243  // Components 0-2 are the GLOBAL coordinates of the integration point
244  // Component 3 is the integration point weight in terms of area (DetJ*Gauss_W)
245  void AddTriangleIntegrationPointData(
246  std::vector<Vector>& rIntegrationPointData,
247  const SplitGeometryType& rGeom,
248  const GeometryData::IntegrationMethod Method) const
249  {
250  const std::size_t num_nodes = rGeom.PointsNumber();
251  const auto& r_integration_points = rGeom.IntegrationPoints(Method);
252  const std::size_t num_gauss = r_integration_points.size();
253  std::size_t new_point_offset = rIntegrationPointData.size();
254 
255  Matrix shape_functions = rGeom.ShapeFunctionsValues(Method);
256  Vector det_j;
257  rGeom.DeterminantOfJacobian(det_j, Method);
258 
259  for (std::size_t g = 0; g < num_gauss; g++) {
260  Vector data = ZeroVector(4);
261  data[3] = r_integration_points[g].Weight() * det_j[g];
262  rIntegrationPointData.push_back(data);
263  }
264 
265  for (std::size_t n = 0; n < num_nodes; n++) {
266  const auto& r_coordinates = rGeom[n].Coordinates();
267  for (std::size_t g = 0; g < num_gauss; g++) {
268  for (std::size_t d = 0; d < 3; d++) {
269  rIntegrationPointData[new_point_offset+g][d] += shape_functions(g, n)*r_coordinates[d];
270  }
271  }
272  }
273  }
