153  // Check if the Orthogonal SubScales (OSS) are active
154  const auto& r_process_info = rModelPart.GetProcessInfo();
155  const bool oss_switch = r_process_info.Has(OSS_SWITCH) ? r_process_info[OSS_SWITCH] : false;
156 
157  // Calculate the OSS projections
158  if (oss_switch) {
159  // Initialize the projection values
160  const array_1d<double,3> aux_zero = ZeroVector(3);
161  block_for_each(rModelPart.Nodes(), aux_zero, [this](Node& rNode, array_1d<double,3>& rAuxZero){
162  for (auto& rp_var : mProjectionScalarVariablesList) {
163  rNode.FastGetSolutionStepValue(*rp_var) = 0.0;
164  }
165  for (auto& rp_var : mProjectionVectorVariablesList) {
166  rNode.FastGetSolutionStepValue(*rp_var) = rAuxZero;
167  }
168  });
169 
170  // Calculate the element residuals projection
171  std::tuple<double, array_1d<double,3>> oss_proj_tls;
172  block_for_each(rModelPart.Elements(), oss_proj_tls, [&, this](Element& rElement, std::tuple<double, array_1d<double,3>>& rOssProjTLS){
173  double& r_scalar_proj = std::get<0>(rOssProjTLS);
174  array_1d<double,3>& r_vector_proj = std::get<1>(rOssProjTLS);
175  for (auto& rp_var : mProjectionScalarVariablesList) {
176  rElement.Calculate(*rp_var, r_scalar_proj, r_process_info);
177  }
178  for (auto& rp_var : mProjectionVectorVariablesList) {
179  rElement.Calculate(*rp_var, r_vector_proj, r_process_info);
180  }
181  });
182 
183  // Do the nodal weighting
184  //TODO: We the weighted L2 projection with the density should be done in the multimaterial case
185  //TODO: Note that this would require customizing the weights calculation in here (standard NODAL_AREA cannot be used)
186  block_for_each(rModelPart.Nodes(), [this](Node& rNode){
187  const double nodal_area = rNode.GetValue(NODAL_AREA);
188  for (auto& rp_var : mProjectionScalarVariablesList) {
189  rNode.FastGetSolutionStepValue(*rp_var) /= nodal_area;
190  }
191  for (auto& rp_var : mProjectionVectorVariablesList) {
