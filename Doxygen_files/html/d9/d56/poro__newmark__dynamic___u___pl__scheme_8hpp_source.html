117  void Predict(
118  ModelPart& r_model_part,
119  DofsArrayType& rDofSet,
120  TSystemMatrixType& A,
121  TSystemVectorType& Dx,
122  TSystemVectorType& b) override
123  {
124  // Predict Displacements on free nodes and update Acceleration, Velocity and DtPressure
125 
126  array_1d<double,3> DeltaDisplacement;
127  double DeltaPressure;
128 
129  const int NNodes = static_cast<int>(r_model_part.Nodes().size());
130  ModelPart::NodesContainerType::iterator node_begin = r_model_part.NodesBegin();
131 
132  #pragma omp parallel for private(DeltaDisplacement,DeltaPressure)
133  for(int i = 0; i < NNodes; i++)
134  {
135  ModelPart::NodesContainerType::iterator itNode = node_begin + i;
136 
137  array_1d<double,3>& CurrentDisplacement = itNode->FastGetSolutionStepValue(DISPLACEMENT);
138  array_1d<double,3>& CurrentAcceleration = itNode->FastGetSolutionStepValue(ACCELERATION);
139  array_1d<double,3>& CurrentVelocity = itNode->FastGetSolutionStepValue(VELOCITY);
140 
141  const array_1d<double,3>& PreviousDisplacement = itNode->FastGetSolutionStepValue(DISPLACEMENT, 1);
142  const array_1d<double,3>& PreviousAcceleration = itNode->FastGetSolutionStepValue(ACCELERATION, 1);
143  const array_1d<double,3>& PreviousVelocity = itNode->FastGetSolutionStepValue(VELOCITY, 1);
144 
145  if (itNode -> IsFixed(ACCELERATION_X))
146  {
147  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime * PreviousVelocity[0] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[0] + mBeta * CurrentAcceleration[0] );
148  }
149  else if (itNode -> IsFixed(VELOCITY_X))
150  {
151  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[0]-PreviousVelocity[0])+PreviousVelocity[0]);
152  }
153  else if (itNode -> IsFixed(DISPLACEMENT_X) == false)
154  {
155  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime * PreviousVelocity[0] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[0];
156  }
157 
158  if (itNode -> IsFixed(ACCELERATION_Y))
159  {
160  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime * PreviousVelocity[1] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[1] + mBeta * CurrentAcceleration[1] );
161  }
162  else if (itNode -> IsFixed(VELOCITY_Y))
163  {
164  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[1]-PreviousVelocity[1])+PreviousVelocity[1]);
165  }
166  else if (itNode -> IsFixed(DISPLACEMENT_Y) == false)
167  {
168  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime * PreviousVelocity[1] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[1];
169  }
170 
171  // For 3D cases
172  if (itNode -> HasDofFor(DISPLACEMENT_Z))
173  {
174  if (itNode -> IsFixed(ACCELERATION_Z))
175  {
176  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime * PreviousVelocity[2] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[2] + mBeta * CurrentAcceleration[2] );
177  }
178  else if (itNode -> IsFixed(VELOCITY_Z))
179  {
180  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[2]-PreviousVelocity[2])+PreviousVelocity[2]);
181  }
182  else if (itNode -> IsFixed(DISPLACEMENT_Z) == false)
183  {
184  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime * PreviousVelocity[2] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[2];
185  }
186  }
187 
188  noalias(DeltaDisplacement) = CurrentDisplacement - PreviousDisplacement;
189 
190  noalias(CurrentAcceleration) = 1.0/(mBeta*mDeltaTime*mDeltaTime)*(DeltaDisplacement - mDeltaTime*PreviousVelocity - (0.5-mBeta)*mDeltaTime*mDeltaTime*PreviousAcceleration);
191  noalias(CurrentVelocity) = PreviousVelocity + (1.0-mGamma)*mDeltaTime*PreviousAcceleration + mGamma*mDeltaTime*CurrentAcceleration;
192 
193  double& CurrentDtPressure = itNode->FastGetSolutionStepValue(DT_LIQUID_PRESSURE);
194  DeltaPressure = itNode->FastGetSolutionStepValue(LIQUID_PRESSURE) - itNode->FastGetSolutionStepValue(LIQUID_PRESSURE, 1);
195  const double& PreviousDtPressure = itNode->FastGetSolutionStepValue(DT_LIQUID_PRESSURE, 1);
196 
197  CurrentDtPressure = 1.0/(mTheta_p*mDeltaTime)*(DeltaPressure - (1.0-mTheta_p)*mDeltaTime*PreviousDtPressure);
198  }
199  }

336  void AddDynamicsToLHS(LocalSystemMatrixType& LHS_Contribution,LocalSystemMatrixType& M,LocalSystemMatrixType& C,const ProcessInfo& rCurrentProcessInfo)
337  {
338  // adding mass contribution
339  if (M.size1() != 0)
340  noalias(LHS_Contribution) += 1.0/(mBeta*mDeltaTime*mDeltaTime)*M;
341 
342  // adding damping contribution
343  if (C.size1() != 0)
344  noalias(LHS_Contribution) += mGamma/(mBeta*mDeltaTime)*C;
345  }

349  void AddDynamicsToRHS(Element& rCurrentElement,LocalSystemVectorType& RHS_Contribution,
350  LocalSystemMatrixType& M,LocalSystemMatrixType& C,const ProcessInfo& rCurrentProcessInfo)
351  {
352  int thread = OpenMPUtils::ThisThread();
353 
354  //adding inertia contribution
355  if (M.size1() != 0)
356  {
357  rCurrentElement.GetSecondDerivativesVector(mAccelerationVector[thread], 0);
358 
359  noalias(RHS_Contribution) -= prod(M, mAccelerationVector[thread]);
360  }
361 
362  //adding damping contribution
363  if (C.size1() != 0)
364  {
365  rCurrentElement.GetFirstDerivativesVector(mVelocityVector[thread], 0);
366 
367  noalias(RHS_Contribution) -= prod(C, mVelocityVector[thread]);
368  }
