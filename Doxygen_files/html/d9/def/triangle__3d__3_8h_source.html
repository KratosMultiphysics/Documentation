937  CoordinatesArrayType& PointLocalCoordinates(
938  CoordinatesArrayType& rResult,
939  const CoordinatesArrayType& rPoint
940  ) const override
941  {
942  // Initialize
943  noalias(rResult) = ZeroVector(3);
944 
945  // Tangent vectors
946  array_1d<double, 3> tangent_xi = this->GetPoint(1) - this->GetPoint(0);
947  tangent_xi /= norm_2(tangent_xi);
948  array_1d<double, 3> tangent_eta = this->GetPoint(2) - this->GetPoint(0);
949  tangent_eta /= norm_2(tangent_eta);
950 
951  // The center of the geometry
952  const auto center = this->Center();
953 
954  // Computation of the rotation matrix
955  BoundedMatrix<double, 3, 3> rotation_matrix = ZeroMatrix(3, 3);
956  for (IndexType i = 0; i < 3; ++i) {
957  rotation_matrix(0, i) = tangent_xi[i];
958  rotation_matrix(1, i) = tangent_eta[i];
959  }
960 
961  // Destination point rotated
962  CoordinatesArrayType aux_point_to_rotate, destination_point_rotated;
963  noalias(aux_point_to_rotate) = rPoint - center.Coordinates();
964  noalias(destination_point_rotated) = prod(rotation_matrix, aux_point_to_rotate) + center.Coordinates();
965 
966  // Points of the geometry
967  array_1d<CoordinatesArrayType, 3> points_rotated;
968  for (IndexType i = 0; i < 3; ++i) {
969  noalias(aux_point_to_rotate) = this->GetPoint(i).Coordinates() - center.Coordinates();
970  noalias(points_rotated[i]) = prod(rotation_matrix, aux_point_to_rotate) + center.Coordinates();
971  }
972 
973  // Compute the Jacobian matrix and its determinant
974  BoundedMatrix<double, 2, 2> J;
975  J(0,0) = points_rotated[1][0] - points_rotated[0][0];
976  J(0,1) = points_rotated[2][0] - points_rotated[0][0];
977  J(1,0) = points_rotated[1][1] - points_rotated[0][1];
978  J(1,1) = points_rotated[2][1] - points_rotated[0][1];
979  const double det_J = J(0,0)*J(1,1) - J(0,1)*J(1,0);
980 
981  // Compute eta and xi
982  const double eta = (J(1,0)*(points_rotated[0][0] - destination_point_rotated[0]) +
983  J(0,0)*(destination_point_rotated[1] - points_rotated[0][1])) / det_J;
984  const double xi = (J(1,1)*(destination_point_rotated[0] - points_rotated[0][0]) +
985  J(0,1)*(points_rotated[0][1] - destination_point_rotated[1])) / det_J;
986 
987  rResult(0) = xi;
988  rResult(1) = eta;
989 
990  return rResult;
991  }
