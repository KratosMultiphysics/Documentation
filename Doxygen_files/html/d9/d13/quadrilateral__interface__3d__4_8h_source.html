518  CoordinatesArrayType& PointLocalCoordinates(
519  CoordinatesArrayType& rResult,
520  const CoordinatesArrayType& rPoint
521  ) const override
522  {
523  BoundedMatrix<double,3,4> X;
524  BoundedMatrix<double,3,2> DN;
525  for(unsigned int i=0; i<this->size();i++) {
526  X(0,i ) = this->GetPoint( i ).X();
527  X(1,i ) = this->GetPoint( i ).Y();
528  X(2,i ) = this->GetPoint( i ).Z();
529  }
530 
531  double tol = 1.0e-8;
532  int maxiter = 1000;
533 
534  Matrix J = ZeroMatrix( 2, 2 );
535  Matrix invJ = ZeroMatrix( 2, 2 );
536 
537  //starting with xi = 0
538  rResult = ZeroVector( 3 );
539  Vector DeltaXi = ZeroVector( 2 );
540  array_1d<double,3> CurrentGlobalCoords;
541 
542 
543  //Newton iteration:
544  for ( int k = 0; k < maxiter; k++ )
545  {
546  noalias(CurrentGlobalCoords) = ZeroVector( 3 );
547  this->GlobalCoordinates( CurrentGlobalCoords, rResult );
548 
549  noalias( CurrentGlobalCoords ) = rPoint - CurrentGlobalCoords;
550 
551 
552  //derivatives of shape functions
553  Matrix shape_functions_gradients;
554  shape_functions_gradients = ShapeFunctionsLocalGradients(shape_functions_gradients, rResult );
555  noalias(DN) = prod(X,shape_functions_gradients);
556 
557  noalias(J) = prod(trans(DN),DN);
558  Vector res = prod(trans(DN),CurrentGlobalCoords);
559 
560  //deteminant of Jacobian
561  const double det_j = J( 0, 0 ) * J( 1, 1 ) - J( 0, 1 ) * J( 1, 0 );
562 
563  //filling matrix
564  invJ( 0, 0 ) = ( J( 1, 1 ) ) / ( det_j );
565  invJ( 1, 0 ) = -( J( 1, 0 ) ) / ( det_j );
566  invJ( 0, 1 ) = -( J( 0, 1 ) ) / ( det_j );
567  invJ( 1, 1 ) = ( J( 0, 0 ) ) / ( det_j );
568 
569 
570  DeltaXi( 0 ) = invJ( 0, 0 ) * res[0] + invJ( 0, 1 ) * res[1];
571  DeltaXi( 1 ) = invJ( 1, 0 ) * res[0] + invJ( 1, 1 ) * res[1];
572 
573  rResult[0] += DeltaXi[0];
574  rResult[1] += DeltaXi[1];
575  rResult[2] = 0.0;
576 
577  if ( norm_2( DeltaXi ) > 300 )
578  {
579  res[0] = 0.0;
580  res[1] = 0.0;
581  std::cout << "detJ =" << det_j << "DeltaX = " << DeltaXi << " stopping calculation and assigning the baricenter" << std::endl;
582  break;
583  //KRATOS_ERROR << "Computation of local coordinates failed at iteration" << k << std::endl;
584  }
585 
586  if ( norm_2( DeltaXi ) < tol )
587  {
588  break;
589  }
590  }
591 
592  return( rResult );
593  }

614  IntegrationMethod ThisMethod,
615  Matrix & DeltaPosition ) const override
616  {
617  array_1d<double, 3> p0 = 0.5 * (BaseType::GetPoint( 0 ) + BaseType::GetPoint( 3 ));
618  array_1d<double, 3> p1 = 0.5 * (BaseType::GetPoint( 1 ) + BaseType::GetPoint( 2 ));
619 
620  array_1d<double, 3> dp0;
621  array_1d<double, 3> dp1;
622  dp0[0] = (DeltaPosition(0,0) + DeltaPosition(3,0))*0.5;
623  dp0[1] = (DeltaPosition(0,1) + DeltaPosition(3,1))*0.5;
624  dp0[2] = (DeltaPosition(0,2) + DeltaPosition(3,2))*0.5;
625  dp1[0] = (DeltaPosition(1,0) + DeltaPosition(2,0))*0.5;
626  dp1[1] = (DeltaPosition(1,1) + DeltaPosition(2,1))*0.5;
627  dp1[2] = (DeltaPosition(1,2) + DeltaPosition(2,2))*0.5;
628 
629  Matrix jacobian( 3, 1 );
630  jacobian( 0, 0 ) = ( p1[0]-dp1[0] - (p0[0]-dp0[0]) ) * 0.5; //on the Gauss points (J is constant at each element)
631  jacobian( 1, 0 ) = ( p1[1]-dp1[1] - (p0[1]-dp0[1]) ) * 0.5;
632  jacobian( 2, 0 ) = ( p1[2]-dp1[2] - (p0[2]-dp0[2]) ) * 0.5;
633 
634  if ( rResult.size() != BaseType::IntegrationPointsNumber( ThisMethod ) )
635  {
636  // KLUDGE: While there is a bug in ublas vector resize, I have to put this beside resizing!!
637  JacobiansType temp( BaseType::IntegrationPointsNumber( ThisMethod ) );
638  rResult.swap( temp );
639  }
640 
641  std::fill( rResult.begin(), rResult.end(), jacobian );
642 
643  return rResult;
644  }
