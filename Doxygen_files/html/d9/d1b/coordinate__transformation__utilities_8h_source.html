138  void LocalRotationOperatorPure(
139  BoundedMatrix<double, 3, 3>& rRot,
140  const GeometryType::PointType& rThisPoint) const
141  {
142  // Get the normal evaluated at the node
143  const array_1d<double, 3>& rNormal = rThisPoint.FastGetSolutionStepValue(NORMAL);
144 
145  double aux = rNormal[0] * rNormal[0] + rNormal[1] * rNormal[1] +
146  rNormal[2] * rNormal[2];
147  aux = sqrt(aux);
148  rRot(0, 0) = rNormal[0] / aux;
149  rRot(0, 1) = rNormal[1] / aux;
150  rRot(0, 2) = rNormal[2] / aux;
151  // Define the new coordinate system, where the first vector is aligned with the normal
152 
153  // To choose the remaining two vectors, we project the first component of the cartesian base to the tangent plane
154  array_1d<double, 3> rT1;
155  rT1(0) = 1.0;
156  rT1(1) = 0.0;
157  rT1(2) = 0.0;
158  double dot = rRot(0, 0); // this->Dot(rN,rT1);
159 
160  // It is possible that the normal is aligned with (1,0,0), resulting in
161  // norm(rT1) = 0 If this is the case, repeat the procedure using (0,1,0)
162  if (fabs(dot) > 0.99) {
163  rT1(0) = 0.0;
164  rT1(1) = 1.0;
165  rT1(2) = 0.0;
166 
167  dot = rRot(0, 1); // this->Dot(rN,rT1);
168  }
169 
170  // calculate projection and normalize
171  rT1[0] -= dot * rRot(0, 0);
172  rT1[1] -= dot * rRot(0, 1);
173  rT1[2] -= dot * rRot(0, 2);
174  Normalize(rT1);
175  rRot(1, 0) = rT1[0];
176  rRot(1, 1) = rT1[1];
177  rRot(1, 2) = rT1[2];
178 
179  // The third base component is choosen as N x T1, which is normalized by construction
180  rRot(2, 0) = rRot(0, 1) * rT1[2] - rRot(0, 2) * rT1[1];
181  rRot(2, 1) = rRot(0, 2) * rT1[0] - rRot(0, 0) * rT1[2];
182  rRot(2, 2) = rRot(0, 0) * rT1[1] - rRot(0, 1) * rT1[0];
183  }

377  << rThisPoint.Coordinates() << " to calculate 3D rotation operator shape sensitivities. [ required number of rows >= "
378  << require_rows
379  << ", available number of rows = " << r_sensitivity_values.size1() << " ].";

382  row(r_sensitivity_values, DerivativeNodeIndex * 3 + DerivativeDirectionIndex);
383 
384  const double nodal_normal_magnitude_derivative = VectorNormDerivative(nodal_normal_magnitude, r_nodal_normal, r_nodal_normal_derivative);
385  const array_1d<double, 3>& unit_normal = r_nodal_normal / nodal_normal_magnitude;
386  const array_1d<double, 3>& unit_normal_derivative = UnitVectorDerivative(nodal_normal_magnitude, nodal_normal_magnitude_derivative, r_nodal_normal, r_nodal_normal_derivative);
387

389  rOutput(0, 1) = unit_normal_derivative[1];
390  rOutput(0, 2) = unit_normal_derivative[2];

392  array_1d<double, 3> rT1(3, 0.0);
393  rT1[0] = 1.0;

395  double dot_derivative = unit_normal_derivative[0];
396 
397  if (std::abs(dot) > 0.99) {
398  rT1[0] = 0.0;

498  TLocalVectorType& rLocalVector,

500  {
501  const unsigned int LocalSize = rLocalVector.size(); // We expect this to work both with elements (4 nodes) and conditions (3 nodes)
502 
503  if (LocalSize > 0)
504  {
505  for(unsigned int itNode = 0; itNode < rGeometry.PointsNumber(); ++itNode)
506  {
507  if( this->IsSlip(rGeometry[itNode]))
508  {
509  // We fix the first dof (normal velocity) for each rotated block
510  unsigned int j = itNode * mBlockSize;
511  //const double k = rLocalMatrix(j,j)+rLocalMatrix(j,j+1)+rLocalMatrix(j,j+2);
512 
513  // If the mesh is moving, we must impose v_normal = vmesh_normal
514  array_1d<double,3> VMesh = rGeometry[itNode].FastGetSolutionStepValue(MESH_VELOCITY);
515  VMesh -= rGeometry[itNode].FastGetSolutionStepValue(VELOCITY);
516  array_1d<double,3> rN = rGeometry[itNode].FastGetSolutionStepValue(NORMAL);
517  this->Normalize(rN);
518 
519  for( unsigned int i = 0; i < j; ++i)// Skip term (i,i)
520  {
521  rLocalMatrix(i,j) = 0.0;
522  rLocalMatrix(j,i) = 0.0;
523  }
524  for( unsigned int i = j+1; i < LocalSize; ++i)
525  {
526  rLocalMatrix(i,j) = 0.0;
527  rLocalMatrix(j,i) = 0.0;
528  }
529 
530  rLocalVector(j) = inner_prod(rN,VMesh);
531  rLocalMatrix(j,j) = 1.0;
532  }

702  template<unsigned int TDim, unsigned int TBlockSize, unsigned int TSkip = 0, bool TRevertRotation = false>
703  void RotateAux(TLocalMatrixType& rLocalMatrix,
704  TLocalVectorType& rLocalVector,
705  GeometryType& rGeometry) const
706  {

708

712  DenseVector<bool> NeedRotation( NumBlocks, false);
713

716

887  template<unsigned int TBlockSize, unsigned int TSkip = 0>
888  void LocalRotationOperator2D(
889  BoundedMatrix<double,TBlockSize,TBlockSize>& rRot,
890  GeometryType::PointType& rThisPoint) const
891  {
892  noalias(rRot) = IdentityMatrix(TBlockSize);
893 
894  // Get the normal evaluated at the node
895  const array_1d<double,3>& rNormal = rThisPoint.FastGetSolutionStepValue(NORMAL);
896 
897  double aux = rNormal[0]*rNormal[0] + rNormal[1]*rNormal[1];
898  aux = sqrt(aux);
899 
900  rRot(TSkip ,TSkip ) = rNormal[0]/aux;
901  rRot(TSkip ,TSkip+1) = rNormal[1]/aux;
902  rRot(TSkip+1,TSkip ) = -rNormal[1]/aux;
903  rRot(TSkip+1,TSkip+1) = rNormal[0]/aux;
904  }

906  template<unsigned int TBlockSize, unsigned int TSkip = 0>
907  void LocalRotationOperator3D(
908  BoundedMatrix<double,TBlockSize,TBlockSize>& rRot,
909  GeometryType::PointType& rThisPoint) const
910  {
911  noalias(rRot) = IdentityMatrix(TBlockSize);
912 
913  // Get the normal evaluated at the node
914  const array_1d<double,3>& rNormal = rThisPoint.FastGetSolutionStepValue(NORMAL);
915 
916  double aux = rNormal[0]*rNormal[0] + rNormal[1]*rNormal[1] + rNormal[2]*rNormal[2];
917  aux = sqrt(aux);
918  rRot(TSkip,TSkip ) = rNormal[0]/aux;
919  rRot(TSkip,TSkip+1) = rNormal[1]/aux;
920  rRot(TSkip,TSkip+2) = rNormal[2]/aux;
921  // Define the new coordinate system, where the first vector is aligned with the normal
922 
923  // To choose the remaining two vectors, we project the first component of the cartesian base to the tangent plane
924  array_1d<double,3> rT1;
925  rT1(0) = 1.0;
926  rT1(1) = 0.0;
927  rT1(2) = 0.0;
928  double dot = rRot(TSkip,TSkip);//this->Dot(rN,rT1);
929 
930  // It is possible that the normal is aligned with (1,0,0), resulting in norm(rT1) = 0
931  // If this is the case, repeat the procedure using (0,1,0)
932  if ( fabs(dot) > 0.99 )
933  {
934  rT1(0) = 0.0;
935  rT1(1) = 1.0;
936  rT1(2) = 0.0;
937 
938  dot = rRot(TSkip,TSkip+1); //this->Dot(rN,rT1);
939  }
940 
941  // calculate projection and normalize
942  rT1[0] -= dot*rRot(TSkip,TSkip);
943  rT1[1] -= dot*rRot(TSkip,TSkip+1);
944  rT1[2] -= dot*rRot(TSkip,TSkip+2);
945  this->Normalize(rT1);
946  rRot(TSkip+1,TSkip ) = rT1[0];
947  rRot(TSkip+1,TSkip+1) = rT1[1];
948  rRot(TSkip+1,TSkip+2) = rT1[2];
949 
950  // The third base component is choosen as N x T1, which is normalized by construction
951  rRot(TSkip+2,TSkip ) = rRot(TSkip,TSkip+1)*rT1[2] - rRot(TSkip,TSkip+2)*rT1[1];
952  rRot(TSkip+2,TSkip+1) = rRot(TSkip,TSkip+2)*rT1[0] - rRot(TSkip,TSkip )*rT1[2];
953  rRot(TSkip+2,TSkip+2) = rRot(TSkip,TSkip )*rT1[1] - rRot(TSkip,TSkip+1)*rT1[0];
954  }
