168  void BlockForEachActiveElement(ModelPart& rModelPart, MemFuncPtr pMemberFunction)
169  {
170  const auto& r_current_process_info = rModelPart.GetProcessInfo();
171  block_for_each(rModelPart.Elements(), [&r_current_process_info, pMemberFunction](auto& rElement) {

330  void CheckAllocatedVariables(const ModelPart& rModelPart) const
331  {
332  for (const auto& r_node : rModelPart.Nodes()) {
333  for (const auto& r_first_order_scalar_variable : mFirstOrderScalarVariables) {
334  this->CheckSolutionStepsData(r_node, r_first_order_scalar_variable.instance);
335  this->CheckSolutionStepsData(r_node, r_first_order_scalar_variable.first_time_derivative);
336  this->CheckDof(r_node, r_first_order_scalar_variable.instance);
337  }
338  }
339 
340  for (const auto& r_node : rModelPart.Nodes()) {
341  for (const auto& r_second_order_vector_variable : this->mSecondOrderVectorVariables) {
342  if (!rModelPart.HasNodalSolutionStepVariable(r_second_order_vector_variable.instance))
343  continue;
344 
345  this->CheckSolutionStepsData(r_node, r_second_order_vector_variable.instance);
346  this->CheckSolutionStepsData(r_node, r_second_order_vector_variable.first_time_derivative);
347  this->CheckSolutionStepsData(r_node, r_second_order_vector_variable.second_time_derivative);
348 
349  // We don't check for "Z", since it is optional (in case of a 2D problem)
350  std::vector<std::string> components{"X", "Y"};
351  for (const auto& component : components) {
352  const auto& variable_component = VariablesUtilities::GetComponentFromVectorVariable(
353  r_second_order_vector_variable.instance.Name(), component);
354  this->CheckDof(r_node, variable_component);
355  }
356  }
357  }
358  }
