162  static void SynchronizePoints(

164  TPointIteratorType itPointEnd,

299  template<typename TPointIteratorType>
300  static bool CheckAllPointsAreTheSameAndCalculateNumberOfPoints(
301  TPointIteratorType itPointBegin,
302  TPointIteratorType itPointEnd,
303  int& rNumberOfPoints,
304  int& rTotalNumberOfPoints,
305  const DataCommunicator& rDataCommunicator
306  )
307  {
308  // Get the World Size in MPI
309  const int world_size = rDataCommunicator.Size();
310 
311  // Getting local number of points
312  rNumberOfPoints = std::distance(itPointBegin, itPointEnd);
313 
314  // First we check if all the partitions have the same number of points
315  rTotalNumberOfPoints = rDataCommunicator.SumAll(rNumberOfPoints);
316  if (rNumberOfPoints * world_size != rTotalNumberOfPoints) {
317  return false;
318  }
319 
320  // Now we check if all the points are the same (we are going to do a gross assumtion that the sum of coordinates in al partitions does not compensate between them)
321  double x_sum, y_sum, z_sum;
322  array_1d<double, 3> coordinates;
323  bool all_points_are_the_same = true;
324  for (auto it_point = itPointBegin ; it_point != itPointEnd ; it_point++) {
325  noalias(coordinates) = it_point->Coordinates();
326  x_sum = rDataCommunicator.SumAll(coordinates[0]);
327  if (std::abs(coordinates[0] - (x_sum/world_size)) > ZeroTolerance) all_points_are_the_same = false;
328  all_points_are_the_same = rDataCommunicator.AndReduceAll(all_points_are_the_same);
329  if (!all_points_are_the_same) break;
330  y_sum = rDataCommunicator.SumAll(coordinates[1]);
331  if (std::abs(coordinates[1] - (y_sum/world_size)) > ZeroTolerance) all_points_are_the_same = false;
332  all_points_are_the_same = rDataCommunicator.AndReduceAll(all_points_are_the_same);
333  if (!all_points_are_the_same) break;
334  z_sum = rDataCommunicator.SumAll(coordinates[2]);
335  if (std::abs(coordinates[2] - (z_sum/world_size)) > ZeroTolerance) all_points_are_the_same = false;
336  all_points_are_the_same = rDataCommunicator.AndReduceAll(all_points_are_the_same);
337  if (!all_points_are_the_same) break;
338  }
339 
340  return all_points_are_the_same;
341  }
