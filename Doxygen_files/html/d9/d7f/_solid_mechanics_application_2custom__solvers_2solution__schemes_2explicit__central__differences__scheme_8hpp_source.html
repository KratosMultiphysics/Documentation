478  void CalculateDeltaTime(ModelPart& rModelPart)
479  {
480 
481  KRATOS_TRY
482 
483  ProcessInfo& rCurrentProcessInfo= rModelPart.GetProcessInfo();
484 
485  const unsigned int NumThreads = ParallelUtilities::GetNumThreads();
486 
487  //most autors recommend a value near 0.80 (Belytschko - Nonlinear FE.. 2000. chap 6. pag. 315)
488  double safety_factor = 0.5;
489  double stable_delta_time = 0.00;
490  Vector delta_times(NumThreads);
491 
492  for(unsigned int i = 0; i < NumThreads; i++)
493  delta_times[i] = mDeltaTime.Maximum/safety_factor;
494 
495  OpenMPUtils::PartitionVector ElementPartition;
496  OpenMPUtils::DivideInPartitions(rModelPart.Elements().size(), NumThreads, ElementPartition);
497 
498  const int nelements = static_cast<int>(rModelPart.Elements().size());
499  ElementsContainerType::iterator ElementBegin = rModelPart.Elements().begin();
500 
501  #pragma omp parallel for firstprivate(ElementBegin) private(stable_delta_time)
502  for(int i = 0; i < nelements; i++)
503  {
504  ElementsContainerType::iterator itElement = ElementBegin + i;
505  //get geometric and material properties
506  double length = (itElement)->GetGeometry().Length();
507  double alpha = (itElement)->GetProperties()[RAYLEIGH_ALPHA];
508  double beta = (itElement)->GetProperties()[RAYLEIGH_BETA];
509  double E = (itElement)->GetProperties()[YOUNG_MODULUS];
510  double v = (itElement)->GetProperties()[POISSON_RATIO];
511  double ro = (itElement)->GetProperties()[DENSITY];
512 
513  //compute courant criterion
514  double bulk = E/(3.0*(1.0-2.0*v));
515  double wavespeed = sqrt(bulk/ro);
516  double w = 2.0*wavespeed/length; //frequency
517 
518  double psi = 0.5*(alpha/w + beta*w); //critical ratio;
519  stable_delta_time = (2.0/w)*(sqrt(1.0 + psi*psi)-psi);
520 
521  if(stable_delta_time > 0.00)
522  {
523  int thread = OpenMPUtils::ThisThread();
524  if(stable_delta_time < delta_times[thread])
525  {
526  delta_times[thread] = stable_delta_time;
527  }
528  }
529 
530  }
531 
532  stable_delta_time = (*std::min_element(delta_times.begin(), delta_times.end()));
533  stable_delta_time *= safety_factor;// * 0.5; //extra factor added to get an stable delta time
534 
535  double current_delta_time = rCurrentProcessInfo[DELTA_TIME];
536 
537  if(stable_delta_time < mDeltaTime.Maximum){
538  rCurrentProcessInfo[DELTA_TIME] = stable_delta_time;
539  }
540  else{
541  if( current_delta_time > mDeltaTime.Maximum/safety_factor )
542  rCurrentProcessInfo[DELTA_TIME] = mDeltaTime.Maximum;
543  }
544 
545  KRATOS_INFO("Time Step prediction") << " [EXPLICIT PREDICTION LEVEL"<<mDeltaTime.PredictionLevel<<"]:(computed stable time step = "<< stable_delta_time <<" s)"<< std::endl;
546  KRATOS_INFO("Using") << rCurrentProcessInfo[DELTA_TIME] <<" s as time step DELTA_TIME)"<< std::endl;
547 
548  KRATOS_CATCH("")
549  }
