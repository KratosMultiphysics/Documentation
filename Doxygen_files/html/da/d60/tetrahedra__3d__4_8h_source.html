572  double Circumradius() const override {
573  // | s10 y10 z10 | | s10 x10 z10 | | s10 x10 y10 | | x10 y10 z10 |
574  // MX | s20 y20 z20 | MY | s20 x20 z20 | MZ | s20 x20 y20 | AL | x20 y20 z20 |
575  // | s30 y30 z30 | | s30 x30 z30 | | s30 x30 y30 | | x30 y30 z30 |
576 
577  const CoordinatesArrayType& rP0 = this->Points()[0].Coordinates();
578  const CoordinatesArrayType& rP1 = this->Points()[1].Coordinates();
579  const CoordinatesArrayType& rP2 = this->Points()[2].Coordinates();
580  const CoordinatesArrayType& rP3 = this->Points()[3].Coordinates();
581 
582  const double aDot = rP0[0] * rP0[0] + rP0[1] * rP0[1] + rP0[2] * rP0[2];
583  const double bDot = rP1[0] * rP1[0] + rP1[1] * rP1[1] + rP1[2] * rP1[2];
584  const double cDot = rP2[0] * rP2[0] + rP2[1] * rP2[1] + rP2[2] * rP2[2];
585  const double dDot = rP3[0] * rP3[0] + rP3[1] * rP3[1] + rP3[2] * rP3[2];
586 
587  // Build the simplified matrices
588  const double s10 = aDot - dDot;
589  const double x10 = rP0[0] - rP3[0];
590  const double y10 = rP0[1] - rP3[1];
591  const double z10 = rP0[2] - rP3[2];
592 
593  const double s20 = bDot - dDot;
594  const double x20 = rP1[0] - rP3[0];
595  const double y20 = rP1[1] - rP3[1];
596  const double z20 = rP1[2] - rP3[2];
597 
598  const double s30 = cDot - dDot;
599  const double x30 = rP2[0] - rP3[0];
600  const double y30 = rP2[1] - rP3[1];
601  const double z30 = rP2[2] - rP3[2];
602 
603  const double detJMX = s10 * y20 * z30 + y10 * z20 * s30 + z10 * s20 * y30 - s30 * y20 * z10 - y30 * z20 * s10 - z30 * s20 * y10;
604  const double detJMY = s10 * x20 * z30 + x10 * z20 * s30 + z10 * s20 * x30 - s30 * x20 * z10 - x30 * z20 * s10 - z30 * s20 * x10;
605  const double detJMZ = s10 * x20 * y30 + x10 * y20 * s30 + y10 * s20 * x30 - s30 * x20 * y10 - x30 * y20 * s10 - y30 * s20 * x10;
606  const double detJAL = x10 * y20 * z30 + y10 * z20 * x30 + z10 * x20 * y30 - x30 * y20 * z10 - y30 * z20 * x10 - z30 * x20 * y10;
607 
608  return std::sqrt( (detJMX*detJMX + detJMY*detJMY + detJMZ*detJMZ)) / (2*std::abs(detJAL));
609  }

616  double Inradius() const override {
617  // | x10 y10 z10 |
618  // AL | x20 y20 z20 |
619  // | x30 y30 z30 |
620 
621  const CoordinatesArrayType& rP0 = this->Points()[0].Coordinates();
622  const CoordinatesArrayType& rP1 = this->Points()[1].Coordinates();
623  const CoordinatesArrayType& rP2 = this->Points()[2].Coordinates();
624  const CoordinatesArrayType& rP3 = this->Points()[3].Coordinates();
625 
626  array_1d<double, 3> c012, c013, c023, c123;
627 
628  MathUtils<double>::CrossProduct(c012, rP2-rP0, rP1-rP0);
629  MathUtils<double>::CrossProduct(c013, rP3-rP0, rP1-rP0);
630  MathUtils<double>::CrossProduct(c023, rP3-rP0, rP2-rP0);
631  MathUtils<double>::CrossProduct(c123, rP3-rP1, rP2-rP1);
632

635  const double n023 = std::sqrt(c023[0]*c023[0] + c023[1]*c023[1] + c023[2]*c023[2]);
636  const double n123 = std::sqrt(c123[0]*c123[0] + c123[1]*c123[1] + c123[2]*c123[2]);

639  const double x10 = rP0[0] - rP3[0];
640  const double y10 = rP0[1] - rP3[1];
641  const double z10 = rP0[2] - rP3[2];
642 
643  const double x20 = rP1[0] - rP3[0];
644  const double y20 = rP1[1] - rP3[1];

915  inline void ComputeDihedralAngles(Vector& rDihedralAngles) const override

917  if(rDihedralAngles.size() != 6)
918  rDihedralAngles.resize(6, false);
919

923  for (unsigned int j = 0; j < 3; j++)

925  }

928  //to do so, we create a list of 3 nodes per angle. face1 is defined with nodes 0,1,2a , and face2 is defined with nodes 0,1,2b

939  for (unsigned int j = 0; j < 3; ++j) {

945  MathUtils<double>::CrossProduct(normal1,edge1,edge2a);
946  MathUtils<double>::CrossProduct(normal2,edge1,edge2b);

1462  const BaseType& tetra, Plane& plane,
1463  std::vector<BaseType>& inside) const

1467  int i = 0;
1468  int positive = 0, negative = 0, zero = 0;
1469  array_1d<double,4> C;
1470  array_1d<int,4> pos, neg, zer;
1471 
1472  noalias(C) = ZeroVector(4);
1473

1476  zer[0] = 0;
1477  pos[1] = 0;
1478  neg[1] = 0;
1479  zer[1] = 0;
1480  pos[2] = 0;
1481  neg[2] = 0;
1482  zer[2] = 0;
1483  pos[3] = 0;

1485  zer[3] = 0;
1486 
1487  for (i = 0; i < 4; ++i) {
1488  const array_1d<double,3>& p = tetra[i].Coordinates();
1489  C[i] = plane.DistanceTo(p);
1490  if (C[i] > 0.00)
1491  pos[positive++] = i;
1492  else if (C[i] < 0.00)
1493  neg[negative++] = i;
1494  else
1495  zer[zero++] = i;
1496  }
1497

1499  // be negative.
1500  if (negative == 0) {
1501  // Tetrahedron is completely on the positive side of plane, full clip.
1502  return;
1503  }
1504

1506  // Tetrahedron is completely on the negative side of plane.
1507  inside.push_back(tetra);

1509  }

1516  // +++-

1518  invCDiff = 1.00/(C[pos[i]] - C[neg[0]]);
1519  w0 = -C[neg[0]]*invCDiff;
1520  w1 = +C[pos[i]]*invCDiff;

1522  }

1532  }

1535  w0 = -C[neg[1]]*invCDiff;
1536  w1 = +C[pos[i]]*invCDiff;
1537  intp[i+2] = w0*tetra[pos[i]].Coordinates() + w1*tetra[neg[1]].Coordinates();
1538  }

1541  V[pos[1]] = intp[1];
