259  double CalculateElementStress(Element& elem_i, const ProcessInfo &rCurrentProcessInfo){
260 
261  std::vector<double> gp_weights_vector;
262  elem_i.CalculateOnIntegrationPoints(INTEGRATION_WEIGHT, gp_weights_vector, rCurrentProcessInfo);
263  std::vector<double> stress_gp_vector;
264 
265  if(mYieldStressType=="PK2_TENS" || mYieldStressType=="PK2_COMP"){
266  std::vector<Vector> PK2_stress_gp_vector;
267  elem_i.CalculateOnIntegrationPoints(PK2_STRESS_VECTOR, PK2_stress_gp_vector, rCurrentProcessInfo);
268  stress_gp_vector.resize(gp_weights_vector.size());
269  for(IndexType i = 0; i < gp_weights_vector.size(); i++)
270  stress_gp_vector[i] = PK2_stress_gp_vector[i][0] + PK2_stress_gp_vector[i][1] + PK2_stress_gp_vector[i][2];
271  }
272  else if(mYieldStressType=="VON_MISES")
273  elem_i.CalculateOnIntegrationPoints(KratosComponents<Variable<double>>::Get("VON_MISES_STRESS"), stress_gp_vector, rCurrentProcessInfo);
274 
275 
276  double elem_value = 0.0;
277  for(IndexType i = 0; i < gp_weights_vector.size(); i++){
278  double gp_stress = stress_gp_vector[i];
279  double ratio = gp_stress/mYieldStressLimit;
280  double pow_val = -2.0 * mBeta * (ratio-1);
281  if(pow_val>700)
282  pow_val = 700;
283  if(pow_val<-700)
284  pow_val = -700;
285  elem_value += gp_weights_vector[i] * (1.0/(1+std::exp(pow_val))) * std::pow(ratio,mHeavisidePenaltyFactor);
286  }
287 
288  return elem_value;
289  }

307  void ComputeAdjointRHS(ModelPart* mpADJModelPart){
308 
309  VariableUtils().SetHistoricalVariableToZero(ADJOINT_RHS, mpADJModelPart->Nodes());
310  for(auto& eval_obj : mrResponseSettings["evaluated_objects"]){
311  ModelPart& r_eval_object = mrModel.GetModelPart(eval_obj.GetString());
312  const ProcessInfo &rCurrentProcessInfo = r_eval_object.GetProcessInfo();
313  for (auto& elem_i : r_eval_object.Elements())
314  {
315 
316  std::vector<double> gp_weights_vector;
317  elem_i.CalculateOnIntegrationPoints(INTEGRATION_WEIGHT, gp_weights_vector, rCurrentProcessInfo);
318  const unsigned int number_integration_points = gp_weights_vector.size();
319  std::vector<double> stress_gp_vector;
320  std::vector<Matrix> stress_gp_tensor;
321  std::vector<double> von_mises_sensitivity_prefactors(number_integration_points);
322 
323  if(mYieldStressType=="PK2_TENS" || mYieldStressType=="PK2_COMP"){
324  std::vector<Vector> PK2_stress_gp_vector;
325  elem_i.CalculateOnIntegrationPoints(PK2_STRESS_VECTOR, PK2_stress_gp_vector, rCurrentProcessInfo);
326  stress_gp_vector.resize(gp_weights_vector.size());
327  for(IndexType i = 0; i < gp_weights_vector.size(); i++)
328  stress_gp_vector[i] = PK2_stress_gp_vector[i][0] + PK2_stress_gp_vector[i][1] + PK2_stress_gp_vector[i][2];
329  }
330  else if(mYieldStressType=="VON_MISES"){
331  elem_i.CalculateOnIntegrationPoints(KratosComponents<Variable<double>>::Get("VON_MISES_STRESS"), stress_gp_vector, rCurrentProcessInfo);
332  elem_i.CalculateOnIntegrationPoints(PK2_STRESS_TENSOR, stress_gp_tensor, rCurrentProcessInfo);
333  for(IndexType k = 0; k < number_integration_points; ++k)
334  {
335  Matrix & PK2_k = stress_gp_tensor[k];
336  double radicant = 0.0;
337  radicant += PK2_k(0,0)*PK2_k(0,0) + PK2_k(1,1)*PK2_k(1,1) + PK2_k(2,2)*PK2_k(2,2);
338  radicant -= PK2_k(0,0)*PK2_k(1,1) + PK2_k(0,0)*PK2_k(2,2) + PK2_k(1,1)*PK2_k(2,2);
339  radicant += 3*PK2_k(0,1)*PK2_k(0,1) + 3*PK2_k(0,2)*PK2_k(0,2) + 3*PK2_k(1,2)*PK2_k(1,2);
340  von_mises_sensitivity_prefactors[k] = 0.5/std::sqrt(radicant);
341  }
342  }
343 
344 
345 
346  // Some working variables
347  const SizeType num_nodes = elem_i.GetGeometry().PointsNumber();
348  const SizeType dimension = elem_i.GetGeometry().WorkingSpaceDimension();
349  const SizeType num_dofs_per_node = dimension;
350  const SizeType num_dofs = num_nodes * num_dofs_per_node;
351 
352 
353  // Build vector of variables containing the DOF-variables of the primal problem
354  std::vector<const Variable<double>*> primal_solution_variable_list {&DISPLACEMENT_X, &DISPLACEMENT_Y, &DISPLACEMENT_Z};
355 
356  // Build vector of variables containing the ADJOINT_RHS
357  std::vector<const Variable<double>*> adj_rhs_variable_list {&ADJOINT_RHS_X, &ADJOINT_RHS_Y, &ADJOINT_RHS_Z};
358 
359  // Store primal results and initialize deformation
360  Vector initial_state_variables;
361  initial_state_variables.resize(num_dofs, false);
362 
363  for (IndexType i = 0; i < num_nodes; ++i)
364  {
365  const IndexType index = i * num_dofs_per_node;
366  for(IndexType j = 0; j < primal_solution_variable_list.size(); ++j)
367  {
368  initial_state_variables[index + j] = elem_i.GetGeometry()[i].FastGetSolutionStepValue(*primal_solution_variable_list[j]);
369  elem_i.GetGeometry()[i].FastGetSolutionStepValue(*primal_solution_variable_list[j]) = 0.0;
370  }
371  }
372 
373 
374  for (IndexType i = 0; i < num_nodes; ++i)
375  {
376  for(IndexType j = 0; j < primal_solution_variable_list.size(); ++j)
377  {
378  elem_i.GetGeometry()[i].FastGetSolutionStepValue(*primal_solution_variable_list[j]) = 1.0;
379  double gp_sensitivity = 0.0;
380  for(IndexType k = 0; k < number_integration_points; ++k){
381  double gp_stress = stress_gp_vector[k];
382  double ratio = gp_stress/mYieldStressLimit;
383  double pow_val = -2.0 * mBeta * (ratio-1);
384  if(pow_val>700)
385  pow_val = 700;
386  if(pow_val<-700)
387  pow_val = -700;
388 
389  //derivative of heaviside
390  double heav_derv_mult = std::pow(ratio,mHeavisidePenaltyFactor);
391  heav_derv_mult *= (1.0/(1+std::exp(pow_val))) * (1.0/(1+std::exp(pow_val)));
392  heav_derv_mult *= std::exp(pow_val);
393  heav_derv_mult *= 2 * mBeta;
394  heav_derv_mult /= mYieldStressLimit;
395 
396  //derivative of penalty term
397  double penal_derv_mult = 0;
398  if(mHeavisidePenaltyFactor>0)
399  penal_derv_mult += (1.0/(1+std::exp(pow_val))) * (mHeavisidePenaltyFactor) * std::pow(ratio,mHeavisidePenaltyFactor-1) / mYieldStressLimit;
400 
401  if(mYieldStressType=="PK2_TENS" || mYieldStressType=="PK2_COMP"){
402  std::vector<Vector> partial_stress_derivatives;
403  elem_i.CalculateOnIntegrationPoints(PK2_STRESS_VECTOR, partial_stress_derivatives, rCurrentProcessInfo);
404  gp_sensitivity += gp_weights_vector[k] * (heav_derv_mult + penal_derv_mult) * (partial_stress_derivatives[k][0] + partial_stress_derivatives[k][1] + partial_stress_derivatives[k][2]);
405  // std::cout<<"heav_derv_mult: "<<heav_derv_mult<<", penal_derv_mult: "<<penal_derv_mult<<", partial_stress_derivatives[k]: "<<partial_stress_derivatives[k]<<std::endl;
406  }
407  else if(mYieldStressType=="VON_MISES"){
408 
409  std::vector<Matrix> partial_stress_derivatives;
410  elem_i.CalculateOnIntegrationPoints(PK2_STRESS_TENSOR, partial_stress_derivatives, rCurrentProcessInfo);
411 
412  double sensitivity_entry_k = 0.0;
413  Matrix & PK2_k = stress_gp_tensor[k];
414 
415  sensitivity_entry_k += 2*PK2_k(0,0)*partial_stress_derivatives[k](0,0);
416  sensitivity_entry_k += 2*PK2_k(1,1)*partial_stress_derivatives[k](1,1);
417  sensitivity_entry_k += 2*PK2_k(2,2)*partial_stress_derivatives[k](2,2);
418 
419  sensitivity_entry_k -= PK2_k(1,1)*partial_stress_derivatives[k](0,0);
420  sensitivity_entry_k -= PK2_k(0,0)*partial_stress_derivatives[k](1,1);
421 
422  sensitivity_entry_k -= PK2_k(0,0)*partial_stress_derivatives[k](2,2);
423  sensitivity_entry_k -= PK2_k(2,2)*partial_stress_derivatives[k](0,0);
424 
425  sensitivity_entry_k -= PK2_k(1,1)*partial_stress_derivatives[k](2,2);
426  sensitivity_entry_k -= PK2_k(2,2)*partial_stress_derivatives[k](1,1);
427 
428  sensitivity_entry_k += 6*PK2_k(0,1)*partial_stress_derivatives[k](0,1);
429  sensitivity_entry_k += 6*PK2_k(0,2)*partial_stress_derivatives[k](0,2);
430  sensitivity_entry_k += 6*PK2_k(1,2)*partial_stress_derivatives[k](1,2);
431 
432  sensitivity_entry_k *= von_mises_sensitivity_prefactors[k];
433  sensitivity_entry_k *= gp_weights_vector[k];
434  sensitivity_entry_k *= (heav_derv_mult + penal_derv_mult);
435  gp_sensitivity += sensitivity_entry_k;
436  }
437  }
438 
439  elem_i.GetGeometry()[i].FastGetSolutionStepValue(*primal_solution_variable_list[j]) = 0.0;
440  elem_i.GetGeometry()[i].FastGetSolutionStepValue(*adj_rhs_variable_list[j]) += gp_sensitivity;
441  }
442  }
443 
444  // Recall primal solution
445  for (IndexType i = 0; i < num_nodes; ++i)
446  {
447  const IndexType index = i * num_dofs_per_node;
448  for(IndexType j = 0; j < primal_solution_variable_list.size(); ++j)
449  elem_i.GetGeometry()[i].FastGetSolutionStepValue(*primal_solution_variable_list[j]) = initial_state_variables[index + j];
450  }
451 
452  }
453  }
454  }

612  void CalculateElementObjMaterialGradients(Element& elem_i, std::string material_gradien_name, const ProcessInfo &rCurrentProcessInfo){
613 
614  // We get the element geometry
615  auto& r_this_geometry = elem_i.GetGeometry();
616  const std::size_t number_of_nodes = r_this_geometry.size();
617 
618  std::vector<double> gp_weights_vector;
619  elem_i.CalculateOnIntegrationPoints(INTEGRATION_WEIGHT, gp_weights_vector, rCurrentProcessInfo);
620  std::vector<double> stress_gp_vector;
621 
622  if(mYieldStressType=="PK2_TENS" || mYieldStressType=="PK2_COMP"){
623  std::vector<Vector> PK2_stress_gp_vector;
624  elem_i.CalculateOnIntegrationPoints(PK2_STRESS_VECTOR, PK2_stress_gp_vector, rCurrentProcessInfo);
625  stress_gp_vector.resize(gp_weights_vector.size());
626  for(IndexType i = 0; i < gp_weights_vector.size(); i++)
627  stress_gp_vector[i] = PK2_stress_gp_vector[i][0] + PK2_stress_gp_vector[i][1] + PK2_stress_gp_vector[i][2];
628  }
629  else if(mYieldStressType=="VON_MISES")
630  elem_i.CalculateOnIntegrationPoints(KratosComponents<Variable<double>>::Get("VON_MISES_STRESS"), stress_gp_vector, rCurrentProcessInfo);
631 
632 
633  double elem_sens = 0.0;
634  for(IndexType i = 0; i < gp_weights_vector.size(); i++){
635  double gp_stress = stress_gp_vector[i];
636  double ratio = gp_stress/mYieldStressLimit;
637  double pow_val = -2.0 * mBeta * (ratio-1);
638  if(pow_val>700)
639  pow_val = 700;
640  if(pow_val<-700)
641  pow_val = -700;
642  elem_sens += gp_weights_vector[i] * (1.0/(1+std::exp(pow_val))) * std::pow(ratio,mHeavisidePenaltyFactor);
643  }
644 
645 
646  for (SizeType i_node = 0; i_node < number_of_nodes; ++i_node){
647  const auto& d_pe_d_fd = r_this_geometry[i_node].FastGetSolutionStepValue(D_PE_D_FD);
648  r_this_geometry[i_node].FastGetSolutionStepValue(KratosComponents<Variable<double>>::Get(material_gradien_name)) += d_pe_d_fd * elem_sens / number_of_nodes;
649  }
