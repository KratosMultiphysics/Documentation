64  void CalculateLocalSystem(MatrixType& rLeftHandSideMatrix,
65  VectorType& rRightHandSideVector,
66  const ProcessInfo& rCurrentProcessInfo) override
67  {
68  KRATOS_TRY
69 
70  Vector det_J_container;
71  GetGeometry().DeterminantOfJacobian(det_J_container, this->GetIntegrationMethod());
72  GeometryType::ShapeFunctionsGradientsType dN_dX_container =
73  GetGeometry().ShapeFunctionsLocalGradients(this->GetIntegrationMethod());
74  std::transform(dN_dX_container.begin(), dN_dX_container.end(), det_J_container.begin(),
75  dN_dX_container.begin(), std::divides<>());
76  const Matrix& r_N_container = GetGeometry().ShapeFunctionsValues(GetIntegrationMethod());
77 
78  const auto integration_coefficients = CalculateIntegrationCoefficients(det_J_container);
79  const auto permeability_matrix = CalculatePermeabilityMatrix(dN_dX_container, integration_coefficients);
80  AddContributionsToLhsMatrix(rLeftHandSideMatrix, permeability_matrix);
81 
82  const auto fluid_body_vector =
83  CalculateFluidBodyVector(r_N_container, dN_dX_container, integration_coefficients);
84  AddContributionsToRhsVector(rRightHandSideVector, permeability_matrix, fluid_body_vector);
85 
86  KRATOS_CATCH("")

182  Vector CalculateIntegrationCoefficients(const Vector& rDetJContainer) const
183  {
184  const auto& r_integration_points = GetGeometry().IntegrationPoints(GetIntegrationMethod());
185 
186  auto result = Vector{r_integration_points.size()};
187  // all governed by PIPE_HEIGHT and element length so without CROSS_AREA
188  std::transform(r_integration_points.begin(), r_integration_points.end(), rDetJContainer.begin(),
189  result.begin(), [](const auto& rIntegrationPoint, const auto& rDetJ) {
190  return rIntegrationPoint.Weight() * rDetJ;
191  });
192  return result;
193  }

200  BoundedMatrix<double, TNumNodes, TNumNodes> CalculatePermeabilityMatrix(
201  const GeometryType::ShapeFunctionsGradientsType& rShapeFunctionGradients,
202  const Vector& rIntegrationCoefficients) const
203  {
204  const auto& r_properties = GetProperties();
205  const double dynamic_viscosity_inverse = 1.0 / r_properties[DYNAMIC_VISCOSITY];
206 
207  auto constitutive_matrix = FillPermeabilityMatrix(r_properties[PIPE_HEIGHT]);
208 
209  auto result = BoundedMatrix<double, TNumNodes, TNumNodes>{ZeroMatrix{TNumNodes, TNumNodes}};
210  for (unsigned int integration_point_index = 0;
211  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
212  ++integration_point_index) {
213  result += GeoTransportEquationUtilities::CalculatePermeabilityMatrix<TDim, TNumNodes>(
214  rShapeFunctionGradients[integration_point_index], dynamic_viscosity_inverse,
215  constitutive_matrix, 1.0, rIntegrationCoefficients[integration_point_index]);
216  }
217 
218  return result;
219  }

231  array_1d<double, TNumNodes> CalculateFluidBodyVector(const Matrix& rNContainer,
232  const GeometryType::ShapeFunctionsGradientsType& rShapeFunctionGradients,
233  const Vector& rIntegrationCoefficients) const
234  {
235  const std::size_t number_integration_points =
236  GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
237  GeometryType::JacobiansType J_container;
238  J_container.resize(number_integration_points, false);
239  for (std::size_t i = 0; i < number_integration_points; ++i) {
240  J_container[i].resize(GetGeometry().WorkingSpaceDimension(),
241  GetGeometry().LocalSpaceDimension(), false);
242  }
243  GetGeometry().Jacobian(J_container, this->GetIntegrationMethod());
244 
245  const auto& r_properties = GetProperties();
246  auto constitutive_matrix = FillPermeabilityMatrix(r_properties[PIPE_HEIGHT]);
247 
248  array_1d<double, TNumNodes * TDim> volume_acceleration;
249  GeoElementUtilities::GetNodalVariableVector<TDim, TNumNodes>(
250  volume_acceleration, GetGeometry(), VOLUME_ACCELERATION);
251 
252  array_1d<double, TNumNodes> fluid_body_vector = ZeroVector(TNumNodes);
253  for (unsigned int integration_point_index = 0;
254  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
255  ++integration_point_index) {
256  array_1d<double, TDim> body_acceleration;
257  GeoElementUtilities::InterpolateVariableWithComponents<TDim, TNumNodes>(
258  body_acceleration, rNContainer, volume_acceleration, integration_point_index);
259 
260  array_1d<double, TDim> tangent_vector = column(J_container[integration_point_index], 0);
261  tangent_vector /= norm_2(tangent_vector);
262 
263  array_1d<double, 1> projected_gravity = ZeroVector(1);
264  projected_gravity(0) = MathUtils<double>::Dot(tangent_vector, body_acceleration);
265  const auto N = Vector{row(rNContainer, integration_point_index)};
266  fluid_body_vector +=
267  r_properties[DENSITY_WATER] *
268  prod(prod(rShapeFunctionGradients[integration_point_index], constitutive_matrix), projected_gravity) *
269  rIntegrationCoefficients[integration_point_index] / r_properties[DYNAMIC_VISCOSITY];
270  }
271  return fluid_body_vector;
272  }
