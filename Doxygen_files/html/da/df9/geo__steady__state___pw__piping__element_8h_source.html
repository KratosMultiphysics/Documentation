129  double CalculateEquilibriumPipeHeight(const PropertiesType& rProp, const GeometryType& rGeom, double)
130  {
131  // calculate head gradient over element ( now without abs in CalculateHeadGradient )
132  const auto head_gradient = CalculateHeadGradient(rProp, rGeom);
133  // return infinite when the head gradient dh/dx is 0
134  if (std::abs(head_gradient) < std::numeric_limits<double>::epsilon()) return 1e10;
135 
136  const auto particle_d = GeoTransportEquationUtilities::CalculateParticleDiameter(rProp);
137  // for a more generic element calculate slope of pipe (in degrees! see formula), currently pipe is assumed to be horizontal
138  constexpr double pipe_slope = 0.0;
139  const auto theta = rProp[PIPE_THETA];
140 
141  return rProp[PIPE_MODEL_FACTOR] * (Globals::Pi / 3.0) * particle_d *
142  (rProp[DENSITY_SOLID] / rProp[DENSITY_WATER] - 1.0) * rProp[PIPE_ETA] *
143  (std::sin(MathUtils<>::DegreesToRadians(theta + pipe_slope)) /
144  std::cos(MathUtils<>::DegreesToRadians(theta))) /
145  std::abs(head_gradient);
146  }

172  void CalculateOnIntegrationPoints(const Variable<array_1d<double, 3>>& rVariable,
173  std::vector<array_1d<double, 3>>& rValues,
174  const ProcessInfo& rCurrentProcessInfo) override
175  {
176  if (rVariable == FLUID_FLUX_VECTOR || rVariable == LOCAL_FLUID_FLUX_VECTOR) {
177  const auto number_of_integration_points =
178  this->GetGeometry().IntegrationPointsNumber(this->GetIntegrationMethod());
179  rValues.resize(number_of_integration_points);
180  const auto dynamic_viscosity_inverse = 1.0 / this->GetProperties()[DYNAMIC_VISCOSITY];
181  const auto permeability_matrix = FillPermeabilityMatrix(this->GetValue(PIPE_HEIGHT));
182  auto local_fluid_flux_vector = array_1d<double, 3>(3, 0.0);
183  local_fluid_flux_vector[0] = -dynamic_viscosity_inverse * permeability_matrix(0, 0) *
184  CalculateHeadGradient(this->GetProperties(), this->GetGeometry());
185  std::fill_n(rValues.begin(), number_of_integration_points, local_fluid_flux_vector);
186 
187  if (rVariable == LOCAL_FLUID_FLUX_VECTOR) return;
188 
189  // For the global fluid flux vector the local fluid flux vector should be rotated to the direction of the element
190  const auto& r_integration_points =
191  this->GetGeometry().IntegrationPoints(this->GetIntegrationMethod());
192  for (std::size_t i = 0; i < number_of_integration_points; ++i) {
193  Matrix jacobian;
194  this->GetGeometry().Jacobian(jacobian, r_integration_points[i]);
195  const auto tangential_vector =
196  GeoMechanicsMathUtilities::Normalized(Vector{column(jacobian, 0)});
197  std::transform(tangential_vector.begin(), tangential_vector.end(),
198  rValues[i].begin(), [&local_fluid_flux_vector](auto component) {
199  return component * local_fluid_flux_vector[0];
200  });
201  }
202  }
203  }

285  double CalculateHeadGradient(const PropertiesType& rProp, const GeometryType& rGeom)
286  {
287  const auto nodal_heads =
288  GeoElementUtilities::CalculateNodalHydraulicHeadFromWaterPressures(rGeom, rProp);
289  Matrix shapefunctions_local_gradient;
290  // iso-parametric derivative
291  GetGeometry().ShapeFunctionsLocalGradients(shapefunctions_local_gradient, GetGeometry().Center());
292  // local derivative
293  shapefunctions_local_gradient /= GetGeometry().DeterminantOfJacobian(GetGeometry().Center());
294  return Vector{prod(trans(shapefunctions_local_gradient), nodal_heads)}[0];
295  }

333  BoundedMatrix<double, TNumNodes, TNumNodes> CalculatePermeabilityMatrix(
334  const GeometryType::ShapeFunctionsGradientsType& rShapeFunctionGradients,
335  const Vector& rIntegrationCoefficients) const
336  {
337  const auto& r_properties = GetProperties();
338  const double dynamic_viscosity_inverse = 1.0 / r_properties[DYNAMIC_VISCOSITY];
339  auto constitutive_matrix = FillPermeabilityMatrix(this->GetValue(PIPE_HEIGHT));
340 
341  auto result = BoundedMatrix<double, TNumNodes, TNumNodes>{ZeroMatrix{TNumNodes, TNumNodes}};
342  for (unsigned int integration_point_index = 0;
343  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
344  ++integration_point_index) {
345  result += GeoTransportEquationUtilities::CalculatePermeabilityMatrix<TDim, TNumNodes>(
346  rShapeFunctionGradients[integration_point_index], dynamic_viscosity_inverse,
347  constitutive_matrix, 1.0, rIntegrationCoefficients[integration_point_index]);
348  }
349 
350  return result;
351  }

363  array_1d<double, TNumNodes> CalculateFluidBodyVector(const Matrix& rNContainer,
364  const GeometryType::ShapeFunctionsGradientsType& rShapeFunctionGradients,
365  const Vector& rIntegrationCoefficients) const
366  {
367  const std::size_t number_integration_points =
368  GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
369  GeometryType::JacobiansType J_container;
370  J_container.resize(number_integration_points, false);
371  for (std::size_t i = 0; i < number_integration_points; ++i) {
372  J_container[i].resize(GetGeometry().WorkingSpaceDimension(),
373  GetGeometry().LocalSpaceDimension(), false);
374  }
375  GetGeometry().Jacobian(J_container, this->GetIntegrationMethod());
376 
377  const auto& r_properties = GetProperties();
378  auto constitutive_matrix = FillPermeabilityMatrix(r_properties[PIPE_HEIGHT]);
379 
380  array_1d<double, TNumNodes * TDim> volume_acceleration;
381  GeoElementUtilities::GetNodalVariableVector<TDim, TNumNodes>(
382  volume_acceleration, GetGeometry(), VOLUME_ACCELERATION);
383 
384  array_1d<double, TNumNodes> fluid_body_vector = ZeroVector(TNumNodes);
385  for (unsigned int integration_point_index = 0;
386  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
387  ++integration_point_index) {
388  array_1d<double, TDim> body_acceleration;
389  GeoElementUtilities::InterpolateVariableWithComponents<TDim, TNumNodes>(
390  body_acceleration, rNContainer, volume_acceleration, integration_point_index);
391 
392  array_1d<double, TDim> tangent_vector = column(J_container[integration_point_index], 0);
393  tangent_vector /= norm_2(tangent_vector);
394 
395  array_1d<double, 1> projected_gravity = ZeroVector(1);
396  projected_gravity(0) = MathUtils<double>::Dot(tangent_vector, body_acceleration);
397  const auto N = Vector{row(rNContainer, integration_point_index)};
398  fluid_body_vector +=
399  r_properties[DENSITY_WATER] *
400  prod(prod(rShapeFunctionGradients[integration_point_index], constitutive_matrix), projected_gravity) *
401  rIntegrationCoefficients[integration_point_index] / r_properties[DYNAMIC_VISCOSITY];
402  }
403  return fluid_body_vector;
404  }
