443 
444  // Delete the new matrix

447  // Already existing matrix
448  if (rA.NumGlobalNonzeros() > 0) {

772  MatrixType& rA,
773  const Matrix& rLHSContribution,
774  const std::vector<std::size_t>& rEquationId
775  )
776  {
777  const unsigned int system_size = Size1(rA);
778 
779  // Count active indices
780  unsigned int active_indices = 0;
781  for (unsigned int i = 0; i < rEquationId.size(); i++)
782  if (rEquationId[i] < system_size)
783  ++active_indices;
784 
785  if (active_indices > 0) {
786  // Size Epetra vectors
787  Epetra_IntSerialDenseVector indices(active_indices);
788  Epetra_SerialDenseMatrix values(active_indices, active_indices);
789 
790  // Fill epetra vectors
791  unsigned int loc_i = 0;
792  for (unsigned int i = 0; i < rEquationId.size(); i++) {
793  if (rEquationId[i] < system_size) {
794  indices[loc_i] = rEquationId[i];
795 
796  unsigned int loc_j = 0;
797  for (unsigned int j = 0; j < rEquationId.size(); j++) {
798  if (rEquationId[j] < system_size) {
799  values(loc_i, loc_j) = rLHSContribution(i, j);
800  ++loc_j;
801  }
802  }
803  ++loc_i;
804  }
805  }
806 
807  const int ierr = rA.SumIntoGlobalValues(indices, values);
808  KRATOS_ERROR_IF(ierr != 0) << "Epetra failure found" << std::endl;
809  }
