438  {
439  // If the linear solver has not been deallocated, clean it before
440  // deallocating mpA. This prevents a memory error with the the ML
441  // solver (which holds a reference to it).
442  // NOTE: The linear solver is hold by the B&S
443  auto p_builder_and_solver = this->GetBuilderAndSolver();
444  if (p_builder_and_solver != nullptr) {
445  p_builder_and_solver->Clear();

448  // Deallocating system vectors to avoid errors in MPI. Clear calls
449  // TrilinosSpace::Clear for the vectors, which preserves the Map of
450  // current vectors, performing MPI calls in the process. Due to the

453  // the pointers here prevents Clear from operating with the
454  // (now deallocated) vectors.

626  void Predict() override
627  {
628  KRATOS_TRY
629  const DataCommunicator &r_comm = BaseType::GetModelPart().GetCommunicator().GetDataCommunicator();
630  //OPERATIONS THAT SHOULD BE DONE ONCE - internal check to avoid repetitions
631  //if the operations needed were already performed this does nothing
632  if (mInitializeWasPerformed == false)
633  Initialize();
634 
635  TSystemMatrixType& rA = *mpA;
636  TSystemVectorType& rDx = *mpDx;
637  TSystemVectorType& rb = *mpb;
638 
639  DofsArrayType& r_dof_set = GetBuilderAndSolver()->GetDofSet();
640 
641  GetScheme()->Predict(BaseType::GetModelPart(), r_dof_set, rA, rDx, rb);
642 
643  // Applying constraints if needed
644  auto& r_constraints_array = BaseType::GetModelPart().MasterSlaveConstraints();
645  const int local_number_of_constraints = r_constraints_array.size();
646  const int global_number_of_constraints = r_comm.SumAll(local_number_of_constraints);
647  if(global_number_of_constraints != 0) {
648  const auto& r_process_info = BaseType::GetModelPart().GetProcessInfo();
649 
650  block_for_each(r_constraints_array, [&r_process_info](MasterSlaveConstraint& rConstraint){
651  rConstraint.ResetSlaveDofs(r_process_info);
652  });
653  block_for_each(r_constraints_array, [&r_process_info](MasterSlaveConstraint& rConstraint){
654  rConstraint.Apply(r_process_info);
655  });
656 
657  // The following is needed since we need to eventually compute time derivatives after applying
658  // Master slave relations
659  TSparseSpace::SetToZero(rDx);
660  this->GetScheme()->Update(BaseType::GetModelPart(), r_dof_set, rA, rDx, rb);
661  }
662 
663  // Move the mesh if needed
664  if (this->MoveMeshFlag() == true)
665  BaseType::MoveMesh();
666 
667  KRATOS_CATCH("")
668  }

920  bool SolveSolutionStep() override
921  {
922  // Pointers needed in the solution
923  ModelPart& r_model_part = BaseType::GetModelPart();
924  typename TSchemeType::Pointer p_scheme = GetScheme();
925  typename TBuilderAndSolverType::Pointer p_builder_and_solver = GetBuilderAndSolver();
926  auto& r_dof_set = p_builder_and_solver->GetDofSet();
927  std::vector<Vector> NonconvergedSolutions;
928 
929  if (mStoreNonconvergedSolutionsFlag) {
930  Vector initial;
931  GetCurrentSolution(r_dof_set,initial);
932  NonconvergedSolutions.push_back(initial);
933  }
934 
935  TSystemMatrixType& rA = *mpA;
936  TSystemVectorType& rDx = *mpDx;
937  TSystemVectorType& rb = *mpb;
938 
939  //initializing the parameters of the Newton-Raphson cycle
940  unsigned int iteration_number = 1;
941  r_model_part.GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
942  bool residual_is_updated = false;
943  p_scheme->InitializeNonLinIteration(r_model_part, rA, rDx, rb);
944  mpConvergenceCriteria->InitializeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
945  bool is_converged = mpConvergenceCriteria->PreCriteria(r_model_part, r_dof_set, rA, rDx, rb);
946 
947  // Function to perform the building and the solving phase.
948  if (BaseType::mRebuildLevel > 0 || BaseType::mStiffnessMatrixIsBuilt == false) {
949  TSparseSpace::SetToZero(rA);
950  TSparseSpace::SetToZero(rDx);
951  TSparseSpace::SetToZero(rb);
952 
953  if (mUseOldStiffnessInFirstIteration){
954  p_builder_and_solver->BuildAndSolveLinearizedOnPreviousIteration(p_scheme, r_model_part, rA, rDx, rb,BaseType::MoveMeshFlag());
955  } else {
956  p_builder_and_solver->BuildAndSolve(p_scheme, r_model_part, rA, rDx, rb);
957  }
958  } else {
959  TSparseSpace::SetToZero(rDx); // Dx = 0.00;
960  TSparseSpace::SetToZero(rb);
961 
962  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
963  }
964 
965  // Debugging info
966  EchoInfo(iteration_number);
967 
968  // Updating the results stored in the database
969  UpdateDatabase(rA, rDx, rb, BaseType::MoveMeshFlag());
970 
971  p_scheme->FinalizeNonLinIteration(r_model_part, rA, rDx, rb);
972  mpConvergenceCriteria->FinalizeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
973 
974  if (mStoreNonconvergedSolutionsFlag) {
975  Vector first;
976  GetCurrentSolution(r_dof_set,first);
977  NonconvergedSolutions.push_back(first);
978  }
979 
980  if (is_converged) {
981  if (mpConvergenceCriteria->GetActualizeRHSflag()) {
982  TSparseSpace::SetToZero(rb);
983 
984  p_builder_and_solver->BuildRHS(p_scheme, r_model_part, rb);
985  }
986 
987  is_converged = mpConvergenceCriteria->PostCriteria(r_model_part, r_dof_set, rA, rDx, rb);
988  }
989 
990  //Iteration Cycle... performed only for NonLinearProblems
991  while (is_converged == false &&
992  iteration_number++ < mMaxIterationNumber)
993  {
994  //setting the number of iteration
995  r_model_part.GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
996 
997  p_scheme->InitializeNonLinIteration(r_model_part, rA, rDx, rb);
998  mpConvergenceCriteria->InitializeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
999 
1000  is_converged = mpConvergenceCriteria->PreCriteria(r_model_part, r_dof_set, rA, rDx, rb);
1001 
1002  //call the linear system solver to find the correction mDx for the
1003  //it is not called if there is no system to solve
1004  if (SparseSpaceType::Size(rDx) != 0)
1005  {
1006  if (BaseType::mRebuildLevel > 1 || BaseType::mStiffnessMatrixIsBuilt == false)
1007  {
1008  if (GetKeepSystemConstantDuringIterations() == false)
1009  {
1010  //A = 0.00;
1011  TSparseSpace::SetToZero(rA);
1012  TSparseSpace::SetToZero(rDx);
1013  TSparseSpace::SetToZero(rb);
1014 
1015  p_builder_and_solver->BuildAndSolve(p_scheme, r_model_part, rA, rDx, rb);
1016  }
1017  else
1018  {
1019  TSparseSpace::SetToZero(rDx);
1020  TSparseSpace::SetToZero(rb);
1021 
1022  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
1023  }
1024  }
1025  else
1026  {
1027  TSparseSpace::SetToZero(rDx);
1028  TSparseSpace::SetToZero(rb);
1029 
1030  p_builder_and_solver->BuildRHSAndSolve(p_scheme, r_model_part, rA, rDx, rb);
1031  }
1032  }
1033  else
1034  {
1035  KRATOS_WARNING("NO DOFS") << "ATTENTION: no free DOFs!! " << std::endl;
1036  }
1037 
1038  // Debugging info
1039  EchoInfo(iteration_number);
1040 
1041  // Updating the results stored in the database
1042  UpdateDatabase(rA, rDx, rb, BaseType::MoveMeshFlag());
1043 
1044  p_scheme->FinalizeNonLinIteration(r_model_part, rA, rDx, rb);
1045  mpConvergenceCriteria->FinalizeNonLinearIteration(r_model_part, r_dof_set, rA, rDx, rb);
1046 
1047  if (mStoreNonconvergedSolutionsFlag == true){
1048  Vector ith;
1049  GetCurrentSolution(r_dof_set,ith);
1050  NonconvergedSolutions.push_back(ith);
1051  }
1052 
1053  residual_is_updated = false;
1054 
1055  if (is_converged == true)
1056  {
1057  if (mpConvergenceCriteria->GetActualizeRHSflag() == true)
1058  {
1059  TSparseSpace::SetToZero(rb);
1060 
1061  p_builder_and_solver->BuildRHS(p_scheme, r_model_part, rb);
1062  residual_is_updated = true;
1063  }
1064 
1065  is_converged = mpConvergenceCriteria->PostCriteria(r_model_part, r_dof_set, rA, rDx, rb);
1066  }
1067  }
1068 
1069  //plots a warning if the maximum number of iterations is exceeded
1070  if (iteration_number >= mMaxIterationNumber) {
1071  MaxIterationsExceeded();
1072  } else {
1073  KRATOS_INFO_IF("ResidualBasedNewtonRaphsonStrategy", this->GetEchoLevel() > 0)
1074  << "Convergence achieved after " << iteration_number << " / "
1075  << mMaxIterationNumber << " iterations" << std::endl;
1076  }
1077 
1078  //recalculate residual if needed
1079  //(note that some convergence criteria need it to be recalculated)
1080  if (residual_is_updated == false)
1081  {
1082  // NOTE:
1083  // The following part will be commented because it is time consuming
1084  // and there is no obvious reason to be here. If someone need this
1085  // part please notify the community via mailing list before uncommenting it.
1086  // Pooyan.
1087 
1088  // TSparseSpace::SetToZero(mb);
1089  // p_builder_and_solver->BuildRHS(p_scheme, r_model_part, mb);
1090  }
1091 
1092  //calculate reactions if required
1093  if (mCalculateReactionsFlag == true)
1094  p_builder_and_solver->CalculateReactions(p_scheme, r_model_part, rA, rDx, rb);
1095 
1096  if (mStoreNonconvergedSolutionsFlag) {
1097  mNonconvergedSolutionsMatrix = Matrix( r_dof_set.size(), NonconvergedSolutions.size() );
1098  for (std::size_t i = 0; i < NonconvergedSolutions.size(); ++i) {
1099  block_for_each(r_dof_set, [&](const auto& r_dof) {
1100  mNonconvergedSolutionsMatrix(r_dof.EquationId(), i) = NonconvergedSolutions[i](r_dof.EquationId());
1101  });
1102  }
1103  }
1104 
1105  return is_converged;
