174  static void IntegrateStressVector(
175  BoundedArrayType& rPredictiveStressVector,
176  Vector& rStrainVector,
177  double& rUniaxialStress,
178  double& rThreshold,
179  double& rPlasticDenominator,
180  BoundedArrayType& rYieldSurfaceDerivative,
181  BoundedArrayType& rDerivativePlasticPotential,
182  double& rPlasticDissipation,
183  BoundedArrayType& rPlasticStrainIncrement,
184  Matrix& rConstitutiveMatrix,
185  Vector& rPlasticStrain,
186  ConstitutiveLaw::Parameters& rValues,
187  const double CharacteristicLength,
188  Vector& rBackStressVector,
189  const Vector& rPreviousStressVector
190  )
191  {
192  // Material properties
193  const Properties& r_material_properties = rValues.GetMaterialProperties();
194 
195  // Defining some variables
196  bool is_converged = false;
197  IndexType iteration = 0, max_iter = r_material_properties.Has(MAX_NUMBER_NL_CL_ITERATIONS) ? r_material_properties.GetValue(MAX_NUMBER_NL_CL_ITERATIONS) : 100;
198  BoundedArrayType delta_sigma;
199  double plastic_consistency_factor_increment, threshold_indicator;
200  BoundedArrayType kin_hard_stress_vector;
201  const bool analytic_tangent = (r_material_properties.Has(TANGENT_OPERATOR_ESTIMATION) && r_material_properties[TANGENT_OPERATOR_ESTIMATION] == 0) ? true : false;
202 
203  // Backward Euler
204  while (!is_converged && iteration <= max_iter) {
205  threshold_indicator = rUniaxialStress - rThreshold;
206  plastic_consistency_factor_increment = threshold_indicator * rPlasticDenominator;
207  plastic_consistency_factor_increment = (plastic_consistency_factor_increment < 0.0) ? 0.0 : plastic_consistency_factor_increment;
208  noalias(rPlasticStrainIncrement) = plastic_consistency_factor_increment * rDerivativePlasticPotential;
209  noalias(rPlasticStrain) += rPlasticStrainIncrement;
210  noalias(delta_sigma) = prod(rConstitutiveMatrix, rPlasticStrainIncrement);
211  noalias(rPredictiveStressVector) -= delta_sigma;
212  CalculateBackStress(rPredictiveStressVector, rValues, rPreviousStressVector, rPlasticStrainIncrement, rBackStressVector);
213  noalias(kin_hard_stress_vector) = rPredictiveStressVector - rBackStressVector;
214  threshold_indicator = CalculatePlasticParameters(kin_hard_stress_vector, rStrainVector, rUniaxialStress, rThreshold, rPlasticDenominator, rYieldSurfaceDerivative, rDerivativePlasticPotential, rPlasticDissipation, rPlasticStrainIncrement,rConstitutiveMatrix, rValues, CharacteristicLength, rPlasticStrain, rBackStressVector);
215 
216 
217  if (std::abs(threshold_indicator) <= std::abs(1.0e-4 * rThreshold)) { // Has converged
218  is_converged = true;
219  } else {
220  iteration++;
221  }
222  }
223  if (analytic_tangent) {
224  Matrix tangent_tensor;
225  tangent_tensor.resize(VoigtSize, VoigtSize, false);
226  CalculateTangentMatrix(tangent_tensor, rConstitutiveMatrix, rYieldSurfaceDerivative, rDerivativePlasticPotential, rPlasticDenominator);
227  noalias(rConstitutiveMatrix) = tangent_tensor;
228  }
229  KRATOS_WARNING_IF("GenericConstitutiveLawIntegratorKinematicPlasticity", iteration > max_iter) << "Maximum number of iterations in plasticity loop reached..." << std::endl;
230  }

332  static void CalculateBackStress(
333  BoundedArrayType& rPredictiveStressVector,
334  ConstitutiveLaw::Parameters& rValues,
335  const Vector& rPreviousStressVector,
336  const Vector& rPlasticStrainIncrement,
337  Vector& rBackStressVector
338  )
339  {
340  const Vector& r_kinematic_parameters = rValues.GetMaterialProperties()[KINEMATIC_PLASTICITY_PARAMETERS];
341  const unsigned int kinematic_hardening_type = rValues.GetMaterialProperties()[KINEMATIC_HARDENING_TYPE];
342 
343  switch (static_cast<KinematicHardeningType>(kinematic_hardening_type))
344  {
345  double pDot, denominator, dot_product_dp;
346  case KinematicHardeningType::LinearKinematicHardening:
347  KRATOS_ERROR_IF(r_kinematic_parameters.size() == 0) << "Kinematic Parameters not defined..." << std::endl;
348  rBackStressVector += 2.0 / 3.0 * r_kinematic_parameters[0] * rPlasticStrainIncrement;
349  break;
350 
351  case KinematicHardeningType::ArmstrongFrederickKinematicHardening:
352  KRATOS_ERROR_IF(r_kinematic_parameters.size() < 2) << "Kinematic Parameters not defined..." << std::endl;
353  dot_product_dp = 0.0;
354  for (IndexType i = 0; i < rPlasticStrainIncrement.size(); ++i) {
355  dot_product_dp += rPlasticStrainIncrement[i] * rPlasticStrainIncrement[i];
356  }
357  pDot = std::sqrt(2.0 / 3.0 * dot_product_dp);
358  denominator = 1.0 + (r_kinematic_parameters[1] * pDot);
359  rBackStressVector = (rBackStressVector + ((2.0 / 3.0 * r_kinematic_parameters[0]) * rPlasticStrainIncrement)) / denominator;
360  break;
361 
362  case KinematicHardeningType::AraujoVoyiadjisKinematicHardening:
363  KRATOS_ERROR_IF(r_kinematic_parameters.size() != 3) << "Kinematic Parameters not defined..." << std::endl;
364  dot_product_dp = 0.0;
365  for (IndexType i = 0; i < rPlasticStrainIncrement.size(); ++i) {
366  dot_product_dp += rPlasticStrainIncrement[i] * rPlasticStrainIncrement[i];
367  }
368  pDot = std::sqrt(2.0 / 3.0 * dot_product_dp);
369  denominator = 1.0 + (r_kinematic_parameters[1] * pDot);
370  if (pDot > tolerance) {
371  rBackStressVector = (rBackStressVector + ((2.0 / 3.0 * r_kinematic_parameters[0]) * rPlasticStrainIncrement)) / denominator;
372  } else {
373  const Vector& r_delta_stress = rPredictiveStressVector - rPreviousStressVector;
374  rBackStressVector = (rBackStressVector + ((2.0 / 3.0 * r_kinematic_parameters[0]) * rPlasticStrainIncrement) +
375  r_kinematic_parameters[2] * r_delta_stress) / denominator;
376  }
377  break;
378  default:
379  KRATOS_ERROR << " The Kinematic hardening type of plasticity is not set or wrong..." << kinematic_hardening_type << std::endl;
380  break;
381  }
382  }

532  static void CalculatePlasticDenominator(
533  const BoundedArrayType& rFFlux,
534  const BoundedArrayType& rGFlux,
535  const Matrix& rConstitutiveMatrix,
536  double& rHardeningParameter,
537  double& rPlasticDenominator,
538  const Vector& rBackStressVector,
539  ConstitutiveLaw::Parameters& rValues
540  )
541  {
542  const Vector& r_kinematic_parameters = rValues.GetMaterialProperties()[KINEMATIC_PLASTICITY_PARAMETERS];
543  const int kinematic_hardening_type = rValues.GetMaterialProperties()[KINEMATIC_HARDENING_TYPE];
544 
545  const BoundedArrayType delta_vector = prod(rGFlux, rConstitutiveMatrix);
546  double A1 = 0.0;
547  for (IndexType i = 0; i < VoigtSize; ++i) {
548  A1 += rFFlux[i] * delta_vector[i];
549  }
550  if (r_kinematic_parameters.size() == 3) {
551  A1 *= (1.0 - r_kinematic_parameters[2]);
552  } // Araujo case with 3 params
553 
554  double dot_fflux_gflux = 0.0, A2;
555  for (IndexType i = 0; i < VoigtSize; ++i) {
556  dot_fflux_gflux += rFFlux[i] * rGFlux[i];
557  }
558  const double two_thirds = 2.0 / 3.0;
559  double dot_fflux_backstress = 0.0, dot_gflux_gflux = 0.0;
560  switch (static_cast<KinematicHardeningType>(kinematic_hardening_type))
561  {
562  case KinematicHardeningType::LinearKinematicHardening:
563  A2 = two_thirds * r_kinematic_parameters[0] * dot_fflux_gflux;
564  break;
565 
566  case KinematicHardeningType::ArmstrongFrederickKinematicHardening:
567  A2 = two_thirds * r_kinematic_parameters[0] * dot_fflux_gflux;
568  for (IndexType i = 0; i < VoigtSize; ++i) {
569  dot_fflux_backstress += rFFlux[i] * rBackStressVector[i];
570  }
571  for (IndexType i = 0; i < VoigtSize; ++i) {
572  dot_gflux_gflux += rGFlux[i] * rGFlux[i];
573  }
574  A2 -= r_kinematic_parameters[1] * dot_fflux_backstress * std::sqrt(two_thirds * dot_gflux_gflux);
575  break;
576 
577  case KinematicHardeningType::AraujoVoyiadjisKinematicHardening:
578  A2 = two_thirds * r_kinematic_parameters[0] * dot_fflux_gflux;
579  for (IndexType i = 0; i < VoigtSize; ++i) {
580  dot_fflux_backstress += rFFlux[i] * rBackStressVector[i];
581  }
582  for (IndexType i = 0; i < VoigtSize; ++i) {
583  dot_gflux_gflux += rGFlux[i] * rGFlux[i];
584  }
585  A2 -= r_kinematic_parameters[1] * dot_fflux_backstress * std::sqrt(two_thirds * dot_gflux_gflux);
586  break;
587 
588  default:
589  KRATOS_ERROR << " The Kinematic hardening type of plasticity is not set or wrong..." << kinematic_hardening_type << std::endl;
590  break;
591  }
592 
593  const double A3 = rHardeningParameter;
594  rPlasticDenominator = 1.0 / (A1 + A2 + A3);
595 
596  if (r_kinematic_parameters.size() == 3) {
597  rPlasticDenominator *= (1.0 - r_kinematic_parameters[2]);
598  } // Araujo case with 3 params
599  }

605  static int Check(const Properties& rMaterialProperties)
606  {
607  // Checking is defined
608  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YOUNG_MODULUS)) << "HARDENING_CURVE is not a defined value" << std::endl;
609  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(HARDENING_CURVE)) << "HARDENING_CURVE is not a defined value" << std::endl;
610  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(FRACTURE_ENERGY)) << "FRACTURE_ENERGY is not a defined value" << std::endl;
611 
612  // Checking curves
613  const int curve_type = rMaterialProperties[HARDENING_CURVE];
614  if (static_cast<HardeningCurveType>(curve_type) == HardeningCurveType::InitialHardeningExponentialSoftening) {
615  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(MAXIMUM_STRESS)) << "MAXIMUM_STRESS is not a defined value" << std::endl;
616  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(MAXIMUM_STRESS_POSITION)) << "MAXIMUM_STRESS_POSITION is not a defined value" << std::endl;
617  } else if (static_cast<HardeningCurveType>(curve_type) == HardeningCurveType::CurveFittingHardening) {
618  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(CURVE_FITTING_PARAMETERS)) << "CURVE_FITTING_PARAMETERS is not a defined value" << std::endl;
619  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(PLASTIC_STRAIN_INDICATORS)) << "PLASTIC_STRAIN_INDICATORS is not a defined value" << std::endl;
620  }
621 
622  if (!rMaterialProperties.Has(YIELD_STRESS)) {
623  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
624  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
625 
626  const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
627  const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
628 
629  KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
630  KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
631  } else {
632  const double yield_stress = rMaterialProperties[YIELD_STRESS];
633 
634  KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
635  }
636 
637  return TYieldSurfaceType::Check(rMaterialProperties);
638  }
