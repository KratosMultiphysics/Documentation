278  void MonotonicityPreserving(
279  TSystemMatrixType& rA,
280  TSystemVectorType& rB
281  )
282  {
283  const auto& r_dof_set = BaseType::GetDofSet();
284  Vector dofs_values = ZeroVector(r_dof_set.size());
285 
286  block_for_each(r_dof_set, [&](Dof<double>& rDof){
287  const std::size_t id = rDof.EquationId();
288  dofs_values[id] = rDof.GetSolutionStepValue();
289  });
290  double *values_vector = rA.value_data().begin();
291  std::size_t *index1_vector = rA.index1_data().begin();
292  std::size_t *index2_vector = rA.index2_data().begin();
293 
294  IndexPartition<std::size_t>(rA.size1()).for_each(
295  [&](std::size_t i)
296  {
297  for (std::size_t k = index1_vector[i]; k < index1_vector[i + 1]; k++) {
298  const double value = values_vector[k];
299  if (value > 0.0) {
300  const auto j = index2_vector[k];
301  if (j > i) {
302  // TODO: Partition in blocks to gain efficiency by avoiding thread locks.
303  // Values conflicting with other threads
304  auto& r_aij = rA(i,j).ref();
305  AtomicAdd(r_aij, -value);
306  auto& r_aji = rA(j,i).ref();
307  AtomicAdd(r_aji, -value);
308  auto& r_aii = rA(i,i).ref();
309  AtomicAdd(r_aii, value);
310  auto& r_ajj = rA(j,j).ref();
311  AtomicAdd(r_ajj, value);
312  auto& r_bi = rB[i];
313  AtomicAdd(r_bi, value*dofs_values[j] - value*dofs_values[i]);
314  auto& r_bj = rB[j];
315  AtomicAdd(r_bj, value*dofs_values[i] - value*dofs_values[j]);
316  }
317  }
318  }
319  }
320  );
321  }

497  void InitializeHROMWeights(ModelPart& rModelPart)

501  using ElementQueue = moodycamel::ConcurrentQueue<Element::Pointer>;
502  using ConditionQueue = moodycamel::ConcurrentQueue<Condition::Pointer>;

506  block_for_each(rModelPart.Elements().GetContainer(),
507  [&](Element::Pointer p_element)

515

517  ConditionQueue condition_queue;
518  block_for_each(rModelPart.Conditions().GetContainer(),
519  [&](Condition::Pointer p_condition)

522  condition_queue.enqueue(std::move(p_condition));
523  } else {
524  p_condition->SetValue(HROM_WEIGHT, 1.0);
525  }
526  });
527 
528  // Dequeueing elements
529  std::size_t err_id;
530  mSelectedElements.reserve(element_queue.size_approx());
531  Element::Pointer p_element;
532  while ( (err_id = element_queue.try_dequeue(p_element)) != 0) {
533  mSelectedElements.push_back(std::move(p_element));
534  }
535 
536  // Dequeueing conditions
537  mSelectedConditions.reserve(condition_queue.size_approx());
538  Condition::Pointer p_condition;
539  while ( (err_id = condition_queue.try_dequeue(p_condition)) != 0) {

550  static DofQueue ExtractDofSet(
551  typename TSchemeType::Pointer pScheme,
552  ModelPart& rModelPart)
553  {
554  KRATOS_TRY
555 
556  DofQueue dof_queue;
557 
558  // Emulates ConcurrentQueue::enqueue_bulk adding move semantics to avoid atomic ops
559  const auto enqueue_bulk_move = [](DofQueue& r_queue, DofsVectorType& r_dof_list) {
560  for(auto& p_dof: r_dof_list) {
561  r_queue.enqueue(std::move(p_dof));
562  }
563  r_dof_list.clear();
564  };
565 
566  // Inspecting elements
567  DofsVectorType tls_dof_list; // Preallocation
568  block_for_each(rModelPart.Elements(), tls_dof_list,
569  [&](const Element& r_element, DofsVectorType& r_dof_list)
570  {
571  pScheme->GetDofList(r_element, r_dof_list, rModelPart.GetProcessInfo());
572  enqueue_bulk_move(dof_queue, r_dof_list);
573  });
574 
575  // Inspecting conditions
576  block_for_each(rModelPart.Conditions(), tls_dof_list,
577  [&](const Condition& r_condition, DofsVectorType& r_dof_list)
578  {
579  pScheme->GetDofList(r_condition, r_dof_list, rModelPart.GetProcessInfo());
580  enqueue_bulk_move(dof_queue, r_dof_list);
581  });
582 
583  // Inspecting master-slave constraints
584  std::pair<DofsVectorType, DofsVectorType> tls_ms_dof_lists; // Preallocation
585  block_for_each(rModelPart.MasterSlaveConstraints(), tls_ms_dof_lists,
586  [&](const MasterSlaveConstraint& r_constraint, std::pair<DofsVectorType, DofsVectorType>& r_dof_lists)
587  {
588  r_constraint.GetDofList(r_dof_lists.first, r_dof_lists.second, rModelPart.GetProcessInfo());
589 
590  enqueue_bulk_move(dof_queue, r_dof_lists.first);
591  enqueue_bulk_move(dof_queue, r_dof_lists.second);
592  });
593 
594  return dof_queue;
595 
596  KRATOS_CATCH("")
597  }

694  LocalSystemVectorType RHS_Contribution = LocalSystemVectorType(0);

697  //terms
698  Element::EquationIdVectorType EquationId;
699

701  const auto timer = BuiltinTimer();
702 
703  #pragma omp parallel firstprivate(nelements,nconditions, LHS_Contribution, RHS_Contribution, EquationId )
704  {
705  # pragma omp for schedule(guided, 512) nowait
706  for (int k = 0; k < nelements; k++) {
707  auto it_elem = el_begin + k;
708 
709  if (it_elem->IsActive()) {
710  // Calculate elemental contribution
711  pScheme->CalculateSystemContributions(*it_elem, LHS_Contribution, RHS_Contribution, EquationId, CurrentProcessInfo);
712 
713  //Get HROM weight and multiply it by its contribution
714  const double h_rom_weight = mHromSimulation ? it_elem->GetValue(HROM_WEIGHT) : 1.0;
715  LHS_Contribution *= h_rom_weight;
716  RHS_Contribution *= h_rom_weight;
717

719  BaseType::Assemble(A, b, LHS_Contribution, RHS_Contribution, EquationId);
720  }
721 
722  }
723 
724  #pragma omp for schedule(guided, 512)
725  for (int k = 0; k < nconditions; k++) {
726  auto it_cond = cond_begin + k;
727

729  // Calculate elemental contribution

731

734  LHS_Contribution *= h_rom_weight;
735  RHS_Contribution *= h_rom_weight;
736 
737  // Assemble the elemental contribution
738  BaseType::Assemble(A, b, LHS_Contribution, RHS_Contribution, EquationId);
739  }
740  }
741  }
742
