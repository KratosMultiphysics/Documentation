331  void Search(unsigned int& ils,Vector& prodr, Vector& eta, unsigned int& ico)
332  {
333  KRATOS_TRY
334 
335  double etaint = 0.00;
336  double etaneg = 1e5;
337  double etmxt = 0.00;
338  double etmaxp = 0.00;
339  double etaalt = 0.00;
340  double etaext = 0.00;
341  unsigned int ipos = 0;
342  unsigned int ineg = 999;
343 
344  for (unsigned int i=0; i<=ils+1; i++)
345  {
346  if (eta(i)>etmaxp)
347  {
348  etmaxp=eta(i);
349  }
350  if (prodr(i)<0 && eta(i)<=etaneg)
351  {
352  etaneg=eta(i);
353  ineg=i;
354  }
355  }
356 
357  //Decide si interpola o extrapola
358  if (ineg !=999)
359  {
360  //Interpola
361  //entontrar ipos=No del S-L con prodr positivo mas cercano a ineg pero con S-L mas pequeo
362  ipos = 0;
363  for( unsigned int i=0; i<=ils+1; i++)
364  {
365  if (prodr(i)>=0 && eta(i)<=eta(ineg) && eta(i)>=eta(ipos))
366  {
367  ipos=i;
368  }
369  }
370  //interpola para encontrar S-L etaint
371  etaint=prodr(ineg)*eta(ipos)-prodr(ipos)*eta(ineg);
372  if ((prodr(ineg)-prodr(ipos))==0)
373  {
374  ico=2;
375  return;
376  }
377  else
378  {
379  etaint=etaint/(prodr(ineg)-prodr(ipos));
380  }
381  //alternativamente encuentra etaalt para asegurar un cambio razonable
382  etaalt = eta(ipos) + 0.2*(eta(ineg)-eta(ipos));
383  //toma el maximo entre etaint y etaalt
384  etaint=std::max(etaint,etaalt);
385  //la longitud de paso minimo
386  if (etaint<metmna)
387  {
388  etaint=metmna;
389  if (ico==1)
390  {
391  ico=2;
392  }
393  else if (ico==0)
394  {
395  ico=1;
396  }
397  }
398 
399  eta(ils+2)=etaint;
400  }
401 
402  else if (ineg==999)
403  {
404 
405  //extrapola
406  //asigna temporalmente la longitud de paso maxima
407  etmxt = mamp*etmaxp;
408  if(etmxt>metmxa)
409  {
410  etmxt=metmxa;
411  }
412  //extrapola la longitud de paso entre el actual y el anterior
413  etaext=prodr(ils+1)*eta(ils)-prodr(ils)*eta(ils+1);
414  if ((prodr(ils+1)-prodr(ils))==0)
415  {
416  ico=2;
417  return;
418  }
419  else
420  {
421  etaext=etaext/(prodr(ils+1)-prodr(ils));
422  }
423 
424  eta(ils+2)=etaext;
425  //se acepta eta si esta dentro de los limites
426  if (etaext<=0 || etaext> etmxt)
427  {
428  eta(ils+2) = etmxt;
429  }
430  if (eta(ils+2)==metmxa && ico==1)
431  {
432  ico=2;
433  return;
434  }
435  if (eta(ils+2)==metmxa)
436  {
437  ico=1;
438  }
439  }
440  KRATOS_CATCH("")
