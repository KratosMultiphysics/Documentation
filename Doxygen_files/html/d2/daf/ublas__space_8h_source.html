704  static double CheckAndCorrectZeroDiagonalValues(
705  const ProcessInfo& rProcessInfo,
706  MatrixType& rA,
707  VectorType& rb,
708  const SCALING_DIAGONAL ScalingDiagonal = SCALING_DIAGONAL::NO_SCALING
709  )
710  {
711  // The system size
712  const std::size_t system_size = rA.size1();
713 
714  // The matrix data
715  auto& r_Avalues = rA.value_data();
716  const auto& r_Arow_indices = rA.index1_data();
717  const auto& r_Acol_indices = rA.index2_data();
718 
719  // Define the iterators
720  const auto it_Acol_indices_begin = r_Acol_indices.begin();
721 
722  // Define zero value tolerance
723  const double zero_tolerance = std::numeric_limits<double>::epsilon();
724 
725  // The diagonal considered
726  const double scale_factor = GetScaleNorm(rProcessInfo, rA, ScalingDiagonal);
727 
728  // Detect if there is a line of all zeros and set the diagonal to a 1 if this happens
729  IndexPartition(system_size).for_each([&](std::size_t Index){
730  bool empty = true;
731 
732  const std::size_t col_begin = r_Arow_indices[Index];
733  const std::size_t col_end = r_Arow_indices[Index + 1];
734 
735  for (std::size_t j = col_begin; j < col_end; ++j) {
736  if(std::abs(r_Avalues[j]) > zero_tolerance) {
737  empty = false;
738  break;
739  }
740  }
741 
742  if(empty) {
743  const auto it_Acol_indices_row_begin = it_Acol_indices_begin + col_begin;
744  const auto it_Acol_indices_row_end = it_Acol_indices_begin + col_end;
745 
746  const auto lower = std::lower_bound(it_Acol_indices_row_begin, it_Acol_indices_row_end, Index);
747  const auto upper = std::upper_bound(it_Acol_indices_row_begin, it_Acol_indices_row_end, Index);
748 
749  if (lower != upper) { // Index was found
750  r_Avalues[std::distance(it_Acol_indices_begin, lower)] = scale_factor;
751  } else {
752  KRATOS_DEBUG_ERROR << "Diagonal term (" << Index << ", " << Index << ") is not defined in the system matrix" << std::endl;
753  KRATOS_WARNING("UblasSpace") << "Diagonal term (" << Index << ", " << Index << ") is not defined in the system matrix" << std::endl;
754  }
755  rb[Index] = 0.0;
756  }
757  });
758 
759  return scale_factor;
760  }

769  static double GetScaleNorm(
770  const ProcessInfo& rProcessInfo,
771  const MatrixType& rA,
772  const SCALING_DIAGONAL ScalingDiagonal = SCALING_DIAGONAL::NO_SCALING
773  )
774  {
775  switch (ScalingDiagonal) {
776  case SCALING_DIAGONAL::NO_SCALING:
777  return 1.0;
778  case SCALING_DIAGONAL::CONSIDER_PRESCRIBED_DIAGONAL: {
779  KRATOS_ERROR_IF_NOT(rProcessInfo.Has(BUILD_SCALE_FACTOR)) << "Scale factor not defined at process info" << std::endl;
780  return rProcessInfo.GetValue(BUILD_SCALE_FACTOR);
781  }
782  case SCALING_DIAGONAL::CONSIDER_NORM_DIAGONAL:
783  return GetDiagonalNorm(rA)/static_cast<double>(rA.size1());
784  case SCALING_DIAGONAL::CONSIDER_MAX_DIAGONAL:
