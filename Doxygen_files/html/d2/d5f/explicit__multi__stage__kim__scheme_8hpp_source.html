383  template <typename TObjectType>
384  void UpdateAccelerationStage(
385  NodeIterator itCurrentNode,
386  const IndexType FieldPosition,
387  const ArrayVarType& rAccelerationVariable,
388  const ArrayVarType& rVelocityVariable,
389  const TObjectType& rDampingVariable,
390  const ArrayVarType& rResidualVariable,
391  const TObjectType& rIntertiaVariable,
392  const DoubleVarType& rFixVariable1,
393  const DoubleVarType& rFixVariable2,
394  const DoubleVarType& rFixVariable3,
395  const SizeType DomainSize = 3
396  )
397  {
398  Double3DArray& r_acceleration = itCurrentNode->FastGetSolutionStepValue(rAccelerationVariable);
399  Double3DArray& r_current_velocity = itCurrentNode->FastGetSolutionStepValue(rVelocityVariable);
400  const Double3DArray& r_current_residual = itCurrentNode->FastGetSolutionStepValue(rResidualVariable);
401 
402  if (std::is_same<TObjectType,DoubleVarType>::value) {
403  const double& r_nodal_inertia = itCurrentNode->GetValue(NODAL_MASS);
404  const double& r_nodal_damping = itCurrentNode->GetValue(NODAL_DISPLACEMENT_DAMPING);
405 
406  // Solution of the explicit equation:
407  if (r_nodal_inertia > numerical_limit)
408  {
409  noalias(r_acceleration) = (r_current_residual - r_nodal_damping * r_current_velocity) / r_nodal_inertia;
410  }
411  else
412  noalias(r_acceleration) = ZeroVector(3);
413  }
414  else if (std::is_same<TObjectType,ArrayVarType>::value) {
415  const Double3DArray& r_nodal_inertia = itCurrentNode->GetValue(NODAL_INERTIA);
416  const Double3DArray& r_nodal_rotational_damping = itCurrentNode->GetValue(NODAL_ROTATION_DAMPING);
417 
418  const IndexType initial_k = DomainSize == 3 ? 0 : 2; // We do this because in 2D only the rotation Z is needed, then we start with 2, instead of 0
419  for (IndexType kk = initial_k; kk < 3; ++kk) {
420  if (r_nodal_inertia[kk] > numerical_limit)
421  r_acceleration[kk] = (r_current_residual[kk] - r_nodal_rotational_damping[kk] * r_current_velocity[kk]) / r_nodal_inertia[kk];
422  else
423  r_acceleration[kk] = 0.0;
424  }
425  }
426  else KRATOS_ERROR << "cannot handle input variable in \"explicit_multi_stage_kim_scheme.hpp\"" << std::endl;
427 
428  std::array<bool, 3> fix_field = {false, false, false};
429  fix_field[0] = (itCurrentNode->GetDof(rFixVariable1, FieldPosition).IsFixed());
430  fix_field[1] = (itCurrentNode->GetDof(rFixVariable2, FieldPosition + 1).IsFixed());
431  if (DomainSize == 3)
432  fix_field[2] = (itCurrentNode->GetDof(rFixVariable3, FieldPosition + 2).IsFixed());
433 
434  for (IndexType j = 0; j < DomainSize; j++) {
435  if (fix_field[j]) {
436  r_acceleration[j] = 0.0;
437  }
438  }
439  }

448  NodeIterator itCurrentNode,

450  const ArrayVarType& rDisplacementVariable,
451  const ArrayVarType& rAccelerationVariable,

548  // The array containing the nodes
549  NodesArrayType& r_nodes = rModelPart.Nodes();
550 
551  // The fisrt node interator
552  const auto it_node_begin = rModelPart.NodesBegin();
553 
554  // Auxiliary zero array
555  const Double3DArray zero_array = ZeroVector(3);
556 
557  // If we consider the rotation DoF
558  const bool has_dof_for_rot_z = it_node_begin->HasDofFor(ROTATION_Z);
559 
560  // Getting dof position
561  const IndexType disppos = it_node_begin->GetDofPosition(DISPLACEMENT_X);
562  const IndexType rotppos = has_dof_for_rot_z ? it_node_begin->GetDofPosition(ROTATION_X) : 0;
563 
564  #pragma omp parallel for schedule(guided,512)
565  for (int i = 0; i < static_cast<int>(r_nodes.size()); ++i) {
566  // Current step information "N+1" (before step update).
567  auto it_node = it_node_begin + i;
568

570  const Double3DArray& r_current_residual = it_node->FastGetSolutionStepValue(FORCE_RESIDUAL);
571  Double3DArray& r_current_acceleration = it_node->FastGetSolutionStepValue(ACCELERATION);
572 
573  // Solution of the explicit equation:
574  if (nodal_mass > numerical_limit) {
575  r_current_acceleration = r_current_residual / nodal_mass;
576  } else {
577  r_current_acceleration = zero_array;
578  }
579 
580  std::array<bool, 3> fix_displacements = {false, false, false};
581 
582  fix_displacements[0] = (it_node->GetDof(DISPLACEMENT_X, disppos).IsFixed());
583  fix_displacements[1] = (it_node->GetDof(DISPLACEMENT_Y, disppos + 1).IsFixed());
584  if (DomainSize == 3)
585  fix_displacements[2] = (it_node->GetDof(DISPLACEMENT_Z, disppos + 2).IsFixed());
586 
587  for (IndexType j = 0; j < DomainSize; j++) {
588  if (fix_displacements[j]) {
589  r_current_acceleration[j] = 0.0;
590  }
591  } // for DomainSize
592 
593 
595  if (has_dof_for_rot_z) {
596  const Double3DArray& nodal_inertia = it_node->GetValue(NODAL_INERTIA);
597  const Double3DArray& r_current_residual_moment = it_node->FastGetSolutionStepValue(MOMENT_RESIDUAL);
598  Double3DArray& r_current_angular_acceleration = it_node->FastGetSolutionStepValue(ANGULAR_ACCELERATION);
599 
600  const IndexType initial_k = DomainSize == 3 ? 0 : 2; // We do this because in 2D only the rotation Z is needed, then we start with 2, instead of 0
601  for (IndexType kk = initial_k; kk < 3; ++kk) {
602  if (nodal_inertia[kk] > numerical_limit) {
603  r_current_angular_acceleration[kk] = r_current_residual_moment[kk] / nodal_inertia[kk];
604  } else {
605  r_current_angular_acceleration[kk] = 0.0;
606  }
607  }
608 
609  std::array<bool, 3> fix_rotation = {false, false, false};
610  if (DomainSize == 3) {
611  fix_rotation[0] = (it_node->GetDof(ROTATION_X, rotppos).IsFixed());
612  fix_rotation[1] = (it_node->GetDof(ROTATION_Y, rotppos + 1).IsFixed());
613  }
614  fix_rotation[2] = (it_node->GetDof(ROTATION_Z, rotppos + 2).IsFixed());
615 
616  for (IndexType j = initial_k; j < 3; j++) {
617  if (fix_rotation[j]) {
618  r_current_angular_acceleration[j] = 0.0;
619  }
620  } // trans DoF
621  } // Rot DoF
622  } // for node parallel
