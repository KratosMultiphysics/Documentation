140  std::vector<Element*> GetPipingElements()
141  {
142  ModelPart& r_current_model_part = this->GetModelPart();
143  std::vector<Element*> pipe_elements;
144  double pipe_element_start_x;
145 
146  for (auto& r_element : r_current_model_part.Elements()) {
147  if (r_element.GetProperties().Has(PIPE_START_ELEMENT)) {
148  pipe_elements.push_back(&r_element);
149 
150  long unsigned int start_element = r_element.GetProperties()[PIPE_START_ELEMENT];
151 
152  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0)
153  << r_element.Id() << " " << start_element << std::endl;
154 
155  if (r_element.Id() == start_element) {
156  pipe_element_start_x = r_element.GetGeometry().GetPoint(0)[0];
157  }
158  }
159  }
160 
161  if (pipe_elements.empty()) {
162  return pipe_elements;
163  }
164 
165  // Get Maximum X Value in Pipe
166  auto right_pipe = std::max_element(pipe_elements.begin(), pipe_elements.end(),
167  [](const Element* a, const Element* b) {
168  return a->GetGeometry().GetPoint(0)[0] < b->GetGeometry().GetPoint(0)[0];
169  });
170 
171  // Get Minimum X Value in Pipe
172  auto left_pipe = std::min_element(pipe_elements.begin(), pipe_elements.end(),
173  [](const Element* a, const Element* b) {
174  return a->GetGeometry().GetPoint(0)[0] < b->GetGeometry().GetPoint(0)[0];
175  });
176 
177  const auto min_x = left_pipe[0]->GetGeometry().GetPoint(0)[0];
178  const auto max_x = right_pipe[0]->GetGeometry().GetPoint(0)[0];
179 
180  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0)
181  << min_x << " " << max_x << " " << pipe_element_start_x << std::endl;
182 
183  if ((min_x != pipe_element_start_x) && (max_x != pipe_element_start_x)) {
184  KRATOS_ERROR << "Unable to determine pipe direction (multiple directions possible) - "
185  "Check PIPE_START_ELEMENT"
186  << std::endl;
187  }
188 
189  if (min_x == pipe_element_start_x) {
190  // Pipe Left -> Right
191  std::sort(pipe_elements.begin(), pipe_elements.end(), [](const Element* lhs, const Element* rhs) {
192  return lhs->GetGeometry().GetPoint(0)[0] < rhs->GetGeometry().GetPoint(0)[0];
193  });
194  } else {
195  // Pipe Right -> Left
196  std::sort(pipe_elements.begin(), pipe_elements.end(), [](const Element* lhs, const Element* rhs) {
197  return lhs->GetGeometry().GetPoint(0)[0] > rhs->GetGeometry().GetPoint(0)[0];
198  });
199  }
200 
201  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0)
202  << "Number of Pipe Elements: " << pipe_elements.size() << std::endl;
203  for (const Element* pipeElement : pipe_elements) {
204  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0)
205  << "PipeElementIDs (in order): " << pipeElement->Id() << std::endl;
206  }
207 
208  return pipe_elements;
209  }

277  template <typename FilteredElementsType>
278  bool CheckPipeEquilibrium(const FilteredElementsType& rOpenPipeElements, double MaxPipeHeight, unsigned int MaxNumberOfPipingIterations)
279  {
280  bool equilibrium = false;
281  bool converged = true;
282  unsigned int piping_iteration = 0;
283 
284  // calculate max pipe height and pipe increment
285  const auto pipe_height_increment =
286  CalculatePipeHeightIncrement(MaxPipeHeight, MaxNumberOfPipingIterations);
287 
288  while (piping_iteration < MaxNumberOfPipingIterations && !equilibrium && converged) {
289  equilibrium = true;
290 
291  // perform a flow calculation and stop growing if the calculation doesn't converge
292  converged = this->Recalculate();
293 
294  if (!converged) continue;
295 
296  // Update depth of open piping Elements
297  equilibrium = true;
298  for (auto p_open_pipe_element : rOpenPipeElements) {
299  // get open pipe element geometry and properties
300  auto& r_geom = p_open_pipe_element->GetGeometry();
301  auto& r_prop = p_open_pipe_element->GetProperties();
302 
303  // calculate equilibrium pipe height and get current pipe height
304  double eq_height = p_open_pipe_element->CalculateEquilibriumPipeHeight(
305  r_prop, r_geom, p_open_pipe_element->GetValue(PIPE_ELEMENT_LENGTH));
306  const auto current_height = p_open_pipe_element->GetValue(PIPE_HEIGHT);
307 
308  // set erosion on true if current pipe height is greater than the equilibrium height
309  p_open_pipe_element->SetValue(PIPE_EROSION, p_open_pipe_element->GetValue(PIPE_EROSION) ||
310  current_height > eq_height);
311  // check this if statement, I don't understand the check for pipe erosion
312  if ((!p_open_pipe_element->GetValue(PIPE_EROSION) || current_height > eq_height) &&
313  current_height < MaxPipeHeight) {
314  p_open_pipe_element->SetValue(PIPE_HEIGHT, current_height + pipe_height_increment);
315  equilibrium = false;
316  }
317 
318  // check if equilibrium height and current pipe heights are diverging, stop
319  // Picard iterations if this is the case and set pipe height on zero
320  if (!p_open_pipe_element->GetValue(PIPE_EROSION) && piping_iteration > 1 &&
321  eq_height - current_height > p_open_pipe_element->GetValue(DIFF_PIPE_HEIGHT)) {
322  p_open_pipe_element->SetValue(PIPE_HEIGHT, mSmallPipeHeight);
323  equilibrium = true;
324  }
325  // calculate difference between equilibrium height and current pipe height
326  p_open_pipe_element->SetValue(DIFF_PIPE_HEIGHT, eq_height - current_height);
327  }
328  ++piping_iteration;
329  }
330 
331  return equilibrium;
332  }

342  template <typename FilteredElementsType>
343  std::tuple<bool, SizeType> CheckStatusTipElement(SizeType NumberOfOpenPipeElements,
344  SizeType NumberOfPipeELements,
345  double MaxPipeHeight,
346  const FilteredElementsType& rPipeElements)
347  {
348  bool grow = true;
349  // check status of tip element, stop growing if pipe_height is zero or greater than maximum
350  // pipe height or if all elements are open
351  if (NumberOfOpenPipeElements < NumberOfPipeELements) {
352  auto p_tip_element = rPipeElements.at(NumberOfOpenPipeElements - 1);
353  const auto pipe_height = p_tip_element->GetValue(PIPE_HEIGHT);
354 
355  if (pipe_height > MaxPipeHeight + std::numeric_limits<double>::epsilon() ||
356  pipe_height < mPipeHeightAccuracy) {
357  // stable element found; pipe length does not increase during current time step
358  grow = false;
359  p_tip_element->SetValue(PIPE_EROSION, false);
360  p_tip_element->SetValue(PIPE_ACTIVE, false);
361  --NumberOfOpenPipeElements;
362 
363  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0)
364  << "Number of Open Pipe Elements: " << NumberOfOpenPipeElements << std::endl;
365  }
366  } else {
367  grow = false;
368  }
369  return std::make_tuple(grow, NumberOfOpenPipeElements);
370  }

390  template <typename FilteredElementsType>
391  void DetermineOpenPipingElements(const FilteredElementsType& rPipingElements)
392  {
393  bool grow = true;
394  const auto number_of_piping_elements = rPipingElements.size();
395  auto number_of_open_piping_elements = this->GetNumberOfActivePipeElements(rPipingElements);
396  auto max_pipe_height = CalculateMaxPipeHeight(rPipingElements);
397  while (grow && (number_of_open_piping_elements < number_of_piping_elements)) {
398  // get tip element and activate
399  auto p_tip_element = rPipingElements.at(number_of_open_piping_elements);
400  ++number_of_open_piping_elements;
401  p_tip_element->SetValue(PIPE_ACTIVE, true);
402 
403  // Get all open pipe elements
404  auto filter = [](auto p_element) {
405  return p_element->Has(PIPE_ACTIVE) && p_element->GetValue(PIPE_ACTIVE);
406  };
407  auto OpenPipeElements = rPipingElements | boost::adaptors::filtered(filter);
408  KRATOS_INFO_IF("PipingLoop", this->GetEchoLevel() > 0 && rank == 0)
409  << "Number of Open Pipe Elements: " << boost::size(OpenPipeElements) << std::endl;
410 
411  // nonlinear Picard iteration, for deepening the pipe
412  // Todo JDN (20220817):: Deal with Equilibrium redundancy
413  CheckPipeEquilibrium(OpenPipeElements, max_pipe_height, mPipingIterations);
414 
415  // check if pipe should grow in length
416  std::tie(grow, number_of_open_piping_elements) = CheckStatusTipElement(
417  number_of_open_piping_elements, number_of_piping_elements, max_pipe_height, rPipingElements);
418 
419  // if n open elements is lower than total pipe elements, save pipe height current
420  // growing iteration or reset to previous iteration in case the pipe should not grow.
421  if (number_of_open_piping_elements < number_of_piping_elements) {
422  SaveOrResetPipeHeights(OpenPipeElements, grow);
423  }
424  // recalculate groundwater flow
425  const auto converged = this->Recalculate();
426 
427  KRATOS_ERROR_IF_NOT(converged) << "Groundwater flow calculation failed to converge." << std::endl;
428  }
