636  CoordinatesArrayType& PointLocalCoordinates(
637  CoordinatesArrayType& rResult,
638  const CoordinatesArrayType& rPoint
639  ) const override
640  {
641  // Define distance_objective as the gradient of ||point - global_coordinate(xi)||**2,
642  // that is (point - local_coordinate(xi))*coordinate_derivative(xi) (times -2, which we ignore)
643  // The zeros of the distance_objective are local extremes of the distance between point and line
644  // We will get the zeros contained in the interval [-1, 1] (if any). If, for any of these zeros,
645  // the point is on the line, the local coordinate is our result.
646  rResult.clear();
647 
648  constexpr double TOLERANCE = 1e-12;
649 
650  const auto& r_p0 = this->GetPoint(0);
651  const auto& r_p1 = this->GetPoint(1);
652  const auto& r_p2 = this->GetPoint(2);
653 
654  // The method is prone to numerical instability close to the ends of the search interval
655  // exit early in that case
656  const array_1d<double,3> d0 = r_p0 - rPoint;
657  if (MathUtils<double>::Dot3(d0, d0) < TOLERANCE) {
658  rResult[0] = -1.0;
659  return rResult;
660  }
661  const array_1d<double,3> d1 = r_p1 - rPoint;
662  if (MathUtils<double>::Dot3(d1, d1) < TOLERANCE) {
663  rResult[0] = 1.0;
664  return rResult;
665  }
666 
667  array_1d<double, 3> c1 = r_p0 + r_p1 - 2*r_p2;
668  array_1d<double, 3> c2 = r_p1 - r_p0;
669  array_1d<double, 3> c3 = r_p2 - rPoint;
670 
671  const double aux1 = MathUtils<double>::Dot3(c1, c1);
672  if (aux1 < TOLERANCE) {
673  // The geometry is a straight line, fall back to Line3D2.h
674  auto line = Line3D2<TPointType>(
675  this->pGetPoint(0), this->pGetPoint(1));
676  return line.PointLocalCoordinates(rResult, rPoint);
677  }
678 
679  const double aux2 = MathUtils<double>::Dot3(c1, c3);
680  if (std::abs(aux2) < TOLERANCE) {
681  // r_p2 == rPoint (we got the center of the line),
682  //the local coordinate is 0
683  return rResult;
684  }
685 
686  PolynomialUtilities::PolynomialType distance_objective{
687  0.5 * aux1,
688  0.75 * MathUtils<double>::Dot3(c1, c2),
689  0.25 * MathUtils<double>::Dot3(c2, c2) + aux2,
690  0.5 * MathUtils<double>::Dot3(c2, c3)
691  };
692 
693  std::vector<PolynomialUtilities::IntervalType> root_ranges;
694  PolynomialUtilities::IsolateRoots(
695  root_ranges, distance_objective,
696  PolynomialUtilities::IntervalType{-1,1});
697 
698  Vector shape;
699  for (const auto& interval: root_ranges) {
700  double root = PolynomialUtilities::FindRoot(distance_objective, interval);
701  // if point == line(root) we found our coordinate;
702  rResult[0] = root;
703  this->ShapeFunctionsValues(shape, rResult);
704  array_1d<double,3> d = (shape[0]*r_p0 + shape[1]*r_p1 + shape[2]*r_p2) - rPoint;
705  if (MathUtils<double>::Dot3(d, d) < TOLERANCE) {
706  return rResult;
707  }
708  }
709 
710  // No points in the interval [-1,1] correspond to our local coordinate
711  rResult[0] = 2.0;
712  return rResult;
713  }
