796  static Matrix CalculateShapeFunctionsIntegrationPointsValues( typename BaseType::IntegrationMethod ThisMethod )
797  {
798  const IntegrationPointsContainerType& all_integration_points = AllIntegrationPoints();
799  const IntegrationPointsArrayType& IntegrationPoints = all_integration_points[static_cast<int>(ThisMethod)];
800  int integration_points_number = IntegrationPoints.size();
801  Matrix N( integration_points_number, 2 );
802 
803  for ( int it_gp = 0; it_gp < integration_points_number; it_gp++ )
804  {
805  double e = IntegrationPoints[it_gp].X();
806  N( it_gp, 0 ) = 0.5 * ( 1 - e );
807  N( it_gp, 1 ) = 0.5 * ( 1 + e );
808  }
809 
810  return N;
811  }
