569  iterator insert(const TPointerType& value)
570  {
571  auto itr_pos = std::lower_bound(mData.begin(), mData.end(), KeyOf(*value), CompareKey());
572  if (itr_pos == mData.end()) {
573  // the position to insert is at the end.
574  mData.push_back(value);
575  mSortedPartSize = mData.size();
576  return iterator(mData.end() - 1);
577  } else if (EqualKeyTo(KeyOf(*value))(*itr_pos)) {
578  // already found existing element with the same key, hence returning the existing element.
579  return iterator(itr_pos);
580  } else {
581  // insert the new value before the itr_pos.
582  mSortedPartSize = mData.size() + 1;
583  return mData.insert(itr_pos, value);
584  }

597  iterator insert(const_iterator position_hint, const TPointerType& value)
598  {
599  if (empty()) {
600  // the dataset is empty. So use push back.
601  mData.push_back(value);
602  mSortedPartSize = mData.size();
603  return iterator(mData.end() - 1);
604  } else if (position_hint == cend()) {
605  // trying to insert at the end.
606  if (KeyOf(*(position_hint - 1)) < KeyOf(*value)) {
607  // key at the position hint is less than the value of key. Hence position hint
608  // is valid. So using the push back.
609  mData.push_back(value);
610  mSortedPartSize = mData.size();
611  return iterator(mData.end() - 1);
612  } else {
613  // given position is invalid. Hence, discarding the hint.
614  return insert(value);
615  }
616  } else if (position_hint == cbegin()) {
617  // trying to insert at the front.
618  if (KeyOf(*value) < KeyOf(*position_hint)) {
619  // key at the position hint is greater than the value of key. Hence position hint
620  // is valid. So using insertion at the beginning.
621  mSortedPartSize = mData.size() + 1;
622  return mData.insert(mData.begin(), value);
623  } else {
624  // given position is invalid. Hence, discarding the hint.
625  return insert(value);
626  }
627  } else {
628  // trying to insert at an arbitrary position.
629  if (KeyOf(*value) < KeyOf(*position_hint) && (KeyOf(*(position_hint - 1)) < KeyOf(*value))) {
630  mSortedPartSize = mData.size() + 1;
631  return mData.insert(mData.begin() + (position_hint - cbegin()), value);
632  } else {
633  // given position is invalid. Hence, discarding the hint.
634  return insert(value);
635  }
636  }

1062  template<class TIteratorType>
1063  void SortedInsert(TIteratorType first, TIteratorType last)
1064  {
1065  if (std::distance(first, last) == 0) {
1066  return;
1067  }
1068 
1069  if (empty()) {
1070  mData.reserve(std::distance(first, last));
1071  for (auto it = first; it != last; ++it) {
1072  mData.push_back(TPointerType(&GetReference(it)));
1073  }
1074  } else {
1075  // first find the largest range
1076  const auto lower_bound_first = std::lower_bound(mData.begin(), mData.end(), KeyOf(GetReference(first)), CompareKey());
1077  const auto upper_bound_last = std::upper_bound(lower_bound_first, mData.end(), KeyOf(GetReference(last-1)), CompareKey());
1078 
1079  // then find the compact sub range
1080  const auto upper_bound_first = std::upper_bound(lower_bound_first, upper_bound_last, KeyOf(GetReference(first)), CompareKey());
1081  const auto lower_bound_last = std::lower_bound(lower_bound_first, upper_bound_last, KeyOf(GetReference(last-1)), CompareKey());
1082 
1083  if (lower_bound_first == lower_bound_last &&
1084  lower_bound_first == upper_bound_first &&
1085  lower_bound_first == upper_bound_last)
1086  {
1087  // all 4 bounds are equal, hence this can be inserted without checking further
1088  mData.reserve(mData.size() + std::distance(first, last));
1089  if (lower_bound_first == mData.end()) {
1090  for (auto it = first; it != last; ++it) {
1091  mData.push_back(TPointerType(&GetReference(it)));
1092  }
1093  } else {
1094  // now if the capacity of the new mData is larger than the existing
1095  // capacity, then the current lower_bound_first is invalidated.
1096  // hence needs to find it again.
1097  const auto new_lower_bound = std::lower_bound(mData.begin(), mData.end(), KeyOf(GetReference(first)), CompareKey());
1098  auto current_pos = new_lower_bound - 1;
1099  for (auto it = first; it != last; ++it) {
1100  current_pos = mData.insert(current_pos + 1, TPointerType(&GetReference(it)));
1101  }
1102  }
1103  } else {
1104  auto p_current_itr = mData.begin();
1105  // now add the new elements
1106  for (auto it = first; it != last; ++it) {
1107  // find the lower bound element.
1108  p_current_itr = std::lower_bound(p_current_itr, mData.end(), KeyOf(GetReference(it)), CompareKey());
1109  if (p_current_itr == mData.end() || !EqualKeyTo(KeyOf(GetReference(it)))(*p_current_itr)) {
1110  p_current_itr = mData.insert(p_current_itr, TPointerType(&GetReference(it)));
1111  }
1112  }
1113  }
1114  }
1115 
1116  // TODO: To be removed once push back is removed.
1117  // insert assumes the PointerVectorSet is already sorted,
1118  // hence mSortedPartSize should be mData.size()
1119  mSortedPartSize = mData.size();
1120  }

1163  template<class TIteratorType>
1164  inline auto& GetReference(TIteratorType Iterator) const
1165  {
1166  // It is difficult to use std::iterator_traits to get the value
1167  // type of the iterator because, boost::indirect has a value type
1168  // which is harder to guess, and cryptic. Hence, using the decltype.
1169  using iterator_value_type = std::decay_t<decltype(*Iterator)>;
1170 
1171  if constexpr(std::is_same_v<iterator_value_type, std::remove_cv_t<value_type>>) {
1172  // in here, std::remove_cv is only used for the value_type because,
1173  // the PointerVectorSet can be with TDataType which is const, but the passed pointers
1174  // must be always TDataType::Pointer which is defined for non cost TDataType. This is
1175  // a valid use case. Other way is not possible, hence std::remove_cv is not used on
1176  // iterator_value_type.
1177  return *Iterator;
1178  } else if constexpr(std::is_same_v<iterator_value_type, pointer>) {
1179  return **Iterator;
1180  } else {
1181  static_assert(!std::is_same_v<TIteratorType, TIteratorType>, "Unsupported iterator type.");
1182  return 0;
1183  }
1184  }
