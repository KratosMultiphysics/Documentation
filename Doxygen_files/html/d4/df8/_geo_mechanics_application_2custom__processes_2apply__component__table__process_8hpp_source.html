88  auto variable_name_1 = mpTable->NameOfX();
89  boost::to_upper(variable_name_1);

91  block_for_each(mrModelPart.Nodes(), [&var, this](Node& rNode) {
92  if (mIsFixed) rNode.Fix(var);
