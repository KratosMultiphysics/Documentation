1202  Vector& rResult,
1203  const LumpingMethods LumpingMethod = LumpingMethods::ROW_SUM
1204  ) const
1205  {

1207  const SizeType local_space_dimension = this->LocalSpaceDimension();
1208 
1209  // Clear lumping factors
1210  if (rResult.size() != number_of_nodes)
1211  rResult.resize(number_of_nodes, false);
1212  noalias(rResult) = ZeroVector(number_of_nodes);
1213 
1214  if (LumpingMethod == LumpingMethods::ROW_SUM) {
1215  const IntegrationMethod integration_method = GetDefaultIntegrationMethod();
1216  const GeometryType::IntegrationPointsArrayType& r_integrations_points = this->IntegrationPoints( integration_method );
1217  const Matrix& r_Ncontainer = this->ShapeFunctionsValues(integration_method);
1218 
1219  // Vector fo jacobians
1220  Vector detJ_vector(r_integrations_points.size());
1221  DeterminantOfJacobian(detJ_vector, integration_method);
1222 
1223  // Iterate over the integration points
1224  double domain_size = 0.0;
1225  for ( IndexType point_number = 0; point_number < r_integrations_points.size(); ++point_number ) {
1226  const double integration_weight = r_integrations_points[point_number].Weight() * detJ_vector[point_number];
1227  const Vector& rN = row(r_Ncontainer,point_number);
1228 
1229  // Computing domain size
1230  domain_size += integration_weight;
1231 
1232  for ( IndexType i = 0; i < number_of_nodes; ++i ) {
1233  rResult[i] += rN[i] * integration_weight;
1234  }
1235  }

1277  PointsLocalCoordinates(local_coordinates);

1283  const double weight = r_integrations_points[0].Weight()/static_cast<double>(number_of_nodes);

1285  for ( IndexType dim = 0; dim < local_space_dimension; ++dim ) {
