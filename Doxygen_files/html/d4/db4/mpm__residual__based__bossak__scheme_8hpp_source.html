165  void Update(
166  ModelPart& rModelPart,
167  DofsArrayType& rDofSet,
168  TSystemMatrixType& rA,
169  TSystemVectorType& rDx,
170  TSystemVectorType& rb ) override
171  {
172  KRATOS_TRY
173 
174  // Rotate the current displacement to the modified coordinate system since rDx is currently at the modified coordinate system
175  mRotationTool.RotateDisplacements(rModelPart);
176 
177  // Update of displacement (by DOF)
178  mpDofUpdater->UpdateDofs(rDofSet, rDx);
179 
180  // Rotate the displacement back to the original coordinate system to calculate the velocity and acceleration
181  mRotationTool.RecoverDisplacements(rModelPart);
182 
183  // Updating time derivatives (nodally for efficiency)
184  block_for_each(rModelPart.Nodes(), [&](Node& rNode)
185  {
186  // In MPM the delta_displacement is the current displacement as the previous_displacement is always reset
187  const array_1d<double, 3 > & r_delta_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT);
188 
189  array_1d<double, 3>& r_current_velocity = rNode.FastGetSolutionStepValue(VELOCITY);
190  const array_1d<double, 3>& r_previous_velocity = rNode.FastGetSolutionStepValue(VELOCITY, 1);
191 
192  array_1d<double, 3>& r_current_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION);
193  const array_1d<double, 3>& r_previous_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION, 1);
194 
195  if (mIsDynamic){
196  BossakBaseType::UpdateVelocity(r_current_velocity, r_delta_displacement, r_previous_velocity, r_previous_acceleration);
197  BossakBaseType::UpdateAcceleration(r_current_acceleration, r_delta_displacement, r_previous_velocity, r_previous_acceleration);
198  }
199  });

214  ModelPart& rModelPart,
215  DofsArrayType& rDofSet,
216  TSystemMatrixType& rA,
217  TSystemVectorType& rDx,
218  TSystemVectorType& rb) override

223  {
224  const array_1d<double, 3 > & r_previous_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT, 1);
225  const array_1d<double, 3 > & r_previous_velocity = rNode.FastGetSolutionStepValue(VELOCITY, 1);
226  const array_1d<double, 3 > & r_previous_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION, 1);

232  r_current_displacement[0] = 0.0;
233  else
234  r_current_displacement[0] = r_previous_displacement[0];
235 
236  if (!(rNode.pGetDof(DISPLACEMENT_Y)->IsFixed()))
237  r_current_displacement[1] = 0.0;
238  else
239  r_current_displacement[1] = r_previous_displacement[1];
240 
241  if (rNode.HasDofFor(DISPLACEMENT_Z))
242  {
243  if (!(rNode.pGetDof(DISPLACEMENT_Z)->IsFixed()))
244  r_current_displacement[2] = 0.0;
245  else
246  r_current_displacement[2] = r_previous_displacement[2];
247  }
248 
249  // Pressure prediction for implicit MPM
250  if (rNode.HasDofFor(PRESSURE))
251  {
252  double& r_current_pressure = rNode.FastGetSolutionStepValue(PRESSURE);
253  const double& r_previous_pressure = rNode.FastGetSolutionStepValue(PRESSURE, 1);
254 
255  if (!(rNode.pGetDof(PRESSURE))->IsFixed())
256  r_current_pressure = r_previous_pressure;
257  }
258 
259  // Updating time derivatives
260  array_1d<double, 3 > & current_velocity = rNode.FastGetSolutionStepValue(VELOCITY);
261  array_1d<double, 3 > & current_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION);
262 
263  if (mIsDynamic){
264  BossakBaseType::UpdateVelocity(current_velocity, r_current_displacement, r_previous_velocity, r_previous_acceleration);
265  BossakBaseType::UpdateAcceleration (current_acceleration, r_current_displacement, r_previous_velocity, r_previous_acceleration);
266  }
267 
268  });

309  void InitializeSolutionStep(
310  ModelPart& rModelPart,
311  TSystemMatrixType& rA,
312  TSystemVectorType& rDx,
313  TSystemVectorType& rb) override
314  {
315  KRATOS_TRY
316 
317  // Loop over the grid nodes performed to clear all nodal information
318  block_for_each(rModelPart.Nodes(), [&](Node& rNode)
319  {
320  // Variables to be cleaned
321  double & r_nodal_mass = rNode.FastGetSolutionStepValue(NODAL_MASS);
322  array_1d<double, 3 > & r_nodal_momentum = rNode.FastGetSolutionStepValue(NODAL_MOMENTUM);
323  array_1d<double, 3 > & r_nodal_inertia = rNode.FastGetSolutionStepValue(NODAL_INERTIA);
324 
325  array_1d<double, 3 > & r_nodal_displacement = rNode.FastGetSolutionStepValue(DISPLACEMENT);
326  array_1d<double, 3 > & r_nodal_velocity = rNode.FastGetSolutionStepValue(VELOCITY,1);
327  array_1d<double, 3 > & r_nodal_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION,1);
328 
329  double & r_nodal_old_pressure = rNode.FastGetSolutionStepValue(PRESSURE,1);
330  double & r_nodal_pressure = rNode.FastGetSolutionStepValue(PRESSURE);
331 
332  // Clear
333  r_nodal_mass = 0.0;
334  r_nodal_momentum.clear();
335  r_nodal_inertia.clear();
336 
337  r_nodal_displacement.clear();
338  r_nodal_velocity.clear();
339  r_nodal_acceleration.clear();
340  r_nodal_old_pressure = 0.0;
341  r_nodal_pressure = 0.0;
342 
343  // Other additional variables
344  if (rNode.SolutionStepsDataHas(NODAL_AREA)){
345  double & r_nodal_area = rNode.FastGetSolutionStepValue(NODAL_AREA);
346  r_nodal_area = 0.0;
347  }
348  if(rNode.SolutionStepsDataHas(NODAL_MPRESSURE)) {
349  double & r_nodal_mpressure = rNode.FastGetSolutionStepValue(NODAL_MPRESSURE);
350  r_nodal_mpressure = 0.0;
351  }
352 
353  // friction-related
354  if(mFrictionIsActive){
355  rNode.FastGetSolutionStepValue(STICK_FORCE).clear();
356  rNode.FastGetSolutionStepValue(FRICTION_STATE) = mRotationTool.GetSlidingState();
357  rNode.SetValue(FRICTION_ASSIGNED, false);
358  }
359  });
360 
361  // Extrapolate from Material Point Elements and Conditions
362  ImplicitBaseType::InitializeSolutionStep(rModelPart,rA,rDx,rb);
363 
364  // Assign nodal variables after extrapolation
365  block_for_each(rModelPart.Nodes(), [&](Node& rNode)
366  {
367  const double & r_nodal_mass = rNode.FastGetSolutionStepValue(NODAL_MASS);
368 
369  if (r_nodal_mass > std::numeric_limits<double>::epsilon())
370  {
371  const array_1d<double, 3 > & r_nodal_momentum = rNode.FastGetSolutionStepValue(NODAL_MOMENTUM);
372  const array_1d<double, 3 > & r_nodal_inertia = rNode.FastGetSolutionStepValue(NODAL_INERTIA);
373 
374  array_1d<double, 3 > & r_nodal_velocity = rNode.FastGetSolutionStepValue(VELOCITY,1);
375  array_1d<double, 3 > & r_nodal_acceleration = rNode.FastGetSolutionStepValue(ACCELERATION,1);
376  double & r_nodal_pressure = rNode.FastGetSolutionStepValue(PRESSURE,1);
377 
378  double delta_nodal_pressure = 0.0;
379 
380  // For mixed formulation
381  if (rNode.HasDofFor(PRESSURE) && rNode.SolutionStepsDataHas(NODAL_MPRESSURE))
382  {
383  double & nodal_mpressure = rNode.FastGetSolutionStepValue(NODAL_MPRESSURE);
384  delta_nodal_pressure = nodal_mpressure/r_nodal_mass;
385  }
386 
387  const array_1d<double, 3 > delta_nodal_velocity = r_nodal_momentum/r_nodal_mass;
388  const array_1d<double, 3 > delta_nodal_acceleration = r_nodal_inertia/r_nodal_mass;
389 
390  r_nodal_velocity += delta_nodal_velocity;
391  r_nodal_acceleration += delta_nodal_acceleration;
392 
393  r_nodal_pressure += delta_nodal_pressure;
394 
395  // mark nodes which have non-zero momentum in the 1st timestep s.t. these nodes can have
396  // an initial friction state of SLIDING instead of STICK
397  if(mFrictionIsActive){
398  const bool has_initial_momentum = (mGridModelPart.GetProcessInfo()[STEP] == 1 && norm_2(r_nodal_momentum) > std::numeric_limits<double>::epsilon());
399  rNode.SetValue(HAS_INITIAL_MOMENTUM, has_initial_momentum);
400  }
401  }
402  });
403 
404  const ProcessInfo& r_current_process_info = rModelPart.GetProcessInfo();
405  const double delta_time = r_current_process_info[DELTA_TIME];
406 
407  // Initializing Bossak constants
408  mBossak.c0 = ( 1.0 / (mBossak.beta * delta_time * delta_time) );
409  mBossak.c1 = ( mBossak.gamma / (mBossak.beta * delta_time) );
410  mBossak.c2 = ( 1.0 / (mBossak.beta * delta_time) );
411  mBossak.c3 = ( 0.5 / (mBossak.beta) - 1.0 );
412  mBossak.c4 = ( (mBossak.gamma / mBossak.beta) - 1.0 );
413  mBossak.c5 = ( delta_time * 0.5 * ( ( mBossak.gamma / mBossak.beta ) - 2.0 ) );
414 
415  KRATOS_CATCH( "" )
416  }

435  const Node& rConstNode = rNode; // const Node reference to avoid issues with previously unset GetValue()
