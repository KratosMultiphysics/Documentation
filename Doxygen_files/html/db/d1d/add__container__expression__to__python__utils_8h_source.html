34 template <class TDataType>
35 pybind11::array_t<TDataType> AllocateNumpyArray(
36  const IndexType NumberOfEntities,
37  const std::vector<IndexType>& rShape)
38 {
39  const IndexType size = std::accumulate(rShape.begin(),
40  rShape.end(),
41  1UL,
42  [](IndexType left, IndexType right) {return left * right;});
43 
44  TDataType* array = new TDataType[size * NumberOfEntities];
45  pybind11::capsule release(array, [](void* a) {
46  delete[] reinterpret_cast<TDataType*>(a);
47  });
48 
49  // we allocate one additional dimension for the shape to hold the
50  // number of entities. So if the shape within kratos is [2, 3] with 70 entities,
51  // then the final shape of the numpy array will be [70,2,3] so it can keep the
52  // existing shape preserved for each entitiy.
53  std::vector<IndexType> c_shape(rShape.size() + 1);
54  c_shape[0] = NumberOfEntities;
55  std::copy(rShape.begin(), rShape.end(), c_shape.begin() + 1);
56 
57  // we have to allocate number of bytes to be skipped to reach next element
58  // in each dimension. So this is calculated backwards.
59  std::vector<IndexType> strides(c_shape.size());
60  IndexType stride_items = 1;
61  for (int i = c_shape.size() - 1; i >= 0; --i) {
62  strides[i] = sizeof(double) * stride_items;
63  stride_items *= c_shape[i];
64  }
65 
66  return ContiguousNumpyArray<TDataType>(
67  c_shape,
68  strides,
69  array,
70  release
71  );
72 }
