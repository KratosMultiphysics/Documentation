184  OpenMPUtils::DivideInPartitions(rDofSet.size(), NumThreads, DofSetPartition);

195  if (itDof->IsFree()) {
196  itDof->GetSolutionStepValue() += TSparseSpace::GetValue(Dv, itDof->EquationId());

200  }
