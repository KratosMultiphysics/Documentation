290  void FinalizeSolutionStep(
291  ModelPart& rModelPart,
292  TSystemMatrixType& A,
293  TSystemVectorType& Dx,
294  TSystemVectorType& b) override
295  {
296  KRATOS_TRY
297 
298  if(rModelPart.GetProcessInfo()[NODAL_SMOOTHING] == true)
299  {
300  const int NNodes = static_cast<int>(rModelPart.Nodes().size());
301  ModelPart::NodesContainerType::iterator node_begin = rModelPart.NodesBegin();
302 
303  // Clear nodal variables
304  #pragma omp parallel for
305  for(int i = 0; i < NNodes; i++)
306  {
307  ModelPart::NodesContainerType::iterator itNode = node_begin + i;
308 
309  itNode->FastGetSolutionStepValue(NODAL_AREA) = 0.0;
310  Matrix& rNodalStress = itNode->FastGetSolutionStepValue(NODAL_EFFECTIVE_STRESS_TENSOR);
311  if(rNodalStress.size1() != 3)
312  rNodalStress.resize(3,3,false);
313  noalias(rNodalStress) = ZeroMatrix(3,3);
314  itNode->FastGetSolutionStepValue(NODAL_JOINT_AREA) = 0.0;
315  itNode->FastGetSolutionStepValue(NODAL_JOINT_WIDTH) = 0.0;
316  itNode->FastGetSolutionStepValue(NODAL_MID_PLANE_LIQUID_PRESSURE) = 0.0;
317  itNode->FastGetSolutionStepValue(NODAL_SLIP_TENDENCY) = 0.0;
318  itNode->FastGetSolutionStepValue(NODAL_JOINT_DAMAGE) = 0.0;
319  }
320 
321  BaseType::FinalizeSolutionStep(rModelPart,A,Dx,b);
322 
323  // Compute smoothed nodal variables
324  #pragma omp parallel for
325  for(int n = 0; n < NNodes; n++)
326  {
327  ModelPart::NodesContainerType::iterator itNode = node_begin + n;
328 
329  const double& NodalArea = itNode->FastGetSolutionStepValue(NODAL_AREA);
330  if (NodalArea>1.0e-20)
331  {
332  const double InvNodalArea = 1.0/NodalArea;
333  Matrix& rNodalStress = itNode->FastGetSolutionStepValue(NODAL_EFFECTIVE_STRESS_TENSOR);
334  for(unsigned int i = 0; i<3; i++)
335  {
336  for(unsigned int j = 0; j<3; j++)
337  {
338  rNodalStress(i,j) *= InvNodalArea;
339  }
340  }
341  }
342 
343  const double& NodalJointArea = itNode->FastGetSolutionStepValue(NODAL_JOINT_AREA);
344  if (NodalJointArea>1.0e-20)
345  {
346  const double InvNodalJointArea = 1.0/NodalJointArea;
347  double& NodalJointWidth = itNode->FastGetSolutionStepValue(NODAL_JOINT_WIDTH);
348  NodalJointWidth *= InvNodalJointArea;
349  double& NodalMidPlaneLiquidPressure = itNode->FastGetSolutionStepValue(NODAL_MID_PLANE_LIQUID_PRESSURE);
350  NodalMidPlaneLiquidPressure *= InvNodalJointArea;
351  double& NodalSlipTendency = itNode->FastGetSolutionStepValue(NODAL_SLIP_TENDENCY);
352  NodalSlipTendency *= InvNodalJointArea;
353  double& NodalJointDamage = itNode->FastGetSolutionStepValue(NODAL_JOINT_DAMAGE);
354  NodalJointDamage *= InvNodalJointArea;
355  }
356  }
357  }
358  else
359  {
360  BaseType::FinalizeSolutionStep(rModelPart,A,Dx,b);
361  }
362 
363  KRATOS_CATCH("")
364  }

531  virtual inline void UpdateVariablesDerivatives(ModelPart& r_model_part)
532  {
533  KRATOS_TRY
534 
535  //Update Acceleration, Velocity and DtPressure
536 
537  array_1d<double,3> DeltaDisplacement;
538  double DeltaPressure;
539 
540  const int NNodes = static_cast<int>(r_model_part.Nodes().size());
541  ModelPart::NodesContainerType::iterator node_begin = r_model_part.NodesBegin();
542 
543  #pragma omp parallel for private(DeltaDisplacement,DeltaPressure)
544  for(int i = 0; i < NNodes; i++)
545  {
546  ModelPart::NodesContainerType::iterator itNode = node_begin + i;
547 
548  array_1d<double,3>& CurrentVelocity = itNode->FastGetSolutionStepValue(VELOCITY);
549  noalias(DeltaDisplacement) = itNode->FastGetSolutionStepValue(DISPLACEMENT) - itNode->FastGetSolutionStepValue(DISPLACEMENT, 1);
550  const array_1d<double,3>& PreviousVelocity = itNode->FastGetSolutionStepValue(VELOCITY, 1);
551 
552  // Note: Since acceleration is not used in the quasi-static scheme, here we use a GN11 scheme to update velocity.
553  // A GN22 is used in the dynamic scheme
554  noalias(CurrentVelocity) = 1.0/(mTheta_u*mDeltaTime)*(DeltaDisplacement - (1.0-mTheta_u)*mDeltaTime*PreviousVelocity);
555 
556  double& CurrentDtPressure = itNode->FastGetSolutionStepValue(DT_LIQUID_PRESSURE);
557  DeltaPressure = itNode->FastGetSolutionStepValue(LIQUID_PRESSURE) - itNode->FastGetSolutionStepValue(LIQUID_PRESSURE, 1);
558  const double& PreviousDtPressure = itNode->FastGetSolutionStepValue(DT_LIQUID_PRESSURE, 1);
559 
560  CurrentDtPressure = 1.0/(mTheta_p*mDeltaTime)*(DeltaPressure - (1.0-mTheta_p)*mDeltaTime*PreviousDtPressure);
561  }
562 
563  KRATOS_CATCH( "" )
564  }
