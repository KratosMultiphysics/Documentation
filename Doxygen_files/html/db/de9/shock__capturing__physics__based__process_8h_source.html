232  template<std::size_t TDim, std::size_t TNumNodes, class TTLSContainerType>
233  void CalculatePhysicsBasedShockCapturingElementContribution(
234  Element &rElement,
235  TTLSContainerType &rShockCapturingTLS)
236  {
237  auto &r_geom = rElement.GetGeometry();
238  const unsigned int n_nodes = r_geom.PointsNumber();
239 
240  // Initialize artificial magnitudes values
241  if (mShockSensor) {rElement.GetValue(ARTIFICIAL_BULK_VISCOSITY) = 0.0;}
242  if (mShearSensor) {rElement.GetValue(ARTIFICIAL_DYNAMIC_VISCOSITY) = 0.0;}
243  if (mThermalSensor || mShockSensor) {rElement.GetValue(ARTIFICIAL_CONDUCTIVITY) = 0.0;} // Note that conductivity is modified in both sensors
244 
245  // Get TLS values geometry values
246  double& r_vol = rShockCapturingTLS.Vol;
247  auto& r_DN_DX = rShockCapturingTLS.DN_DX;
248  auto& r_N = rShockCapturingTLS.N;
249  auto& r_metric_tensor = rShockCapturingTLS.MetricTensor;
250  auto& r_inv_metric_tensor = rShockCapturingTLS.InverseMetricTensor;
251 
252  // Calculate geometry data
253  GeometryUtils::CalculateGeometryData(r_geom, r_DN_DX, r_N, r_vol);
254 
255  // Calculate the geometry metric
256  double h_ref, metric_tensor_inf, metric_tensor_sup;
257  GeometryMetricCalculator<TDim,TNumNodes>::CalculateMetricTensorDimensionless(
258  r_geom,
259  r_metric_tensor,
260  h_ref,
261  metric_tensor_inf,
262  metric_tensor_sup);
263 
264  // Inverse metric tensor calculation
265  double aux_det;
266  MathUtils<double>::InvertMatrix(r_metric_tensor, r_inv_metric_tensor, aux_det);
267 
268  // Get fluid physical properties
269  const auto& r_prop = rElement.GetProperties();
270  const double c_v = r_prop.GetValue(SPECIFIC_HEAT);
271  const double gamma = r_prop.GetValue(HEAT_CAPACITY_RATIO);
272 
273  // Set auxiliary constants
274  const double k = 1.0; // Polynomial order of the numerical simulation
275  const double eps = std::numeric_limits<double>::epsilon(); // Small constant to avoid division by 0
276 
277  // Calculate the midpoint values
278  double midpoint_rho, formulation_midpoint_temp;
279  auto& r_midpoint_v = rShockCapturingTLS.MidpointVelocity;
280  if (mThermallyCoupledFormulation) {
281  // Get required midpoint values
282  double midpoint_temp;
283  FluidCalculationUtilities::EvaluateInPoint(r_geom, r_N, std::tie(r_midpoint_v, VELOCITY), std::tie(midpoint_rho, DENSITY), std::tie(midpoint_temp, TEMPERATURE));
284  // If the formulation is thermally coupled, the temperature for the stagnation temperature is the midpoint value
285  formulation_midpoint_temp = midpoint_temp;
286  } else {
287  // Get required midpoint values
288  double midpoint_p;
289  FluidCalculationUtilities::EvaluateInPoint(r_geom, r_N, std::tie(r_midpoint_v, VELOCITY), std::tie(midpoint_p, PRESSURE), std::tie(midpoint_rho, DENSITY));
290  // If the formulation is not energy coupled, the temperature for the stagnation temperature it's calculated by the state equation with the midpoint values
291  formulation_midpoint_temp = midpoint_p / (midpoint_rho * (gamma - 1.0) * c_v);
292  }
293 
294  // Calculate common values
295  const double v_norm_pow = SquaredArrayNorm(r_midpoint_v);
296  const double stagnation_temp = formulation_midpoint_temp + 0.5 * inner_prod(r_midpoint_v,r_midpoint_v)/(gamma * c_v);
297  const double critical_temp = stagnation_temp * (2.0 / (gamma + 1.0));
298  const double c_star = std::sqrt(gamma * (gamma - 1.0) * c_v * critical_temp); // Critical speed of sound
299  const double ref_mom_norm = midpoint_rho * std::sqrt(v_norm_pow + std::pow(c_star, 2));
300 
301  // Shock sensor values
302  if (mShockSensor) {
303  // Calculate the required differential operators
304  double div_v, mach;
305  auto& r_grad_rho = rShockCapturingTLS.DensityGradient;
306  auto& r_rot_v = rShockCapturingTLS.VelocityRotational;
307  CalculateShockSensorValues<TDim,TNumNodes>(r_geom, r_N, r_DN_DX, mach, div_v, r_grad_rho, r_rot_v);
308 
309  // Characteristic element size along the direction of the density gradient
310  const double h_beta = h_ref * norm_2(r_grad_rho) / std::sqrt(CalculateProjectedInverseMetricElementSize(r_inv_metric_tensor, r_grad_rho) + eps);
311 
312  // Dilatation sensor (activates in shock waves)
313  // Critical speed of sound might be zero if zero initial condition is prescribed (i.e. thermally uncoupled framework)
314  const double s_omega = c_star > 0.0 ? -h_beta * div_v / (k * c_star) : 0.0;
315 
316  // Vorticity sensor (vanishes in vorticity dominated regions)
317  const double div_v_pow = std::pow(div_v, 2);
318  const double rot_v_norm_pow = SquaredArrayNorm(r_rot_v);
319  const double s_w = div_v_pow / (div_v_pow + rot_v_norm_pow + eps);
320 
321  // Calculate limited shock sensor
322  const double s_beta_0 = 0.01;
323  const double s_beta_max = 2.0 / std::sqrt(std::pow(gamma, 2) - 1.0);
324  const double s_beta = s_omega * s_w;
325  const double s_beta_hat = SmoothedLimitingFunction(s_beta, s_beta_0, s_beta_max);
326  rElement.GetValue(SHOCK_SENSOR) = s_beta_hat;
327 
328  // Calculate elemental artificial bulk viscosity
329  const double elem_b_star = (mArtBulkViscosityConstant * h_beta / k) * ref_mom_norm * s_beta_hat;
330  rElement.GetValue(ARTIFICIAL_BULK_VISCOSITY) = elem_b_star;
331 
332  // Calculate elemental artificial conductivity (dilatancy)
333  // Note that this is only required if the formulation is thermally coupled
334  if (mThermallyCoupledFormulation) {
335  const double Pr_beta_min = 0.9;
336  const double alpha_pr_beta = 2.0;
337  const double mach_threshold = 3.0;
338  double ArtPrandtlNumber_beta;
339  if (mFarFieldPrandtlNumber < std::numeric_limits<double>::epsilon()) {
340  ArtPrandtlNumber_beta = Pr_beta_min * (1.0 + std::exp(-2.0 * alpha_pr_beta * (mach - mach_threshold)));
341  } else {
342  ArtPrandtlNumber_beta = mFarFieldPrandtlNumber;
343  }
344  const double elem_k1_star = (gamma * c_v / ArtPrandtlNumber_beta) * elem_b_star;
345  rElement.GetValue(ARTIFICIAL_CONDUCTIVITY) += elem_k1_star;
346  }
347  }
348 
349  if (mThermalSensor || mShearSensor) {
350  // Calculate Jacobian matrix (non-required for the shock sensor)
351  Matrix mid_pt_jacobian; //FIXME: We should use a bounded matrix in here
352  r_geom.Jacobian(mid_pt_jacobian, 0, GeometryData::IntegrationMethod::GI_GAUSS_1);
353 
354  // Thermal sensor values
355  if (mThermalSensor) {
356  // Calculate temperature gradients
357  auto& r_grad_temp = rShockCapturingTLS.TemperatureGradient;
358  auto& r_grad_temp_local = rShockCapturingTLS.TemperatureLocalGradient;
359  CalculateTemperatureGradients(r_geom, r_DN_DX, mid_pt_jacobian, r_grad_temp, r_grad_temp_local);
360 
361  // Characteristic element size along the direction of the temperature gradient
362  const double h_kappa = h_ref * norm_2(r_grad_temp) / std::sqrt(CalculateProjectedInverseMetricElementSize(r_inv_metric_tensor, r_grad_temp) + eps);
363 
364  // Thermal sensor (detect thermal gradients that are larger than possible with the grid resolution)
365  const double s_kappa_0 = 1.0;
366  const double s_kappa_max = 2.0;
367  const double s_kappa = h_ref * norm_2(r_grad_temp_local) / k / stagnation_temp;
368  const double s_kappa_hat = SmoothedLimitingFunction(s_kappa, s_kappa_0, s_kappa_max);
369  rElement.GetValue(THERMAL_SENSOR) = s_kappa_hat;
370 
371  // Calculate elemental artificial conductivity (thermal sensor)
372  const double elem_k2_star = (gamma * c_v) * (mArtConductivityConstant * h_kappa / k) * ref_mom_norm * s_kappa_hat;
373  rElement.GetValue(ARTIFICIAL_CONDUCTIVITY) += elem_k2_star;
374  }
375 
376  // Shear sensor values
377  if (mShearSensor) {
378  // Calculate shear sensor values
379  double r_c;
380  auto& r_local_shear_grad_v = rShockCapturingTLS.VelocityShearLocalGradient;
381  CalculateShearSensorValues(r_geom, r_N, r_DN_DX, mid_pt_jacobian, r_local_shear_grad_v, r_c);
382  BoundedMatrix<double,TDim,TDim> eigen_vect_mat, eigen_val_mat;
383  MathUtils<double>::GaussSeidelEigenSystem(r_local_shear_grad_v, eigen_vect_mat, eigen_val_mat);
384  double shear_spect_norm = 0.0;
385  for (unsigned int d = 0; d < eigen_val_mat.size1(); ++d) {
386  if (eigen_val_mat(d, d) > shear_spect_norm) {
387  shear_spect_norm = eigen_val_mat(d, d);
388  }
389  }
390  shear_spect_norm = std::sqrt(shear_spect_norm);
391 
392  // Characteristic element size for the shear sensor
393  const double h_mu = h_ref * metric_tensor_inf;
394 
395  // Shear sensor (detect velocity gradients that are larger than possible with the grid resolution)
396  const double isentropic_max_vel = std::sqrt(v_norm_pow + (2.0 / (gamma - 1.0)) * std::pow(r_c, 2));
397  const double s_mu_0 = 1.0;
398  const double s_mu_max = 2.0;
399  const double s_mu = h_ref * shear_spect_norm / isentropic_max_vel / k;
400  const double s_mu_hat = SmoothedLimitingFunction(s_mu, s_mu_0, s_mu_max);
401  rElement.GetValue(SHEAR_SENSOR) = s_mu_hat;
402 
403  // Calculate elemental artificial dynamic viscosity
404  const double elem_mu_star = (mArtDynViscosityConstant * h_mu / k) * ref_mom_norm * s_mu_hat;
405  rElement.GetValue(ARTIFICIAL_DYNAMIC_VISCOSITY) = elem_mu_star;
406  }
407  }
408 
409  // Project the shock capturing magnitudes to the nodes
410  const double aux_weight = r_vol / static_cast<double>(n_nodes);
411  for (unsigned int i_node = 0; i_node < n_nodes; ++i_node) {
412  auto &r_node = r_geom[i_node];
413  if (mShockSensor) {AtomicAdd(r_node.GetValue(ARTIFICIAL_BULK_VISCOSITY), aux_weight * rElement.GetValue(ARTIFICIAL_BULK_VISCOSITY));}
414  if (mShearSensor) {AtomicAdd(r_node.GetValue(ARTIFICIAL_DYNAMIC_VISCOSITY), aux_weight * rElement.GetValue(ARTIFICIAL_DYNAMIC_VISCOSITY));}
415  if (mShockSensor || mThermalSensor) {AtomicAdd(r_node.GetValue(ARTIFICIAL_CONDUCTIVITY), aux_weight * rElement.GetValue(ARTIFICIAL_CONDUCTIVITY));}
416  }
417  }
