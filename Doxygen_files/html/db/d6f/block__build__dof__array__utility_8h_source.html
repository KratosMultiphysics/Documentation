75  KRATOS_TRY;
76 
77  // Check that the provided DOF set is empty
78  if (rDofArray.size() != 0) {
79  KRATOS_WARNING_IF("BlockBuildDofArrayUtility", (EchoLevel > 1 && rModelPart.GetCommunicator().MyPID() == 0)) << "Provided DOF set is not empty. About to clear it." << std::endl;
80  rDofArray.clear();
81  }
82 
83  // Allocate auxiliary arrays
84  DofsVectorType dof_list;
85  DofsVectorType second_dof_list; // NOTE: The second dof list is only used on constraints to include master/slave relations
86 
87  AuxiliaryDofsSetType dof_global_set;
88  dof_global_set.reserve(rModelPart.NumberOfElements() * 20);
89 
90  KRATOS_INFO_IF("BlockBuildDofArrayUtility", (EchoLevel > 1 && rModelPart.GetCommunicator().MyPID() == 0)) << "Setting up the DOFs" << std::endl;
91  KRATOS_INFO_IF("BlockBuildDofArrayUtility", (EchoLevel > 2 && rModelPart.GetCommunicator().MyPID() == 0)) << "Number of threads: " << ParallelUtilities::GetNumThreads() << std::endl;
92 
93  // Fill the global DOF set array
94  #pragma omp parallel firstprivate(dof_list, second_dof_list)
95  {
96  const auto& r_current_process_info = rModelPart.GetProcessInfo();
97 
98  // We create the temporal set in current thread and we reserve some space on it
99  AuxiliaryDofsSetType dofs_tmp_set;
100  dofs_tmp_set.reserve(20000);
101 
102  // Add the DOFs from the model part elements
103  const auto& r_elements_array = rModelPart.Elements();
104  const int number_of_elements = static_cast<int>(r_elements_array.size());
105  KRATOS_INFO_IF("BlockBuildDofArrayUtility", EchoLevel > 2) << "Initializing elements loop" << std::endl;
106  #pragma omp for schedule(guided, 512) nowait
107  for (int i = 0; i < number_of_elements; ++i) {
108  // Get current element iterator
109  const auto it_elem = r_elements_array.cbegin() + i;
110 
111  // Gets list of DOF involved on every element
112  it_elem->GetDofList(dof_list, r_current_process_info);
113  dofs_tmp_set.insert(dof_list.begin(), dof_list.end());
114  }
115 
116  // Add the DOFs from the model part conditions
117  const auto& r_conditions_array = rModelPart.Conditions();
118  const int number_of_conditions = static_cast<int>(r_conditions_array.size());
119  KRATOS_INFO_IF("BlockBuildDofArrayUtility", EchoLevel > 2) << "Initializing conditions loop" << std::endl;
120  #pragma omp for schedule(guided, 512) nowait
121  for (int i = 0; i < number_of_conditions; ++i) {
122  // Get current condition iterator
123  const auto it_cond = r_conditions_array.cbegin() + i;
124 
125  // Gets list of DOF involved on every condition
126  it_cond->GetDofList(dof_list, r_current_process_info);
127  dofs_tmp_set.insert(dof_list.begin(), dof_list.end());
128  }
129 
130  // Add the DOFs from the model part constraints
131  const auto& r_constraints_array = rModelPart.MasterSlaveConstraints();
132  const int number_of_constraints = static_cast<int>(r_constraints_array.size());
133  KRATOS_INFO_IF("BlockBuildDofArrayUtility", EchoLevel > 2) << "Initializing constraints loop" << std::endl;
134  #pragma omp for schedule(guided, 512) nowait
135  for (int i = 0; i < number_of_constraints; ++i) {
136  // Get current constraint iterator
137  const auto it_const = r_constraints_array.cbegin() + i;
138 
139  // Gets list of DOF involved on every constraint
140  it_const->GetDofList(dof_list, second_dof_list, r_current_process_info);
141  dofs_tmp_set.insert(dof_list.begin(), dof_list.end());
142  dofs_tmp_set.insert(second_dof_list.begin(), second_dof_list.end());
143  }
144 
145  // Merge all the sets in one thread
146  #pragma omp critical
147  {
148  dof_global_set.insert(dofs_tmp_set.begin(), dofs_tmp_set.end());
149  }
150  }
151 
152  // Fill and sort the provided DOF array from the auxiliary global DOFs set
153  KRATOS_INFO_IF("BlockBuildDofArrayUtility", EchoLevel > 2) << "Initializing ordered array filling" << std::endl;
154  rDofArray.reserve(dof_global_set.size());
155  for (auto it = dof_global_set.begin(); it!= dof_global_set.end(); it++) {
156  rDofArray.push_back(*it);
157  }
158  rDofArray.Sort();
159 
160  // Throw an exception if there are no DOFs involved in the analysis
161  KRATOS_ERROR_IF(rDofArray.size() == 0) << "No degrees of freedom in model part: " << rModelPart.FullName() << std::endl;
162  KRATOS_INFO_IF("ResidualBasedBlockBuilderAndSolver", EchoLevel > 2 && rModelPart.GetCommunicator().MyPID() == 0) << "Finished setting up the DOFs" << std::endl;
163 
164 #ifdef KRATOS_DEBUG
165  // If reactions are to be calculated, we check if all the dofs have reactions defined
166  // This is to be done only in debug mode
167  if (CheckReactionDofs) {
168  for (auto& r_dof : rDofArray) {
169  KRATOS_ERROR_IF_NOT(r_dof.HasReaction())
170  << "Reaction variable not set for the following: " << std::endl
171  << "- Node: "<< r_dof.Id() << std::endl
172  << "- DOF: " << r_dof << std::endl;
173  }
