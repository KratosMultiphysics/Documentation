888  CoordinatesArrayType& PointLocalCoordinates(
889  CoordinatesArrayType& rResult,
890  const CoordinatesArrayType& rPoint
891  ) const override
892  {
893  rResult.clear();
894 
895  const TPointType& r_first_point = BaseType::GetPoint(0);
896  const TPointType& r_second_point = BaseType::GetPoint(1);
897 
898  // Project point
899  const double tolerance = 1e-14; // Tolerance
900 
901  const double length = Length();
902 
903  const double length_1 = std::sqrt( std::pow(rPoint[0] - r_first_point[0], 2)
904  + std::pow(rPoint[1] - r_first_point[1], 2) + std::pow(rPoint[2] - r_first_point[2], 2));
905 
906  const double length_2 = std::sqrt( std::pow(rPoint[0] - r_second_point[0], 2)
907  + std::pow(rPoint[1] - r_second_point[1], 2) + std::pow(rPoint[2] - r_second_point[2], 2));
908 
909  if (length_1 <= (length + tolerance) && length_2 <= (length + tolerance)) {
910  rResult[0] = 2.0 * length_1/(length + tolerance) - 1.0;
911  } else if (length_1 > (length + tolerance)) {
912  rResult[0] = 2.0 * length_1/(length + tolerance) - 1.0; // NOTE: The same value as before, but it will be > than 1
913  } else if (length_2 > (length + tolerance)) {
914  rResult[0] = 1.0 - 2.0 * length_2/(length + tolerance);
915  } else {
916  rResult[0] = 2.0; // Out of the line!!!
917  }
918 
919  return rResult ;
920  }
