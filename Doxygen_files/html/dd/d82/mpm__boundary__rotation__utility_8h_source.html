138  void ApplySlipCondition(TLocalMatrixType& rLocalMatrix,
139  TLocalVectorType& rLocalVector,
140  GeometryType& rGeometry) const override
141  {
142  const unsigned int LocalSize = rLocalVector.size();
143 
144  if (LocalSize > 0)
145  {
146  for(unsigned int itNode = 0; itNode < rGeometry.PointsNumber(); ++itNode)
147  {
148  if(this->IsSlip(rGeometry[itNode]) )
149  {
150  // We fix the first displacement dof (normal component) for each rotated block
151  unsigned int j = itNode * this->GetBlockSize();
152 
153  // Get the displacement of the boundary mesh, this does not assume that the mesh is moving.
154  // If the mesh is moving, need to consider the displacement of the moving mesh into account.
155  const array_1d<double,3> & displacement = rGeometry[itNode].FastGetSolutionStepValue(DISPLACEMENT);
156 
157  // Get Normal Vector of the boundary
158  array_1d<double,3> rN = rGeometry[itNode].FastGetSolutionStepValue(NORMAL);
159  this->Normalize(rN);
160 
161  // Zero out row/column corresponding to normal displacement DoF except diagonal term (set to 1)
162  // Applied IFF the local matrix passed is not empty [otherwise does nothing -- RHS only case]
163  if (rLocalMatrix.size1() != 0) {
164  for( unsigned int i = 0; i < LocalSize; ++i)
165  {
166  rLocalMatrix(i,j) = 0.0;
167  rLocalMatrix(j,i) = 0.0;
168  }
169  rLocalMatrix(j, j) = 1.0; // set diagonal term to 1.0
170  }
171 
172  // Modifications to introduce friction force
173  const double mu = rGeometry[itNode].GetValue(FRICTION_COEFFICIENT);
174  const double tangential_penalty_factor = rGeometry[itNode].GetValue(TANGENTIAL_PENALTY_FACTOR);
175 
176  if (mu > 0 && tangential_penalty_factor > 0) { // Friction active
177  array_1d<double,3> & r_stick_force = rGeometry[itNode].FastGetSolutionStepValue(STICK_FORCE);
178 
179  // accumulate normal forces (RHS vector) for subsequent re-determination of friction state
180  // and check if friction force has been computed for the current node
181  rGeometry[itNode].SetLock();
182  rGeometry[itNode].FastGetSolutionStepValue(STICK_FORCE)[0] -= rLocalVector[j];
183  const bool friction_assigned = rGeometry[itNode].GetValue(FRICTION_ASSIGNED);
184  rGeometry[itNode].SetValue(FRICTION_ASSIGNED, true);
185  rGeometry[itNode].UnSetLock();
186 
187  if (!friction_assigned) {
188  // obtain displacement in normal-tangential coordinates
189  // [ use a copy and not reference to avoid rotating the displacement in SolutionStepValue ]
190  array_1d<double,3> displacement_copy = rGeometry[itNode].FastGetSolutionStepValue(DISPLACEMENT);
191  RotateVector(displacement_copy, rGeometry[itNode]);
192 
193  // determine penalty-based sticking force in the tangential direction for the current node
194  // [ displacement in MPM is the displacement update -> penalize any and all displacement
195  // updates in the tangential direction [this assumes a stationary background grid]
196  for( unsigned int i = 1; i < this->GetDomainSize(); ++i) {
197  r_stick_force[i] = -tangential_penalty_factor * displacement_copy[i];
198  }
199 
200  const int friction_state = rGeometry[itNode].FastGetSolutionStepValue(FRICTION_STATE);
201 
202  if(friction_state == STICK) {
203  if (rLocalMatrix.size1() != 0) {
204  for (unsigned int i = 1; i < this->GetDomainSize(); ++i)
205  rLocalMatrix(j + i, j + i) += tangential_penalty_factor;
206  }
207  } // else, sliding: nothing needed for LHS
208 
209  // add friction to RHS
210  const array_1d<double,3> & r_friction_force = rGeometry[itNode].FastGetSolutionStepValue(REACTION);
211 
212  for( unsigned int i = 1; i < this->GetDomainSize(); ++i)
213  rLocalVector[j + i] += r_friction_force[i];
214  }
215  }
216 
217  // Set value of normal displacement at node directly to the normal displacement of the boundary mesh
218  rLocalVector[j] = inner_prod(rN,displacement);
219  }
220  }
221  }

259  // An extra function to distinguish the application of slip in condition considering penalty imposition
260  void ConditionApplySlipCondition(TLocalMatrixType& rLocalMatrix,
261  TLocalVectorType& rLocalVector,
262  GeometryType& rGeometry) const
263  {
264  // If it is not a penalty condition, do as standard
265  if (!this->IsPenalty(rGeometry))
266  {
267  this->ApplySlipCondition(rLocalMatrix, rLocalVector, rGeometry);
268  }
269  // Otherwise, do the following modification
270  else
271  {
272  const unsigned int LocalSize = rLocalVector.size();
273 
274  if (LocalSize > 0)
275  {
276  const unsigned int block_size = this->GetBlockSize();
277  TLocalMatrixType temp_matrix = ZeroMatrix(rLocalMatrix.size1(),rLocalMatrix.size2());
278  for(unsigned int itNode = 0; itNode < rGeometry.PointsNumber(); ++itNode)
279  {
280  if(this->IsSlip(rGeometry[itNode]) )
281  {
282  // We fix the first displacement dof (normal component) for each rotated block
283  unsigned int j = itNode * block_size;
284 
285  // Copy all normal value in LHS to the temp_matrix
286  // [ does nothing for dummy rLocalMatrix (size1() == 0) -- RHS only case ]
287  for (unsigned int i = j; i < rLocalMatrix.size1(); i+= block_size)
288  {
289  temp_matrix(i,j) = rLocalMatrix(i,j);
290  temp_matrix(j,i) = rLocalMatrix(j,i);
291  }
292 
293  // Remove all other value in RHS than the normal component
294  for(unsigned int i = j; i < (j + block_size); ++i)
295  {
296  if (i!=j) rLocalVector[i] = 0.0;
297  }
298  }
299  }
300  // All entries in penalty matrix zeroed out except for normal component
301  // [ no effect in case of empty dummy rLocalMatrix ]
302  rLocalMatrix = temp_matrix;
303  }
304  }
305  }

447  {

449  {
450  BoundedMatrix<double,3,3> rRot;
451  this->LocalRotationOperatorPure(rRot,*itNode);

454  // rotate reaction to local frame

456  noalias(local_reaction) = prod(rRot,reaction);

460  local_reaction[2]=0.0;

463  // save values
464  for(unsigned int i = 0; i < 3; i++) rReaction[i] = global_reaction[i];
465  }

468  BoundedMatrix<double,2,2> rRot;
469  this->LocalRotationOperatorPure(rRot,*itNode);
470 
471  array_1d<double,3>& rReaction = itNode->FastGetSolutionStepValue(REACTION);
472  // rotate reaction to local frame

527  // Sets FRICTION_STATE for a SLIP node to indicate its stick/sliding state and stores the friction force in REACTION
528  void ComputeFrictionAndResetFlags(ModelPart& rModelPart) const {
529  const bool is_initial_loop = !rModelPart.GetProcessInfo()[INITIAL_LOOP_COMPLETE];
530 
531  block_for_each(rModelPart.Nodes(), [&](NodeType& rNode)
532  {
533  const Node& rConstNode = rNode; // const Node reference to avoid issues with previously unset GetValue()
534 
535  int& r_friction_state = rNode.FastGetSolutionStepValue(FRICTION_STATE, 0);
536  const double mu = rConstNode.GetValue(FRICTION_COEFFICIENT);
537 
538  array_1d<double, 3>& r_friction_force = rNode.FastGetSolutionStepValue(REACTION);
539 
540  // Limit tangential forces for friction
541  if (mu > 0) {
542  // obtain normal and tangent forces assoc. with node at the desired timestep
543  const double normal_force = rNode.FastGetSolutionStepValue(STICK_FORCE_X, 1);
544  const double tangent_force1 = rNode.FastGetSolutionStepValue(STICK_FORCE_Y, 0);
545  const double tangent_force2 = rNode.FastGetSolutionStepValue(STICK_FORCE_Z, 0);
546 
547  // forces unmodified in normal direction
548  r_friction_force[0] = normal_force;
549 
550  // [note: no friction if normal component < 0 [direction chosen to be consistent with contact algo]]
551  // [since normal point away from the boundary/interface, a normal component < 0 indicates a force
552  // pulling towards the normal (i.e. tensile force)]
553  const double normal_force_norm = fmax(normal_force, 0.0);
554 
555  const double tangent_force_norm = sqrt(tangent_force1 * tangent_force1 + tangent_force2 * tangent_force2);
556  const double max_tangent_force_norm = normal_force_norm * mu;
557 
558  // special treatment for initial loop
559  if (is_initial_loop) {
560  if (rConstNode.GetValue(HAS_INITIAL_MOMENTUM)) {
561  r_friction_state = SLIDING;
562  }
563  else {
564  r_friction_state = STICK;
565  }
566  } else {
567  r_friction_state = (tangent_force_norm >= max_tangent_force_norm) ? SLIDING : STICK;
568  }
569 
570  if (r_friction_state == SLIDING) {
571  double tangent_force_dir1 = 0.0;
572  double tangent_force_dir2 = 0.0;
573 
574  if (tangent_force_norm > std::numeric_limits<double>::epsilon()) {
575  tangent_force_dir1 = tangent_force1 / tangent_force_norm;
576  tangent_force_dir2 = tangent_force2 / tangent_force_norm;
577  }
578 
579  r_friction_force[1] = tangent_force_dir1 * max_tangent_force_norm;
580  r_friction_force[2] = tangent_force_dir2 * max_tangent_force_norm;
581  }
582  else { // STICK
583  r_friction_force[1] = tangent_force1;
584  r_friction_force[2] = tangent_force2;
585  }
586 
587  // reset friction-related flags/variables
588  rNode.SetValue(FRICTION_ASSIGNED, false);
589  rNode.FastGetSolutionStepValue(STICK_FORCE).clear();
590  }
591  });
592  }
