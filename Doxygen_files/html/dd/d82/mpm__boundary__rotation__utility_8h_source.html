149  void ApplySlipCondition(TLocalMatrixType& rLocalMatrix,
150  TLocalVectorType& rLocalVector,
151  GeometryType& rGeometry) const override
152  {
153  const unsigned int LocalSize = rLocalVector.size();
154 
155  if (LocalSize > 0)
156  {
157  for(unsigned int itNode = 0; itNode < rGeometry.PointsNumber(); ++itNode)
158  {
159  // Checks for conforming slip
160  // [ non-conforming SLIP BCs are treated within the resp. condition itself ]
161  if( this->IsConformingSlip(rGeometry[itNode]) )
162  {
163  // We fix the first displacement dof (normal component) for each rotated block
164  unsigned int j = itNode * this->GetBlockSize();
165 
166  // Get the displacement of the boundary mesh, this does not assume that the mesh is moving.
167  // If the mesh is moving, need to consider the displacement of the moving mesh into account.
168  const array_1d<double,3> & displacement = rGeometry[itNode].FastGetSolutionStepValue(DISPLACEMENT);
169 
170  // Get Normal Vector of the boundary
171  array_1d<double,3> rN = rGeometry[itNode].FastGetSolutionStepValue(NORMAL);
172  this->Normalize(rN);
173 
174  // Zero out row/column corresponding to normal displacement DoF except diagonal term (set to 1)
175  // Applied IFF the local matrix passed is not empty [otherwise does nothing -- RHS only case]
176  if (rLocalMatrix.size1() != 0) {
177  for( unsigned int i = 0; i < LocalSize; ++i)
178  {
179  rLocalMatrix(i,j) = 0.0;
180  rLocalMatrix(j,i) = 0.0;
181  }
182  rLocalMatrix(j, j) = 1.0; // set diagonal term to 1.0
183  }
184 
185  // Modifications to introduce friction force
186  const double mu = rGeometry[itNode].GetValue(FRICTION_COEFFICIENT);
187  const double tangential_penalty_factor = rGeometry[itNode].GetValue(TANGENTIAL_PENALTY_FACTOR);
188 
189  if (mu > 0 && tangential_penalty_factor > 0) { // Friction active
190  array_1d<double,3> & r_stick_force = rGeometry[itNode].FastGetSolutionStepValue(STICK_FORCE);
191 
192  // accumulate normal forces (RHS vector) for subsequent re-determination of friction state
193  // and check if friction force has been computed for the current node
194  rGeometry[itNode].SetLock();
195  rGeometry[itNode].FastGetSolutionStepValue(STICK_FORCE)[0] -= rLocalVector[j];
196  const bool friction_assigned = rGeometry[itNode].GetValue(FRICTION_ASSIGNED);
197  rGeometry[itNode].SetValue(FRICTION_ASSIGNED, true);
198  rGeometry[itNode].UnSetLock();
199 
200  if (!friction_assigned) {
201  // obtain displacement in normal-tangential coordinates
202  // [ use a copy and not reference to avoid rotating the displacement in SolutionStepValue ]
203  array_1d<double,3> displacement_copy = rGeometry[itNode].FastGetSolutionStepValue(DISPLACEMENT);
204  RotateVector(displacement_copy, rGeometry[itNode]);
205 
206  // determine penalty-based sticking force in the tangential direction for the current node
207  // [ displacement in MPM is the displacement update -> penalize any and all displacement
208  // updates in the tangential direction [this assumes a stationary background grid]
209  for( unsigned int i = 1; i < this->GetDomainSize(); ++i) {
210  r_stick_force[i] = -tangential_penalty_factor * displacement_copy[i];
211  }
212 
213  const int friction_state = rGeometry[itNode].FastGetSolutionStepValue(FRICTION_STATE);
214 
215  if(friction_state == STICK) {
216  if (rLocalMatrix.size1() != 0) {
217  for (unsigned int i = 1; i < this->GetDomainSize(); ++i)
218  rLocalMatrix(j + i, j + i) += tangential_penalty_factor;
219  }
220  } // else, sliding: nothing needed for LHS
221 
222  // add friction to RHS
223  const array_1d<double,3> & r_friction_force = rGeometry[itNode].FastGetSolutionStepValue(REACTION);
224 
225  for( unsigned int i = 1; i < this->GetDomainSize(); ++i)
226  rLocalVector[j + i] += r_friction_force[i];
227  }
228  }
229 
230  // Set value of normal displacement at node directly to the normal displacement of the boundary mesh
231  rLocalVector[j] = inner_prod(rN,displacement);
232  }
233  }
234  }

360  void CalculateReactionForces(ModelPart& rModelPart) const
361  {
362  TLocalVectorType global_reaction(this->GetDomainSize());
363  TLocalVectorType local_reaction(this->GetDomainSize());
364  TLocalVectorType reaction(this->GetDomainSize());
365 
366  ModelPart::NodeIterator it_begin = rModelPart.NodesBegin();
367  #pragma omp parallel for firstprivate(reaction,local_reaction,global_reaction)
368  for(int iii=0; iii<static_cast<int>(rModelPart.Nodes().size()); iii++)
369  {
370  ModelPart::NodeIterator itNode = it_begin+iii;
371  const double nodal_mass = itNode->FastGetSolutionStepValue(NODAL_MASS, 0);
372 
373  if( this->IsConformingSlip(*itNode) && nodal_mass > std::numeric_limits<double>::epsilon())
374  {
375  if(this->GetDomainSize() == 3)
376  {
377  BoundedMatrix<double,3,3> rRot;
378  this->LocalRotationOperatorPure(rRot,*itNode);
379 
380  array_1d<double,3>& rReaction = itNode->FastGetSolutionStepValue(REACTION);
381  // rotate reaction to local frame
382  for(unsigned int i = 0; i < 3; i++) reaction[i] = rReaction[i];
383  noalias(local_reaction) = prod(rRot,reaction);
384 
385  // remove tangential directions
386  local_reaction[1]=0.0;
387  local_reaction[2]=0.0;
388  // rotate reaction to global frame
389  noalias(global_reaction) = prod(trans(rRot),local_reaction);
390  // save values
391  for(unsigned int i = 0; i < 3; i++) rReaction[i] = global_reaction[i];
392  }
393  else
394  {
395  BoundedMatrix<double,2,2> rRot;
396  this->LocalRotationOperatorPure(rRot,*itNode);
397 
398  array_1d<double,3>& rReaction = itNode->FastGetSolutionStepValue(REACTION);
399  // rotate reaction to local frame
400  for(unsigned int i = 0; i < 2; i++) reaction[i] = rReaction[i];
401  noalias(local_reaction) = prod(rRot,reaction);
402 
403  // remove tangential direction
404  local_reaction[1]=0.0;
405  // rotate reaction to global frame
406  noalias(global_reaction) = prod(trans(rRot),local_reaction);

454  // Sets FRICTION_STATE for a SLIP node to indicate its stick/sliding state and stores the friction force in REACTION
455  void ComputeFrictionAndResetFlags(ModelPart& rModelPart) const {
456  const bool is_initial_loop = !rModelPart.GetProcessInfo()[INITIAL_LOOP_COMPLETE];
457 
458  block_for_each(rModelPart.Nodes(), [&](NodeType& rNode)
459  {
460  const Node& rConstNode = rNode; // const Node reference to avoid issues with previously unset GetValue()
461 
462  int& r_friction_state = rNode.FastGetSolutionStepValue(FRICTION_STATE, 0);
463  const double mu = rConstNode.GetValue(FRICTION_COEFFICIENT);
464 
465  array_1d<double, 3>& r_friction_force = rNode.FastGetSolutionStepValue(REACTION);
466 
467  // Limit tangential forces for friction
468  if (mu > 0) {
469  // obtain normal and tangent forces assoc. with node at the desired timestep
470  const double normal_force = rNode.FastGetSolutionStepValue(STICK_FORCE_X, 1);
471  const double tangent_force1 = rNode.FastGetSolutionStepValue(STICK_FORCE_Y, 0);
472  const double tangent_force2 = rNode.FastGetSolutionStepValue(STICK_FORCE_Z, 0);
473 
474  // forces unmodified in normal direction
475  r_friction_force[0] = normal_force;
476 
477  // [note: no friction if normal component < 0 [direction chosen to be consistent with contact algo]]
478  // [since normal point away from the boundary/interface, a normal component < 0 indicates a force
479  // pulling towards the normal (i.e. tensile force)]
480  const double normal_force_norm = fmax(normal_force, 0.0);
481 
482  const double tangent_force_norm = sqrt(tangent_force1 * tangent_force1 + tangent_force2 * tangent_force2);
483  const double max_tangent_force_norm = normal_force_norm * mu;
484 
485  // special treatment for initial loop
486  if (is_initial_loop) {
487  if (rConstNode.GetValue(HAS_INITIAL_MOMENTUM)) {
488  r_friction_state = SLIDING;
489  }
490  else {
491  r_friction_state = STICK;
492  }
493  } else {
494  r_friction_state = (tangent_force_norm >= max_tangent_force_norm) ? SLIDING : STICK;
495  }
496 
497  if (r_friction_state == SLIDING) {
498  double tangent_force_dir1 = 0.0;
499  double tangent_force_dir2 = 0.0;
500 
501  if (tangent_force_norm > std::numeric_limits<double>::epsilon()) {
502  tangent_force_dir1 = tangent_force1 / tangent_force_norm;
503  tangent_force_dir2 = tangent_force2 / tangent_force_norm;
504  }
505 
506  r_friction_force[1] = tangent_force_dir1 * max_tangent_force_norm;
507  r_friction_force[2] = tangent_force_dir2 * max_tangent_force_norm;
508  }
509  else { // STICK
510  r_friction_force[1] = tangent_force1;
511  r_friction_force[2] = tangent_force2;
512  }
513 
514  // reset friction-related flags/variables
515  rNode.SetValue(FRICTION_ASSIGNED, false);
516  rNode.FastGetSolutionStepValue(STICK_FORCE).clear();
517  }
518  });
519  }
