177  static inline void InvertMatrix2(BoundedMatrix<double, 2, 2>& rInvertedMatrix,
178  const BoundedMatrix<double, 2, 2>& InputMatrix,
179  double& InputMatrixDet)
180  {
181  KRATOS_TRY
182 
183  const double numerical_limit = std::numeric_limits<double>::epsilon();
184 
185  InputMatrixDet = InputMatrix(0, 0) * InputMatrix(1, 1) - InputMatrix(0, 1) * InputMatrix(1, 0);
186 
187  if (InputMatrixDet < numerical_limit) {
188  KRATOS_ERROR << "determinant zero or negative" << std::endl;
189  }
190 
191  rInvertedMatrix(0, 0) = InputMatrix(1, 1) / InputMatrixDet;
192  rInvertedMatrix(0, 1) = -InputMatrix(0, 1) / InputMatrixDet;
193  rInvertedMatrix(1, 0) = -InputMatrix(1, 0) / InputMatrixDet;
194  rInvertedMatrix(1, 1) = InputMatrix(0, 0) / InputMatrixDet;
195 
196  KRATOS_CATCH("")
197  }

324  static inline void CalculateEquallyDistributedPointsLineShapeFunctions3N(Matrix& NContainer)
325  {
326  // Line 3-noded
327  // nodes:
328  // 0------2------1
329  const unsigned int NumNodes = 3;
330  const std::vector<double> Xi{-1.0 / 3.0, 1.0 / 3.0};
331 
332  if (NContainer.size1() != Xi.size() || NContainer.size2() != NumNodes)
333  NContainer.resize(Xi.size(), NumNodes, false);
334 
335  for (unsigned int integrationPoint = 0; integrationPoint < Xi.size(); ++integrationPoint) {
336  const double& X = Xi[integrationPoint];
337  NContainer(integrationPoint, 0) = -0.5 * (1.0 - X) * X;
338  NContainer(integrationPoint, 1) = 0.5 * (1.0 + X) * X;
339  NContainer(integrationPoint, 2) = (1.0 + X) * (1.0 - X);

616  static Vector CalculateNodalHydraulicHeadFromWaterPressures(const GeometryType& rGeom, const Properties& rProp)
617  {
618  const auto NumericalLimit = std::numeric_limits<double>::epsilon();
619  // Defining necessary variables
620 
621  Vector nodal_hydraulic_heads(rGeom.PointsNumber());
622  for (unsigned int node = 0; node < rGeom.PointsNumber(); ++node) {
623  array_1d<double, 3> node_volume_acceleration;
624  noalias(node_volume_acceleration) = rGeom[node].FastGetSolutionStepValue(VOLUME_ACCELERATION, 0);
625  const double g = norm_2(node_volume_acceleration);
626  if (g > NumericalLimit) {
627  const double FluidWeight = g * rProp[DENSITY_WATER];
628 
629  array_1d<double, 3> NodeCoordinates;
630  noalias(NodeCoordinates) = rGeom[node].Coordinates();
631  array_1d<double, 3> NodeVolumeAccelerationUnitVector;
632  noalias(NodeVolumeAccelerationUnitVector) = node_volume_acceleration / g;
633 
634  const double WaterPressure = rGeom[node].FastGetSolutionStepValue(WATER_PRESSURE);
635  nodal_hydraulic_heads[node] = -inner_prod(NodeCoordinates, NodeVolumeAccelerationUnitVector) -
636  PORE_PRESSURE_SIGN_FACTOR * WaterPressure / FluidWeight;
637  } else {
638  nodal_hydraulic_heads[node] = 0.0;
639  }
640  }
641  return nodal_hydraulic_heads;
642  }
