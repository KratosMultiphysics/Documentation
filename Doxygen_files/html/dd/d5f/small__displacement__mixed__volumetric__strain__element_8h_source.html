796  template<class TType>
797  void CalculateOnConstitutiveLaw(
798  const Variable<TType>& rVariable,
799  std::vector<TType>& rOutput,
800  const ProcessInfo& rCurrentProcessInfo)
801  {
802  const auto& r_geometry = GetGeometry();
803  const SizeType n_nodes = r_geometry.size();
804  const SizeType dim = r_geometry.WorkingSpaceDimension();
805  const SizeType strain_size = mConstitutiveLawVector[0]->GetStrainSize();
806  const SizeType n_gauss = r_geometry.IntegrationPointsNumber(GetIntegrationMethod());
807  const auto& r_integration_points = r_geometry.IntegrationPoints(GetIntegrationMethod());
808 
809  // Create the kinematics container and fill the nodal data
810  KinematicVariables kinematic_variables(strain_size, dim, n_nodes);
811  for (IndexType i_node = 0; i_node < n_nodes; ++i_node) {
812  const auto& r_disp = r_geometry[i_node].FastGetSolutionStepValue(DISPLACEMENT);
813  for (IndexType d = 0; d < dim; ++d) {
814  kinematic_variables.Displacements(i_node * dim + d) = r_disp[d];
815  }
816  kinematic_variables.VolumetricNodalStrains[i_node] = r_geometry[i_node].FastGetSolutionStepValue(VOLUMETRIC_STRAIN);
817  }
818 
819  // Create the constitutive variables and values containers
820  ConstitutiveVariables constitutive_variables(strain_size);
821  ConstitutiveLaw::Parameters cons_law_values(r_geometry, GetProperties(), rCurrentProcessInfo);
822  auto& r_cons_law_options = cons_law_values.GetOptions();
823  r_cons_law_options.Set(ConstitutiveLaw::COMPUTE_STRESS, true);
824  r_cons_law_options.Set(ConstitutiveLaw::USE_ELEMENT_PROVIDED_STRAIN, true);
825  r_cons_law_options.Set(ConstitutiveLaw::COMPUTE_CONSTITUTIVE_TENSOR, false);
826 
827  // Call the initialize material response
828  for (IndexType i_gauss = 0; i_gauss < n_gauss; ++i_gauss) {
829  // Recompute the kinematics
830  CalculateKinematicVariables(kinematic_variables, i_gauss, GetIntegrationMethod());
831 
832  // Set the constitutive variables
833  SetConstitutiveVariables(kinematic_variables, constitutive_variables, cons_law_values, i_gauss, r_integration_points);
834 
835  // Calculate the output value
836  rOutput[i_gauss] = mConstitutiveLawVector[i_gauss]->CalculateValue(cons_law_values, rVariable, rOutput[i_gauss] );
837  }
838  }
