118  double CalculateElementValue(Element& elem_i, const std::size_t DomainSize){
119  // We get the element geometry
120  auto& r_this_geometry = elem_i.GetGeometry();
121  const std::size_t number_of_nodes = r_this_geometry.size();
122 
123  // We copy the current coordinates and move the coordinates to the initial configuration
124  std::vector<array_1d<double, 3>> current_coordinates(number_of_nodes);
125  for (std::size_t i_node = 0; i_node < number_of_nodes; ++i_node) {

129

131  double element_val = elem_i.GetGeometry().Volume() * std::pow((element_density) * (1.0-element_density),2);

133  // We restore the current configuration

175  double curr_density = elem_i.GetProperties().GetValue(DENSITY);
