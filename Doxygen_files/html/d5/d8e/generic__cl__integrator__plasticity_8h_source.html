160  static void IntegrateStressVector(
161  array_1d<double, VoigtSize>& rPredictiveStressVector,
162  Vector& rStrainVector,
163  double& rUniaxialStress,
164  double& rThreshold,
165  double& rPlasticDenominator,
166  array_1d<double, VoigtSize>& rFflux,
167  array_1d<double, VoigtSize>& rGflux,
168  double& rPlasticDissipation,
169  array_1d<double, VoigtSize>& rPlasticStrainIncrement,
170  Matrix& rConstitutiveMatrix,
171  Vector& rPlasticStrain,
172  ConstitutiveLaw::Parameters& rValues,
173  const double CharacteristicLength
174  )
175  {
176  // Material properties
177  const Properties& r_material_properties = rValues.GetMaterialProperties();
178 
179  bool is_converged = false;
180  IndexType iteration = 0, max_iter = r_material_properties.Has(MAX_NUMBER_NL_CL_ITERATIONS)
181  ? r_material_properties.GetValue(MAX_NUMBER_NL_CL_ITERATIONS) : 100;
182  array_1d<double, VoigtSize> delta_sigma;
183  double plastic_consistency_factor_increment;
184  double F = rUniaxialStress - rThreshold;
185  const bool analytic_tangent = (r_material_properties.Has(TANGENT_OPERATOR_ESTIMATION) && r_material_properties[TANGENT_OPERATOR_ESTIMATION] == 0) ? true : false;
186 
187  // Backward Euler
188  while (!is_converged && iteration <= max_iter) {
189  plastic_consistency_factor_increment = F * rPlasticDenominator;
190  plastic_consistency_factor_increment = (plastic_consistency_factor_increment < 0.0) ? 0.0 : plastic_consistency_factor_increment;
191  noalias(rPlasticStrainIncrement) = plastic_consistency_factor_increment * rGflux;
192  noalias(rPlasticStrain) += rPlasticStrainIncrement;
193  noalias(delta_sigma) = prod(rConstitutiveMatrix, rPlasticStrainIncrement);
194 
195  noalias(rPredictiveStressVector) -= delta_sigma;
196 
197  F = CalculatePlasticParameters(rPredictiveStressVector, rStrainVector, rUniaxialStress, rThreshold, rPlasticDenominator, rFflux, rGflux, rPlasticDissipation, rPlasticStrainIncrement, rConstitutiveMatrix, rValues, CharacteristicLength, rPlasticStrain);
198 
199  if (F <= std::abs(1.0e-4 * rThreshold)) { // Has converged
200  is_converged = true;
201  } else {
202  iteration++;
203  }
204  }
205  if (analytic_tangent) {
206  Matrix tangent_tensor;
207  tangent_tensor.resize(VoigtSize, VoigtSize, false);
208  CalculateTangentMatrix(tangent_tensor, rConstitutiveMatrix, rFflux, rGflux, rPlasticDenominator);
209  noalias(rConstitutiveMatrix) = tangent_tensor;
210  }
211  KRATOS_WARNING_IF("GenericConstitutiveLawIntegratorPlasticity", iteration > max_iter) << "Maximum number of iterations in plasticity loop reached..." << std::endl;
212  }

329  static void CalculateIndicatorsFactors(
330  const array_1d<double, VoigtSize>& rPredictiveStressVector,
331  double& rTensileIndicatorFactor,
332  double& rCompressionIndicatorFactor
333  )
334  {
335  // We do an initial check
336  if (norm_2(rPredictiveStressVector) < 1.0e-8) {
337  rTensileIndicatorFactor = 1.0;
338  rCompressionIndicatorFactor = 0.0;
339  return;
340  }
341 
342  // We proceed as usual
343  array_1d<double, Dimension> principal_stresses = ZeroVector(Dimension);
344  AdvancedConstitutiveLawUtilities<VoigtSize>::CalculatePrincipalStresses(principal_stresses, rPredictiveStressVector);
345 
346  double suma = 0.0, sumb = 0.0, sumc = 0.0;
347  double aux_sa;
348 
349  for (IndexType i = 0; i < Dimension; ++i) {
350  aux_sa = std::abs(principal_stresses[i]);
351  suma += aux_sa;
352  sumb += 0.5 * (principal_stresses[i] + aux_sa);
353  sumc += 0.5 * (-principal_stresses[i] + aux_sa);
354  }
355 
356  if (std::abs(suma) > tolerance) {
357  rTensileIndicatorFactor = sumb / suma;
358  rCompressionIndicatorFactor = sumc / suma;
359  } else {
360  rTensileIndicatorFactor = sumb;
361  rCompressionIndicatorFactor = sumc;
362  }
363 
364  // Final check
365  if ((std::abs(rTensileIndicatorFactor) + std::abs(rCompressionIndicatorFactor)) < tolerance) {
366  rTensileIndicatorFactor = 0.0;
367  rCompressionIndicatorFactor = 0.0;
368  return;
369  }
370  }

383  static void CalculatePlasticDissipation(
384  const array_1d<double, VoigtSize>& rPredictiveStressVector,
385  const double TensileIndicatorFactor,
386  const double CompressionIndicatorFactor,
387  const Vector& PlasticStrainInc,
388  double& rPlasticDissipation,
389  array_1d<double, VoigtSize>& rHCapa,
390  ConstitutiveLaw::Parameters& rValues,
391  const double CharacteristicLength
392  )
393  {
394  const Properties& r_material_properties = rValues.GetMaterialProperties();
395 
396  const double young_modulus = r_material_properties[YOUNG_MODULUS];
397  const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
398  const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
399  const double yield_tension = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
400  const double n = yield_compression / yield_tension;
401  const double fracture_energy_tension = r_material_properties[FRACTURE_ENERGY]; // Frac energy in tension
402  const double fracture_energy_compression = r_material_properties[FRACTURE_ENERGY] * std::pow(n, 2); // Frac energy in compression
403 
404  const double characteristic_fracture_energy_tension = fracture_energy_tension / CharacteristicLength;
405  const double characteristic_fracture_energy_compression = fracture_energy_compression / CharacteristicLength;
406 
407  const double hlim = 2.0 * young_modulus * fracture_energy_compression / (std::pow(yield_compression, 2));
408  KRATOS_ERROR_IF(CharacteristicLength > hlim) << "The Fracture Energy is to low: " << characteristic_fracture_energy_compression << std::endl;
409 
410  double constant0 = 0.0, constant1 = 0.0, dplastic_dissipation = 0.0;
411  if (characteristic_fracture_energy_tension > 0.000001) {
412  constant0 = TensileIndicatorFactor / characteristic_fracture_energy_tension;
413  constant1 = CompressionIndicatorFactor / characteristic_fracture_energy_compression;
414  }
415  const double constant = constant0 + constant1;
416 
417  for (IndexType i = 0; i < VoigtSize; ++i) {
418  rHCapa[i] = constant * rPredictiveStressVector[i];
419  dplastic_dissipation += rHCapa[i] * PlasticStrainInc[i];
420  }
421 
422  if (dplastic_dissipation < 0.0 || dplastic_dissipation > 1.0)
423  dplastic_dissipation = 0.0;
424 
425  rPlasticDissipation += dplastic_dissipation;
426  if (rPlasticDissipation >= 0.9999)
427  rPlasticDissipation = 0.9999;
428  else if (rPlasticDissipation < 0.0)
429  rPlasticDissipation = 0.0;
430 
431  // We add a check
432  KRATOS_DEBUG_ERROR_IF(std::isnan(rPlasticDissipation)) << "rPlasticDissipation is nan" << std::endl;
433  }

446  static void CalculateEquivalentStressThreshold(
447  const double PlasticDissipation,
448  const double TensileIndicatorFactor,
449  const double CompressionIndicatorFactor,
450  double& rEquivalentStressThreshold,
451  double& rSlope,
452  ConstitutiveLaw::Parameters& rValues,
453  const double EquivalentPlasticStrain,
454  const double CharacteristicLength
455  )
456  {
457  const Properties& r_material_properties = rValues.GetMaterialProperties();
458  const int curve_type = r_material_properties[HARDENING_CURVE];
459  BoundedVector<double, 2> slopes, eq_thresholds;
460 
461  for (IndexType i = 0; i < 2; ++i) { // i:0 Tension ; i:1 compression
462  switch (static_cast<HardeningCurveType>(curve_type))
463  {
464  case HardeningCurveType::LinearSoftening:
465  CalculateEquivalentStressThresholdHardeningCurveLinearSoftening(
466  PlasticDissipation, TensileIndicatorFactor,
467  CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
468  rValues);
469  break;
470 
471  case HardeningCurveType::ExponentialSoftening:
472  CalculateEquivalentStressThresholdHardeningCurveExponentialSoftening(
473  PlasticDissipation, TensileIndicatorFactor,
474  CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
475  rValues, CharacteristicLength);
476  break;
477 
478  case HardeningCurveType::InitialHardeningExponentialSoftening:
479  CalculateEquivalentStressThresholdHardeningCurveInitialHardeningExponentialSoftening(
480  PlasticDissipation, TensileIndicatorFactor,
481  CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
482  rValues);
483  break;
484 
485  case HardeningCurveType::PerfectPlasticity:
486  CalculateEquivalentStressThresholdHardeningCurvePerfectPlasticity(
487  PlasticDissipation, TensileIndicatorFactor,
488  CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
489  rValues);
490  break;
491 
492  case HardeningCurveType::CurveFittingHardening: // Only for VonMises + Tresca!
493  CalculateEquivalentStressThresholdCurveFittingHardening(
494  PlasticDissipation, TensileIndicatorFactor,
495  CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
496  rValues, EquivalentPlasticStrain, CharacteristicLength);
497  break;
498 
499  case HardeningCurveType::LinearExponentialSoftening:
500  CalculateEquivalentStressThresholdHardeningCurveLinearExponentialSoftening(
501  PlasticDissipation, TensileIndicatorFactor,
502  CompressionIndicatorFactor, eq_thresholds[i], slopes[i], CharacteristicLength,
503  rValues);
504  break;
505 
506  case HardeningCurveType::CurveDefinedByPoints:
507  CalculateEquivalentStressThresholdHardeningCurveDefinedByPoints(
508  PlasticDissipation, TensileIndicatorFactor,
509  CompressionIndicatorFactor, eq_thresholds[i], slopes[i],
510  rValues, CharacteristicLength);
511  break;
512 
513  // Add more cases...
514  default:
515  KRATOS_ERROR << " The HARDENING_CURVE of plasticity is not set or wrong..." << curve_type << std::endl;
516  break;
517  }
518  }
519 
520  rEquivalentStressThreshold = TensileIndicatorFactor * eq_thresholds[0] + CompressionIndicatorFactor * eq_thresholds[1];
521  rSlope = rEquivalentStressThreshold * ((TensileIndicatorFactor * slopes[0] / eq_thresholds[0]) + (CompressionIndicatorFactor * slopes[1] / eq_thresholds[1]));
522  }

533  static void CalculateEquivalentStressThresholdHardeningCurveLinearSoftening(
534  const double PlasticDissipation,
535  const double TensileIndicatorFactor,
536  const double CompressionIndicatorFactor,
537  double& rEquivalentStressThreshold,
538  double& rSlope,
539  ConstitutiveLaw::Parameters& rValues
540  )
541  {
542  const Properties& r_material_properties = rValues.GetMaterialProperties();
543  const bool has_plastic_dissipation_limit = r_material_properties.Has(PLASTIC_DISSIPATION_LIMIT_LINEAR_SOFTENING);
544  const double plastic_dissipation_limit = has_plastic_dissipation_limit ? r_material_properties[PLASTIC_DISSIPATION_LIMIT_LINEAR_SOFTENING] : 0.99;
545  double initial_threshold;
546  GetInitialUniaxialThreshold(rValues, initial_threshold);
547 
548  if (PlasticDissipation <= plastic_dissipation_limit){ //Linear branch
549  rEquivalentStressThreshold = initial_threshold * std::sqrt(1.0 - PlasticDissipation);
550  rSlope = -0.5 * (std::pow(initial_threshold, 2.0) / (rEquivalentStressThreshold));
551  } else { //Exponential branch included to achieve consistent results after full plasticity scenarios
552  rEquivalentStressThreshold = (initial_threshold / std::sqrt(1.0 - plastic_dissipation_limit)) * (1.0 - PlasticDissipation);
553  rSlope = - (initial_threshold / std::sqrt(1.0 - plastic_dissipation_limit));
554  }
555  }

567  static void CalculateEquivalentStressThresholdHardeningCurveExponentialSoftening(
568  const double PlasticDissipation,
569  const double TensileIndicatorFactor,
570  const double CompressionIndicatorFactor,
571  double& rEquivalentStressThreshold,
572  double& rSlope,
573  ConstitutiveLaw::Parameters& rValues,
574  const double CharacteristicLength
575  )
576  {
577  const Properties& r_material_properties = rValues.GetMaterialProperties();
578 
579  const double young_modulus = r_material_properties[YOUNG_MODULUS];
580  const bool has_symmetric_yield_stress = r_material_properties.Has(YIELD_STRESS);
581  const double yield_compression = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_COMPRESSION];
582  const double yield_tension = has_symmetric_yield_stress ? r_material_properties[YIELD_STRESS] : r_material_properties[YIELD_STRESS_TENSION];
583  const double n = yield_compression / yield_tension;
584  const double fracture_energy_compression = r_material_properties[FRACTURE_ENERGY] * std::pow(n, 2); // Frac energy in compression
585  const double characteristic_fracture_energy_compression = fracture_energy_compression / CharacteristicLength;
586 
587  const double minimum_characteristic_fracture_energy_exponential_softening = (std::pow(yield_compression, 2)) / young_modulus;
588 
589  const bool has_total_or_plastic_strain_space = r_material_properties.Has(TOTAL_OR_PLASTIC_STRAIN_SPACE);
590  const bool total_or_plastic_strain_space = has_total_or_plastic_strain_space ? r_material_properties[TOTAL_OR_PLASTIC_STRAIN_SPACE] : false; //Default value = plastic strain space
591 
592  double initial_threshold;
593  GetInitialUniaxialThreshold(rValues, initial_threshold);
594  KRATOS_ERROR_IF(characteristic_fracture_energy_compression < minimum_characteristic_fracture_energy_exponential_softening) << "The Fracture Energy is to low: " << characteristic_fracture_energy_compression << std::endl;
595  if (total_or_plastic_strain_space) { // Curve built in the total strain space
596  rEquivalentStressThreshold = (young_modulus / initial_threshold) * ((0.5 * std::pow(initial_threshold, 2.0) / young_modulus - characteristic_fracture_energy_compression)
597  + std::sqrt(std::pow((0.5 * std::pow(initial_threshold, 2.0) / young_modulus + characteristic_fracture_energy_compression), 2.0)
598  - 2.0 * std::pow(initial_threshold, 2.0) / young_modulus * characteristic_fracture_energy_compression * PlasticDissipation));
599  rSlope = - initial_threshold * characteristic_fracture_energy_compression * std::pow(std::pow((0.5 * std::pow(initial_threshold, 2.0) / young_modulus + characteristic_fracture_energy_compression), 2.0)
600  - 2.0 * std::pow(initial_threshold, 2.0) / young_modulus * characteristic_fracture_energy_compression * PlasticDissipation, -0.5);
601  } else { // Curve built in the plastic strain space
602  rEquivalentStressThreshold = initial_threshold * (1.0 - PlasticDissipation);
603  rSlope = - initial_threshold;
604  }
605  }

616  static void CalculateEquivalentStressThresholdHardeningCurveInitialHardeningExponentialSoftening(
617  const double PlasticDissipation,
618  const double TensileIndicatorFactor,
619  const double CompressionIndicatorFactor,
620  double& rEquivalentStressThreshold,
621  double& rSlope,
622  ConstitutiveLaw::Parameters& rValues
623  )
624  {
625  const Properties& r_material_properties = rValues.GetMaterialProperties();
626 
627  double initial_threshold;
628  GetInitialUniaxialThreshold(rValues, initial_threshold);
629  const double ultimate_stress = r_material_properties[MAXIMUM_STRESS]; // sikpi
630  const double max_stress_position = r_material_properties[MAXIMUM_STRESS_POSITION]; // cappi
631 
632  if (PlasticDissipation < 1.0) {
633  const double ro = std::sqrt(1.0 - initial_threshold / ultimate_stress);
634  double alpha = std::log((1.0 - (1.0 - ro) * (1.0 - ro)) / ((3.0 - ro) * (1.0 + ro) * max_stress_position));
635  alpha = std::exp(alpha / (1.0 - max_stress_position));
636  const double phi = std::pow((1.0 - ro), 2.0) + ((3.0 - ro) * (1.0 + ro) * PlasticDissipation * (std::pow(alpha, (1.0 - PlasticDissipation))));
637 
638  rEquivalentStressThreshold = ultimate_stress * (2.0 * std::sqrt(phi) - phi);
639  rSlope = ultimate_stress * ((1.0 / std::sqrt(phi)) - 1.0) * (3.0 - ro) * (1.0 + ro) * (std::pow(alpha, (1.0 - PlasticDissipation))) *
640  (1.0 - std::log(alpha) * PlasticDissipation);
641  } else {
642  KRATOS_ERROR << "PlasticDissipation > 1.0 " << PlasticDissipation << std::endl;
643  }
644  }

693  const Properties& r_material_properties = rValues.GetMaterialProperties();
694  const Vector& curve_fitting_parameters = r_material_properties[CURVE_FITTING_PARAMETERS];
695 
696  const bool has_tangency_linear_region = r_material_properties.Has(TANGENCY_REGION2);
697  const bool tangency_linear_region = has_tangency_linear_region ? r_material_properties[TANGENCY_REGION2] : false;
698 
699  const Vector& plastic_strain_indicators = r_material_properties[PLASTIC_STRAIN_INDICATORS];
700  const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
701  const double volumetric_fracture_energy = fracture_energy / CharacteristicLength;
702 
703  const SizeType order_polinomial = curve_fitting_parameters.size();
704  const double plastic_strain_indicator_1 = plastic_strain_indicators[0];
705  const double plastic_strain_indicator_2 = plastic_strain_indicators[1];
706 
707  // Compute the initial and the final stresses
708  double stress_indicator_1 = curve_fitting_parameters[0];
709  double dS_dEp = 0.0;
710 
711  for (IndexType i = 1; i < order_polinomial; ++i) {
712  stress_indicator_1 += curve_fitting_parameters[i] * (std::pow(plastic_strain_indicator_1, i));
713  dS_dEp += i * curve_fitting_parameters[i] * std::pow(plastic_strain_indicator_1, i - 1);
714  }
715 
716  double dKp_dEp = stress_indicator_1 / volumetric_fracture_energy;
717  if (!tangency_linear_region){
718  dS_dEp = 0.0; // initial slope is zero, else the slope is the tangent of the polinomial region.
719  }
720 
721  const double stress_indicator_2 = stress_indicator_1 + dS_dEp * (plastic_strain_indicator_2 - plastic_strain_indicator_1);
722 
723  // Compute volumetric fracture energies of each region
724  double Gt1 = 0.0;
725  for (IndexType i = 0; i < order_polinomial; ++i) {
726  Gt1 += curve_fitting_parameters[i] * (std::pow(plastic_strain_indicator_1, i + 1)) / (i + 1);
727  }
728  const double Gt2 = (stress_indicator_1 + stress_indicator_2) * (plastic_strain_indicator_2 - plastic_strain_indicator_1) * 0.5;
729  const double Gt3 = volumetric_fracture_energy - Gt2 - Gt1;
730 
731  KRATOS_ERROR_IF(Gt3 < 0.0) << "Fracture energy too low in CurveFittingHardening of plasticity..." << std::endl;
732 
733  // Compute segment threshold
734  const double segment_threshold = (Gt2 + Gt1) / volumetric_fracture_energy;
735 
736  if (PlasticDissipation <= segment_threshold) {
737  const double Eps = EquivalentPlasticStrain;
738 
739  if (EquivalentPlasticStrain < plastic_strain_indicator_1) { // Polinomial region
740  double S_Ep = curve_fitting_parameters[0];
741  double dS_dEp = 0.0;
742  for (IndexType i = 1; i < order_polinomial; ++i) {
743  S_Ep += curve_fitting_parameters[i] * std::pow(Eps, i);
744  dS_dEp += i * curve_fitting_parameters[i] * std::pow(Eps, i - 1);
745  }
746  dKp_dEp = S_Ep / volumetric_fracture_energy;
747 
748  rEquivalentStressThreshold = S_Ep;
749  rSlope = dS_dEp / dKp_dEp;
750  } else { // Linear region
751  const double S_Ep = stress_indicator_1 + (stress_indicator_2 - stress_indicator_1) / (plastic_strain_indicator_2 - plastic_strain_indicator_1) * (Eps - plastic_strain_indicator_1);
752  double dS_dEp = (stress_indicator_2 - stress_indicator_1) / (plastic_strain_indicator_2 - plastic_strain_indicator_1);
753  dKp_dEp = S_Ep / volumetric_fracture_energy;
754 
755  rEquivalentStressThreshold = S_Ep;
756  rSlope = dS_dEp / dKp_dEp;
757  }
758  } else { // Exponential softening
759  const double Eps = EquivalentPlasticStrain;
760  const double alpha = std::pow(stress_indicator_1, 2);
761  const double beta = (std::pow(stress_indicator_2, 2) - alpha) / (plastic_strain_indicator_2 - plastic_strain_indicator_1);
762 
763  const double S_Ep = std::sqrt(alpha + beta * (Eps - plastic_strain_indicator_1));
764  const double plastic_dissipation_region_3 = PlasticDissipation - segment_threshold;
765 
766  const double beta2 = 1.5 * S_Ep / Gt3;
767  const double alpha2 = std::sqrt((plastic_dissipation_region_3 * 2.0 * beta2 * volumetric_fracture_energy / S_Ep) + 1.0);
768  rEquivalentStressThreshold = S_Ep * alpha2 * (2.0 - alpha2);
769  rSlope = 2.0 * beta2 * volumetric_fracture_energy * (1.0 / alpha2 - 1.0);
770  }
771 
772  }

784  static void CalculateEquivalentStressThresholdHardeningCurveLinearExponentialSoftening(
785  const double PlasticDissipation,
786  const double TensileIndicatorFactor,
787  const double CompressionIndicatorFactor,
788  double& rEquivalentStressThreshold,
789  double& rSlope,
790  const double CharacteristicLength,
791  ConstitutiveLaw::Parameters& rValues
792  )
793  {
794  const Properties& r_material_properties = rValues.GetMaterialProperties();
795  const bool has_plastic_dissipation_limit = r_material_properties.Has(PLASTIC_DISSIPATION_LIMIT_LINEAR_SOFTENING);
796 
797  const double plastic_dissipation_limit = has_plastic_dissipation_limit ? r_material_properties[PLASTIC_DISSIPATION_LIMIT_LINEAR_SOFTENING] : 0.9;
798  const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
799  const double volumetric_fracture_energy = fracture_energy / CharacteristicLength;
800  double initial_threshold;
801  GetInitialUniaxialThreshold(rValues, initial_threshold);
802 
803  const double volumetric_fracture_energy_linear_branch = 0.5 * volumetric_fracture_energy * (plastic_dissipation_limit + 1.0);
804 
805  if (PlasticDissipation <= plastic_dissipation_limit){ //Linear branch
806  rEquivalentStressThreshold = initial_threshold * std::sqrt(1.0 - PlasticDissipation * volumetric_fracture_energy / volumetric_fracture_energy_linear_branch);
807  rSlope = - 0.5 * initial_threshold * (volumetric_fracture_energy / volumetric_fracture_energy_linear_branch) * std::pow(1.0 - PlasticDissipation * volumetric_fracture_energy / volumetric_fracture_energy_linear_branch, -0.5);
808  } else { //Exponential branch included to achieve consistent results after full plasticity scenarios
809  const double volumetric_fracture_energy_exponential_branch = volumetric_fracture_energy * (1.0 - plastic_dissipation_limit) * std::exp((plastic_dissipation_limit + 1.0) / (std::sqrt(1.0 - std::pow(plastic_dissipation_limit, 2.0))) - 1.0);
810  const double initial_threshold_exponential = initial_threshold * volumetric_fracture_energy_exponential_branch / volumetric_fracture_energy * std::sqrt(1.0 - plastic_dissipation_limit * volumetric_fracture_energy / volumetric_fracture_energy_linear_branch) / (1.0 - plastic_dissipation_limit);
811  rEquivalentStressThreshold = initial_threshold_exponential * (1.0 - PlasticDissipation) * volumetric_fracture_energy / volumetric_fracture_energy_exponential_branch;
812  rSlope = - initial_threshold_exponential * volumetric_fracture_energy / volumetric_fracture_energy_exponential_branch;
813  }
814  }

828  static void CalculateEquivalentStressThresholdHardeningCurveDefinedByPoints(
829  const double PlasticDissipation,
830  const double TensileIndicatorFactor,
831  const double CompressionIndicatorFactor,
832  double& rEquivalentStressThreshold,
833  double& rSlope,
834  ConstitutiveLaw::Parameters& rValues,
835  const double CharacteristicLength
836  )
837  {
838  const Properties& r_material_properties = rValues.GetMaterialProperties();
839  const Vector& equivalent_stress_vector = r_material_properties[EQUIVALENT_STRESS_VECTOR_PLASTICITY_POINT_CURVE];
840  const bool has_plastic_strain_vector = r_material_properties.Has(PLASTIC_STRAIN_VECTOR_PLASTICITY_POINT_CURVE);
841  const double young_modulus = r_material_properties[YOUNG_MODULUS];
842  Vector plastic_strain_vector;
843  if (has_plastic_strain_vector) { // Strain input is equivalent plastic strain
844  plastic_strain_vector = r_material_properties[PLASTIC_STRAIN_VECTOR_PLASTICITY_POINT_CURVE];
845  } else {
846  const Vector& total_strain_vector = r_material_properties[TOTAL_STRAIN_VECTOR_PLASTICITY_POINT_CURVE];
847  plastic_strain_vector = total_strain_vector - 1.0 / young_modulus * equivalent_stress_vector;
848  }
849  const double fracture_energy = r_material_properties[FRACTURE_ENERGY];
850  const double volumetric_fracture_energy = fracture_energy / CharacteristicLength;
851  const SizeType points_hardening_curve = equivalent_stress_vector.size();
852 
853  // Compute volumetric fracture energies of each region
854  double Gt1 = 0.0;
855  for (IndexType i = 1; i < points_hardening_curve; ++i) {
856  Gt1 += 0.5 * (equivalent_stress_vector(i - 1) + equivalent_stress_vector(i)) * (plastic_strain_vector(i) - plastic_strain_vector(i - 1));
857  }
858  const double Gt2 = volumetric_fracture_energy - Gt1;
859 
860  KRATOS_ERROR_IF(Gt2 < 0.0) << "Fracture energy too low in CurveDefinedByPoints of plasticity..." << std::endl;
861 
862  // Compute segment threshold
863  const double segment_threshold = (Gt1) / volumetric_fracture_energy;
864  if (PlasticDissipation < segment_threshold) {
865  IndexType i = 0;
866  double gf_point_region = 0.0;
867  double plastic_dissipation_previous_point = 0.0;
868  while (PlasticDissipation >= gf_point_region / volumetric_fracture_energy) {
869  i += 1;
870  plastic_dissipation_previous_point = gf_point_region / volumetric_fracture_energy;
871  gf_point_region += 0.5 * (equivalent_stress_vector(i - 1) + equivalent_stress_vector(i)) * (plastic_strain_vector(i) - plastic_strain_vector(i - 1));
872  }
873  const double plastic_dissipation_next_point = gf_point_region / volumetric_fracture_energy;
874 
875  // Stress is computed using an equivalent equation to the one used for the linear softening curve, i.e. f(S) = a * (1.0 - b * kp)
876  const double b = (std::pow(equivalent_stress_vector(i), 2.0) - std::pow(equivalent_stress_vector(i - 1), 2.0)) / (plastic_dissipation_previous_point * std::pow(equivalent_stress_vector(i), 2.0) - plastic_dissipation_next_point * std::pow(equivalent_stress_vector(i - 1), 2.0));
877  const double a = equivalent_stress_vector(i - 1) / std::sqrt(1.0 - b * plastic_dissipation_previous_point);
878  rEquivalentStressThreshold = a * std::sqrt(1.0 - b * PlasticDissipation);
879  rSlope = - 0.5 * std::pow(a, 2.0) * b / rEquivalentStressThreshold;
880 
881  } else { // Exponential branch included to achieve consistent results after full plasticity scenarios
882  const bool has_total_or_plastic_strain_space = r_material_properties.Has(TOTAL_OR_PLASTIC_STRAIN_SPACE);
883  const bool total_or_plastic_strain_space = has_total_or_plastic_strain_space ? r_material_properties[TOTAL_OR_PLASTIC_STRAIN_SPACE] : false; // Default value = plastic strain space
884  if (total_or_plastic_strain_space) { // Curve built in the total strain space
885  const double yield_strain = equivalent_stress_vector(0) / young_modulus;
886  const double a = (0.5 * equivalent_stress_vector(points_hardening_curve - 1) * yield_strain + equivalent_stress_vector(0) / equivalent_stress_vector(points_hardening_curve - 1)
887  * volumetric_fracture_energy * (segment_threshold - 1.0)) / yield_strain;
888 
889  rEquivalentStressThreshold = a + std::sqrt(std::pow(a, 2.0) + 2.0 * equivalent_stress_vector(0) * volumetric_fracture_energy * (1.0 - PlasticDissipation) / yield_strain);
890  rSlope = - equivalent_stress_vector(0) * volumetric_fracture_energy / (yield_strain * std::sqrt(std::pow(a, 2.0) + 2.0 * equivalent_stress_vector(0) * volumetric_fracture_energy * (1.0 - PlasticDissipation) / yield_strain));
891 
892  } else { // Curve built in the plastic strain space
893  const double a = equivalent_stress_vector(points_hardening_curve - 1) / (1.0 - segment_threshold);
894  rEquivalentStressThreshold = a * (1.0 - PlasticDissipation);
895  rSlope = - a;
896  }
897  }
898  }

999  static int Check(const Properties& rMaterialProperties)
1000  {

1002  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YOUNG_MODULUS)) << "HARDENING_CURVE is not a defined value" << std::endl;

1005 
1006  // Checking curves
1007  const int curve_type = rMaterialProperties[HARDENING_CURVE];
1008  if (static_cast<HardeningCurveType>(curve_type) == HardeningCurveType::InitialHardeningExponentialSoftening) {
1009  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(MAXIMUM_STRESS)) << "MAXIMUM_STRESS is not a defined value" << std::endl;
1010  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(MAXIMUM_STRESS_POSITION)) << "MAXIMUM_STRESS_POSITION is not a defined value" << std::endl;
1011  } else if (static_cast<HardeningCurveType>(curve_type) == HardeningCurveType::CurveFittingHardening) {
1012  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(CURVE_FITTING_PARAMETERS)) << "CURVE_FITTING_PARAMETERS is not a defined value" << std::endl;
1013  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(PLASTIC_STRAIN_INDICATORS)) << "PLASTIC_STRAIN_INDICATORS is not a defined value" << std::endl;
1014  }
1015 
1016  if (!rMaterialProperties.Has(YIELD_STRESS)) {
1017  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_TENSION)) << "YIELD_STRESS_TENSION is not a defined value" << std::endl;
1018  KRATOS_ERROR_IF_NOT(rMaterialProperties.Has(YIELD_STRESS_COMPRESSION)) << "YIELD_STRESS_COMPRESSION is not a defined value" << std::endl;
1019 
1020  const double yield_compression = rMaterialProperties[YIELD_STRESS_COMPRESSION];
1021  const double yield_tension = rMaterialProperties[YIELD_STRESS_TENSION];
1022 
1023  KRATOS_ERROR_IF(yield_compression < tolerance) << "Yield stress in compression almost zero or negative, include YIELD_STRESS_COMPRESSION in definition";
1024  KRATOS_ERROR_IF(yield_tension < tolerance) << "Yield stress in tension almost zero or negative, include YIELD_STRESS_TENSION in definition";
1025  } else {
1026  const double yield_stress = rMaterialProperties[YIELD_STRESS];
1027 
1028  KRATOS_ERROR_IF(yield_stress < tolerance) << "Yield stress almost zero or negative, include YIELD_STRESS in definition";
1029  }
1030 
1031  return TYieldSurfaceType::Check(rMaterialProperties);
