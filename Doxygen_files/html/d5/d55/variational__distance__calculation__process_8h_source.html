284  void Execute() override
285  {
286  KRATOS_TRY;
287 
288  if(mDistancePartIsInitialized == false){
289  ReGenerateDistanceModelPart(mrBaseModelPart);
290  }
291 
292  ModelPart& r_distance_model_part = mrModel.GetModelPart( mAuxModelPartName );
293 
294  // TODO: check flag PERFORM_STEP1
295  // Step1 - solve a poisson problem with a source term which depends on the sign of the existing distance function
296  r_distance_model_part.pGetProcessInfo()->SetValue(FRACTIONAL_STEP,1);
297 
298  // Unfix the distances
299  const int nnodes = static_cast<int>(r_distance_model_part.NumberOfNodes());
300 
301  block_for_each(r_distance_model_part.Nodes(), [](Node& rNode){
302  double& d = rNode.FastGetSolutionStepValue(DISTANCE);
303 
304  // Free the DISTANCE values
305  rNode.Free(DISTANCE);
306  // Set the fix flag to 0
307  rNode.Set(BLOCKED, false);
308 
309  // Save the distances
310  rNode.SetValue(DISTANCE, d);
311 
312  if(d == 0){
313  d = 1.0e-15;
314  rNode.Set(BLOCKED, true);
315  rNode.Fix(DISTANCE);
316  } else {
317  if(d > 0.0){
318  d = 1.0e15; // Set to a large number, to make sure that that the minimal distance is computed according to CaculateTetrahedraDistances
319  } else {
320  d = -1.0e15;
321  }
322  }
323  });
324 
325  block_for_each(r_distance_model_part.Elements(), [this](Element& rElem){
326  array_1d<double,TDim+1> distances;
327  auto& geom = rElem.GetGeometry();
328 
329  for(unsigned int i=0; i<TDim+1; i++){
330  distances[i] = geom[i].GetValue(DISTANCE);
331  }
332 
333  const array_1d<double,TDim+1> original_distances = distances;
334 
335  // The element is cut by the interface
336  if(this->IsSplit(distances)){
337  // Compute the unsigned distance using GeometryUtils
338  if (mOptions.Is(CALCULATE_EXACT_DISTANCES_TO_PLANE)) {
339  GeometryUtils::CalculateExactDistancesToPlane(geom, distances);
340  }
341  else {
342  if constexpr (TDim==3){
343  GeometryUtils::CalculateTetrahedraDistances(geom, distances);
344  }
345  else {
346  GeometryUtils::CalculateTriangleDistances(geom, distances);
347  }
348  }
349 
350  // Assign the sign using the original distance values
351  for(unsigned int i = 0; i < TDim+1; ++i){
352  if(original_distances[i] < 0){
353  distances[i] = -distances[i];
354  }
355  }
356 
357  for(unsigned int i = 0; i < TDim+1; ++i){
358  double &d = geom[i].FastGetSolutionStepValue(DISTANCE);
359  geom[i].SetLock();
360  if(std::abs(d) > std::abs(distances[i])){
361  d = distances[i];
362  }
363  geom[i].Set(BLOCKED, true);
364  geom[i].Fix(DISTANCE);
365  geom[i].UnSetLock();
366  }
367  }
368  });
369 
370  // SHALL WE SYNCHRONIZE SOMETHING IN HERE???? WE'VE CHANGED THE NODAL DISTANCE VALUES FROM THE ELEMENTS...
371  this->SynchronizeFixity();
372  this->SynchronizeDistance();
373 
374  // Compute the maximum and minimum distance for the fixed nodes
375  double max_dist = 0.0;
376  double min_dist = 0.0;
377  for(int i_node = 0; i_node < nnodes; ++i_node){
378  auto it_node = r_distance_model_part.NodesBegin() + i_node;
379  if(it_node->IsFixed(DISTANCE)){
380  const double& d = it_node->FastGetSolutionStepValue(DISTANCE);
381  if(d > max_dist){
382  max_dist = d;
383  }
384  if(d < min_dist){
385  min_dist = d;
386  }
387  }
388  }
389 
390  // Synchronize the maximum and minimum distance values
391  const auto &r_communicator = r_distance_model_part.GetCommunicator().GetDataCommunicator();
392  max_dist = r_communicator.MaxAll(max_dist);
393  min_dist = r_communicator.MinAll(min_dist);
394 
395  // Assign the max dist to all of the non-fixed positive nodes
396  // and the minimum one to the non-fixed negatives
397  block_for_each(r_distance_model_part.Nodes(), [&min_dist, &max_dist](Node& rNode){
398  if(!rNode.IsFixed(DISTANCE)){
399  double& d = rNode.FastGetSolutionStepValue(DISTANCE);
400  if(d>0){
401  d = max_dist;
402  } else {
403  d = min_dist;
404  }
405  }
406  });
407  mpSolvingStrategy->Solve();
408 
409  // Step2 - minimize the target residual
410  r_distance_model_part.pGetProcessInfo()->SetValue(FRACTIONAL_STEP,2);
411  for(unsigned int it = 0; it<mMaxIterations; it++){
412  mpSolvingStrategy->Solve();
413  }
414 
415  // Unfix the distances
416  VariableUtils().ApplyFixity(DISTANCE, false, r_distance_model_part.Nodes());
417  VariableUtils().SetFlag(BOUNDARY, false, r_distance_model_part.Nodes());
418  VariableUtils().SetFlag(BLOCKED, false, r_distance_model_part.Nodes());
419 
420  KRATOS_CATCH("")
421  }

557  virtual void ReGenerateDistanceModelPart(ModelPart& rBaseModelPart)
558  {
559  KRATOS_TRY
560 
561  if(mrModel.HasModelPart( mAuxModelPartName ))
562  mrModel.DeleteModelPart( mAuxModelPartName );
563 
564  // Ensure that the nodes have distance as a DOF
565  VariableUtils().AddDof<Variable<double> >(DISTANCE, rBaseModelPart);
566 
567  // Generate
568  ModelPart& r_distance_model_part = mrModel.CreateModelPart( mAuxModelPartName );
569 
570  Element::Pointer p_distance_element = Kratos::make_intrusive<DistanceCalculationElementSimplex<TDim> >();
571 
572  r_distance_model_part.GetNodalSolutionStepVariablesList() = rBaseModelPart.GetNodalSolutionStepVariablesList();
573 
574  ConnectivityPreserveModeler modeler;
575  modeler.GenerateModelPart(rBaseModelPart, r_distance_model_part, *p_distance_element);
576 
577  // Using the conditions to mark the boundary with the flag boundary
578  // Note that we DO NOT add the conditions to the model part
579  VariableUtils().SetFlag<ModelPart::NodesContainerType>(BOUNDARY, false, r_distance_model_part.Nodes());
580  // Note that above we have assigned the same geometry. Thus the flag is
581  // set in the distance model part despite we are iterating the base one
582  for (auto it_cond = rBaseModelPart.ConditionsBegin(); it_cond != rBaseModelPart.ConditionsEnd(); ++it_cond){
583  Geometry< Node >& geom = it_cond->GetGeometry();
584  for(unsigned int i=0; i<geom.size(); i++){
585  geom[i].Set(BOUNDARY,true);
586  }
587  }
588 
589  rBaseModelPart.GetCommunicator().SynchronizeOrNodalFlags(BOUNDARY);
590 
591  r_distance_model_part.GetProcessInfo().SetValue(VARIATIONAL_REDISTANCE_COEFFICIENT_FIRST, mCoefficient1);
592  r_distance_model_part.GetProcessInfo().SetValue(VARIATIONAL_REDISTANCE_COEFFICIENT_SECOND, mCoefficient2);
593 
594  mDistancePartIsInitialized = true;
595 
596  KRATOS_CATCH("")
597  }

646  void SynchronizeDistance(){
647  ModelPart& r_distance_model_part = mrModel.GetModelPart( mAuxModelPartName );
648  auto &r_communicator = r_distance_model_part.GetCommunicator();
649 
650  // Only required in the MPI case
651  if(r_communicator.TotalProcesses() != 1){
652  int nnodes = static_cast<int>(r_distance_model_part.NumberOfNodes());
653 
654  // Set the distance absolute value
655  #pragma omp parallel for
656  for(int i_node = 0; i_node < nnodes; ++i_node){
657  auto it_node = r_distance_model_part.NodesBegin() + i_node;
658  it_node->FastGetSolutionStepValue(DISTANCE) = std::abs(it_node->FastGetSolutionStepValue(DISTANCE));
659  }
660 
661  // Synchronize the unsigned value to minimum
662  r_communicator.SynchronizeCurrentDataToMin(DISTANCE);
663 
664  // Set the distance sign again by retrieving it from the non-historical database
665  #pragma omp parallel for
666  for(int i_node = 0; i_node < nnodes; ++i_node){
667  auto it_node = r_distance_model_part.NodesBegin() + i_node;
668  if(it_node->GetValue(DISTANCE) < 0.0){
669  it_node->FastGetSolutionStepValue(DISTANCE) = -it_node->FastGetSolutionStepValue(DISTANCE);
670  }
671  }
672  }

676  ModelPart& r_distance_model_part = mrModel.GetModelPart( mAuxModelPartName );

678 
679  // Only required in the MPI case
680  if(r_communicator.TotalProcesses() != 1){

682 
683  // Synchronize the fixity flag variable to minium
684  // (true means fixed and false means free)
685  r_communicator.SynchronizeOrNodalFlags(BLOCKED);
686
