216  {

228  BoundedMatrix<double, TNumNodes, TNumNodes> CalculatePermeabilityMatrix(
229  const GeometryType::ShapeFunctionsGradientsType& rShapeFunctionGradients,
230  const Vector& rIntegrationCoefficients) const
231  {
232  RetentionLaw::Parameters RetentionParameters(GetProperties());
233  BoundedMatrix<double, 1, 1> constitutive_matrix;
234  const auto& r_properties = GetProperties();
235  GeoElementUtilities::FillPermeabilityMatrix(constitutive_matrix, r_properties);
236 
237  auto result = BoundedMatrix<double, TNumNodes, TNumNodes>{ZeroMatrix{TNumNodes, TNumNodes}};
238  for (unsigned int integration_point_index = 0;
239  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
240  ++integration_point_index) {
241  const double RelativePermeability =
242  mRetentionLawVector[integration_point_index]->CalculateRelativePermeability(RetentionParameters);
243  double dynamic_viscosity_inverse = 1.0 / r_properties[DYNAMIC_VISCOSITY];
244  result += GeoTransportEquationUtilities::CalculatePermeabilityMatrix<TDim, TNumNodes>(
245  rShapeFunctionGradients[integration_point_index], dynamic_viscosity_inverse, constitutive_matrix,
246  RelativePermeability, rIntegrationCoefficients[integration_point_index]);
247  }
248  return result;
249  }

295  double CalculateBiotModulusInverse(const unsigned int integrationPointIndex) const
296  {
297  const auto& r_properties = GetProperties();
298  const double biot_coefficient = r_properties[BIOT_COEFFICIENT];
299 
300  double bulk_fluid = TINY;
301  if (!r_properties[IGNORE_UNDRAINED]) {
302  bulk_fluid = r_properties[BULK_MODULUS_FLUID];
303  }
304  double result = (biot_coefficient - r_properties[POROSITY]) / r_properties[BULK_MODULUS_SOLID] +
305  r_properties[POROSITY] / bulk_fluid;
306 
307  RetentionLaw::Parameters RetentionParameters(GetProperties());
308  const double degree_of_saturation =
309  mRetentionLawVector[integrationPointIndex]->CalculateSaturation(RetentionParameters);
310  const double derivative_of_saturation =
311  mRetentionLawVector[integrationPointIndex]->CalculateDerivativeOfSaturation(RetentionParameters);
312 
313  result *= degree_of_saturation;
314  result -= derivative_of_saturation * r_properties[POROSITY];
315  return result;
316  }

334  array_1d<double, TNumNodes> CalculateFluidBodyVector(const Matrix& rNContainer,
335  const GeometryType::ShapeFunctionsGradientsType& rShapeFunctionGradients,
336  const Vector& rIntegrationCoefficients) const
337  {
338  const std::size_t number_integration_points =
339  GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
340  GeometryType::JacobiansType J_container;
341  J_container.resize(number_integration_points, false);
342  for (std::size_t i = 0; i < number_integration_points; ++i) {
343  J_container[i].resize(GetGeometry().WorkingSpaceDimension(),
344  GetGeometry().LocalSpaceDimension(), false);
345  }
346  GetGeometry().Jacobian(J_container, this->GetIntegrationMethod());
347 
348  const auto& r_properties = GetProperties();
349  BoundedMatrix<double, 1, 1> constitutive_matrix;
350  GeoElementUtilities::FillPermeabilityMatrix(constitutive_matrix, r_properties);
351 
352  RetentionLaw::Parameters RetentionParameters(GetProperties());
353 
354  array_1d<double, TNumNodes * TDim> volume_acceleration;
355  GeoElementUtilities::GetNodalVariableVector<TDim, TNumNodes>(
356  volume_acceleration, GetGeometry(), VOLUME_ACCELERATION);
357  array_1d<double, TDim> body_acceleration;
358 
359  array_1d<double, TNumNodes> fluid_body_vector = ZeroVector(TNumNodes);
360  for (unsigned int integration_point_index = 0;
361  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
362  ++integration_point_index) {
363  GeoElementUtilities::InterpolateVariableWithComponents<TDim, TNumNodes>(
364  body_acceleration, rNContainer, volume_acceleration, integration_point_index);
365 
366  array_1d<double, TDim> tangent_vector = column(J_container[integration_point_index], 0);
367  tangent_vector /= norm_2(tangent_vector);
368 
369  array_1d<double, 1> projected_gravity = ZeroVector(1);
370  projected_gravity(0) = MathUtils<double>::Dot(tangent_vector, body_acceleration);
371  const auto N = Vector{row(rNContainer, integration_point_index)};
372  double RelativePermeability =
373  mRetentionLawVector[integration_point_index]->CalculateRelativePermeability(RetentionParameters);
374  fluid_body_vector +=
375  r_properties[DENSITY_WATER] * RelativePermeability *
376  prod(prod(rShapeFunctionGradients[integration_point_index], constitutive_matrix), projected_gravity) *
377  rIntegrationCoefficients[integration_point_index] / r_properties[DYNAMIC_VISCOSITY];
378  }
379  return fluid_body_vector;
380  }
