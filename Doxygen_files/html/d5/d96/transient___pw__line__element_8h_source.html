262  array_1d<double, TNumNodes> CalculateFluidBodyVector() const
263  {
264  Vector det_J_container;
265  GetGeometry().DeterminantOfJacobian(det_J_container, this->GetIntegrationMethod());
266  GeometryType::ShapeFunctionsGradientsType shape_function_gradients =
267  GetGeometry().ShapeFunctionsLocalGradients(this->GetIntegrationMethod());
268  std::transform(shape_function_gradients.begin(), shape_function_gradients.end(),
269  det_J_container.begin(), shape_function_gradients.begin(), std::divides<>());
270  const Matrix& N_container = GetGeometry().ShapeFunctionsValues(GetIntegrationMethod());
271 
272  const auto integration_coefficients = CalculateIntegrationCoefficients(det_J_container);
273 
274  const std::size_t number_integration_points =
275  GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
276  GeometryType::JacobiansType J_container;
277  J_container.resize(number_integration_points, false);
278  for (std::size_t i = 0; i < number_integration_points; ++i) {
279  J_container[i].resize(GetGeometry().WorkingSpaceDimension(),
280  GetGeometry().LocalSpaceDimension(), false);
281  }
282  GetGeometry().Jacobian(J_container, this->GetIntegrationMethod());
283 
284  const auto& r_properties = GetProperties();
285  BoundedMatrix<double, 1, 1> constitutive_matrix;
286  GeoElementUtilities::FillPermeabilityMatrix(constitutive_matrix, r_properties);
287 
288  RetentionLaw::Parameters RetentionParameters(GetProperties());
289 
290  array_1d<double, TNumNodes * TDim> volume_acceleration;
291  GeoElementUtilities::GetNodalVariableVector<TDim, TNumNodes>(
292  volume_acceleration, GetGeometry(), VOLUME_ACCELERATION);
293 
294  array_1d<double, TNumNodes> fluid_body_vector = ZeroVector(TNumNodes);
295  for (unsigned int integration_point_index = 0;
296  integration_point_index < GetGeometry().IntegrationPointsNumber(GetIntegrationMethod());
297  ++integration_point_index) {
298  array_1d<double, TDim> body_acceleration;
299  GeoElementUtilities::InterpolateVariableWithComponents<TDim, TNumNodes>(
300  body_acceleration, N_container, volume_acceleration, integration_point_index);
301 
302  array_1d<double, TDim> tangent_vector = column(J_container[integration_point_index], 0);
303  tangent_vector /= norm_2(tangent_vector);
304 
305  array_1d<double, 1> projected_gravity = ZeroVector(1);
306  projected_gravity(0) = MathUtils<double>::Dot(tangent_vector, body_acceleration);
307  const auto N = Vector{row(N_container, integration_point_index)};
308  double RelativePermeability =
309  mRetentionLawVector[integration_point_index]->CalculateRelativePermeability(RetentionParameters);
310  fluid_body_vector +=
311  r_properties[DENSITY_WATER] * RelativePermeability *
312  prod(prod(shape_function_gradients[integration_point_index], constitutive_matrix), projected_gravity) *
313  integration_coefficients[integration_point_index] / r_properties[DYNAMIC_VISCOSITY];
314  }
315  return fluid_body_vector;
316  }
