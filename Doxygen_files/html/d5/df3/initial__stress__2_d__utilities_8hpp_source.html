128  void NodalVariablesMapping(

130  ModelPart& rInitialModelPart,

132  {
133  // Define ElementOld Cell matrix
134  std::vector< std::vector< std::vector<Element::Pointer> > > ElementOldCellMatrix;
135  ElementOldCellMatrix.resize(AuxVariables.NRows);
136  for(int i = 0; i < AuxVariables.NRows; i++) ElementOldCellMatrix[i].resize(AuxVariables.NColumns);

145 
146  #pragma omp parallel for private(X_me,Y_me,PointsNumber)
147  for(int i = 0; i < NElems; i++)
148  {
149  ModelPart::ElementsContainerType::iterator itElemOld = el_begin + i;
150 
151  double X_left = itElemOld->GetGeometry().GetPoint(0).X0();
152  double X_right = X_left;
153  double Y_top = itElemOld->GetGeometry().GetPoint(0).Y0();
154  double Y_bot = Y_top;
155  PointsNumber = itElemOld->GetGeometry().PointsNumber();
156 
157  for(int j = 1; j < PointsNumber; j++)
158  {
159  X_me = itElemOld->GetGeometry().GetPoint(j).X0();
160  Y_me = itElemOld->GetGeometry().GetPoint(j).Y0();
161 
162  if(X_me > X_right) X_right = X_me;
163  else if(X_me < X_left) X_left = X_me;
164  if(Y_me > Y_top) Y_top = Y_me;
165  else if(Y_me < Y_bot) Y_bot = Y_me;
166  }
167 
168  int Column_left = int((X_left-AuxVariables.X_min)/AuxVariables.ColumnSize);
169  int Column_right = int((X_right-AuxVariables.X_min)/AuxVariables.ColumnSize);
170  int Row_top = int((AuxVariables.Y_max-Y_top)/AuxVariables.RowSize);
171  int Row_bot = int((AuxVariables.Y_max-Y_bot)/AuxVariables.RowSize);
172 
173  if(Column_left < 0) Column_left = 0;
174  else if(Column_left >= AuxVariables.NColumns) Column_left = AuxVariables.NColumns-1;
175  if(Column_right >= AuxVariables.NColumns) Column_right = AuxVariables.NColumns-1;
176  else if(Column_right < 0) Column_right = 0;
177 
178  if(Row_top < 0) Row_top = 0;
179  else if(Row_top >= AuxVariables.NRows) Row_top = AuxVariables.NRows-1;
180  if(Row_bot >= AuxVariables.NRows) Row_bot = AuxVariables.NRows-1;
181  else if(Row_bot < 0) Row_bot = 0;
182 
183  for(int k = Row_top; k <= Row_bot; k++)
184  {
185  for(int l = Column_left; l <= Column_right; l++)
186  {
187  #pragma omp critical
188  {
189  ElementOldCellMatrix[k][l].push_back((*(itElemOld.base())));
190  }
191  }
192  }
193  }

197  ModelPart::NodesContainerType::iterator node_begin = rCurrentModelPart.NodesBegin();
198 
199  array_1d<double,3> GlobalCoordinates;

205  ModelPart::NodesContainerType::iterator itNodeNew = node_begin + i;
206

209

212 
213  if(Column >= AuxVariables.NColumns) Column = AuxVariables.NColumns-1;
214  else if(Column < 0) Column = 0;
215  if(Row >= AuxVariables.NRows) Row = AuxVariables.NRows-1;
216  else if(Row < 0) Row = 0;
217 
218  noalias(GlobalCoordinates) = itNodeNew->Coordinates(); //Coordinates of new nodes are still in the original position
219  bool IsInside = false;
220  Element::Pointer pElementOld;
221 
222  for(unsigned int m = 0; m < (ElementOldCellMatrix[Row][Column]).size(); m++)
223  {
224  pElementOld = ElementOldCellMatrix[Row][Column][m];
225  IsInside = pElementOld->GetGeometry().IsInside(GlobalCoordinates,LocalCoordinates); //Checks whether the global coordinates fall inside the original old element
226  if(IsInside) break;
227  }
228  if(IsInside==false)
229  {
230  for(unsigned int m = 0; m < (ElementOldCellMatrix[Row][Column]).size(); m++)

255  for (int k= 0; k < 3; k++) { // voigt_size
256  ComponentsNodalVariableVector[k][j] = nodal_initial_stress_vector[k];
257  }
258  }
259  for (int k= 0; k < 3; k++) { // voigt_size
260  nodal_initial_stress_vector[k] = inner_prod(ShapeFunctionsValuesVector,ComponentsNodalVariableVector[k]);
261  }
262  noalias(nodal_initial_stress_tensor) = MathUtils<double>::StressVectorToTensor(nodal_initial_stress_vector);
263  Matrix& rNodalStress = itNodeNew->FastGetSolutionStepValue(INITIAL_STRESS_TENSOR);
264  if(rNodalStress.size1() != 2) // Dimension
265  rNodalStress.resize(2,2,false);
266  noalias(rNodalStress) = nodal_initial_stress_tensor;
267  }

300  void ComputeCellMatrixDimensions(
301  UtilityVariables& rAuxVariables,
302  ModelPart& rModelPart)
303  {
304  // Compute X and Y limits of the current geometry
305  unsigned int NumThreads = ParallelUtilities::GetNumThreads();
306  std::vector<double> X_max_partition(NumThreads);
307  std::vector<double> X_min_partition(NumThreads);
308  std::vector<double> Y_max_partition(NumThreads);
309  std::vector<double> Y_min_partition(NumThreads);
310 
311  const int NNodes = static_cast<int>(rModelPart.Nodes().size());
312  ModelPart::NodesContainerType::iterator node_begin = rModelPart.NodesBegin();
313 
314  #pragma omp parallel
315  {
316  int k = OpenMPUtils::ThisThread();
317 
318  X_max_partition[k] = node_begin->X();
319  X_min_partition[k] = X_max_partition[k];
320  Y_max_partition[k] = node_begin->Y();
321  Y_min_partition[k] = Y_max_partition[k];
322 
323  double X_me, Y_me;
324 
325  #pragma omp for
326  for(int i = 0; i < NNodes; i++)
327  {
328  ModelPart::NodesContainerType::iterator itNode = node_begin + i;
329 
330  X_me = itNode->X();
331  Y_me = itNode->Y();
332 
333  if( X_me > X_max_partition[k] ) X_max_partition[k] = X_me;
334  else if( X_me < X_min_partition[k] ) X_min_partition[k] = X_me;
335 
336  if( Y_me > Y_max_partition[k] ) Y_max_partition[k] = Y_me;
337  else if( Y_me < Y_min_partition[k] ) Y_min_partition[k] = Y_me;
338  }
339  }
340 
341  rAuxVariables.X_max = X_max_partition[0];
342  rAuxVariables.X_min = X_min_partition[0];
343  rAuxVariables.Y_max = Y_max_partition[0];
344  rAuxVariables.Y_min = Y_min_partition[0];
345 
346  for(unsigned int i=1; i < NumThreads; i++)
347  {
348  if(X_max_partition[i] > rAuxVariables.X_max) rAuxVariables.X_max = X_max_partition[i];
349  if(X_min_partition[i] < rAuxVariables.X_min) rAuxVariables.X_min = X_min_partition[i];
350  if(Y_max_partition[i] > rAuxVariables.Y_max) rAuxVariables.Y_max = Y_max_partition[i];
351  if(Y_min_partition[i] < rAuxVariables.Y_min) rAuxVariables.Y_min = Y_min_partition[i];
352  }
353 
354  // Calculate Average Element Length
355  double AverageElementLength = 0.0;
356 
357  int NElems = static_cast<int>(rModelPart.Elements().size());
358  ModelPart::ElementsContainerType::iterator el_begin = rModelPart.ElementsBegin();
359 
360  #pragma omp parallel for reduction(+:AverageElementLength)
361  for(int i = 0; i < NElems; i++)
362  {
363  ModelPart::ElementsContainerType::iterator itElem = el_begin + i;
364 
365  AverageElementLength += itElem->GetGeometry().Length();
366  }
367 
368  AverageElementLength = AverageElementLength/NElems;
369 
370  // Compute FracturePoints CellMatrix dimensions
371  rAuxVariables.NRows = int((rAuxVariables.Y_max-rAuxVariables.Y_min)/(AverageElementLength));
372  rAuxVariables.NColumns = int((rAuxVariables.X_max-rAuxVariables.X_min)/(AverageElementLength));
373  if(rAuxVariables.NRows <= 0) rAuxVariables.NRows = 1;
374  if(rAuxVariables.NColumns <= 0) rAuxVariables.NColumns = 1;
375  rAuxVariables.RowSize = (rAuxVariables.Y_max-rAuxVariables.Y_min)/rAuxVariables.NRows;
376  rAuxVariables.ColumnSize = (rAuxVariables.X_max-rAuxVariables.X_min)/rAuxVariables.NColumns;
377  }

381  void CalculateNodalStresses(ModelPart& rCurrentModelPart) {
382 
383  // Clear nodal Stresses
384  const int NNodes = static_cast<int>(rCurrentModelPart.Nodes().size());
385  ModelPart::NodesContainerType::iterator node_begin = rCurrentModelPart.NodesBegin();
386 
387  #pragma omp parallel for
388  for(int i = 0; i < NNodes; i++)
389  {
390  ModelPart::NodesContainerType::iterator it_node = node_begin + i;
391  it_node->FastGetSolutionStepValue(NODAL_AREA) = 0.0;
392  Matrix& rNodalStress = it_node->FastGetSolutionStepValue(NODAL_EFFECTIVE_STRESS_TENSOR);
393  if(rNodalStress.size1() != 3)
394  rNodalStress.resize(3,3,false);
395  noalias(rNodalStress) = ZeroMatrix(3,3);
396  }
397 
398  // Calculate and Extrapolate Stresses
399  const ProcessInfo& r_current_process_info = rCurrentModelPart.GetProcessInfo();
400  const auto it_elem_begin = rCurrentModelPart.ElementsBegin();
401  #pragma omp parallel for
402  for(int i=0; i<static_cast<int>(rCurrentModelPart.Elements().size()); ++i) {
403  auto it_elem = it_elem_begin + i;
404  it_elem->FinalizeSolutionStep(r_current_process_info);
405  }
406 
407  // Compute smoothed Stresses
408  #pragma omp parallel for
409  for(int n = 0; n < NNodes; n++)
410  {
411  ModelPart::NodesContainerType::iterator it_node = node_begin + n;
412  const double& NodalArea = it_node->FastGetSolutionStepValue(NODAL_AREA);
413  if (NodalArea>1.0e-20)
414  {
415  const double InvNodalArea = 1.0/NodalArea;
416  Matrix& rNodalStress = it_node->FastGetSolutionStepValue(NODAL_EFFECTIVE_STRESS_TENSOR);
417  for(unsigned int i = 0; i<3; i++) // Dimension
418  {
419  for(unsigned int j = 0; j<3; j++)
420  {
421  rNodalStress(i,j) *= InvNodalArea;
422  }
423  }
424  }
425  }
426  }
