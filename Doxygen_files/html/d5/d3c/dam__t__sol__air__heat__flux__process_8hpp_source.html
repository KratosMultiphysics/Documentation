112  // Computing the t_soil_air according to t_sol_air criteria
113  double t_sol_air = mAmbientTemperature + (mAbsorption_index * mTotalInsolation / mH0) - (mEmisivity * mDeltaR / mH0);
114 
115  if (nnodes != 0)
