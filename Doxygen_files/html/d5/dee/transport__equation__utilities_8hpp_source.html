128  [[nodiscard]] static double CalculateBulkModulus(const Matrix& rConstitutiveMatrix)
129  {
130  KRATOS_ERROR_IF(rConstitutiveMatrix.size1() == 0)
131  << "Constitutive matrix is empty, aborting bulk modulus calculation.\n";

168  [[nodiscard]] static double CalculateInverseBiotModulus(double BiotCoefficient,
169  double DegreeOfSaturation,
170  double DerivativeOfSaturation,
171  const Properties& rProperties)
172  {
173  const auto bulk_modulus_fluid = rProperties[IGNORE_UNDRAINED] ? TINY : rProperties[BULK_MODULUS_FLUID];
174  double result = (BiotCoefficient - rProperties[POROSITY]) / rProperties[BULK_MODULUS_SOLID] +
175  rProperties[POROSITY] / bulk_modulus_fluid;
176 
177  result *= DegreeOfSaturation;
178  result -= DerivativeOfSaturation * rProperties[POROSITY];
179 
180  return result;
181  }

183  [[nodiscard]] static double CalculatePermeabilityUpdateFactor(const Vector& rStrainVector,
184  const Properties& rProperties)
185  {
186  if (rProperties[PERMEABILITY_CHANGE_INVERSE_FACTOR] > 0.0) {
187  const double InverseCK = rProperties[PERMEABILITY_CHANGE_INVERSE_FACTOR];
188  const double epsV = StressStrainUtilities::CalculateTrace(rStrainVector);
189  const double ePrevious = rProperties[POROSITY] / (1.0 - rProperties[POROSITY]);
190  const double eCurrent = (1.0 + ePrevious) * std::exp(epsV) - 1.0;
191  const double permLog10 = (eCurrent - ePrevious) * InverseCK;
192  return std::pow(10.0, permLog10);
193  }
194 
195  return 1.0;
