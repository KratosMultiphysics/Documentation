171  [[nodiscard]] static double CalculateInverseBiotModulus(double BiotCoefficient,
172  double DegreeOfSaturation,
173  double DerivativeOfSaturation,
174  const Properties& rProperties)
175  {
176  const auto bulk_modulus_fluid = rProperties[IGNORE_UNDRAINED] ? TINY : rProperties[BULK_MODULUS_FLUID];
177  double result = (BiotCoefficient - rProperties[POROSITY]) / rProperties[BULK_MODULUS_SOLID] +
178  rProperties[POROSITY] / bulk_modulus_fluid;
179 
180  result *= DegreeOfSaturation;
181  result -= DerivativeOfSaturation * rProperties[POROSITY];
182 
183  return result;
184  }

186  [[nodiscard]] static double CalculatePermeabilityUpdateFactor(const Vector& rStrainVector,
187  const Properties& rProperties)
188  {
189  if (rProperties[PERMEABILITY_CHANGE_INVERSE_FACTOR] > 0.0) {
190  const double InverseCK = rProperties[PERMEABILITY_CHANGE_INVERSE_FACTOR];
191  const double epsV = StressStrainUtilities::CalculateTrace(rStrainVector);
192  const double ePrevious = rProperties[POROSITY] / (1.0 - rProperties[POROSITY]);
193  const double eCurrent = (1.0 + ePrevious) * std::exp(epsV) - 1.0;
194  const double permLog10 = (eCurrent - ePrevious) * InverseCK;
195  return std::pow(10.0, permLog10);
196  }
197 
198  return 1.0;
