154 
155  for (int i=0; i<sizeOfArray; ++i) {
156  input_file >> data[i];

158

161  RemoveFile(rFileName);
162 
163  EMPIRE_API_LOG(2) << "Finished receiving array" << std::endl;

218 static void EMPIRE_API_Connect(const char* inputFileName)
219 {
220  const std::string file_name(inputFileName);
221 
222  if (EMPIRE_API_helpers::StringEndsWith(file_name, ".xml")) {
223  EMPIRE_API_LOG(0) << "Called \"EMPIRE_API_Connect\" with an xml-file, which is no longer supported.\nPlease pass a config file instead" << std::endl;
224  } else {
225  if (EMPIRE_API_helpers::FileExists(file_name)) {
226  // reading config-file
227  std::ifstream input_file(file_name);
228  EMPIRE_API_helpers::CheckStream(input_file, file_name);
229 
230  // reading file
231  std::string current_line;
232 
233  while (std::getline(input_file, current_line)) {
234  if (current_line.find("EchoLevel") != std::string::npos) {
235  EMPIRE_API_helpers::ReadNumberAfterKeyword("EchoLevel", current_line, EMPIRE_API_helpers::EchoLevel);
236  }
237  if (current_line.find("PrintTiming") != std::string::npos) {
238  EMPIRE_API_helpers::ReadNumberAfterKeyword("PrintTiming", current_line, EMPIRE_API_helpers::PrintTiming);
239  }
240  }
241  } else {
242  EMPIRE_API_LOG(0) << "Config-file \"" << file_name << "\" not found!" << std::endl;
243  }
244  }
245 
246  EMPIRE_API_LOG(0) << "Configuration:\n"
247  << " EchoLevel: " << EMPIRE_API_helpers::EchoLevel << "\n"
248  << " PrintTiming: " << EMPIRE_API_helpers::PrintTiming << std::endl;
249 }

263  * \brief Send the mesh to the server
264  * \param[in] name name of the mesh
265  * \param[in] numNodes number of nodes
266  * \param[in] numElems number of elements
267  * \param[in] nodes coordinates of all nodes
268  * \param[in] nodeIDs IDs of all nodes
269  * \param[in] numNodesPerElem number of nodes per element
270  * \param[in] elems connectivity table of all elements
271  ***********/
272 static void EMPIRE_API_sendMesh(const char *name, const int numNodes, const int numElems, const double *nodes, const int *nodeIDs, const int *numNodesPerElem, const int *elems)
273 {
274  EMPIRE_API_LOG(2) << "Attempting to send mesh \"" << std::string(name) << "\" with " << numNodes << " Nodes | " << numElems << " Elements ..." << std::endl;
275 
276  const auto start_time(std::chrono::steady_clock::now());
277 
278  const std::string file_name(EMPIRE_API_helpers::GetFullPath("EMPIRE_mesh_" + std::string(name) + ".vtk"));
279 
280  std::ofstream output_file;
281  output_file.open(EMPIRE_API_helpers::GetTempFileName(file_name));
282  EMPIRE_API_helpers::CheckStream(output_file, file_name);
283 
284  output_file << std::scientific << std::setprecision(7); // TODO maybe this should be configurable
285 
286  // write file header
287  output_file << "# vtk DataFile Version 4.0\n";
288  output_file << "vtk output\n";
289  output_file << "ASCII\n";
290  output_file << "DATASET UNSTRUCTURED_GRID\n\n";
291 
292  // write nodes
293  int vtk_id = 0;
294  std::unordered_map<int, int> node_vtk_id_map;
295  output_file << "POINTS " << numNodes << " float\n";
296  for (int i=0; i<numNodes; ++i) {
297  output_file << nodes[i*3] << " " << nodes[i*3+1] << " " << nodes[i*3+2] << "\n";
298  node_vtk_id_map[nodeIDs[i]] = vtk_id++;
299  }
300  output_file << "\n";
301 
302  // write cells connectivity
303  int cell_list_size = 0;
304  for (int i=0; i<numElems; ++i) {
305  cell_list_size += numNodesPerElem[i] + 1;
306  }
307 
308  int counter=0;
309  output_file << "CELLS " << numElems << " " << cell_list_size << "\n";
310  for (int i=0; i<numElems; ++i) {
311  const int num_nodes_elem = numNodesPerElem[i];
312  output_file << num_nodes_elem << " ";
313  for (int j=0; j<num_nodes_elem; ++j) {
314  output_file << node_vtk_id_map.at(elems[counter++]);
315  if (j<num_nodes_elem-1) output_file << " "; // not adding a whitespace after last number
316  }
317  output_file << "\n";
318  }
319 
320  output_file << "\n";
321 
322  // write cell types
323  output_file << "CELL_TYPES " << numElems << "\n";
324  for (int i=0; i<numElems; ++i) {
325  output_file << EMPIRE_API_helpers::GetVtkCellType(numNodesPerElem[i]) << "\n";
326  }
327 
328  output_file.close();
329  EMPIRE_API_helpers::MakeFileVisible(file_name);
330 
331  EMPIRE_API_LOG(2) << "Finished sending mesh" << std::endl;
332 
333  if (EMPIRE_API_helpers::PrintTiming) {
334  EMPIRE_API_LOG(0) << "Sending Mesh \"" << file_name << "\" took: " << EMPIRE_API_helpers::ElapsedSeconds(start_time) << " [sec]" << std::endl;
335  }
336 }

339  * \brief Recieve mesh from the server
340  * \param[in] name name of the mesh
341  * \param[in] numNodes number of nodes
342  * \param[in] numElems number of elements
343  * \param[in] nodes coordinates of all nodes
344  * \param[in] nodeIDs IDs of all nodes
345  * \param[in] numNodesPerElem number of nodes per element
346  * \param[in] elems connectivity table of all elements
347  ***********/
348 template <typename TDouble, typename TInt>
349 static void EMPIRE_API_recvMesh(const char *name, int *numNodes, int *numElems, TDouble* nodes, TInt* nodeIDs, TInt* numNodesPerElem, TInt* elem)
350 {
351  EMPIRE_API_LOG(2) << "Attempting to receive mesh \"" << std::string(name) << "\" ..." << std::endl;
352 
353  const std::string file_name(EMPIRE_API_helpers::GetFullPath("EMPIRE_mesh_" + std::string(name) + ".vtk"));
354 
355  EMPIRE_API_helpers::WaitForFile(file_name);
356 
357  const auto start_time(std::chrono::steady_clock::now());
358 
359  std::ifstream input_file(file_name);
360  EMPIRE_API_helpers::CheckStream(input_file, file_name);
361 
362  // reading file
363  std::string current_line;
364  bool nodes_read = false;
365 
366  while (std::getline(input_file, current_line)) {
367  // reading nodes
368  if (current_line.find("POINTS") != std::string::npos) {
369  if (nodes_read) throw std::runtime_error("The nodes were read already!");
370  nodes_read = true;
371 
372  EMPIRE_API_helpers::ReadNumberAfterKeyword("POINTS", current_line, *numNodes);
373 
374  EMPIRE_API_LOG(2) << "Mesh contains " << *numNodes << " Nodes" << std::endl;
375 
376  // allocating memory for nodes
377  // note that this has to be deleted by the client!
378  EMPIRE_API_helpers::AllocateMemory(nodes, (*numNodes) * 3); // *nodes = new double[(*numNodes) * 3];
379  EMPIRE_API_helpers::AllocateMemory(nodeIDs, *numNodes); // *nodeIDs = new int[*numNodes];
380 
381  for (int i=0; i<*numNodes*3; ++i) {
382  input_file >> (*nodes)[i];
383  }
384 
385  for (int i=0; i<*numNodes; ++i) {
386  (*nodeIDs)[i] = i+1; // Node Ids have an offset of 1 from Kratos to VTK
387  }
388  }
389 
390  // reading elements
391  if (current_line.find("CELLS") != std::string::npos) {
392  if (!nodes_read) throw std::runtime_error("The nodes were not yet read!");
393 
394  int num_nodes_per_elem, node_id, cell_list_size;
395  current_line = current_line.substr(current_line.find("CELLS") + 6); // removing "CELLS"
396  std::istringstream line_stream(current_line);
397  line_stream >> *numElems;
398  line_stream >> cell_list_size;
399 
400  EMPIRE_API_LOG(2) << "Mesh contains " << *numElems << " Elements" << std::endl;
401 
402  // allocating memory for elements
403  // note that this has to be deleted by the client!
404  EMPIRE_API_helpers::AllocateMemory(numNodesPerElem, *numElems); // *numNodesPerElem = new int[*numElems];
405  EMPIRE_API_helpers::AllocateMemory(elem, cell_list_size-*numElems); // *elem = new int[cell_list_size-*numElems];
406 
407  int counter=0;
408  for (int i=0; i<*numElems; ++i) {
409  input_file >> num_nodes_per_elem;
410  (*numNodesPerElem)[i] = num_nodes_per_elem;
411  for (int j=0; j<num_nodes_per_elem; ++j) {
412  input_file >> node_id;
413  (*elem)[counter++] = node_id+1; // Node Ids have an offset of 1 from Kratos to VTK
414  }
415  }
416  break; // no further information reading required => CELL_TYPES are not used here
417  }
418  }
419 
420  input_file.close();
421 
422  EMPIRE_API_helpers::RemoveFile(file_name);
423 
424  EMPIRE_API_LOG(2) << "Finished receiving mesh" << std::endl;
425 
426  if (EMPIRE_API_helpers::PrintTiming) {
427  EMPIRE_API_LOG(0) << "Receiving Mesh \"" << file_name << "\" took: " << EMPIRE_API_helpers::ElapsedSeconds(start_time) << " [sec]" << std::endl;
428  }
429 }
