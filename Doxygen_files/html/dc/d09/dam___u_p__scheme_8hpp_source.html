182  void Predict(
183  ModelPart& r_model_part,
184  DofsArrayType& rDofSet,
185  TSystemMatrixType& A,
186  TSystemVectorType& Dx,
187  TSystemVectorType& b) override
188  {
189 
190  KRATOS_TRY
191 
192  // Updating Velocities, Acclerations, DtPressure and Dt2Pressure
193  array_1d<double,3> DeltaDisplacement;
194  double DeltaPressure;
195 
196  const int NNodes = static_cast<int>(r_model_part.Nodes().size());
197  ModelPart::NodesContainerType::iterator node_begin = r_model_part.NodesBegin();
198 
199  #pragma omp parallel for private(DeltaDisplacement,DeltaPressure)
200  for(int i = 0; i < NNodes; i++)
201  {
202  ModelPart::NodesContainerType::iterator itNode = node_begin + i;
203 
204  // Termes related to displacement field
205 
206  array_1d<double,3>& CurrentDisplacement = itNode->FastGetSolutionStepValue(DISPLACEMENT);
207  array_1d<double,3>& CurrentAcceleration = itNode->FastGetSolutionStepValue(ACCELERATION);
208  array_1d<double,3>& CurrentVelocity = itNode->FastGetSolutionStepValue(VELOCITY);
209 
210  const array_1d<double,3>& PreviousDisplacement = itNode->FastGetSolutionStepValue(DISPLACEMENT, 1);
211  const array_1d<double,3>& PreviousAcceleration = itNode->FastGetSolutionStepValue(ACCELERATION, 1);
212  const array_1d<double,3>& PreviousVelocity = itNode->FastGetSolutionStepValue(VELOCITY, 1);
213 
214  if (itNode -> IsFixed(ACCELERATION_X))
215  {
216  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime * PreviousVelocity[0] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[0] + mBeta * CurrentAcceleration[0] );
217  }
218  else if (itNode -> IsFixed(VELOCITY_X))
219  {
220  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[0]-PreviousVelocity[0])+PreviousVelocity[0]);
221  }
222  else if (itNode -> IsFixed(DISPLACEMENT_X) == false)
223  {
224  CurrentDisplacement[0] = PreviousDisplacement[0] + mDeltaTime * PreviousVelocity[0] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[0];
225  }
226 
227  if (itNode -> IsFixed(ACCELERATION_Y))
228  {
229  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime * PreviousVelocity[1] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[1] + mBeta * CurrentAcceleration[1] );
230  }
231  else if (itNode -> IsFixed(VELOCITY_Y))
232  {
233  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[1]-PreviousVelocity[1])+PreviousVelocity[1]);
234  }
235  else if (itNode -> IsFixed(DISPLACEMENT_Y) == false)
236  {
237  CurrentDisplacement[1] = PreviousDisplacement[1] + mDeltaTime * PreviousVelocity[1] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[1];
238  }
239 
240  // For 3D cases
241  if (itNode -> HasDofFor(DISPLACEMENT_Z))
242  {
243  if (itNode -> IsFixed(ACCELERATION_Z))
244  {
245  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime * PreviousVelocity[2] + std::pow(mDeltaTime, 2) * ( ( 0.5 - mBeta) * PreviousAcceleration[2] + mBeta * CurrentAcceleration[2] );
246  }
247  else if (itNode -> IsFixed(VELOCITY_Z))
248  {
249  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime*(mBeta/mGamma*(CurrentVelocity[2]-PreviousVelocity[2])+PreviousVelocity[2]);
250  }
251  else if (itNode -> IsFixed(DISPLACEMENT_Z) == false)
252  {
253  CurrentDisplacement[2] = PreviousDisplacement[2] + mDeltaTime * PreviousVelocity[2] + 0.5 * std::pow(mDeltaTime, 2) * PreviousAcceleration[2];
254  }
255  }
256 
257  noalias(DeltaDisplacement) = CurrentDisplacement - PreviousDisplacement;
258 
259  noalias(CurrentAcceleration) = 1.0/(mBeta*mDeltaTime*mDeltaTime)*(DeltaDisplacement - mDeltaTime*PreviousVelocity - (0.5-mBeta)*mDeltaTime*mDeltaTime*PreviousAcceleration);
260  noalias(CurrentVelocity) = PreviousVelocity + (1.0-mGamma)*mDeltaTime*PreviousAcceleration + mGamma*mDeltaTime*CurrentAcceleration;
261 
262  // Terms related to Pressure field
263 
264  double& CurrentDt2Pressure = itNode->FastGetSolutionStepValue(Dt2_PRESSURE);
265  double& CurrentDtPressure = itNode->FastGetSolutionStepValue(Dt_PRESSURE);
266  DeltaPressure = itNode->FastGetSolutionStepValue(PRESSURE) - itNode->FastGetSolutionStepValue(PRESSURE, 1);
267  const double& PreviousDt2Pressure = itNode->FastGetSolutionStepValue(Dt2_PRESSURE, 1);
268  const double& PreviousDtPressure = itNode->FastGetSolutionStepValue(Dt_PRESSURE, 1);
269 
270  CurrentDt2Pressure = 1.0/(mBeta*mDeltaTime*mDeltaTime)*(DeltaPressure - mDeltaTime*PreviousDtPressure - (0.5-mBeta)*mDeltaTime*mDeltaTime*PreviousDt2Pressure);
271  CurrentDtPressure = PreviousDtPressure + (1.0-mGamma)*mDeltaTime*PreviousDt2Pressure + mGamma*mDeltaTime*CurrentDt2Pressure;
272 
273  }
274 
275  KRATOS_CATCH( "" )
276  }

586  void AddDynamicsToLHS(LocalSystemMatrixType& LHS_Contribution, LocalSystemMatrixType& M, LocalSystemMatrixType& C, const ProcessInfo& CurrentProcessInfo)
587  {
588  // adding mass contribution
589  if (M.size1() != 0)
590  noalias(LHS_Contribution) += 1.0/(mBeta*mDeltaTime*mDeltaTime)*M;
591 
592  // adding damping contribution
593  if (C.size1() != 0)
594  noalias(LHS_Contribution) += mGamma/(mBeta*mDeltaTime)*C;
595  }
