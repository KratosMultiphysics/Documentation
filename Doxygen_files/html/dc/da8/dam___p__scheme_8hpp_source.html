58  int Check(ModelPart& r_model_part) override
59  {
60  KRATOS_TRY
61 
62  //check that variables are correctly allocated
63  for(ModelPart::NodesContainerType::iterator it=r_model_part.NodesBegin(); it!=r_model_part.NodesEnd(); it++)
64  {
65  if(it->SolutionStepsDataHas(PRESSURE) == false)
66  KRATOS_THROW_ERROR( std::logic_error, "PRESSURE variable is not allocated for node ", it->Id() )
67  if(it->SolutionStepsDataHas(Dt_PRESSURE) == false)
68  KRATOS_THROW_ERROR( std::logic_error, "Dt_PRESSURE variable is not allocated for node ", it->Id() )
69  if(it->SolutionStepsDataHas(Dt2_PRESSURE) == false)
70  KRATOS_THROW_ERROR( std::logic_error, "Dt2_PRESSURE variable is not allocated for node ", it->Id() )
71 
72  if(it->HasDofFor(PRESSURE) == false)
73  KRATOS_THROW_ERROR( std::invalid_argument,"missing PRESSURE dof on node ",it->Id() )
74  }
75 
76  //check for minimum value of the buffer index.
77  if (r_model_part.GetBufferSize() < 2)
78  KRATOS_THROW_ERROR( std::logic_error, "insufficient buffer size. Buffer size should be greater than 2. Current size is", r_model_part.GetBufferSize() )
79 
80  // Check beta, gamma and theta
81  if(mBeta <= 0.0 || mGamma<= 0.0)
82  KRATOS_THROW_ERROR( std::invalid_argument,"Some of the scheme variables: beta or gamma has an invalid value ", "" )
83 
84  return 0;
85 
86  KRATOS_CATCH( "" )
