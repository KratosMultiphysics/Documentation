784  double deltaY = rGeom[idB].Y() - rGeom[idA].Y();
785  double elementSize = sqrt(deltaX * deltaX + deltaY * deltaY);
786  if (fabs(deltaX) > fabs(deltaY))
787  { //to avoid division by zero or small numbers
788  NormalVector[0] = -deltaY / deltaX;
789  NormalVector[1] = 1.0;
790  double normNormal = NormalVector[0] * NormalVector[0] + NormalVector[1] * NormalVector[1];
791  NormalVector *= 1.0 / sqrt(normNormal);
792  }
793  else
794  {
795  NormalVector[0] = 1.0;
796  NormalVector[1] = -deltaX / deltaY;
797  double normNormal = NormalVector[0] * NormalVector[0] + NormalVector[1] * NormalVector[1];
798  NormalVector *= 1.0 / sqrt(normNormal);
799  }

804  const double normA = DistanceA[0] * DistanceA[0] + DistanceA[1] * DistanceA[1];
805  const double normB = DistanceB[0] * DistanceB[0] + DistanceB[1] * DistanceB[1];
