90 ModelPart& Determine2DModelPart(
91  ModelPart& rFirstModelPart,
92  ModelPart& rSecondModelPart
93  )
94 {
95  // Getting the maximum local space dimension
96  const unsigned int max_local_space_dimension_1 = DetermineModelPartMaximumLocalDimension(rFirstModelPart);
97  const unsigned int max_local_space_dimension_2 = DetermineModelPartMaximumLocalDimension(rSecondModelPart);
98  KRATOS_ERROR_IF(max_local_space_dimension_1 == 3 && max_local_space_dimension_2 == 3) << "Both model parts are 3D" << std::endl;
99  KRATOS_ERROR_IF(max_local_space_dimension_1 == 1 || max_local_space_dimension_2 == 1) << "One model part is 1D, not compatible" << std::endl;
100  KRATOS_ERROR_IF(max_local_space_dimension_1 == 0 || max_local_space_dimension_2 == 0) << "Impossible to determine local space dimension in at least one model part" << std::endl;
101  if (max_local_space_dimension_1 == 2) {
102  return rFirstModelPart;
103  } else if (max_local_space_dimension_2 == 2) {
104  return rSecondModelPart;
105  } else { // Corner case a priori impossible
106  KRATOS_ERROR << "Impossible to detect 2D model part" << std::endl;
107  }
108  return rFirstModelPart;

451  void GetNormalAndReferencePlane()
452  {
453  KRATOS_TRY;
454 
455  // We retrieve the values of interest
456  const auto& r_2d_model_part = this->Get2DModelPart();
457  const bool is_distributed = r_2d_model_part.IsDistributed();
458  auto p_geometry = GetGeometryFromModelPart(r_2d_model_part);
459 
460  // MPI data
461  const auto& r_communicator = r_2d_model_part.GetCommunicator();
462  const auto& r_data_communicator = r_communicator.GetDataCommunicator();
463  const int mpi_rank = r_data_communicator.Rank();
464  const int mpi_size = r_data_communicator.Size();
465 
466  // Getting the partition with entities index
467  const int partition_entity = DeterminePartitionWithEntities(r_2d_model_part);
468 
469  // Define the send tag
470  const int tag_send_normal = 1;
471  const int tag_send_point = 2;
472 
473  // Getting from parameters if not elements or conditions
474  if (partition_entity != mpi_rank) {
475  // Now transfer the normal plane and the point plane between partitions
476  if (is_distributed) {
477  // The partitions that receive
478  r_data_communicator.Recv(mNormalPlane, partition_entity, tag_send_normal);
479  r_data_communicator.Recv(mPointPlane.Coordinates(), partition_entity, tag_send_point);
480  }
481  } else {
482  GeometryType::CoordinatesArrayType aux_coords;
483  noalias(mPointPlane.Coordinates()) = p_geometry->Center();
484  p_geometry->PointLocalCoordinates(aux_coords, mPointPlane);
485  const bool is_pure_2d_geometry = p_geometry->LocalSpaceDimension() == p_geometry->WorkingSpaceDimension() ? true : false;
486  if (is_pure_2d_geometry) {
487  mNormalPlane[0] = 0.0;
488  mNormalPlane[1] = 0.0;
489  mNormalPlane[2] = 1.0;
490  } else {
491  noalias(mNormalPlane) = p_geometry->UnitNormal(aux_coords);
492  }
493 
494  // Doing a check that all normals are aligned
495  if (!is_pure_2d_geometry) {
496  std::size_t check_normal;
497  const double numerical_limit = std::numeric_limits<double>::epsilon() * 1.0e4;
498  struct normal_check {
499  normal_check(array_1d<double, 3>& rNormal) : reference_normal(rNormal) {};
500  array_1d<double, 3> reference_normal;
501  GeometryType::CoordinatesArrayType aux_coords;
502  };
503  if (mEntityTypeMesh == EntityTypeMesh::CONDITIONS) {
504  check_normal = block_for_each<SumReduction<std::size_t>>(r_2d_model_part.Conditions(), normal_check(mNormalPlane), [&numerical_limit](auto& r_cond, normal_check& nc) {
505  auto& r_geom = r_cond.GetGeometry();
506  r_geom.PointLocalCoordinates(nc.aux_coords, r_geom.Center());
507  const auto normal = r_geom.UnitNormal(nc.aux_coords);
508  return (norm_2(normal - nc.reference_normal) > numerical_limit);
509  });
510  } else {
511  check_normal = block_for_each<SumReduction<std::size_t>>(r_2d_model_part.Elements(), normal_check(mNormalPlane), [&numerical_limit](auto& r_elem, normal_check& nc) {
512  auto& r_geom = r_elem.GetGeometry();
513  r_geom.PointLocalCoordinates(nc.aux_coords, r_geom.Center());
514  const auto normal = r_geom.UnitNormal(nc.aux_coords);
515  return (norm_2(normal - nc.reference_normal) > numerical_limit);
516  });
517  }
518  KRATOS_ERROR_IF_NOT(check_normal == 0) << "The 2D reference model part has not consistent normals. Please check that is properly aligned" << std::endl;
519  }
520 
521  // The partition that sends
522  if (is_distributed) {
523  const auto& r_point_coordinates = mPointPlane.Coordinates();
524  for (int i_rank = 0; i_rank < mpi_size; ++i_rank) {
525  if (i_rank != partition_entity) {
526  r_data_communicator.Send(mNormalPlane, i_rank, tag_send_normal);
527  r_data_communicator.Send(r_point_coordinates, i_rank, tag_send_point);
528  }
529  }
530  }
531  }
532 
533  KRATOS_CATCH("");
534  }
