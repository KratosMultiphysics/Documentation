90 ModelPart& Determine2DModelPart(ModelPart& rFirstModelPart, ModelPart& rSecondModelPart)
91 {
92  // Getting the maximum local space dimension
93  const unsigned int max_local_space_dimension_1 = DetermineModelPartMaximumLocalDimension(rFirstModelPart);
94  const unsigned int max_local_space_dimension_2 = DetermineModelPartMaximumLocalDimension(rSecondModelPart);
95  KRATOS_ERROR_IF(max_local_space_dimension_1 == 3 && max_local_space_dimension_2 == 3) << "Both model parts are 3D" << std::endl;
96  KRATOS_ERROR_IF(max_local_space_dimension_1 == 1 || max_local_space_dimension_2 == 1) << "One model part is 1D, not compatible" << std::endl;
97  KRATOS_ERROR_IF(max_local_space_dimension_1 == 0 || max_local_space_dimension_2 == 0) << "Impossible to determine local space dimension in at least one model part" << std::endl;
98  if (max_local_space_dimension_1 == 2) {
99  return rFirstModelPart;
100  } else if (max_local_space_dimension_2 == 2) {
101  return rSecondModelPart;
102  } else { // Corner case a priori impossible
103  KRATOS_ERROR << "Impossible to detect 2D model part" << std::endl;
104  }
105  return rFirstModelPart;
106 }

445  void GetNormalAndReferencePlane()
446  {
447  KRATOS_TRY;
448 
449  // We retrieve the values of interest
450  const auto& r_2d_model_part = this->Get2DModelPart();
451  const bool is_distributed = r_2d_model_part.IsDistributed();
452  auto p_geometry = GetGeometryFromModelPart(r_2d_model_part);
453 
454  // MPI data
455  const auto& r_communicator = r_2d_model_part.GetCommunicator();
456  const auto& r_data_communicator = r_communicator.GetDataCommunicator();
457  const int mpi_rank = r_data_communicator.Rank();
458  const int mpi_size = r_data_communicator.Size();
459 
460  // Getting the partition with entities index
461  const int partition_entity = DeterminePartitionWithEntities(r_2d_model_part);
462 
463  // Define the send tag
464  const int tag_send_normal = 1;
465  const int tag_send_point = 2;
466 
467  // Getting from parameters if not elements or conditions
468  if (partition_entity != mpi_rank) {
469  // Now transfer the normal plane and the point plane between partitions
470  if (is_distributed) {
471  // The partitions that receive
472  r_data_communicator.Recv(mNormalPlane, partition_entity, tag_send_normal);
473  r_data_communicator.Recv(mPointPlane.Coordinates(), partition_entity, tag_send_point);
474  }
475  } else {
476  GeometryType::CoordinatesArrayType aux_coords;
477  noalias(mPointPlane.Coordinates()) = p_geometry->Center();
478  p_geometry->PointLocalCoordinates(aux_coords, mPointPlane);
479  noalias(mNormalPlane) = p_geometry->UnitNormal(aux_coords);
480 
481  // Doing a check that all normals are aligned
482  std::size_t check_normal;
483  const double numerical_limit = std::numeric_limits<double>::epsilon() * 1.0e4;
484  struct normal_check {
485  normal_check(array_1d<double, 3>& rNormal) : reference_normal(rNormal) {};
486  array_1d<double, 3> reference_normal;
487  GeometryType::CoordinatesArrayType aux_coords;
488  };
489  if (mEntityTypeMesh == EntityTypeMesh::CONDITIONS) {
490  check_normal = block_for_each<SumReduction<std::size_t>>(r_2d_model_part.Conditions(), normal_check(mNormalPlane), [&numerical_limit](auto& r_cond, normal_check& nc) {
491  auto& r_geom = r_cond.GetGeometry();
492  r_geom.PointLocalCoordinates(nc.aux_coords, r_geom.Center());
493  const auto normal = r_geom.UnitNormal(nc.aux_coords);
494  return (norm_2(normal - nc.reference_normal) > numerical_limit);
495  });
496  } else {
497  check_normal = block_for_each<SumReduction<std::size_t>>(r_2d_model_part.Elements(), normal_check(mNormalPlane), [&numerical_limit](auto& r_elem, normal_check& nc) {
498  auto& r_geom = r_elem.GetGeometry();
499  r_geom.PointLocalCoordinates(nc.aux_coords, r_geom.Center());
500  const auto normal = r_geom.UnitNormal(nc.aux_coords);
501  return (norm_2(normal - nc.reference_normal) > numerical_limit);
502  });
503  }
504  KRATOS_ERROR_IF_NOT(check_normal == 0) << "The 2D reference model part has not consistent normals. Please check that is properly aligned" << std::endl;
505 
506  // The partition that sends
507  if (is_distributed) {
508  const auto& r_point_coordinates = mPointPlane.Coordinates();
509  for (int i_rank = 0; i_rank < mpi_size; ++i_rank) {
510  if (i_rank != partition_entity) {
511  r_data_communicator.Send(mNormalPlane, i_rank, tag_send_normal);
512  r_data_communicator.Send(r_point_coordinates, i_rank, tag_send_point);
513  }
514  }
515  }
516  }
517 
518  KRATOS_CATCH("");
519  }
