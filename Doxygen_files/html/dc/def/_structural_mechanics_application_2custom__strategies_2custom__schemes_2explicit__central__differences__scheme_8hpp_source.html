411  void UpdateTranslationalDegreesOfFreedom(
412  NodeIterator itCurrentNode,
413  const IndexType DisplacementPosition,
414  const SizeType DomainSize = 3
415  )
416  {
417 
418  const double nodal_mass = itCurrentNode->GetValue(NODAL_MASS);
419  const double nodal_displacement_damping = itCurrentNode->GetValue(NODAL_DISPLACEMENT_DAMPING);
420  const array_1d<double, 3>& r_current_residual = itCurrentNode->FastGetSolutionStepValue(FORCE_RESIDUAL);
421 
422 
423  array_1d<double, 3>& r_current_velocity = itCurrentNode->FastGetSolutionStepValue(VELOCITY);
424  array_1d<double, 3>& r_current_displacement = itCurrentNode->FastGetSolutionStepValue(DISPLACEMENT);
425  array_1d<double, 3>& r_middle_velocity = itCurrentNode->FastGetSolutionStepValue(MIDDLE_VELOCITY);
426 
427  array_1d<double, 3>& r_current_acceleration = itCurrentNode->FastGetSolutionStepValue(ACCELERATION);
428 
429  const array_1d<double, 3>& r_previous_displacement = itCurrentNode->FastGetSolutionStepValue(DISPLACEMENT, 1);
430  const array_1d<double, 3>& r_previous_middle_velocity = itCurrentNode->FastGetSolutionStepValue(MIDDLE_VELOCITY, 1);
431  // Solution of the explicit equation:
432  if (nodal_mass > numerical_limit)
433  noalias(r_current_acceleration) = (r_current_residual - nodal_displacement_damping * r_current_velocity) / nodal_mass;
434  else
435  noalias(r_current_acceleration) = ZeroVector(3);
436 
437  std::array<bool, 3> fix_displacements = {false, false, false};
438 
439  fix_displacements[0] = (itCurrentNode->GetDof(DISPLACEMENT_X, DisplacementPosition).IsFixed());
440  fix_displacements[1] = (itCurrentNode->GetDof(DISPLACEMENT_Y, DisplacementPosition + 1).IsFixed());
441  if (DomainSize == 3)
442  fix_displacements[2] = (itCurrentNode->GetDof(DISPLACEMENT_Z, DisplacementPosition + 2).IsFixed());
443 
444  for (IndexType j = 0; j < DomainSize; j++) {
445  if (fix_displacements[j]) {
446  r_current_acceleration[j] = 0.0;
447  r_middle_velocity[j] = 0.0;
448  }
449 
450  r_current_velocity[j] = r_previous_middle_velocity[j] + (mTime.Previous - mTime.PreviousMiddle) * r_current_acceleration[j]; //+ actual_velocity;
451  r_middle_velocity[j] = r_current_velocity[j] + (mTime.Middle - mTime.Previous) * r_current_acceleration[j];
452  r_current_displacement[j] = r_previous_displacement[j] + mTime.Delta * r_middle_velocity[j];
453  } // for DomainSize
454  }

462  void UpdateRotationalDegreesOfFreedom(
463  NodeIterator itCurrentNode,
464  const IndexType RotationPosition,
465  const SizeType DomainSize = 3
466  )
467  {
469  const array_1d<double, 3>& nodal_inertia = itCurrentNode->GetValue(NODAL_INERTIA);
470  const array_1d<double, 3>& nodal_rotational_damping = itCurrentNode->GetValue(NODAL_ROTATION_DAMPING);
471  const array_1d<double, 3>& r_current_residual_moment = itCurrentNode->FastGetSolutionStepValue(MOMENT_RESIDUAL);
472  array_1d<double, 3>& r_current_angular_velocity = itCurrentNode->FastGetSolutionStepValue(ANGULAR_VELOCITY);
473  array_1d<double, 3>& r_current_rotation = itCurrentNode->FastGetSolutionStepValue(ROTATION);
474  array_1d<double, 3>& r_middle_angular_velocity = itCurrentNode->FastGetSolutionStepValue(MIDDLE_ANGULAR_VELOCITY);
475  array_1d<double, 3>& r_current_angular_acceleration = itCurrentNode->FastGetSolutionStepValue(ANGULAR_ACCELERATION);
476 
477 
478  const array_1d<double, 3>& r_previous_rotation = itCurrentNode->FastGetSolutionStepValue(ROTATION, 1);
479  const array_1d<double, 3>& r_previous_middle_angular_velocity = itCurrentNode->FastGetSolutionStepValue(MIDDLE_ANGULAR_VELOCITY, 1);
480 
481  const IndexType initial_k = DomainSize == 3 ? 0 : 2; // We do this because in 2D only the rotation Z is needed, then we start with 2, instead of 0
482  for (IndexType kk = initial_k; kk < 3; ++kk) {
483  if (nodal_inertia[kk] > numerical_limit)
484  r_current_angular_acceleration[kk] = (r_current_residual_moment[kk] - nodal_rotational_damping[kk] * r_current_angular_velocity[kk]) / nodal_inertia[kk];
485  else
486  r_current_angular_acceleration[kk] = 0.0;
487  }
488 
489  std::array<bool, 3> fix_rotation = {false, false, false};
490  if (DomainSize == 3) {
491  fix_rotation[0] = (itCurrentNode->GetDof(ROTATION_X, RotationPosition).IsFixed());
492  fix_rotation[1] = (itCurrentNode->GetDof(ROTATION_Y, RotationPosition + 1).IsFixed());
493  }
494  fix_rotation[2] = (itCurrentNode->GetDof(ROTATION_Z, RotationPosition + 2).IsFixed());
495 
496  for (IndexType j = initial_k; j < 3; j++) {
497  if (fix_rotation[j]) {
498  r_current_angular_acceleration[j] = 0.0;
499  r_middle_angular_velocity[j] = 0.0;
500  }
501  r_current_angular_velocity[j] = r_previous_middle_angular_velocity[j] + (mTime.Previous - mTime.PreviousMiddle) * r_current_angular_acceleration[j];
502  r_middle_angular_velocity[j] = r_current_angular_velocity[j] + (mTime.Middle - mTime.Previous) * r_current_angular_acceleration[j];
503  r_current_rotation[j] = r_previous_rotation[j] + mTime.Delta * r_middle_angular_velocity[j];
504  }
505  }
