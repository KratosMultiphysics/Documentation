149  ShellQ4_LocalCoordinateSystem CreateLocalCoordinateSystem() const override
150  {
151  const GeometryType & geom = GetGeometry();
152 
153  //return ShellQ4_LocalCoordinateSystem(geom[0], geom[1], geom[2], geom[3]);
154 
155  // reference coordinate system
156  ShellQ4_LocalCoordinateSystem a( CreateReferenceCoordinateSystem() );
157 
158  // current coordinate system using the 1-2 side alignement
159  ShellQ4_LocalCoordinateSystem b( geom[0], geom[1], geom[2], geom[3] );
160 
161  double aX1 = a.X1(); double aY1 = a.Y1();
162  double bX1 = b.X1(); double bY1 = b.Y1();
163  double aX2 = a.X2(); double aY2 = a.Y2();
164  double bX2 = b.X2(); double bY2 = b.Y2();
165  double aX3 = a.X3(); double aY3 = a.Y3();
166  double bX3 = b.X3(); double bY3 = b.Y3();
167  double aX4 = a.X4(); double aY4 = a.Y4();
168  double bX4 = b.X4(); double bY4 = b.Y4();
169 
170  // now we are in the local coordinate systems (reference and current), i.e. we are looking in the local Z direction
171  // which is the same for both coordinate systems.
172  // now we can compute the 2D deformation gradient between the 2 configurations, at the element center.
173 
174  double C1 = 1/(aX1*aY2 - aX2*aY1 - aX1*aY4 + aX2*aY3 - aX3*aY2 + aX4*aY1 + aX3*aY4 - aX4*aY3);
175  double C2 = bY1/4 + bY2/4 - bY3/4 - bY4/4;
176  double C3 = bY1/4 - bY2/4 - bY3/4 + bY4/4;
177  double C4 = bX1/4 + bX2/4 - bX3/4 - bX4/4;
178  double C5 = bX1/4 - bX2/4 - bX3/4 + bX4/4;
179  double C6 = aX1 + aX2 - aX3 - aX4;
180  double C7 = aX1 - aX2 - aX3 + aX4;
181  double C8 = aY1 + aY2 - aY3 - aY4;
182  double C9 = aY1 - aY2 - aY3 + aY4;
183  double f11 = 2*C1*C5*C8 - 2*C1*C4*C9;
184  double f12 = 2*C1*C4*C7 - 2*C1*C5*C6;
185  double f21 = 2*C1*C3*C8 - 2*C1*C2*C9;
186  double f22 = 2*C1*C2*C7 - 2*C1*C3*C6;
187 
188  // now we can extrapolate the rotation angle that makes this deformation gradient symmetric.
189  // F = R*U -> find R such that R'*F = U
190  double alpha = std::atan2( f21 - f12, f11 + f22 );
191 
192  // this final coordinate system is the one in which
193  // the deformation gradient is equal to the stretch tensor
194  return ShellQ4_LocalCoordinateSystem( geom[0], geom[1], geom[2], geom[3], alpha );
195  }

197  VectorType CalculateLocalDisplacements(const ShellQ4_LocalCoordinateSystem & LCS,
198  const VectorType & globalDisplacements) override
199  {
200  const GeometryType & geom = GetGeometry();
201 
202  Vector3Type deformationalDisplacements;
203 
204  QuaternionType Q = QuaternionType::FromRotationMatrix( LCS.Orientation() );
205 
206  const Vector3Type & C = LCS.Center();
207 
208  VectorType localDisplacements(24);
209 
210  TransformationMatrixType T, T0;
211  Q.ToRotationMatrix( T );
212  mQ0.ToRotationMatrix( T0 );
213 
214  for(unsigned int i = 0; i < 4; i++)
215  {
216  unsigned int index = i * 6;
217 
218  // get deformational displacements
219 
220  noalias( deformationalDisplacements ) = prod( T , geom[i] - C );
221  noalias( deformationalDisplacements ) -= prod( T0, geom[i].GetInitialPosition() - mC0 );
222 
223  localDisplacements[index] = deformationalDisplacements[0];
224  localDisplacements[index + 1] = deformationalDisplacements[1];
225  localDisplacements[index + 2] = deformationalDisplacements[2];
226 
227  // get deformational rotations
228 
229  QuaternionType Qd = Q * mQN[i] * mQ0.conjugate();
230 
231  Qd.ToRotationVector( localDisplacements[index + 3],
232  localDisplacements[index + 4],
233  localDisplacements[index + 5] );
234  }
235 
236  return localDisplacements;

239  void FinalizeCalculations(const ShellQ4_LocalCoordinateSystem & LCS,
240  const VectorType & globalDisplacements,
241  const VectorType & localDisplacements,
242  MatrixType & rLeftHandSideMatrix,
243  VectorType & rRightHandSideVector,
244  const bool RHSrequired,
245  const bool LHSrequired) override
246  {
247  // Get the total rotation matrix (local - to - global)
248  // Note: do NOT include the warpage correction matrix!
249  // Explanation:
250  // The Warpage correction matrix computed by the LocalCoordinateSystem is a Linear Projector.
251  // It should be used in a LinearCoordinateTransformation.
252  // Here instead we already calculate a nonlinear Projector (P = Pu - S * G)!
253 
254  MatrixType T(24, 24);
255  LCS.ComputeTotalRotationMatrix( T );
256 
257  // Form all matrices:
258  // S: Spin-Fitter matrix
259  // G: Spin-Lever matrix
260  // P: Projector (Translational & Rotational)
261 
262  MatrixType P( EICR::Compute_Pt(4) );
263  MatrixType S( EICR::Compute_S(LCS.Nodes()) );
264  //MatrixType G( Compute_G(LCS.P1(), LCS.P2(), LCS.P3(), LCS.P4(), LCS.Area()) );
265  MatrixType G( RotationGradient() );
266  noalias( P ) -= prod( S, G );
267 
268  // Compute the projected local forces ( pe = P' * RHS ).
269  // Note: here the RHS is already given as a residual vector -> - internalForces -> (pe = - Ke * U)
270  // so projectedLocalForces = - P' * Ke * U
271 
272  VectorType projectedLocalForces( prod( trans( P ), rRightHandSideVector ) );
273 
274  // Compute the Right-Hand-Side vector in global coordinate system (- T' * P' * Km * U).
275  // At this point the computation of the Right-Hand-Side is complete.
276 
277  noalias( rRightHandSideVector ) = prod( trans( T ), projectedLocalForces );
278 
279  // Begin the computation of the Left-Hand-Side Matrix :
280 
281  if(!LHSrequired) return; // avoid useless calculations!
282 
283  // This is a temporary matrix to store intermediate values
284  // to avoid extra dynamic memory allocations!
285 
286  MatrixType temp(24, 24);
287 
288  // H: Axial Vector Jacobian
289  MatrixType H( EICR::Compute_H(localDisplacements) );
290 
291  // Step 1: ( K.M : Material Stiffness Matrix )
292  // Apply the projector to the Material Stiffness Matrix (Ke = P' * Km * H * P)
293  // At this point 'LHS' contains the 'projected' Material Stiffness matrix
294  // in local corotational coordinate system
295 
296  noalias( temp ) = prod( rLeftHandSideMatrix, H );
297  noalias( rLeftHandSideMatrix ) = prod( temp, P );
298  noalias( temp ) = prod( trans( P ), rLeftHandSideMatrix );
299  rLeftHandSideMatrix.swap( temp );
300 
301  // Step 2: ( K.GP: Equilibrium Projection Geometric Stiffness Matrix )
302  // First assemble the 'Fnm' matrix with the Spins of the nodal forces.
303  // Actually at this point the 'Fnm' Matrix is the 'Fn' Matrix,
304  // because it only contains the spins of the 'translational' forces.
305  // At this point 'LHS' contains also this term of the Geometric stiffness
306  // (Ke = (P' * Km * H * P) - (G' * Fn' * P))
307 
308  MatrixType Fnm(24, 3, 0.0);
309 
310  EICR::Spin_AtRow( projectedLocalForces, Fnm, 0 );
311  EICR::Spin_AtRow( projectedLocalForces, Fnm, 6 );
312  EICR::Spin_AtRow( projectedLocalForces, Fnm, 12 );
313  EICR::Spin_AtRow( projectedLocalForces, Fnm, 18 );
314 
315  noalias( temp ) = prod( trans( G ), trans( Fnm ) );
316  noalias( rLeftHandSideMatrix ) += prod( temp, P ); // note: '+' not '-' because the RHS already has the negative sign
317 
318  // Step 3: ( K.GR: Rotational Geometric Stiffness Matrix )
319  // Add the Spins of the nodal moments to 'Fnm'.
320  // At this point 'LHS' contains also this term of the Geometric stiffness
321  // (Ke = (P' * Km * H * P) - (G' * Fn' * P) - (Fnm * G))
322 
323  EICR::Spin_AtRow( projectedLocalForces, Fnm, 3 );
324  EICR::Spin_AtRow( projectedLocalForces, Fnm, 9 );
325  EICR::Spin_AtRow( projectedLocalForces, Fnm, 15 );
326  EICR::Spin_AtRow( projectedLocalForces, Fnm, 21 );
327 
328  noalias( rLeftHandSideMatrix ) += prod( Fnm, G ); // note: '+' not '-' because the RHS already has the negative sign
329 
330  // Step 4: (Global Stiffness Matrix)
331  // Transform the LHS to the Global coordinate system.
332  // T' * [(P' * Km * H * P) - (G' * Fn' * P) - (Fnm * G)] * T
333  noalias( temp ) = prod( rLeftHandSideMatrix, T );
334  noalias( rLeftHandSideMatrix ) = prod( trans( T ), temp );
335  }

352  MatrixType GetNodalDeformationalRotationTensor(const ShellQ4_LocalCoordinateSystem & LCS,
353  const Vector& globalDisplacements,
354  const Vector& N) override
355  {
356  QuaternionType Q = QuaternionType::FromRotationMatrix( LCS.Orientation() );
357 
358  RealType qx(0.0);
359  RealType qy(0.0);
360  RealType qz(0.0);
361  RealType qw(0.0);
362 
363  for(int i = 0; i < 4; i++)
364  {
365  QuaternionType iQd = Q * mQN[i] * mQ0.conjugate();
366  iQd.normalize();
367  RealType iN = N[i];
368  qx += iN * iQd.X();
369  qy += iN * iQd.Y();
370  qz += iN * iQd.Z();
371  qw += iN * iQd.W();
372  }
373 
374  MatrixType R(3, 3);
375 
376  QuaternionType Qd(qw, qx, qy, qz);
377  Qd.normalize();
378  Qd.ToRotationMatrix(R);
379 
380  return R;
381  }

398  inline MatrixType Compute_G(const Vector3Type & P1,
399  const Vector3Type & P2,
400  const Vector3Type & P3,
401  const Vector3Type & P4,
402  RealType area)
403  {
404  RealType Ap = 2.0 * area;
405  RealType m = 1.0 / Ap;
406 
407  Vector3Type D12( P2 - P1 );
408  Vector3Type D24( P4 - P2 );
409  Vector3Type D13( P3 - P1 );
410 
411  RealType x42 = D24(0);
412  RealType x24 = - x42;
413  RealType y42 = D24(1);
414  RealType y24 = - y42;
415  RealType x31 = D13(0);
416  RealType x13 = - x31;
417  RealType y31 = D13(1);
418  RealType y13 = - y31;
419 
420  // Note, assuming the input vectors are in local CR,
421  // l12 is the length of the side 1-2 projected onto the xy plane.
422  RealType l12 = std::sqrt( D12(0)*D12(0) + D12(1)*D12(1) );
423 
424  MatrixType G(3, 24, 0.0);
425 
426  // G1
427 
428  G(0, 2) = x42 * m;
429  G(1, 2) = y42 * m;
430  G(2, 1) = - 1.0 / l12;
431 
432  // G2
433 
434  G(0, 8) = x13 * m;
435  G(1, 8) = y13 * m;
436  G(2, 7) = 1.0 / l12;
437 
438  // G3
439 
440  G(0, 14) = x24 * m;
441  G(1, 14) = y24 * m;
442 
443  // G4
444 
445  G(0, 20) = x31 * m;
446  G(1, 20) = y31 * m;
447 
448  return G;
449  }

460  inline MatrixType RotationGradient()
461  {
462  Matrix G(3, 24, 0.0);
463 
464  ShellQ4_LocalCoordinateSystem a( CreateReferenceCoordinateSystem() ); // current coordinate system
465 
466  ShellQ4_LocalCoordinateSystem::Vector3ContainerType nodes = a.Nodes();
467 
468  double aX1 = a.X1(); double aY1 = a.Y1();
469  double aX2 = a.X2(); double aY2 = a.Y2();
470  double aX3 = a.X3(); double aY3 = a.Y3();
471  double aX4 = a.X4(); double aY4 = a.Y4();
472 
473  double pert = std::sqrt(a.Area()) * 1.0E-2;
474 
475  Vector3Type e3;
476  Vector3Type e1;
477 
478  for(int i = 0; i < 4; i++) // for each node
479  {
480  Vector3Type& iNode = nodes[i];
481 
482  int index = i * 6;
483 
484  for(int j = 0; j < 3; j++) // for each component in [x, y, z]
485  {
486  double saved_coord = iNode[j]; // save the current coordinate
487 
488  iNode[j] += pert; // apply perturbation
489 
490  ShellQ4_LocalCoordinateSystem b( nodes[0], nodes[1], nodes[2], nodes[3] ); // perturbed coordinate system (1-2 side alignement)
491 
492  double bX1 = b.X1(); double bY1 = b.Y1();
493  double bX2 = b.X2(); double bY2 = b.Y2();
494  double bX3 = b.X3(); double bY3 = b.Y3();
495  double bX4 = b.X4(); double bY4 = b.Y4();
496 
497  double C1 = 1/(aX1*aY2 - aX2*aY1 - aX1*aY4 + aX2*aY3 - aX3*aY2 + aX4*aY1 + aX3*aY4 - aX4*aY3);
498  double C2 = bY1/4 + bY2/4 - bY3/4 - bY4/4;
499  double C3 = bY1/4 - bY2/4 - bY3/4 + bY4/4;
500  double C4 = bX1/4 + bX2/4 - bX3/4 - bX4/4;
501  double C5 = bX1/4 - bX2/4 - bX3/4 + bX4/4;
502  double C6 = aX1 + aX2 - aX3 - aX4;
503  double C7 = aX1 - aX2 - aX3 + aX4;
504  double C8 = aY1 + aY2 - aY3 - aY4;
505  double C9 = aY1 - aY2 - aY3 + aY4;
506  double f11 = 2*C1*C5*C8 - 2*C1*C4*C9;
507  double f12 = 2*C1*C4*C7 - 2*C1*C5*C6;
508  double f21 = 2*C1*C3*C8 - 2*C1*C2*C9;
509  double f22 = 2*C1*C2*C7 - 2*C1*C3*C6;
510 
511  double alpha = std::atan2( f21 - f12, f11 + f22 );
512 
513  ShellQ4_LocalCoordinateSystem c( nodes[0], nodes[1], nodes[2], nodes[3], alpha ); // perturbed coordinate system (polar decomposition)
514 
515  // save the (numerical) rotation gradient
516 
517  noalias( e3 ) = c.Vz();
518  noalias( e1 ) = c.Vx();
519 
520  G(0, index + j) = -e3(1) / pert; // - d(e3.y)/dxj , where xj is x,y,z for j=0,1,2
521  G(1, index + j) = e3(0) / pert; // + d(e3.x)/dxj , where xj is x,y,z for j=0,1,2
522  G(2, index + j) = e1(1) / pert; // + d(e1.y)/dxj , where xj is x,y,z for j=0,1,2
523 
524  iNode[j] = saved_coord; // restore the current coordinate
525  }
526  }
527 
528  return G;
